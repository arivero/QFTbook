\chapter{Maxwell Theory, Constraints, and Gauge Fixing}
\label{ch:maxwell-theory-constraints-gauge-fixing}
\ConventionsLorentzian


\section{Representative Fields and Field Strength}

\begin{definition}[Local Maxwell representative datum]
\label{def:local-maxwell-representative-datum}
Work on four-dimensional Minkowski space with mostly-plus metric. A Maxwell
configuration in this local discussion is represented by a real one-form
\[
  A=A_\mu(x)\,\dd x^\mu .
\]
Two representatives related by
\[
  A_\mu(x)\longmapsto A_\mu(x)+\partial_\mu\xi(x)
\]
describe the same electromagnetic field. Here \(\xi\) is a real scalar
gauge parameter in the same boundary-condition class as \(A_\mu\). We assume
throughout this chapter that fields are either smeared with compactly
supported test functions or obey boundary conditions for which integrations
by parts have no boundary term. Zero modes of the gauge-fixing operators are
then removed separately; this is part of the definition of the local gauge
slice used below.

The local gauge-invariant field strength is
\[
  F_{\mu\nu}
  =
  \partial_\mu A_\nu-\partial_\nu A_\mu .
\]
\end{definition}

\begin{proposition}[Maxwell action and Euler--Lagrange equation]
\label{prop:maxwell-action-euler-lagrange}
The free Maxwell action is
\[
  S[A]
  =
  -\frac14\int\dd^4x\,F_{\mu\nu}F^{\mu\nu}.
\]
Its Euler--Lagrange equation is
\[
  \partial_\mu F^{\mu\nu}=0 .
\]
\end{proposition}

\begin{proof}
The variation of the field strength is
\(\delta F_{\mu\nu}=\partial_\mu\delta A_\nu-\partial_\nu\delta A_\mu\).
Using antisymmetry of \(F^{\mu\nu}\),
\[
  \delta S
  =
  -\frac12\int\dd^4x\,F^{\mu\nu}\delta F_{\mu\nu}
  =
  -\int\dd^4x\,F^{\mu\nu}\partial_\mu\delta A_\nu .
\]
The boundary condition convention of
Definition~\ref{def:local-maxwell-representative-datum} gives
\[
  \delta S
  =
  \int\dd^4x\,(\partial_\mu F^{\mu\nu})\delta A_\nu .
\]
Since \(\delta A_\nu\) is arbitrary within the chosen test-function class, the
Euler--Lagrange equation is \(\partial_\mu F^{\mu\nu}=0\).  The same
calculation shows gauge invariance of \(S[A]\), because
\(\delta_\xi F_{\mu\nu}=0\) for \(\delta_\xi A_\mu=\partial_\mu\xi\).
\end{proof}

With \(x^0=t\) and spatial indices \(i,j=1,2,3\), the Lagrangian density is
\[
  \mathcal L
  =
  \frac12\sum_{i=1}^3(\partial_0A_i-\partial_iA_0)^2
  -
  \frac14\sum_{i,j=1}^3(\partial_iA_j-\partial_jA_i)^2 .
\]
The absence of \(\partial_0A_0\) is the first sign that the vector
potential is a redundant representative rather than four independent
propagating fields.

\section{Canonical Constraint Analysis}

\begin{definition}[Maxwell canonical Poisson structure]
\label{def:maxwell-canonical-poisson-structure}
At fixed time let \(A_\mu(t,\vec x)\) be canonical coordinates. For
functionals \(F,G\) of \(A_\mu,\Pi^\mu\), use the equal-time Poisson bracket
\[
  \{F,G\}_{\mathrm P}
  =
  \int\dd^3\vec x\,
  \left(
    \frac{\delta F}{\delta A_\mu(\vec x)}
    \frac{\delta G}{\delta\Pi^\mu(\vec x)}
    -
    \frac{\delta F}{\delta\Pi^\mu(\vec x)}
    \frac{\delta G}{\delta A_\mu(\vec x)}
  \right).
\]
Equivalently,
\[
  \{A_\mu(\vec x),\Pi^\nu(\vec y)\}_{\mathrm P}
  =
  \delta_\mu{}^\nu\delta^{(3)}(\vec x-\vec y),
\]
with all \(A\)-\(A\) and \(\Pi\)-\(\Pi\) brackets zero.
\end{definition}

\begin{proposition}[Primary constraint, Gauss constraint, and canonical Hamiltonian]
\label{prop:maxwell-primary-gauss-canonical-hamiltonian}
The canonical momenta are
\[
  \Pi^\mu(\vec x)
  =
  \frac{\partial\mathcal L}
       {\partial(\partial_0A_\mu(t,\vec x))},
\]
so
\[
  \Pi^0(\vec x)=0,
  \qquad
  \Pi^i(\vec x)=\partial_0A_i-\partial_iA_0 .
\]
The equation
\[
  \chi_1(\vec x)=\Pi^0(\vec x)=0
\]
is the primary constraint. The canonical Hamiltonian is
\[
  H_{\mathrm c}
  =
  \int\dd^3\vec x\,
  \left[
    \frac12\Pi^i\Pi^i
    +
    \frac14F_{ij}F_{ij}
    -
    A_0\,\partial_i\Pi^i
  \right],
\]
after integrating \(\Pi^i\partial_iA_0\) by parts. Preserving the primary
constraint under time evolution gives
\[
  \dot\Pi^0(\vec x)
  =
  \{\Pi^0(\vec x),H_{\mathrm c}\}_{\mathrm P}
  =
  \partial_i\Pi^i(\vec x),
\]
hence the secondary constraint
\[
  \chi_2(\vec x)=\partial_i\Pi^i(\vec x)=0 .
\]
This is Gauss' law in the absence of charges.
\end{proposition}

\begin{proof}
The Lagrangian density contains \(\partial_0A_i\) only through
\(\partial_0A_i-\partial_iA_0\), and contains no \(\partial_0A_0\).  This
gives the displayed momenta and the primary constraint \(\Pi^0=0\).  Solving
\(\partial_0A_i=\Pi^i+\partial_iA_0\), the Legendre transform is
\[
  \Pi^i\partial_0A_i-\mathcal L
  =
  \frac12\Pi^i\Pi^i+\frac14F_{ij}F_{ij}+\Pi^i\partial_iA_0 .
\]
Integrating the last term by parts gives \(H_{\rm c}\).  Finally,
\[
  \{\Pi^0(\vec x),H_{\rm c}\}_{\rm P}
  =
  -\frac{\delta H_{\rm c}}{\delta A_0(\vec x)}
  =
  \partial_i\Pi^i(\vec x),
\]
so preservation of \(\Pi^0=0\) imposes Gauss' law.
\end{proof}

\begin{proposition}[First-class constraints generate Maxwell gauge transformations]
\label{prop:maxwell-first-class-gauge-generator}
The constraints obey
\[
  \{\chi_a(\vec x),\chi_b(\vec y)\}_{\mathrm P}=0,
  \qquad
  a,b=1,2.
\]
They are first-class constraints. Their relation to gauge redundancy is
direct. For a spacetime function \(\zeta\), define at time \(t\)
\[
  G[\zeta]
  =
  \int\dd^3\vec y\,
  \left[
    (\partial_0\zeta)(t,\vec y)\,\Pi^0(\vec y)
    -
    \zeta(t,\vec y)\,\partial_i\Pi^i(\vec y)
  \right].
\]
Then
\[
  \{A_0(\vec x),G[\zeta]\}_{\mathrm P}
  =
  \partial_0\zeta(t,\vec x),
  \qquad
  \{A_i(\vec x),G[\zeta]\}_{\mathrm P}
  =
  \partial_i\zeta(t,\vec x).
\]
Thus the first-class constraints generate
\[
  \delta_\zeta A_\mu(t,\vec x)=\partial_\mu\zeta(t,\vec x)
\]
on the vector-potential representative. The gauge-invariant phase space is
obtained by restricting to the constraint surface and quotienting by these
gauge orbits, or equivalently by choosing a regular slice through the orbits.
\end{proposition}

\begin{proof}
The constraints depend only on momenta, hence their mutual Poisson brackets
vanish.  For the generator \(G[\zeta]\),
\[
  \{A_0(\vec x),G[\zeta]\}_{\rm P}
  =
  \int\dd^3\vec y\,\partial_0\zeta(t,\vec y)
  \delta^{(3)}(\vec x-\vec y)
  =
  \partial_0\zeta(t,\vec x).
\]
For \(A_i\), integrate the derivative in the Gauss term against the delta
function:
\[
  \{A_i(\vec x),G[\zeta]\}_{\rm P}
  =
  -\int\dd^3\vec y\,\zeta(t,\vec y)
  \partial_i^{\vec y}\delta^{(3)}(\vec x-\vec y)
  =
  \partial_i\zeta(t,\vec x).
\]
The transformed field strength is unchanged because mixed partial
derivatives commute.  Thus the Hamiltonian first-class flow is precisely the
Maxwell representative redundancy.
\end{proof}

\section{Axial Gauge and Reduced Phase Space}

\begin{definition}[Axial gauge local chart]
\label{def:axial-gauge-local-chart}
Axial gauge chooses the representative satisfying
\[
  A_3=0 .
\]
Locally along an orbit,
\[
  A_3\longmapsto A_3+\partial_3\xi,
\]
so this is a regular gauge condition for modes on which \(\partial_3\) is
invertible. Modes with \(\partial_3\xi=0\) are residual gauge parameters for
this slice and must be treated separately by boundary conditions or an
additional gauge condition.
\end{definition}

\begin{proposition}[Axial gauge leaves two canonical degrees of freedom]
\label{prop:axial-gauge-two-canonical-degrees}
There are two equivalent local descriptions. In canonical language the
condition \(A_3=0\) pairs with Gauss' law and allows one to solve
\[
  \partial_3\Pi^3=-\partial_a\Pi^a,
  \qquad a=1,2,
\]
for the longitudinal momentum \(\Pi^3\), again modulo \(\partial_3\) zero
modes. In Lagrangian language one imposes \(A_3=0\) before eliminating the
non-dynamical field \(A_0\). The gauge-fixed Lagrangian density is
\[
\begin{aligned}
  \mathcal L_{\mathrm{ax}}
  &=
  \frac12\sum_{a=1}^2(\partial_0A_a-\partial_aA_0)^2
  +
  \frac12(\partial_3A_0)^2  \\
  &\quad
  -
  \frac12(\partial_1A_2-\partial_2A_1)^2
  -
  \frac12\sum_{a=1}^2(\partial_3A_a)^2 .
\end{aligned}
\]
The \(A_0\) equation is
\[
  \sum_{a=1}^2\partial_a(\partial_aA_0-\partial_0A_a)
  +\partial_3^2A_0=0.
\]
With
\[
  \Pi^a=\partial_0A_a-\partial_aA_0,
\]
this becomes
\[
  \sum_{a=1}^2\partial_a\Pi^a+\partial_3^2A_0=0.
\]
The pair
\[
  \Pi^0(\vec x)=0,
  \qquad
  \sum_{a=1}^2\partial_a\Pi^a(\vec x)+\partial_3^2A_0(\vec x)=0
\]
is second-class for nonzero \(k_3\), since
\[
  \left\{
    \Pi^0(\vec x),
    \sum_{a=1}^2\partial_a\Pi^a(\vec y)+\partial_3^2A_0(\vec y)
  \right\}_{\mathrm P}
  =
  \partial_3^2\delta^{(3)}(\vec x-\vec y).
\]
Hence one may either introduce the corresponding Dirac bracket or solve
\[
  \partial_3^2A_0
  =
  -\sum_{a=1}^2\partial_a\Pi^a
\]
and eliminate \(A_0,\Pi^0\). For the regular modes of the axial slice the
remaining canonical variables are
\[
  A_a(\vec x),\Pi^a(\vec x),
  \qquad a=1,2.
\]
They are the two canonical degrees of freedom of the free photon.
\end{proposition}

\begin{proof}
On modes where \(\partial_3\) is invertible, Gauss' law gives
\(\Pi^3=-\partial_3^{-1}\partial_a\Pi^a\), so \((A_3,\Pi^3)\) is removed
after imposing the axial condition and the constraint.  In the Lagrangian
description, \(A_0\) has no time derivative.  Its equation of motion is the
displayed elliptic equation along the \(x^3\)-direction once \(A_3=0\) is
imposed.  The Poisson bracket
\[
  \{\Pi^0(\vec x),\partial_3^2A_0(\vec y)\}_{\rm P}
  =
  \partial_3^2\delta^{(3)}(\vec x-\vec y)
\]
is invertible on the same nonzero-\(k_3\) modes, so
\((A_0,\Pi^0)\) is a second-class pair there and may be eliminated.  The only
regular canonical variables left are \(A_a,\Pi^a\), \(a=1,2\).
\end{proof}

\section{Gauge Slices and the Faddeev--Popov Factor}

\begin{definition}[Local Faddeev--Popov chart]
\label{def:local-faddeev-popov-chart-maxwell}
Before gauge fixing, the path-integral shorthand
\[
  Z_{\mathrm{unfixed}}
  =
  \int DA_\mu\,\ee^{\ii S[A]}
\]
integrates over all representatives on each gauge orbit. Let
\(\mathcal G(A)\) be a gauge condition. Locally in field space, and after
removing zero modes, the Faddeev--Popov identity is
\[
  1
  =
  \Delta_{\mathcal G}[A]
  \int D\xi\,\delta\bigl(\mathcal G(A+\partial\xi)\bigr),
\]
where
\[
  \Delta_{\mathcal G}[A]
  =
  \det M_{\mathcal G}[A],
  \qquad
  M_{\mathcal G}[A]\xi
  =
  \left.
  \frac{\dd}{\dd s}
  \mathcal G(A+s\,\partial\xi)
  \right|_{s=0}.
\]
\end{definition}

\paragraph{Abelian linear FP determinant.}
For abelian Maxwell theory with a linear gauge condition, \(M_{\mathcal G}\)
is independent of \(A_\mu\). Its determinant is then an overall
normalization in the free theory. This special simplification is not a
definition of gauge fixing; it is a property of the abelian linear examples
used in this chapter. The corresponding nonabelian construction is developed
later as a local coordinate system on gauge orbits. There the analogue of
\(M_{\mathcal G}\) depends on the gauge potential, can acquire zero modes, and
therefore cannot be promoted to a single global Faddeev--Popov chart without
additional geometric data.
If \(\mathcal G\) is linear, then
\[
  \mathcal G(A+\partial\xi)=\mathcal G(A)+\mathcal G(\partial\xi).
\]
Therefore the derivative with respect to \(s\) at \(s=0\) is simply
\(M_{\mathcal G}\xi=\mathcal G(\partial\xi)\), independent of \(A\).  The
determinant is consequently independent of the integration variable in the
free abelian representative integral and factors from normalized correlators.

\begin{example}[Finite-dimensional gauge-slice independence model]
\label{prop:finite-dimensional-fp-slice-independence-model}
A finite-dimensional model isolates the logic. Let
\[
  S(x,y)=(x-y)^2
\]
on \(\R^2\), with gauge orbits
\[
  (x,y)\longmapsto(x+\alpha,y+\alpha).
\]
The ungauged Gaussian integral diverges by the orbit volume. Choose the
linear slice
\[
  \ell(x,y)=y+ax+b=0,
  \qquad a\ne -1.
\]
The Faddeev--Popov factor is the absolute value of the derivative of
\(\ell(x+\alpha,y+\alpha)\) with respect to \(\alpha\),
\[
  \Delta_\ell=|1+a|.
\]
For a gauge-invariant observable \(O(x,y)=O(x+\alpha,y+\alpha)\),
\[
  \frac{
    \int\dd x\,\Delta_\ell\,
    \ee^{-(x+ax+b)^2}O(x,-ax-b)
  }{
    \int\dd x\,\Delta_\ell\,
    \ee^{-(x+ax+b)^2}
  }
\]
is independent of \(a,b\). This is the finite-dimensional form of
gauge-slice independence for gauge-invariant quantities in a region where the
slice is regular.
On the slice \(y=-ax-b\), the gauge-invariant coordinate is
\[
  u=x-y=(1+a)x+b .
\]
The measure changes by \(\dd u=|1+a|\dd x\).  The factor
\(\Delta_\ell=|1+a|\) therefore cancels the Jacobian in both numerator and
denominator.  Because \(O\) is constant on gauge orbits, \(O(x,-ax-b)\)
depends only on \(u\).  Both normalized integrals are thus the same Gaussian
average in \(u\), independent of \(a,b\).
\end{example}

\section{Axial-Gauge Gaussian and Propagator}

\paragraph{Axial-gauge quadratic inverse.}
For axial gauge,
\[
  \mathcal G_{\mathrm{ax}}(A)=A_3,
  \qquad
  M_{\mathrm{ax}}=\partial_3 .
\]
After removing the \(\partial_3\) zero modes, the determinant is independent
of \(A_\mu\), and the gauge-fixed Maxwell integral may be written as
\[
  Z_{\mathrm{ax}}
  =
  \int DA_\mu\,\delta(A_3)\,\ee^{\ii S[A]} .
\]
Integrating the Maxwell action by parts gives
\[
  S[A]
  =
  \frac12\int\dd^4x\,
  A_\mu
  \left(\eta^{\mu\nu}\Box-\partial^\mu\partial^\nu\right)
  A_\nu ,
  \qquad
  \Box=\partial^\rho\partial_\rho .
\]
The unfixed quadratic operator has zero directions \(A_\mu=\partial_\mu\xi\).
On the axial slice \(A_3=0\), the remaining block has indices
\(\mu,\nu=0,1,2\). With
\[
  A_\mu(x)
  =
  \int\frac{\dd^4k}{(2\pi)^4}\,
  \ee^{\ii k\cdot x}\widetilde A_\mu(k),
\]
the quadratic form is
\[
  S_{\mathrm{ax}}
  =
  \frac12
  \int\frac{\dd^4k}{(2\pi)^4}\,
  \widetilde A_\mu(-k)\,
  \widetilde D_{\mathrm{ax}}^{\mu\nu}(k)\,
  \widetilde A_\nu(k),
\]
where
\[
  \widetilde D_{\mathrm{ax}}^{\mu\nu}(k)
  =
  -\eta^{\mu\nu}k^2+k^\mu k^\nu,
  \qquad
  \mu,\nu=0,1,2.
\]
For \(k^2\ne0\) and \(k_3\ne0\), its inverse on this \(3\times3\) block is
\[
  \bigl(\widetilde D_{\mathrm{ax}}(k)^{-1}\bigr)_{\mu\nu}
  =
  -\frac{1}{k^2}
  \left(
    \eta_{\mu\nu}+\frac{k_\mu k_\nu}{k_3^2}
  \right),
  \qquad
  \mu,\nu=0,1,2.
\]

The abelian Faddeev--Popov operator is
\(M_{\rm ax}\xi=\partial_3\xi\), so its determinant is field independent
after the \(\partial_3\) zero modes are removed.  Integration by parts gives
the displayed quadratic operator.  On the axial slice \(A_3=0\), only the
\(\mu,\nu=0,1,2\) block remains.  Multiplying the proposed inverse by the
quadratic block gives
\[
\begin{aligned}
  &\sum_{\rho=0}^2
  \bigl(-\eta^{\mu\rho}k^2+k^\mu k^\rho\bigr)
  \left[
    -\frac1{k^2}
    \left(\eta_{\rho\nu}+\frac{k_\rho k_\nu}{k_3^2}\right)
  \right]  \\
  &\qquad
  =
  \delta^\mu{}_\nu
  -
  \frac{k^\mu k_\nu}{k^2}
  +
  \frac{k^\mu k_\nu}{k_3^2}
  -
  \frac{k^\mu k_\nu}{k^2k_3^2}
  \sum_{\rho=0}^2 k^\rho k_\rho
  =
  \delta^\mu{}_\nu,
\end{aligned}
\]
because
\[
  \sum_{\rho=0}^2 k^\rho k_\rho=k^2-k_3^2 .
\]

\begin{corollary}[Axial-gauge representative propagator and field strength]
\label{cor:axial-gauge-field-strength-propagator}
The Lorentzian Feynman boundary value therefore gives, for
\(\mu,\nu=0,1,2\),
\[
\begin{aligned}
  &\bra{\vac}
  \operatorname T\{\widehat A_\mu(x)\widehat A_\nu(y)\}
  \ket{\vac}_{\mathrm{ax}}  \\
  &=
  \int\frac{\dd^4k}{(2\pi)^4}\,
  \ee^{\ii k\cdot(x-y)}
  \frac{-\ii}{k^2-\ii\epsilon}
  \left(
    \eta_{\mu\nu}+\frac{k_\mu k_\nu}{k_3^2}
  \right),
\end{aligned}
\]
with components involving \(A_3\) set to zero in this gauge. The displayed
kernel is singular at \(k_3=0\), precisely where the axial slice has residual
gauge directions.

The field-strength two-point function is obtained by antisymmetrizing
momentum factors. Away from coincident points, equivalently modulo local
contact terms fixed by the definition of time ordering, the \(k_\mu k_\nu\)
part of the axial propagator cancels:
\[
\begin{aligned}
  &\bra{\vac}
  \operatorname T\{
    \widehat F_{\mu\nu}(x)\widehat F_{\rho\sigma}(y)
  \}
  \ket{\vac}  \\
  &=
  \int\frac{\dd^4k}{(2\pi)^4}\,
  \ee^{\ii k\cdot(x-y)}
  \frac{-\ii}{k^2-\ii\epsilon}
  \bigl(
    k_\mu k_\rho\eta_{\nu\sigma}
    -k_\nu k_\rho\eta_{\mu\sigma}
    -k_\mu k_\sigma\eta_{\nu\rho}
    +k_\nu k_\sigma\eta_{\mu\rho}
  \bigr).
\end{aligned}
\]
Thus the local gauge-invariant correlator has a Lorentz-covariant kernel
although the axial representative propagator does not.
\end{corollary}

\section{Covariant Gauges}

\begin{definition}[Lorenz gauge and covariant gauge parameter]
\label{def:lorenz-gauge-covariant-parameter}
For perturbative calculations it is often useful to preserve manifest
Lorentz covariance in the representative propagator. The Lorenz gauge
condition is
\[
  \mathcal G_{\mathrm L}(A)=\partial_\mu A^\mu .
\]
Along a gauge orbit,
\[
  \partial_\mu(A^\mu+\partial^\mu\xi)
  =
  \partial_\mu A^\mu+\Box\xi,
\]
so the abelian Faddeev--Popov operator is \(M_{\mathrm L}=\Box\). In the
free abelian theory its determinant is field independent.
\end{definition}

\begin{remark}[Gaussian averaging of the Lorenz gauge condition]
\label{prop:lorenz-gauge-gaussian-averaging}
Instead of imposing \(\partial_\mu A^\mu=0\) sharply, introduce an auxiliary
function \(\varphi\) and average the condition
\[
  \partial_\mu A^\mu=\varphi
\]
with a Gaussian weight. In Euclidean signature the insertion is
\[
  \int D\varphi\,
  \exp\left[
    -\frac{1}{2\xi_{\mathrm g}}
    \int\dd^4x_E\,\varphi^2
  \right]
  \delta(\partial_\mu A^\mu-\varphi),
  \qquad
  \xi_{\mathrm g}>0.
\]
Equivalently, writing the delta-functional with a real Lagrange multiplier
\(\lambda\),
\[
\begin{aligned}
  &\int D\varphi\,D\lambda\,
  \exp\left[
    -\frac{1}{2\xi_{\mathrm g}}\int\varphi^2
    +
    \ii\int\lambda(\partial_\mu A^\mu-\varphi)
  \right]  \\
  &\qquad
  \propto
  \exp\left[
    -\frac{1}{2\xi_{\mathrm g}}
    \int(\partial_\mu A^\mu)^2
  \right].
\end{aligned}
\]
The Euclidean gauge-fixing term is therefore
\[
  \Delta S_E
  =
  \frac{1}{2\xi_{\mathrm g}}
  \int\dd^4x_E\,(\partial_\mu A^\mu)^2 .
\]
The delta-functional sets \(\varphi=\partial_\mu A^\mu\).  Integrating first
over \(\lambda\) implements this constraint, and the remaining Gaussian
weight in \(\varphi\) becomes exactly the displayed quadratic term in
\(\partial_\mu A^\mu\).  Conversely, integrating first over \(\varphi\)
completes the square and produces the same local quadratic functional of the
Lagrange multiplier; the two orders express the same finite-mode Gaussian
identity before the regulator is removed.
\end{remark}

\paragraph{Covariant-gauge quadratic inverse.}
The corresponding Lorentzian gauge-fixed action is
\[
  S_{\xi_{\mathrm g}}[A]
  =
  \int\dd^4x\,
  \left[
    -\frac14F_{\mu\nu}F^{\mu\nu}
    -
    \frac{1}{2\xi_{\mathrm g}}(\partial_\mu A^\mu)^2
  \right].
\]
After integration by parts,
\[
  S_{\xi_{\mathrm g}}[A]
  =
  \frac12\int\dd^4x\,
  A_\mu
  \left[
    \eta^{\mu\nu}\Box
    -
    \left(1-\frac1{\xi_{\mathrm g}}\right)
    \partial^\mu\partial^\nu
  \right]
  A_\nu .
\]

In momentum space,
\[
  \widetilde D_{\xi_{\mathrm g}}^{\mu\nu}(k)
  =
  -\eta^{\mu\nu}k^2
  +
  \left(1-\frac1{\xi_{\mathrm g}}\right)k^\mu k^\nu .
\]
For \(\xi_{\mathrm g}\ne0\), its inverse is
\[
  \bigl(\widetilde D_{\xi_{\mathrm g}}^{-1}(k)\bigr)_{\mu\nu}
  =
  -\frac{\eta_{\mu\nu}}{k^2}
  +
  (1-\xi_{\mathrm g})\frac{k_\mu k_\nu}{(k^2)^2}.
\]

The gauge-fixed action is obtained from the Maxwell action and the
Lorentzian continuation of the Euclidean term in
Proposition~\ref{prop:lorenz-gauge-gaussian-averaging}.  Integrating by
parts gives the displayed quadratic operator.  The inverse is verified by
\[
\begin{aligned}
  &\left[
    -\eta^{\mu\rho}k^2
    +
    \left(1-\frac1{\xi_{\mathrm g}}\right)k^\mu k^\rho
  \right]
  \left[
    -\frac{\eta_{\rho\nu}}{k^2}
    +
    (1-\xi_{\mathrm g})\frac{k_\rho k_\nu}{(k^2)^2}
  \right]  \\
  &\qquad
  =
  \delta^\mu{}_\nu .
\end{aligned}
\]

\begin{corollary}[Covariant representative propagator]
\label{cor:covariant-representative-propagator}
The Feynman two-point function of the representative field is
\[
\begin{aligned}
  &\bra{\vac}
  \operatorname T\{\widehat A_\mu(x)\widehat A_\nu(y)\}
  \ket{\vac}_{\xi_{\mathrm g}}  \\
  &=
  \int\frac{\dd^4k}{(2\pi)^4}
  \ee^{\ii k\cdot(x-y)}
  \frac{-\ii}{k^2-\ii\epsilon}
  \left[
    \eta_{\mu\nu}
    -
    (1-\xi_{\mathrm g})
    \frac{k_\mu k_\nu}{k^2-\ii\epsilon}
  \right].
\end{aligned}
\]
The value \(\xi_{\mathrm g}=1\) is Feynman gauge:
\[
  \begin{tikzpicture}[baseline=-0.5ex,scale=0.75]
    \draw[thick,decorate,decoration={snake,amplitude=1.2pt,segment length=5pt}]
      (0,0)--(2.1,0);
    \node[blue!70!black] at (0.15,0.28) {\(\mu\)};
    \node[blue!70!black] at (1.95,0.28) {\(\nu\)};
    \draw[->,blue!70!black] (1.55,0.35)--(0.75,0.35);
    \node[blue!70!black] at (1.1,0.62) {\(k\)};
  \end{tikzpicture}
  \quad
  =
  \frac{-\ii\eta_{\mu\nu}}{k^2-\ii\epsilon}.
\]
The value \(\xi_{\mathrm g}\to0\) is Landau gauge.
\end{corollary}

\section{Gupta--Bleuler Fock Space in Lorenz Gauge}

\begin{definition}[Covariant Maxwell Krein-Fock representative]
\label{def:covariant-maxwell-krein-fock-representative}
The covariant representative propagator has a direct operator realization in
a Krein-Fock space.  The construction keeps Lorentz covariance manifest at
the level of representative fields and recovers the positive physical photon
space by a null quotient.

Let
\[
  \dd\mu_0(k)
  =
  \frac{\dd^3\vec k}{(2\pi)^3\,2|\vec k|},
  \qquad
  k^0=|\vec k|,
\]
be the positive-energy massless measure.  Define the one-particle covariant
test space \(\mathcal K_1^0\) to consist of smooth rapidly decreasing
complex vector wavefunctions \(f_\mu(k)\) on the forward light cone.  It is
equipped with the indefinite Hermitian form
\[
  [f,g]_1
  =
  \int \dd\mu_0(k)\,
  \eta^{\mu\nu}\overline{f_\mu(k)}\,g_\nu(k).
\]
The time-like component contributes with the opposite sign because
\(\eta^{00}=-1\).  Choosing a time-like polarization and three space-like
polarizations gives a fundamental symmetry and hence a Hilbert majorant
topology; the completion is a Krein space \(\mathcal K_1\).  The covariant
Fock space is the symmetric Krein-Fock space
\[
  \mathcal K=\mathcal F_s(\mathcal K_1).
\]
On the algebraic finite-particle domain define annihilation and creation
operator-valued distributions by
\[
  [a_\mu(\vec k),a_\nu^\dagger(\vec q)]
  =
  (2\pi)^3\,2|\vec k|\,\eta_{\mu\nu}\,
  \delta^{(3)}(\vec k-\vec q),
  \qquad
  a_\mu(\vec k)\Omega=0 .
\]
The Lorenz-gauge representative field in Feynman gauge is
\[
  A_\mu(x)
  =
  \int\dd\mu_0(k)\,
  \left[
    a_\mu(\vec k)\ee^{-\ii k\cdot x}
    +
    a_\mu^\dagger(\vec k)\ee^{\ii k\cdot x}
  \right].
\]
It satisfies the free wave equation \(\Box A_\mu=0\) and has two-point
function
\[
  \langle\Omega|
  \operatorname T A_\mu(x)A_\nu(y)
  |\Omega\rangle
  =
  \int\frac{\dd^4k}{(2\pi)^4}\,
  \ee^{\ii k\cdot(x-y)}
  \frac{-\ii\eta_{\mu\nu}}{k^2-\ii0},
\]
which is the \(\xi_{\mathrm g}=1\) propagator above.  The price of this
covariant representative is the indefinite form on \(\mathcal K\).
\end{definition}

\begin{definition}[Gupta--Bleuler physical subspace]
\label{def:gupta-bleuler-physical-subspace}
Write
\[
  B(x)=\partial^\mu A_\mu(x),
  \qquad
  B^{(+)}(x)
  =
  -\ii\int\dd\mu_0(k)\,
  k^\mu a_\mu(\vec k)\ee^{-\ii k\cdot x}.
\]
The Gupta--Bleuler physical subspace is
\[
  \mathcal K_{\rm GB}
  =
  \{\Psi\in\mathcal K:\ B^{(+)}(f)\Psi=0
  \text{ for every test function } f\}.
\]
Equivalently, on wavefunctions this condition is the mass-shell Lorenz
condition
\[
  k^\mu f_\mu(k)=0 .
\]
\end{definition}

\begin{remark}[Weak Lorenz condition]
\label{rem:gupta-bleuler-weak-lorenz-condition}
If \(\Phi,\Psi\in\mathcal K_{\rm GB}\), then
\[
  \langle\Phi|B(x)|\Psi\rangle=0 .
\]
Indeed \(B^{(+)}(x)\Psi=0\) by the definition of
\(\mathcal K_{\rm GB}\), while \(B^{(-)}(x)\) is the Krein adjoint of
\(B^{(+)}(x)\), so \(\langle\Phi|B^{(-)}(x)=0\) for
\(\Phi\in\mathcal K_{\rm GB}\).  Since \(B=B^{(+)}+B^{(-)}\), the Lorenz
condition is imposed weakly, as a condition on matrix elements between
physical vectors.
\end{remark}

\begin{construction}[One-particle Gupta--Bleuler quotient]
\label{cons:gupta-bleuler-one-particle-quotient}
The quotient that removes null gauge vectors can be described already at
one-particle level.  Let
\[
  \mathcal K_{1,{\rm GB}}
  =
  \{f\in\mathcal K_1:\ k^\mu f_\mu(k)=0\}.
\]
The subspace
\[
  \mathcal N_1
  =
  \{f_\mu(k)=k_\mu h(k)\}
\]
lies in \(\mathcal K_{1,{\rm GB}}\) and is null:
\[
  [kh,kh']_1
  =
  \int\dd\mu_0(k)\,k^2\,\overline{h(k)}h'(k)=0 .
\]
It is also orthogonal to every \(f\in\mathcal K_{1,{\rm GB}}\), since
\[
  [kh,f]_1
  =
  \int\dd\mu_0(k)\,\overline{h(k)}\,k^\mu f_\mu(k)=0 .
\]
For each null \(k\), the quotient \(k^\perp/\mathbb C k\) is a two-dimensional
positive Hermitian space.  Choosing two transverse polarization
representatives \(\epsilon_\mu^h(k)\), \(h=\pm1\), gives the isometry
\[
  [f]\longmapsto
  f_h(k)=\epsilon^{h\mu}(k)f_\mu(k)
\]
from
\[
  \mathcal H_1^\gamma
  :=
  \mathcal K_{1,{\rm GB}}/\mathcal N_1
\]
onto the two-helicity one-photon Hilbert space
\[
  \bigoplus_{h=\pm1}L^2(\mathcal N_+,\dd\mu_0).
\]
The positive inner product is
\[
  \langle[f],[g]\rangle_{\gamma}
  =
  \sum_{h=\pm1}
  \int\dd\mu_0(k)\,
  \overline{f_h(k)}g_h(k).
\]
\end{construction}

\begin{proposition}[Gupta--Bleuler quotient as the two-helicity photon Fock space]
\label{prop:gupta-bleuler-quotient-two-helicity-fock}
With the one-particle quotient \(\mathcal H_1^\gamma\) constructed in
Construction~\ref{cons:gupta-bleuler-one-particle-quotient}, the many-particle
physical Hilbert space is
\[
  \Hilb_{\rm GB}
  =
  \mathcal K_{\rm GB}/\mathcal N
  \simeq
  \mathcal F_s(\mathcal H_1^\gamma),
\]
where \(\mathcal N\) is the closed subspace generated by finite-particle
vectors with at least one one-particle factor in \(\mathcal N_1\).
Equivalently, on every \(n\)-particle sector,
\[
  \operatorname{Sym}^n(\mathcal K_{1,{\rm GB}})/
  \sum_{r=1}^n
  \operatorname{Sym}^{r-1}(\mathcal K_{1,{\rm GB}})
  \odot\mathcal N_1\odot
  \operatorname{Sym}^{n-r}(\mathcal K_{1,{\rm GB}})
  \cong
  \operatorname{Sym}^n(\mathcal H_1^\gamma).
\]
Thus Gupta--Bleuler quantization and reduced two-helicity quantization give
the same positive free-photon Hilbert space after quotienting null vectors.
\end{proposition}

\begin{proof}
Construction~\ref{cons:gupta-bleuler-one-particle-quotient} identifies
\(\mathcal K_{1,{\rm GB}}/\mathcal N_1\) with the two-helicity one-photon
Hilbert space.  The symmetric tensor functor sends a quotient by a subspace
to the quotient of each symmetric tensor power by the sum of tensors with one
factor in that subspace:
\[
  \operatorname{Sym}^n(V/N)
  \cong
  \operatorname{Sym}^n(V)/
  \sum_{r=1}^n
  \operatorname{Sym}^{r-1}(V)\odot N\odot\operatorname{Sym}^{n-r}(V).
\]
Taking \(V=\mathcal K_{1,{\rm GB}}\) and \(N=\mathcal N_1\) gives the
displayed \(n\)-particle identification.  The finite-particle direct sum
therefore quotients to the symmetric Fock space over
\(\mathcal H_1^\gamma\), and completion gives
\(\mathcal F_s(\mathcal H_1^\gamma)\).
\end{proof}

\begin{proposition}[Field strengths descend to the Gupta--Bleuler quotient]
\label{prop:field-strength-descends-gupta-bleuler}
Let \(u_{\mu\nu}(x)\) be a compactly supported test two-form and smear the
field strength:
\[
  F(u)
  =
  \int\dd^4x\,u^{\mu\nu}(x)
  \bigl(\partial_\mu A_\nu-\partial_\nu A_\mu\bigr)(x).
\]
Then \(F(u)\Omega\) lies in \(\mathcal K_{1,{\rm GB}}\), and \(F(u)\)
descends to a well-defined operator on
\(\Hilb_{\rm GB}\simeq\mathcal F_s(\mathcal H_1^\gamma)\).  Every free
time-ordered correlator of field strengths computed in the covariant
Gupta--Bleuler Fock representation agrees with the corresponding correlator
computed in the reduced positive two-helicity Fock representation.
\end{proposition}

\begin{proof}
The one-particle wavefunction of \(F(u)\Omega\) is proportional on the light
cone to
\[
  f_\nu(k)
  =
  2\ii\,k_\mu\widetilde u^{\mu}{}_{\nu}(k),
  \qquad
  k^\nu f_\nu(k)=0,
\]
because \(k^\mu k^\nu\widetilde u_{\mu\nu}=0\).  Therefore \(F(u)\Omega\)
lies in \(\mathcal K_{1,{\rm GB}}\).  If the representative field is shifted
by \(A_\mu\mapsto A_\mu+\partial_\mu\xi\), the field-strength smearing
changes by zero; at the one-particle level this is the statement that adding
\(k_\mu h(k)\) changes only the null class.  Consequently field-strength
operators descend to well-defined operators on
\(\Hilb_{\rm GB}\simeq\mathcal F_s(\mathcal H_1^\gamma)\).

The equality of observable correlators follows by Wick's theorem from the
two-point statement.  The Feynman-gauge vector propagator and the
two-helicity time-ordered vector kernel differ by terms whose external vector
indices are longitudinal on the light cone, together with local contact terms
fixed by the time-ordering prescription.  Applying the antisymmetric
derivatives that form \(F_{\mu\nu}\) kills the longitudinal terms.  Hence
every free correlator of field strengths computed in the covariant
Gupta--Bleuler Fock representation agrees with the correlator computed in the
reduced positive two-helicity Fock representation.  The same cancellation
explains why the \(\xi_{\mathrm g}\)-dependent term in the covariant
path-integral propagator drops out of gauge-invariant field-strength
correlators.
\end{proof}

\section{Field Strength and One-Photon States}

\begin{proposition}[Field-strength two-point function and photon helicity data]
\label{prop:maxwell-field-strength-one-photon-spectral-data}
The gauge-fixed path integral defines Green functions of the representative
variable \(A_\mu\). In a positive-metric physical Hilbert space the local
observable obtained from the free Maxwell field is the field strength
\(\widehat F_{\mu\nu}\). The field-strength two-point function is independent
of \(\xi_{\mathrm g}\), because the gauge-parameter term in the representative
propagator is proportional to \(k_\mu k_\nu\) and is killed by
antisymmetrization.

The Wightman two-point function has the one-photon spectral form
\[
\begin{aligned}
  &\bra{\vac}
  \widehat F_{\mu\nu}(x)\widehat F_{\rho\sigma}(y)
  \ket{\vac}  \\
  &=
  \sum_{h=\pm1}
  \int
  \frac{\dd^3\vec k}{(2\pi)^3\,2|\vec k|}
  \bra{\vac}\widehat F_{\mu\nu}(x)|k,h\rangle
  \langle k,h|\widehat F_{\rho\sigma}(y)\ket{\vac} ,
\end{aligned}
\]
where the sharp states are generalized one-particle vectors in the sense of
Section~\ref{sec:rigged-hilbert-continuous-spectrum}. With the helicity
representatives of
Chapter~\ref{ch:massless-particles-helicity-gauge-redundancy},
\[
  \langle k,h|\widehat F_{\mu\nu}(x)\ket{\vac}
  =
  Z_A^{1/2}\ee^{-\ii k\cdot x}
  \left(
    -\ii k_\mu\epsilon_\nu^{h*}(k)
    +\ii k_\nu\epsilon_\mu^{h*}(k)
  \right).
\]
The representative shift
\[
  \epsilon_\mu^h(k)\longmapsto \epsilon_\mu^h(k)+c(k)k_\mu
\]
does not change this matrix element.

For a fixed null momentum \(k\), choose a null covector \(C_\mu(k)\) satisfying
\[
  C^2=0,
  \qquad
  k^\mu C_\mu=-1.
\]
The two transverse helicity representatives may be normalized so that
\[
  \epsilon_\mu^h(k)\epsilon^{h'*\mu}(k)=\delta_{hh'},
  \qquad
  k^\mu\epsilon_\mu^h(k)=0,
\]
and obey the completeness relation
\[
  \sum_{h=\pm1}
  \epsilon_\mu^h(k)\epsilon_\nu^{h*}(k)
  =
  \eta_{\mu\nu}+k_\mu C_\nu+k_\nu C_\mu .
\]
For example, when \(k^\mu=(|\vec k|,\vec k)\), one may take
\[
  C^\mu(k)
  =
  \frac{1}{2|\vec k|^2}\bigl(|\vec k|,-\vec k\bigr).
\]
In the field-strength spectral sum all terms involving \(C_\mu\) cancel by
antisymmetrization. Comparison with the explicit Wightman boundary value of
the free field-strength correlator fixes the free normalization
\[
  Z_A=1.
\]
\end{proposition}

\begin{proof}
The \(\xi_{\mathrm g}\)-dependent term of the representative propagator is
proportional to \(k_\alpha k_\beta\).  Applying the two field-strength
antisymmetrizers gives the factor
\[
  (k_\mu\delta_\nu{}^\alpha-k_\nu\delta_\mu{}^\alpha)
  k_\alpha k_\beta
  (k_\rho\delta_\sigma{}^\beta-k_\sigma\delta_\rho{}^\beta),
\]
which vanishes because \(k_\mu k_\nu-k_\nu k_\mu=0\).  Thus the
gauge-parameter part of the representative two-point function is invisible
to \(F_{\mu\nu}\).

The spectral representation is the one-particle part of the Wightman
two-point function for the positive photon Hilbert space constructed in
Proposition~\ref{prop:gupta-bleuler-quotient-two-helicity-fock}; sharp
\(|k,h\rangle\) are used as generalized vectors in the rigged-Hilbert-space
sense.  The matrix element of the field strength is obtained by applying
\(\partial_\mu A_\nu-\partial_\nu A_\mu\) to a one-photon vector-potential
matrix element.  The replacement
\(\epsilon_\mu^h\mapsto\epsilon_\mu^h+c\,k_\mu\) changes the parent vector
potential representative, but it contributes
\[
  -\ii c\,k_\mu k_\nu+\ii c\,k_\nu k_\mu=0
\]
to the field-strength matrix element.

For the completeness relation, the tensor
\(\eta_{\mu\nu}+k_\mu C_\nu+k_\nu C_\mu\) annihilates \(k^\nu\), has zero
components along the null complement modulo the line \(\mathbb Ck\), and
restricts to the positive Euclidean metric on the two-dimensional transverse
quotient \(k^\perp/\mathbb Ck\).  Hence it is the sum of the two normalized
helicity projectors.  Substituting this completeness relation into the
field-strength spectral sum, every term containing \(C\) has two symmetric
factors of \(k\) entering one of the antisymmetrizers, so those terms vanish.
The remaining \(\eta_{\mu\nu}\) term is exactly the explicit free
field-strength Wightman kernel obtained from the Feynman-gauge representative
propagator.  With the normalization of the photon creation and annihilation
operators in Definition~\ref{def:covariant-maxwell-krein-fock-representative},
this comparison gives \(Z_A=1\).
\end{proof}

The next chapter couples this representative gauge field to a charged spinor
field. The perturbative Green functions will be written using gauge-fixed
representatives, while local gauge-invariant operators and external photon
states are read through the field strength and helicity data constructed
here.
