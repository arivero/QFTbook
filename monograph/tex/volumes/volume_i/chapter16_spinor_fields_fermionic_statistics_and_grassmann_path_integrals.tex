\section{The Free Dirac Field}

\begin{definition}[Spinor convention and massive intertwiners]
\label{def:free-dirac-spinor-intertwiner-data}
Work in \(D=4\) with
\[
  \eta_{\mu\nu}=\operatorname{diag}(-,+,+,+),
  \qquad
  \{\gamma^\mu,\gamma^\nu\}=2\eta^{\mu\nu}.
\]
The convention is the one fixed globally in
Section~\ref{sec:spinor-conventions}.  In particular, the Dirac adjoint,
chirality matrix, Majorana conjugation, and later anomaly traces are not
independent sign choices.
Set
\[
  \not\partial=\gamma^\mu\partial_\mu,
  \qquad
  \not p=p_\mu\gamma^\mu,
  \qquad
  \beta=\ii\gamma^0 .
\]
For a spinor \(s\), its Dirac adjoint is
\[
  \bar s=s^\dagger\beta .
\]

Let \(u^\sigma(p)\) and \(v^\sigma(p)\), \(\sigma=1,2\), be the spinor
intertwiners on the positive-energy mass shell \(\Sigma_m^+\) constructed in
Chapter~\ref{ch:massive-particles-spin}:
\[
  (\not p-\ii m)u^\sigma(p)=0,
  \qquad
  (\not p+\ii m)v^\sigma(p)=0.
\]
The delta-normalized spinors used below inherit the normalization
\[
  \bar u^\sigma(p)u^\tau(p)=\frac{m}{\omega_{\vec p}}\delta^{\sigma\tau},
  \qquad
  \bar v^\sigma(p)v^\tau(p)=-\frac{m}{\omega_{\vec p}}\delta^{\sigma\tau},
  \qquad
  \bar u^\sigma(p)v^\tau(p)=0,
\]
where \(\bar u=u^\dagger\beta\), \(\bar v=v^\dagger\beta\), and
\(p^0=\omega_{\vec p}\) on \(\Sigma_m^+\).
The corresponding individual completeness relations are
\begin{equation}
  \sum_{\sigma=1}^2
  u_\alpha^\sigma(p)\bar u^{\sigma,\beta}(p)
  =
  \frac{(m-\ii\not p)_\alpha{}^\beta}{2\omega_{\vec p}},
  \qquad
  \sum_{\sigma=1}^2
  v_\alpha^\sigma(p)\bar v^{\sigma,\beta}(p)
  =
  -\frac{(m+\ii\not p)_\alpha{}^\beta}{2\omega_{\vec p}} .
  \label{eq:dirac-individual-spin-sums}
\end{equation}
At the reference momentum \(p_R=(m,\vec0)\), \(\not p_R=\ii m\beta\), so
\[
  \sum_\sigma u^\sigma(p_R)\bar u^\sigma(p_R)
  =
  \frac{1+\beta}{2},
  \qquad
  \sum_\sigma v^\sigma(p_R)\bar v^\sigma(p_R)
  =
  -\frac{1-\beta}{2}.
\]
These rest-frame matrices obey
\[
  (\not p_R-\ii m)\frac{1+\beta}{2}=0,
  \qquad
  (\not p_R+\ii m)\left[-\frac{1-\beta}{2}\right]=0,
\]
and have traces \(2\) and \(-2\), reproducing the summed Dirac-adjoint
normalizations at \(p^0=m\).  Lorentz covariance of the intertwiners transports
the identities to all \(p\in\Sigma_m^+\).
Let
\[
  \dd\mu_m(p)
  =
  \frac{\dd^3\vec p}{(2\pi)^3\,2\omega_{\vec p}},
  \qquad
  \omega_{\vec p}=\sqrt{\vec p^{\,2}+m^2},
  \qquad
  p^0=\omega_{\vec p}.
\]
\end{definition}

\begin{definition}[Free charged Dirac field and CAR]
\label{def:free-charged-dirac-field-car}
Introduce two independent families of fermionic creation and annihilation
operators
\[
  b_\sigma(\vec p),\ b_\sigma^\dagger(\vec p),
  \qquad
  d_\sigma(\vec p),\ d_\sigma^\dagger(\vec p),
\]
with canonical anticommutation relations
\[
  \{b_\sigma(\vec p),b_\tau^\dagger(\vec q)\}
  =
  (2\pi)^3\,2\omega_{\vec p}\,
  \delta_{\sigma\tau}\delta^{(3)}(\vec p-\vec q),
\]
\[
  \{d_\sigma(\vec p),d_\tau^\dagger(\vec q)\}
  =
  (2\pi)^3\,2\omega_{\vec p}\,
  \delta_{\sigma\tau}\delta^{(3)}(\vec p-\vec q),
\]
all other anticommutators being zero. The vacuum vector \(\vac\) is
annihilated by all \(b_\sigma(\vec p)\) and \(d_\sigma(\vec p)\).

The charged Dirac field is the operator-valued distribution
\[
  \widehat\psi_\alpha(x)
  =
  \sum_{\sigma=1}^2
  \int_{\Sigma_m^+}\dd\mu_m(p)\,
  \left[
    v_\alpha^\sigma(p)b_\sigma^\dagger(\vec p)\ee^{-\ii p\cdot x}
    +
    u_\alpha^\sigma(p)d_\sigma(\vec p)\ee^{\ii p\cdot x}
  \right].
\]
Its adjoint field is
\[
  \widehat{\bar\psi}^{\alpha}(x)
  =
  \bigl(\widehat\psi^\dagger(x)\beta\bigr)^\alpha .
\]
\end{definition}

\begin{proposition}[Free Dirac equations and charge convention]
\label{prop:free-dirac-equations-charge-convention}
Because of the two momentum-space Dirac equations above,
\[
  (\not\partial+m)\widehat\psi(x)=0,
  \qquad
  \widehat{\bar\psi}(x)(-\overleftarrow{\not\partial}+m)=0,
\]
where the arrow indicates that the derivative acts on the field to its left.

The convention in this chapter assigns charge \(+1\) to the one-particle
states created by \(b^\dagger\) and charge \(-1\) to the states created by
\(d^\dagger\). If
\[
  U_\theta=\ee^{\ii\theta \widehat Q},
  \qquad
  U_\theta\vac=\vac ,
\]
then
\[
  U_\theta\widehat\psi_\alpha(x)U_\theta^{-1}
  =
  \ee^{\ii\theta}\widehat\psi_\alpha(x),
  \qquad
  U_\theta\widehat{\bar\psi}^{\alpha}(x)U_\theta^{-1}
  =
  \ee^{-\ii\theta}\widehat{\bar\psi}^{\alpha}(x).
\]
\end{proposition}

\begin{proof}
For the creation part of \(\widehat\psi\),
\[
  (\not\partial+m)\bigl(v^\sigma(p)\ee^{-\ii p\cdot x}\bigr)
  =
  (-\ii\not p+m)v^\sigma(p)\ee^{-\ii p\cdot x}
  =
  0,
\]
because \((\not p+\ii m)v^\sigma(p)=0\).  For the annihilation part,
\[
  (\not\partial+m)\bigl(u^\sigma(p)\ee^{\ii p\cdot x}\bigr)
  =
  (\ii\not p+m)u^\sigma(p)\ee^{\ii p\cdot x}
  =
  0,
\]
because \((\not p-\ii m)u^\sigma(p)=0\).  The adjoint equation follows by
taking the Dirac adjoint and using
\((\gamma^\mu)^\dagger\beta=-\beta\gamma^\mu\), with the derivative acting on
the field to its left.

The charge convention is fixed on the oscillator algebra by
\[
  [\widehat Q,b_\sigma^\dagger(\vec p)]=b_\sigma^\dagger(\vec p),
  \qquad
  [\widehat Q,d_\sigma^\dagger(\vec p)]=-d_\sigma^\dagger(\vec p).
\]
Taking adjoints gives
\[
  [\widehat Q,b_\sigma(\vec p)]=-b_\sigma(\vec p),
  \qquad
  [\widehat Q,d_\sigma(\vec p)]=d_\sigma(\vec p).
\]
Both terms in \(\widehat\psi\) therefore have charge \(+1\), while
\(\widehat{\bar\psi}\) has charge \(-1\). Exponentiating the commutators gives
the displayed \(U_\theta\)-transformation laws.
\end{proof}

\section{The Statistics Sign from Locality}

\begin{proposition}[Free spinor locality selects the CAR sign]
\label{prop:free-dirac-locality-selects-car}
Once the spin-\(\frac12\) one-particle representation and the covariant spinor
intertwiners have been chosen, the local free field is obtained from
canonical anticommutators rather than ordinary commutators. Use the
delta-normalized polarizations \(\mathcal U,\mathcal V\) of
Chapter~\ref{ch:massive-particles-spin}. Let
\[
  \Delta_+(z;m^2)
  =
  \int_{\Sigma_m^+}\dd\mu_m(p)\,\ee^{\ii p\cdot z}
\]
be the positive-frequency scalar distribution, and form the trial field
\[
  \widehat\Phi_\alpha(x)
  =
  \sum_{\sigma=1}^2
  \int\frac{d^3\vec p}{(2\pi)^{3/2}}
  \left[
    \mathcal V_\alpha^\sigma(p)a_\sigma^\dagger(\vec p)
      \ee^{-\ii p\cdot x}
    +
    \mathcal U_\alpha^\sigma(p)c_\sigma(\vec p)
      \ee^{\ii p\cdot x}
  \right].
\]
Suppose first that the oscillator algebra used ordinary commutators,
\[
  [a_\sigma(\vec p),a_\tau^\dagger(\vec q)]
  =
  [c_\sigma(\vec p),c_\tau^\dagger(\vec q)]
  =
  \delta_{\sigma\tau}\delta^{(3)}(\vec p-\vec q),
\]
with all other commutators zero. Using
\eqref{eq:dirac-individual-spin-sums} gives
\[
\begin{aligned}
  [\widehat\Phi_\alpha(x),\widehat{\bar\Phi}^{\beta}(y)]
  &=
  \int\frac{d^3\vec p}{(2\pi)^3}\,
  \left[
    \ee^{-\ii p\cdot(x-y)}
    \frac{m+\ii\not p}{2p^0}
    +
    \ee^{\ii p\cdot(x-y)}
    \frac{m-\ii\not p}{2p^0}
  \right]_\alpha{}^\beta .
\end{aligned}
\]
The scalar part contains
\[
  m\bigl(\Delta_+(x-y;m^2)+\Delta_+(y-x;m^2)\bigr),
\]
which has nonzero spacelike values. For example, at equal times and
\(\vec x\ne\vec y\) it equals the distribution
\[
  2m\int\frac{d^3\vec p}{(2\pi)^3\,2\omega_{\vec p}}\,
  \ee^{\ii\vec p\cdot(\vec x-\vec y)} .
\]
Thus the field obtained with the commutator sign is nonlocal.

If instead the oscillator algebra is the canonical anticommutator algebra,
\eqref{eq:dirac-individual-spin-sums} gives
\[
  \{\widehat\psi_\alpha(x),\widehat{\bar\psi}^{\beta}(y)\}
  =
  (-\not\partial_x+m)_\alpha{}^\beta
  \left[
    \Delta_+(x-y;m^2)-\Delta_+(y-x;m^2)
  \right].
\]
The bracket is the Pauli--Jordan distribution. It vanishes for spacelike
separation, so the spinor field is local as an odd field in a graded local
algebra. This computation is the free-field derivation of the statistics sign
used in the rest of the chapter.
\end{proposition}

\begin{proof}
Only the oscillator pairings \(aa^\dagger\) and \(cc^\dagger\), or their CAR
counterparts, contribute.  Inserting the mode expansion and using the spin
sums \eqref{eq:dirac-individual-spin-sums} gives the two displayed kernels.
For ordinary commutators the scalar part is the even combination
\[
  m\bigl(\Delta_+(x-y;m^2)+\Delta_+(y-x;m^2)\bigr).
\]
At equal time and \(\vec x\ne\vec y\) this is the Fourier transform of the
positive mass-shell density
\((2\omega_{\vec p})^{-1}d^3\vec p\) plus its reflected copy.  It is not the
zero distribution; for example smearing by a nonzero real test function
supported in an equal-time ball gives a strictly positive quadratic form.
Thus ordinary commutators violate locality.

For CAR, the creation and annihilation contributions enter with the relative
minus sign shown above, and the scalar factor is the Pauli--Jordan
distribution
\[
  \Delta(x-y;m^2)=\Delta_+(x-y;m^2)-\Delta_+(y-x;m^2).
\]
The scalar chapter proves that \(\Delta(z;m^2)\) has support in the closed
light cone. Applying the local differential operator
\((-\not\partial_x+m)_\alpha{}^\beta\) cannot enlarge support, so the
anticommutator vanishes at spacelike separation. This is precisely graded
locality for an odd field.
\end{proof}

\section{Locality and the CAR}

With the same positive-frequency scalar distribution
\[
  \Delta_+(x;m^2)
  =
  \int_{\Sigma_m^+}\dd\mu_m(p)\,\ee^{\ii p\cdot x}.
\]
With the spinor normalizations fixed by the one-particle inner product, the
field anticommutator is
\[
  \{\widehat\psi_\alpha(x),\widehat{\bar\psi}^{\beta}(y)\}
  =
  (-\not\partial_x+m)_\alpha{}^\beta
  \left[
    \Delta_+(x-y;m^2)-\Delta_+(y-x;m^2)
  \right].
\]
The bracketed distribution is the Pauli--Jordan distribution in the
normalization convention already used for the scalar field. It vanishes at
spacelike separation. Hence
\[
  \{\widehat\psi_\alpha(x),\widehat{\bar\psi}^{\beta}(y)\}=0
  \qquad\text{when }(x-y)^2>0.
\]
The equal-time anticommutator is obtained by differentiating the
Pauli--Jordan distribution:
\[
  \{\widehat\psi_\alpha(t,\vec x),
    \widehat{\bar\psi}^{\beta}(t,\vec y)\}
  =
  \ii(\gamma^0)_\alpha{}^\beta\delta^{(3)}(\vec x-\vec y),
\]
and
\[
  \{\widehat\psi_\alpha(t,\vec x),\widehat\psi_\beta(t,\vec y)\}=0,
  \qquad
  \{\widehat{\bar\psi}^{\alpha}(t,\vec x),
    \widehat{\bar\psi}^{\beta}(t,\vec y)\}=0.
\]

The local algebra is therefore a graded local algebra. If \(|A|\in\{0,1\}\)
is the fermion parity of a homogeneous operator \(A\), the graded commutator is
\[
  [A,B]_{\mathrm g}
  =
  AB-(-1)^{|A||B|}BA .
\]
Locality is the vanishing of this graded commutator for spacelike separated
smeared fields. The Dirac field has parity \(1\), while bosonic local fields
have parity \(0\).

\begin{remark}
The preceding computation is the free spin-\(\frac12\) instance of the
spin-statistics connection.  A theorem-level spin-statistics statement
requires a specified Wightman framework: a common invariant dense domain for
the smeared fields, a positive-energy unitary representation of the spin
cover of the proper orthochronous Poincare group, covariance of the fields as
operator-valued tempered distributions, locality on the common domain,
positivity of the Hilbert-space inner product, and the stated adjoint
compatibility between fields and their conjugates.  Under those hypotheses,
integer-spin fields have even local commutation and half-integer-spin fields
have odd local anticommutation, except for fields that vanish in the physical
Hilbert space.  This chapter does not use that theorem as an input; the
displayed anticommutator is the direct verification needed for the free Dirac
field used below.
\end{remark}

\section{Dirac, Weyl, and Majorana Fields}

\begin{definition}[Dirac, Weyl, and Majorana local spinor data]
\label{def:dirac-weyl-majorana-local-spinor-data}
The Dirac field transforms in the spinor representation:
\[
  U(\Lambda)\widehat\psi_\alpha(x)U(\Lambda)^{-1}
  =
  R_\alpha{}^\beta(\Lambda)\widehat\psi_\beta(\Lambda x),
\]
where
\[
  R(\Lambda)
  =
  \exp\!\left(-\frac{\ii}{2}\omega_{\mu\nu}S^{\mu\nu}\right),
  \qquad
  S^{\mu\nu}=-\frac{\ii}{4}[\gamma^\mu,\gamma^\nu],
\]
for \(\Lambda=\exp\omega\). The field \(\widehat\psi\) has four complex
components before additional Lorentz-covariant conditions are imposed.

Define
\[
  \gamma_5=-\ii\gamma^0\gamma^1\gamma^2\gamma^3 .
\]
It obeys
\[
  \gamma_5^2=1,
  \qquad
  \{\gamma_5,\gamma^\mu\}=0,
  \qquad
  [\gamma_5,S^{\mu\nu}]=0.
\]
Therefore
\[
  P_\pm=\frac{1\pm\gamma_5}{2}
\]
are Lorentz-covariant projectors. A Weyl spinor field is a Dirac spinor field
subject to
\[
  \widehat\chi_\pm=P_\pm\widehat\chi_\pm .
\]
In the explicit Clifford basis used in Chapter~\ref{ch:massive-particles-spin},
\[
  \gamma_5
  =
  \begin{pmatrix}
    I_2&0\\0&-I_2
  \end{pmatrix},
  \qquad
  P_+
  =
  \begin{pmatrix}
    I_2&0\\0&0
  \end{pmatrix},
  \qquad
  P_-
  =
  \begin{pmatrix}
    0&0\\0&I_2
  \end{pmatrix}.
\]
Thus a Weyl condition removes two of the four complex Dirac components while
preserving Lorentz covariance.
The two chiralities are inequivalent complex representations of
\(\operatorname{Spin}^+(1,3)\). The Dirac mass term couples them because
\(\gamma^\mu\) maps one chirality to the other.

A Majorana condition is a Lorentz-covariant reality condition. Suppose a
matrix \(B\) is chosen so that
\[
  (\gamma^\mu)^*=B\gamma^\mu B^{-1},
  \qquad
  B^{-1}(B^{-1})^*=1.
\]
Here \( * \) denotes the antilinear operation on spinor components, combined
with componentwise adjunction when it is applied to operator-valued spinor
fields. In the displayed gamma basis one may take \(B=\gamma_2\). Since
\(\sigma_2^*=-\sigma_2\), this choice obeys the two identities above. Then
\[
  \widehat\psi^c=B^{-1}\widehat\psi^*
\]
transforms in the same Lorentz representation as \(\widehat\psi\). A Majorana
field satisfies
\[
  \widehat\xi=\widehat\xi^c .
\]
\end{definition}

\begin{proposition}[Lorentz covariance of chiral and Majorana reductions]
\label{prop:weyl-majorana-covariant-reductions}
The projectors \(P_\pm=(1\pm\gamma_5)/2\) define Lorentz-covariant Weyl
subfields. If \(B\) obeys the two identities in
Definition~\ref{def:dirac-weyl-majorana-local-spinor-data}, then charge
conjugation \(\psi\mapsto\psi^c=B^{-1}\psi^*\) is Lorentz covariant and
involutive. In four dimensions a Weyl field \(\widehat\chi_+\) determines a
Majorana field
\[
  \widehat\xi
  =
  \widehat\chi_+
  +
  B^*\widehat\chi_+^*,
\]
whose \(P_+\)-projection is \(\widehat\chi_+\).
\end{proposition}

\begin{proof}
The commutator \([\gamma_5,S^{\mu\nu}]=0\) implies
\([P_\pm,R(\Lambda)]=0\) for the connected spin group, hence the Weyl
conditions \(P_\pm\chi_\pm=\chi_\pm\) are preserved by Lorentz
transformations.  The charge-conjugation covariance check is direct:
\[
\begin{aligned}
  U(\Lambda)\widehat\psi^c(x)U(\Lambda)^{-1}
  &=
  B^{-1}
  \left(R(\Lambda)\widehat\psi(\Lambda x)\right)^*      \\
  &=
  B^{-1}BR(\Lambda)B^{-1}\widehat\psi^*(\Lambda x)
   =
  R(\Lambda)\widehat\psi^c(\Lambda x).
\end{aligned}
\]
The second identity for \(B\) gives \((\psi^c)^c=\psi\), so the Majorana
condition is a genuine reality condition on the complex Dirac module.
If \(\widehat\chi_+=P_+\widehat\chi_+\) is a Weyl field, then
\[
  \widehat\xi
  =
  \widehat\chi_+
  +
  B^*\widehat\chi_+^*
\]
is Majorana, and
\[
  P_+\widehat\xi=\widehat\chi_+ .
\]
Indeed \(P_+B^*=B^*P_-^*\), so
\[
  P_+B^*\widehat\chi_+^*
  =
  B^*P_-^*\widehat\chi_+^*
  =
  B^*(P_-\widehat\chi_+)^*
  =
  0.
\]
Moreover \(B^*\widehat\chi_+^*\) is anti-chiral:
\[
  \gamma_5B^*\widehat\chi_+^*
  =
  -B^*\gamma_5^*\widehat\chi_+^*
  =
  -B^*\widehat\chi_+^* .
\]
\end{proof}

Thus in four dimensions a Majorana spinor is equivalent, as Lorentz
representation data, to one Weyl spinor together with its conjugate. These
are statements about local field representations. Particle content is
determined separately by the spectral and asymptotic data of the theory.

\section{Odd Field Variables and Configuration Superspaces}

\begin{definition}[Finite fermionic configuration superspace]
\label{def:finite-fermionic-configuration-superspace}
There are two distinct meanings of the word ``fermionic field'' in this
chapter. The operator field \(\widehat\psi_\alpha(x)\) is an
operator-valued distribution on a \(\mathbb Z_2\)-graded Hilbert space. After
smearing by test functions, it gives conventional linear operators on a
common dense domain, with parity \(1\). Its locality property is the
vanishing of the graded commutator at spacelike separation. This is a
statement about the local operator algebra.

The variables \(\psi_\alpha(x)\), \(\bar\psi^\alpha(x)\) appearing in a
fermionic path integral are different objects. In a finite-dimensional
regularization they are odd coordinates on a supermanifold. Let \(V\) be a
finite-dimensional complex vector space carrying the cutoff spinor labels,
including spacetime lattice points, spinor indices, and any internal indices.
The purely odd affine superspace with those labels is \(\Pi V\). As a
locally super-ringed space it is
\[
  \Pi V=\bigl(\{\ast\},\mathcal O_{\Pi V}\bigr),
  \qquad
  \mathcal O_{\Pi V}=\Lambda(V^\vee),
\]
where \(V^\vee\) is placed in odd degree. Equivalently, for every
supercommutative test algebra
\(R=R_{\overline{0}}\oplus R_{\overline{1}}\), its
\(R\)-points are
\[
  (\Pi V)(R)=R_{\overline{1}}\otimes_{\mathbb C} V .
\]
Thus a Grassmann field variable is an odd coordinate in the structure sheaf,
or equivalently an \(R_{\overline{1}}\)-valued component in the
functor-of-points description. It is not an ordinary function on spacetime
and it is not a point of an underlying set of configurations. For independent
Dirac variables the finite-dimensional configuration superspace is
\[
  \Pi(V_\psi\oplus V_{\bar\psi}),
  \qquad
  \mathcal O=\Lambda(V_\psi^\vee\oplus V_{\bar\psi}^\vee),
\]
with \(\psi\) and \(\bar\psi\) treated as independent odd coordinates until a
real or adjoint structure is imposed by the operator theory.

If bosonic cutoff fields are present, their configuration space is an
ordinary affine or smooth space \(B\). Fermionic directions are then encoded
by an odd vector bundle \(E\to B\), and the corresponding configuration
supermanifold is the locally super-ringed space
\[
  \bigl(B,\mathcal O_B\otimes \Lambda(E^\vee)\bigr)
\]
in a local trivialization. This description is the finite-dimensional
geometric meaning of the formal phrase ``integration over fermionic
fields.''
\end{definition}

\begin{proposition}[Purely odd Berezinian transformation]
\label{prop:purely-odd-berezinian-transformation}
Berezin integration is the associated algebraic integration on the
Berezinian line. For a purely odd coordinate system
\((\eta_1,\ldots,\eta_N)\), choosing the ordered Berezinian element
\(\dd\eta_N\cdots\dd\eta_1\) defines the linear functional which extracts the
coefficient of \(\eta_1\cdots\eta_N\). Under a change of odd coordinates
\(\eta'=A\eta\), this ordered Berezinian transforms as
\[
  \dd\eta'_N\cdots\dd\eta'_1
  =
  (\det A)^{-1}\dd\eta_N\cdots\dd\eta_1 .
\]
This inverse determinant is the odd part of the Berezinian transformation
law. The Berezin integral is therefore neither a positive measure nor a
countably additive Borel measure on a topological space; it is a coordinate
independent algebraic functional on Berezinian densities. The continuum
symbol
\[
  \int[\dd\bar\psi\,\dd\psi]\;(\cdots)
\]
used below denotes a compatible family of such finite-dimensional Berezin
integrals after a regulator has supplied finitely many odd coordinates. All
determinants, Pfaffians, source contractions, and fermionic signs in the
formal continuum notation are inherited from these finite-dimensional
Berezinian operations.
\end{proposition}

\begin{proof}
The ordered integral with density
\(\dd\eta_N\cdots\dd\eta_1\) extracts the coefficient of the ordered monomial
\(\eta_1\cdots\eta_N\). If \(\eta'_a=A_a{}^b\eta_b\), then antisymmetry of
the odd coordinates gives
\[
  \eta'_1\cdots\eta'_N=(\det A)\eta_1\cdots\eta_N .
\]
For the coefficient extracted from \(f(\eta)\dd\eta_N\cdots\dd\eta_1\) to be
coordinate independent, the ordered Berezinian element must transform by the
inverse factor:
\[
  \dd\eta'_N\cdots\dd\eta'_1
  =
  (\det A)^{-1}\dd\eta_N\cdots\dd\eta_1 .
\]
This is the odd part of the Berezinian.  The construction uses only the
finite-dimensional exterior algebra and its top coefficient.  No underlying
set of odd values, sigma-algebra, or positivity structure enters.
\end{proof}

\section{Classical Grassmann Mechanics and Dirac Brackets}

\begin{definition}[Finite odd first-order mechanical datum]
\label{def:finite-odd-first-order-mechanics}
The finite-dimensional model underlying fermion path integrals is specified
by its coordinate superalgebra. Let \(\eta_\alpha\), \(\alpha=1,\ldots,N\),
be odd generators of a Grassmann algebra,
\[
  \eta_\alpha\eta_\beta=-\eta_\beta\eta_\alpha,
  \qquad
  \eta_\alpha^2=0.
\]
Functions of the formal coordinates are elements of this algebra. Consider
the first-order action
\[
  S[\eta]
  =
  \int\dd t\,
  M_{\alpha\beta}\eta_\alpha\dot\eta_\beta,
  \qquad
  M_{\alpha\beta}=M_{\beta\alpha},
\]
where \(M\) is nondegenerate. The symmetry of \(M\) is compatible with the
odd nature of the coordinates because integration by parts gives
\[
  -\int\dd t\,M_{\alpha\beta}\dot\eta_\alpha\eta_\beta
  =
  \int\dd t\,M_{\alpha\beta}\eta_\beta\dot\eta_\alpha .
\]
Using the right derivative with respect to \(\dot\eta_\alpha\), the canonical
momenta are
\[
  \pi_\alpha
  =
  L\frac{\overleftarrow\partial}{\partial\dot\eta_\alpha}
  =
  M_{\alpha\beta}\eta_\beta .
\]
Thus the phase-space variables obey primary constraints
\[
  \chi_\alpha=\pi_\alpha-M_{\alpha\beta}\eta_\beta=0 .
\]
For odd canonical variables use the graded Poisson brackets
\[
  \{\eta_\alpha,\eta_\beta\}_{\mathrm P}=0,
  \qquad
  \{\pi_\alpha,\pi_\beta\}_{\mathrm P}=0,
  \qquad
  \{\eta_\alpha,\pi_\beta\}_{\mathrm P}
  =
  \{\pi_\beta,\eta_\alpha\}_{\mathrm P}
  =
  \delta_{\alpha\beta}.
\]
\end{definition}

\begin{proposition}[Odd second-class constraints and Dirac brackets]
\label{prop:odd-second-class-dirac-brackets}
For the datum of
Definition~\ref{def:finite-odd-first-order-mechanics}, the constraints
\(\chi_\alpha=\pi_\alpha-M_{\alpha\beta}\eta_\beta\) are second class with
constraint matrix
\[
  K_{\alpha\beta}
  =
  \{\chi_\alpha,\chi_\beta\}_{\mathrm P}
  =
  -2M_{\alpha\beta},
\]
and the Dirac bracket
\[
  \{f,g\}_{\mathrm D}
  =
  \{f,g\}_{\mathrm P}
  -
  \{f,\chi_\alpha\}_{\mathrm P}
  (K^{-1})^{\alpha\beta}
  \{\chi_\beta,g\}_{\mathrm P}.
\]
annihilates the constraints:
\[
  \{f,\chi_\alpha\}_{\mathrm D}=0.
\]
For the odd coordinates,
\[
\begin{aligned}
  \{\eta_\alpha,\eta_\beta\}_{\mathrm D}
  &=
  - (K^{-1})_{\alpha\beta}
   =
  \frac12(M^{-1})_{\alpha\beta},\\
  \{\eta_\alpha,\pi_\beta\}_{\mathrm D}
  &=
  \frac12\delta_{\alpha\beta}.
\end{aligned}
\]
In the \(N=2\) model below, quantization gives a two-state Hilbert space with
Hamiltonian eigenvalues \(-\omega/4\) and \(+\omega/4\).
\end{proposition}

\begin{proof}
Using the graded Poisson brackets in the definition,
\[
\begin{aligned}
  \{\chi_\alpha,\chi_\beta\}_{\mathrm P}
  &=
  \{\pi_\alpha-M_{\alpha\gamma}\eta_\gamma,
    \pi_\beta-M_{\beta\delta}\eta_\delta\}_{\mathrm P} \\
  &=
  -M_{\beta\alpha}-M_{\alpha\beta}
   =
  -2M_{\alpha\beta},
\end{aligned}
\]
because \(M_{\alpha\beta}=M_{\beta\alpha}\).  Nondegeneracy of \(M\) makes
\(K\) invertible, hence the constraints are second class.  The displayed
Dirac bracket is the usual Schur-complement correction of the graded Poisson
bracket by the inverse constraint matrix.  Substituting one argument
\(\chi_\alpha\) gives zero because
\(\{\chi_\rho,\chi_\alpha\}_{\mathrm P}=K_{\rho\alpha}\).
For the coordinates,
\(\{\eta_\alpha,\chi_\rho\}_{\mathrm P}=\delta_{\alpha\rho}\), so
\[
  \{\eta_\alpha,\eta_\beta\}_{\mathrm D}
  =
  -\delta_{\alpha\rho}(K^{-1})^{\rho\sigma}
  \delta_{\sigma\beta}
  =
  \frac12(M^{-1})_{\alpha\beta}.
\]
The same substitution gives
\(\{\eta_\alpha,\pi_\beta\}_{\mathrm D}=\delta_{\alpha\beta}/2\).

For \(N=2\), take
\[
  S=\int\dd t\,\ii(\eta_1\dot\eta_1+\eta_2\dot\eta_2).
\]
Equivalently, with
\[
  \eta=\eta_1+\ii\eta_2,
  \qquad
  \chi=\eta_1-\ii\eta_2,
\]
the difference between
\(\int\dd t\,\ii(\eta_1\dot\eta_1+\eta_2\dot\eta_2)\) and
\(\int\dd t\,\ii\chi\dot\eta\) is a total derivative. The endpoint convention
fixes which representative of the first-order action is used. The Dirac
brackets are
\[
  \{\eta,\chi\}_{\mathrm D}=-\ii,
  \qquad
  \{\eta,\eta\}_{\mathrm D}=0=\{\chi,\chi\}_{\mathrm D}.
\]
Quantization replaces \(\eta_\alpha\) by operators satisfying
\[
  \{\widehat\eta_\alpha,\widehat\eta_\beta\}
  =
  \frac{\ii\hbar}{2}(M^{-1})_{\alpha\beta}.
\]
In units \(\hbar=1\), the oscillator with
\[
  L=\ii\eta_1\dot\eta_1+\ii\eta_2\dot\eta_2-\ii\omega\eta_1\eta_2
\]
has Hamiltonian \(H=\ii\omega\eta_1\eta_2\). A minimal representation is
\[
  \widehat\eta_1
  =
  \frac12
  \begin{pmatrix}0&1\\1&0\end{pmatrix},
  \qquad
  \widehat\eta_2
  =
  \frac12
  \begin{pmatrix}0&-\ii\\ \ii&0\end{pmatrix},
\]
so that
\[
  \widehat H
  =
  \ii\omega\widehat\eta_1\widehat\eta_2
  =
  \begin{pmatrix}
    -\omega/4&0\\0&\omega/4
  \end{pmatrix}.
\]
The finite Grassmann system therefore quantizes to a two-state Hilbert space.
\end{proof}

\section{Berezin Calculus}

\begin{convention}[Berezin normalization and ordering]
\label{conv:berezin-normalization-ordering}
For one odd coordinate \(\theta\), the Berezin integral is normalized by
\[
  \int\dd\theta\,1=0,
  \qquad
  \int\dd\theta\,\theta=1.
\]
For ordered odd coordinates \((\theta_1,\ldots,\theta_N)\), the ordered
Berezinian density \(\dd\theta_N\cdots\dd\theta_1\) is normalized by
\[
  \int\dd\theta_N\cdots\dd\theta_1\,
  \theta_1\theta_2\cdots\theta_N
  =
  1.
\]
All signs below, including the sign of a closed fermion loop and the determinant
obtained from a fermionic Gaussian, use this ordering together with the Koszul
rule for moving odd variables and odd differentials past one another.
\end{convention}

\begin{definition}[Finite Grassmann algebra and Berezin derivative]
\label{def:finite-grassmann-algebra-berezin-derivative}
Let \(\Lambda_N\) be the complex Grassmann algebra generated by
\[
  \eta_1,\ldots,\eta_N
\]
with relations
\[
  \eta_a\eta_b=-\eta_b\eta_a,
  \qquad
  \eta_a^2=0.
\]
A homogeneous monomial has parity equal to its degree modulo \(2\). If \(X\)
and \(Y\) are homogeneous, then
\[
  XY=(-1)^{|X||Y|}YX
\]
when every generator appearing in \(X\) is distinct from every generator
appearing in \(Y\). Berezin differentials are treated as odd symbols, so
\[
  \eta_a\,\dd\eta_b=-\dd\eta_b\,\eta_a .
\]

The left derivative \(\partial^L/\partial\eta_a\) is defined by
\[
  \frac{\partial^L}{\partial\eta_a}\eta_b=\delta_{ab},
  \qquad
  \frac{\partial^L}{\partial\eta_a}(XY)
  =
  \frac{\partial^L X}{\partial\eta_a}Y
  +
  (-1)^{|X|}X\frac{\partial^L Y}{\partial\eta_a}
\]
for homogeneous \(X\). Berezin integration is the same algebraic operation, with
the normalization of Convention~\ref{conv:berezin-normalization-ordering}:
\[
  \int\dd\eta_a\,1=0,
  \qquad
  \int\dd\eta_a\,\eta_a=1.
\]
For \(N\) generators the ordered integral is fixed by
\[
  \int\dd\eta_N\cdots\dd\eta_1\,
  \eta_1\eta_2\cdots\eta_N
  =
  1.
\]
\end{definition}

\begin{proposition}[Finite Berezin Gaussian and contractions]
\label{prop:finite-berezin-gaussian-contractions}
For independent pairs \(\eta_a,\bar\eta_a\), \(a=1,\ldots,N\), and an
invertible \(N\times N\) complex matrix \(A\), use the ordered measure
\[
  [\dd\bar\eta\,\dd\eta]
  =
  \prod_{a=1}^N\dd\bar\eta_a\,\dd\eta_a .
\]
With this convention,
\[
  \int[\dd\bar\eta\,\dd\eta]\,
  \exp\left(-\bar\eta_a A^a{}_b\eta_b\right)
  =
  \det A .
\]
Adding Grassmann sources \(J,\bar J\),
\[
\begin{aligned}
  &\int[\dd\bar\eta\,\dd\eta]\,
  \exp\left(
    -\bar\eta_a A^a{}_b\eta_b
    +\bar J\eta+\bar\eta J
  \right)  \\
  &\hspace{2.2cm}
  =
  \det A\,
  \exp\left(\bar J A^{-1}J\right).
\end{aligned}
\]
The normalized two-point contraction is
\[
  \langle\eta_a\bar\eta_b\rangle_A
  =
  (A^{-1})_a{}^b .
\]
This identity follows from Berezin integration by parts. Since
\[
  0=
  \int[\dd\bar\eta\,\dd\eta]\,
  \frac{\partial^L}{\partial\bar\eta_b}
  \left[
    \bar\eta_c
    \exp\!\left(-\bar\eta_aA^a{}_d\eta_d\right)
  \right],
\]
the normalized integral obeys
\[
  \delta_c{}^b
  =
  A^b{}_d
  \langle\eta_d\bar\eta_c\rangle_A .
\]
Higher Gaussian moments are obtained by repeated contractions, with the sign
determined by the number of odd variables moved past one another to bring a
contracted pair together.
\end{proposition}

\begin{proof}
The Gaussian identity is an identity in the finite exterior algebra generated
by the \(2N\) odd variables.  The Berezin integral extracts one fixed top
coefficient.  The coefficient of the top monomial in
\(\exp(-\bar\eta_a A^a{}_b\eta_b)\) is multilinear and alternating in the
columns of \(A\), and the declared ordering normalizes the identity matrix
case to \(1\).  Hence the coefficient is \(\det A\).

For the source identity, use the odd translation
\[
  \eta\mapsto \eta-A^{-1}J,
  \qquad
  \bar\eta\mapsto \bar\eta-\bar J A^{-1}.
\]
Berezinian densities are invariant under translations, and the exponent
becomes
\[
  -\bar\eta A\eta+\bar J A^{-1}J .
\]
The source-free Gaussian integral then gives the factor
\(\det A\exp(\bar J A^{-1}J)\).  Differentiating the source identity with the
same left/right derivative convention gives
\(\langle\eta_a\bar\eta_b\rangle_A=(A^{-1})_a{}^b\). Equivalently, the
integration-by-parts identity displayed above gives
\[
  \delta_c{}^b=A^b{}_d\langle\eta_d\bar\eta_c\rangle_A ,
\]
which is the same statement after multiplication by \(A^{-1}\). Repeated
differentiation gives Wick contractions, with Koszul signs determined by the
permutation needed to place each contracted pair next to the source
derivatives.
\end{proof}

\section{The Fermionic Coherent-State Path Integral}

The one-mode complex fermionic system has two operators
\[
  \widehat\eta,\qquad \widehat\chi,
\]
obeying
\[
  \{\widehat\eta,\widehat\chi\}=1,
  \qquad
  \widehat\eta^2=\widehat\chi^2=0.
\]
On the two-dimensional Hilbert space with basis \(\ket{0},\ket{1}\), choose
\[
  \widehat\eta\ket{0}=0,\qquad
  \widehat\eta\ket{1}=\ket{0},
  \qquad
  \widehat\chi\ket{0}=\ket{1},\qquad
  \widehat\chi\ket{1}=0.
\]

\begin{proposition}[Fermionic coherent-state resolution]
\label{prop:fermionic-coherent-state-resolution}
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=0.95]
  \node[gray!70] (nullL) at (-1.25,0) {\(0\)};
  \node[draw,circle,minimum size=0.95cm] (zero) at (0,0) {\(\ket{0}\)};
  \node[draw,circle,minimum size=0.95cm] (one) at (4,0) {\(\ket{1}\)};
  \node[gray!70] (nullR) at (5.25,0) {\(0\)};
  \draw[->,thick,blue!70!black]
    (zero) to[bend left=28] node[above] {\(\widehat\chi\)} (one);
  \draw[->,thick,green!50!black]
    (one) to[bend left=28] node[below] {\(\widehat\eta\)} (zero);
  \draw[->,gray!65] (zero) -- (nullL)
    node[midway,above] {\(\widehat\eta\)};
  \draw[->,gray!65] (one) -- (nullR)
    node[midway,above] {\(\widehat\chi\)};
\end{tikzpicture}
\end{center}

For a Grassmann parameter \(\eta\), define the right coherent ket
\[
  \ket{\eta}=\ket{0}+\ket{1}\eta ,
\]
so that
\[
  \widehat\eta\ket{\eta}=\ket{\eta}\eta .
\]
Define the dual coherent bra by
\[
  \langle\!\langle\chi|
  =
  -\bra{1}+\chi\bra{0}.
\]
It satisfies
\[
  \langle\!\langle\chi|\eta\rangle=\chi-\eta .
\]
It also gives the left action
\[
  \langle\!\langle\chi|\widehat\eta
  =
  -\chi\,\langle\!\langle\chi| .
\]
The resolution of the identity is
\[
  \int\ket{\eta}\,\dd\eta\,\langle\!\langle\eta|
  =
  \mathbf 1 .
\]
The sign in this formula is part of the ordered Berezin convention. Written
with the differential moved to the left, it is
\[
\begin{aligned}
  \int\ket{\eta}\,\dd\eta\,\langle\!\langle\eta|
  &=
  \int\dd\eta\,\ket{-\eta}\,\langle\!\langle\eta|       \\
  &=
  \int\dd\eta\,
  (-\ket{1}\eta+\ket{0})(-\bra{1}+\eta\bra{0})          \\
  &=
  \ket{1}\bra{1}+\ket{0}\bra{0}.
\end{aligned}
\]

Let
\[
  U(T)=\ee^{-\ii \widehat H T}.
\]
Insert the coherent-state resolution of identity at times
\[
  0=t_0<t_1<\cdots<t_N=T,
  \qquad
  t_{n+1}-t_n=\varepsilon=T/N.
\]
\begin{center}
\begin{tikzpicture}[>=Stealth,scale=0.92]
  \foreach \x/\lab in {0/{\eta_0},1.8/{\eta_1},3.6/{\eta_2},5.4/{\eta_3},7.2/{\eta_N}} {
    \fill[black] (\x,0) circle (1.6pt);
    \node[below] at (\x,-0.03) {\(\lab\)};
  }
  \draw[thick] (0,0)--(7.2,0);
  \draw[->,thick] (7.2,0)--(7.85,0) node[right] {\(t\)};
  \foreach \x/\lab in {0.9/{\chi_0},2.7/{\chi_1},4.5/{\chi_2},6.3/{\chi_{N-1}}} {
    \draw[blue!70!black,thick] (\x,-0.16)--(\x,0.16);
    \node[above,blue!70!black] at (\x,0.16) {\(\lab\)};
  }
  \node[below] at (0,-0.45) {\(t_0\)};
  \node[below] at (7.2,-0.45) {\(t_N\)};
\end{tikzpicture}
\end{center}
The one-step overlap identities are
\[
  \langle\!\langle\eta_{n+1}|\eta_n\rangle
  =
  -\int\dd\chi_n\,
  \exp\!\left[-\chi_n(\eta_{n+1}-\eta_n)\right],
\]
and
\[
  \langle\!\langle\eta_{n+1}|\widehat\chi|\eta_n\rangle
  =
  -\int\dd\chi_n\,\chi_n\,
  \exp\!\left[-\chi_n(\eta_{n+1}-\eta_n)\right].
\]
The endpoint convention fixes the signs in these formulas. With
\(\eta_0=\eta_i\), \(\eta_N=\eta_f\), and the ordered measure displayed
below, the finite-\(N\) kernel is
\[
\begin{aligned}
  K_N(\eta_f,T;\eta_i,0)
  &=
  -\int
  \dd\chi_0
  \prod_{n=1}^{N-1}\dd\eta_n\,\dd\chi_n                 \\
  &\quad\times
  \exp
  \sum_{n=0}^{N-1}
  \left[
    -\chi_n(\eta_{n+1}-\eta_n)
    -\ii\varepsilon H(\chi_n,\eta_n)
  \right].
\end{aligned}
\]
The kernel itself is the continuum limit of these regulated Berezin
integrals,
\[
  \langle\!\langle\eta_f|U(T)|\eta_i\rangle
  =
  \lim_{N\to\infty}K_N(\eta_f,T;\eta_i,0),
\]
when that limit exists in the chosen topology. Equivalently, the global sign
may be absorbed into the definition of the ordered finite-dimensional measure.
The continuum notation for the same construction is
\[
  \int D\chi\,D\eta\,
  \exp\left\{
    \int_0^T\dd t\,
    \left[-\chi\dot\eta-\ii H(\chi,\eta)\right]
  \right\}.
\]
Equivalently, with the first-order action
\[
  S[\chi,\eta]
  =
  \int_0^T\dd t\,\left[\ii\chi\dot\eta-H(\chi,\eta)\right],
\]
the continuum Grassmann path-integral notation assigns the weight
\(\ee^{\ii S}\) to the kernel. This expression denotes the limit, when it exists,
of the finite time-sliced Berezin integrals displayed above; the time-sliced
expression is the regulated meaning of the notation.
In this first-order odd system the Hamiltonian and Lagrangian forms coincide:
\(\chi\) is the odd conjugate variable of \(\eta\), and it remains in the
first-order action rather than being removed by a nondegenerate even Gaussian
integration.
\end{proposition}

\begin{proof}
The eigenket identity follows directly from the two-state action:
\[
  \widehat\eta(\ket0+\ket1\eta)=\ket0\eta=(\ket0+\ket1\eta)\eta,
\]
because \(\eta^2=0\).  The dual coherent bra gives
\[
  \langle\!\langle\chi|\eta\rangle
  =
  (-\bra1+\chi\bra0)(\ket0+\ket1\eta)
  =
  \chi-\eta,
\]
and
\[
  \langle\!\langle\chi|\widehat\eta=-\chi\langle\!\langle\chi|
\]
by the same basis calculation.  The displayed Berezin computation of
\(\int\ket\eta\,\dd\eta\,\langle\!\langle\eta|\) extracts the coefficient of
\(\eta\) and gives \(\ket0\bra0+\ket1\bra1\).  Inserting this identity between
short-time evolution factors gives the finite-\(N\) kernel. The overlap
identities follow by applying the same one-variable Berezin integral to
\(\langle\!\langle\eta_{n+1}|\eta_n\rangle\) and
\(\langle\!\langle\eta_{n+1}|\widehat\chi|\eta_n\rangle\).
\end{proof}

\section{Fermionic Traces and Thermal Boundary Conditions}

\begin{proposition}[Fermionic trace endpoint signs]
\label{prop:fermionic-trace-boundary-signs}
The trace of a fermionic Hilbert-space operator carries one extra sign from
the odd coherent-state resolution. For inverse temperature
\(\beta_{\mathrm T}>0\),
\[
\begin{aligned}
  Z(\beta_{\mathrm T})
  &=
  \operatorname{Tr}_{\mathcal H}\ee^{-\beta_{\mathrm T}\widehat H}        \\
  &=
  \int
  \operatorname{Tr}_{\mathcal H}
  \left(
    \ket{\eta}\,\dd\eta\,\langle\!\langle\eta|\,
    \ee^{-\beta_{\mathrm T}\widehat H}
  \right)                                      \\
  &=
  \int\dd\eta\,
  \langle\!\langle\eta|
  \ee^{-\beta_{\mathrm T}\widehat H}
  \ket{-\eta}.
\end{aligned}
\]
The state \(\ket{-\eta}\) appears because moving the odd ket
\(\ket{\eta}=\ket0+\ket1\eta\) through the odd differential \(\dd\eta\)
changes the sign of its odd component. After Euclidean time slicing this
identity becomes the finite-dimensional Berezin integral with boundary
condition
\[
  \eta(\tau+\beta_{\mathrm T})=-\eta(\tau).
\]
Thus ordinary thermal traces impose anti-periodic boundary conditions on
fermionic variables. The supertrace
\[
  \operatorname{Tr}_{\mathcal H}
  \left((-1)^F\ee^{-\beta_{\mathrm T}\widehat H}\right)
\]
replaces the final \(\ket{-\eta}\) by \(\ket{\eta}\), and therefore imposes
periodic fermionic boundary conditions. This distinction is the
finite-dimensional origin of the spin-structure choice for fermionic fields
on a Euclidean time circle.
\end{proposition}

\begin{proof}
For the trace sign, let \(A\) be the matrix
\[
  A=
  \begin{pmatrix} a&b\\ c&d\end{pmatrix}
\]
in the ordered basis \(\ket0,\ket1\). Then
\[
  \int\dd\eta\,\langle\!\langle\eta|A|-\eta\rangle=a+d
  =
  \operatorname{Tr}A,
\]
whereas
\[
  \int\dd\eta\,\langle\!\langle\eta|A|\eta\rangle=a-d
  =
  \operatorname{Tr}((-1)^F A).
\]
Thus an ordinary trace identifies the endpoint coherent variable with a minus
sign, while the insertion of \((-1)^F\) removes that sign. Time slicing carries
this endpoint algebra into anti-periodic and periodic Euclidean fermion
boundary conditions, respectively.
\end{proof}

\section{Berezin Integration as Functorial Pushforward}

\begin{definition}[Finite Berezin pushforward]
\label{def:finite-berezin-pushforward}
Let \(V\) be a finite-dimensional complex vector space and let
\(\Pi V\) denote the purely odd affine superspace with structure sheaf
\(\Lambda(V^\vee)\).  A basis \(e_1,\ldots,e_N\) of \(V\) determines odd
coordinates \(\theta_1,\ldots,\theta_N\) and the ordered Berezinian density
\[
  [\dd\theta]=\dd\theta_N\cdots\dd\theta_1 .
\]
The integral of \(f(\theta)[\dd\theta]\) is the coefficient of
\(\theta_1\cdots\theta_N\) in \(f\).  If
\(\theta'_a=A_a{}^b\theta_b\), then
\[
  \theta'_1\cdots\theta'_N=(\det A)\theta_1\cdots\theta_N,
  \qquad
  [\dd\theta']=(\det A)^{-1}[\dd\theta].
\]
Thus the product \(f(\theta)[\dd\theta]\) is the coordinate-independent
object, and Berezin integration is the linear functional on compactly
supported algebraic Berezinian densities in this finite model.

For a direct sum \(V\oplus W\), choose ordered coordinates \(\theta\) on
\(\Pi V\) and \(\eta\) on \(\Pi W\), with total density
\[
  [\dd\eta][\dd\theta]
  =
  \dd\eta_M\cdots\dd\eta_1\,\dd\theta_N\cdots\dd\theta_1 .
\]
The projection
\[
  p:\Pi(V\oplus W)\longrightarrow \Pi V
\]
has a pushforward
\[
  p_!:\Gamma(\operatorname{Ber}_{\Pi(V\oplus W)})
  \longrightarrow
  \Gamma(\operatorname{Ber}_{\Pi V})
\]
defined by
\[
  p_!\bigl(F(\theta,\eta)[\dd\eta][\dd\theta]\bigr)
  =
  \left(
    \int\dd\eta_M\cdots\dd\eta_1\,F(\theta,\eta)
  \right)
  [\dd\theta].
\]
\end{definition}

This finite-dimensional pushforward is the prototype of a fermionic path
integral: it is a fiberwise algebraic pushforward of Berezinian densities.

\begin{proposition}[Functoriality of finite Berezin pushforward]
\label{prop:finite-berezin-pushforward-functoriality}
For
\[
  \Pi(V\oplus W\oplus U)
  \xrightarrow{q}
  \Pi(V\oplus W)
  \xrightarrow{p}
  \Pi V ,
\]
the equality
\[
  (p\circ q)_!=p_!\circ q_!
\]
holds when the Berezinian orders are the induced product orders.  Thus a
finite regulator may integrate fermionic modes in stages without changing the
value, provided the induced Berezinian order is carried along.
\end{proposition}

\begin{proof}
Choose ordered odd coordinates \(\theta\) on \(V\), \(\eta\) on \(W\), and
\(\zeta\) on \(U\), with total Berezinian density
\([\dd\zeta][\dd\eta][\dd\theta]\).  Applying \((p\circ q)_!\) extracts first
the coefficient of \(\zeta_1\cdots\zeta_{\dim U}\eta_1\cdots\eta_{\dim W}\)
from the coefficient polynomial over \(\Lambda(V^\vee)\).  Applying \(q_!\)
extracts the \(\zeta\)-top coefficient and leaves a Berezinian density on
\(\Pi(V\oplus W)\); applying \(p_!\) then extracts the \(\eta\)-top
coefficient.  Since coefficient extraction in a finite exterior algebra is
associative once the ordered monomial is fixed, the two operations agree.
\end{proof}

\section{Supersymmetric Quantum Mechanics and the Witten Index}

\begin{definition}[One-dimensional supersymmetric quantum-mechanics datum]
\label{def:susy-qm-witten-index-datum}
Let \(W:\mathbb R\to\mathbb R\) be a smooth function for which the operators
below are essentially self-adjoint on the Schwartz domain and have discrete
low-energy spectrum.  The Hilbert space is
\[
  \mathcal H=L^2(\mathbb R)\otimes\Lambda^\bullet\C,
\]
with fermion operators \(\psi,\psi^\dagger\) obeying
\[
  \{\psi,\psi^\dagger\}=1,\qquad
  F=\psi^\dagger\psi .
\]
Define
\[
  Q=\psi^\dagger(\partial_x+W'(x)),
  \qquad
  Q^\dagger=\psi(-\partial_x+W'(x)),
  \qquad
  H=\frac12\{Q,Q^\dagger\}.
\]
Then \(Q^2=(Q^\dagger)^2=0\), \([H,Q]=[H,Q^\dagger]=0\), and
\[
  H
  =
  \frac12
  \left[
    -\partial_x^2+W'(x)^2+W''(x)(2F-1)
  \right].
  \label{eq:susy-qm-hamiltonian}
\]
The Witten index is
\[
  I_W(\beta_{\mathrm T})
  =
  \operatorname{Tr}_{\mathcal H}
  \left((-1)^F\ee^{-\beta_{\mathrm T}H}\right)
\]
whenever this trace is defined by the finite-volume or confining
regularization under discussion.
\end{definition}

\begin{proposition}[Pairing of positive-energy states]
\label{prop:susy-qm-witten-index-pairing}
Assume \(H\ge0\), the nonzero spectrum in the regulated problem is discrete,
and \(Q,Q^\dagger\) preserve the domain of \(H\).  Then
\[
  I_W(\beta_{\mathrm T})
  =
  \dim\ker H|_{F=0}-\dim\ker H|_{F=1},
  \qquad
  \frac{d}{d\beta_{\mathrm T}}I_W(\beta_{\mathrm T})=0 .
\]
\end{proposition}

\begin{proof}
Let \(E>0\) and let \(\mathcal H_E\) be the finite-dimensional eigenspace of
energy \(E\).  Since \(H=\frac12(QQ^\dagger+Q^\dagger Q)\),
a vector \(\Psi\in\mathcal H_E\) with \(E>0\) cannot be annihilated by both
\(Q\) and \(Q^\dagger\).  The maps \(Q\) and \(Q^\dagger\) commute with \(H\)
and change fermion parity.  The operator
\[
  \frac{1}{2E}(QQ^\dagger+Q^\dagger Q)
\]
is the identity on \(\mathcal H_E\), so the even and odd parts of
\(\mathcal H_E\) have equal dimension: \(Q+Q^\dagger\) gives an invertible
odd map on \(\mathcal H_E\) after division by \(\sqrt{2E}\).  Therefore the
supertrace of \(\ee^{-\beta_{\mathrm T}H}\) vanishes on every
positive-energy eigenspace.  Only \(\ker H\) remains, where
\(Q\Psi=Q^\dagger\Psi=0\).  The displayed formula and
\(\beta_{\mathrm T}\)-independence follow.
\end{proof}

For the harmonic superpotential
\[
  W(x)=\frac{m}{2}x^2,\qquad m>0,
\]
\begin{proposition}[Harmonic supersymmetric oscillator index]
\label{prop:harmonic-susy-oscillator-index}
For \(W(x)=mx^2/2\) with \(m>0\), the regulated Witten index is
\[
  I_W=1.
\]
\end{proposition}

\begin{proof}
The zero-mode equations are
\[
  (\partial_x+mx)f_0(x)=0,
  \qquad
  (-\partial_x+mx)f_1(x)=0.
\]
Their solutions are
\[
  f_0(x)=C_0\ee^{-mx^2/2},
  \qquad
  f_1(x)=C_1\ee^{mx^2/2}.
\]
Only \(f_0\) lies in \(L^2(\mathbb R)\).  Hence
\[
  I_W=1 .
\]
The oscillator spectrum makes the cancellation explicit.  With
\[
  q=\ee^{-\beta_{\mathrm T}m},
\]
the even contribution is
\[
  Z_{\mathrm{even}}=\sum_{n\ge0}q^n=\frac1{1-q},
\]
and the odd contribution is
\[
  Z_{\mathrm{odd}}=\sum_{n\ge0}q^{n+1}=\frac{q}{1-q}.
\]
Thus
\[
  \operatorname{Str}\ee^{-\beta_{\mathrm T}H}
  =
  Z_{\mathrm{even}}-Z_{\mathrm{odd}}=1.
\]
In the coherent-state path integral this is the same supertrace discussed in
the previous section, so the fermionic variable is periodic on the Euclidean
time circle.
\end{proof}

\section{Worldline Index Density from Periodic Fermions}

\begin{definition}[Twisted Dirac index and periodic worldline datum]
\label{def:twisted-dirac-index-periodic-worldline-datum}
Let \(M\) be a compact oriented Riemannian spin manifold of dimension
\(2n\), let \(E\to M\) be a Hermitian vector bundle with unitary connection,
and let
\[
  D_E:\Gamma(S^+\otimes E)\longrightarrow\Gamma(S^-\otimes E)
\]
be the chiral Dirac operator.  The analytic index is
\[
  \operatorname{ind}D_E
  =
  \dim\ker D_E-\dim\ker D_E^\dagger
  =
  \operatorname{Str}_{L^2(S\otimes E)}
  \ee^{-\beta_{\mathrm T}D_E^2}.
\]
The last equality follows from the same positive-spectrum pairing as in
Proposition~\ref{prop:susy-qm-witten-index-pairing}, with
\((-1)^F\) replaced by chirality.

The worldline path integral for this supertrace has periodic bosonic and
fermionic loops:
\[
  x(\tau+\beta_{\mathrm T})=x(\tau),
  \qquad
  \psi^a(\tau+\beta_{\mathrm T})=\psi^a(\tau),
\]
where \(a\) is an orthonormal tangent-frame index.  In normal coordinates at
a point \(x_0\in M\), split
\[
  x(\tau)=x_0+\sqrt{\beta_{\mathrm T}}\,y(\tau),
  \qquad
  \psi^a(\tau)=\psi_0^a+\psi_{\perp}^a(\tau),
\]
with \(y\) and \(\psi_\perp\) having zero average.  The finite-mode Gaussian
calculation is performed around this zero-mode decomposition.
\end{definition}

\begin{proposition}[Periodic-fermion worldline index density]
\label{prop:periodic-fermion-worldline-index-density}
For the preceding periodic worldline datum, the finite-mode nonzero-mode
calculation and the fermion zero-mode Berezin integral give the local index
density
\[
  \left[
    \widehat A(TM)\operatorname{ch}(E)
  \right]_{2n}.
\]
After integration over the bosonic zero mode, this gives
\[
  \operatorname{ind}D_E
  =
  \int_M\widehat A(TM)\operatorname{ch}(E).
\]
\end{proposition}

\begin{proof}
The finite-mode Gaussian calculation of the nonzero modes gives one factor
for each skew-eigenvalue
\(r_j\) of the curvature two-form \(R/(2\pi\ii)\):
\begin{equation}
  \prod_{j=1}^n
  \frac{r_j/2}{\sinh(r_j/2)}.
  \label{eq:worldline-ahat-product}
\end{equation}
The derivation is the product identity
\[
  \frac{\sinh z}{z}
  =
  \prod_{k=1}^\infty
  \left(1+\frac{z^2}{\pi^2k^2}\right)
\]
applied to the ratio of the periodic bosonic and fermionic nonzero-mode
determinants in each curvature two-plane.  The connection on \(E\) contributes
\[
  \operatorname{tr}_E\exp\!\left(\frac{F_E}{2\pi\ii}\right),
\]
because the fermion zero modes represent exterior multiplication by tangent
one-forms and the Berezin integral over
\(\psi_0^1,\ldots,\psi_0^{2n}\) extracts the top-degree component.  Therefore
the local density produced by the worldline supertrace is
\begin{equation}
  \left[
    \widehat A(TM)\operatorname{ch}(E)
  \right]_{2n},
  \qquad
  \widehat A(TM)=
  \prod_{j=1}^n\frac{r_j/2}{\sinh(r_j/2)},
  \qquad
  \operatorname{ch}(E)=
  \operatorname{tr}_E\exp\!\left(\frac{F_E}{2\pi\ii}\right).
  \label{eq:worldline-index-density}
\end{equation}
Integrating over the bosonic zero mode \(x_0\) gives the
Atiyah--Singer formula
\[
  \operatorname{ind}D_E
  =
  \int_M\widehat A(TM)\operatorname{ch}(E).
\]
\end{proof}

The analytic content behind the last equality is the heat-kernel asymptotic
expansion for \(D_E^2\) on a compact manifold and the McKean--Singer
supertrace identity.  The path-integral calculation above identifies the
local coefficient by finite-dimensional Gaussian and Berezinian operations.

\section{Dirac Functional Integrals and Propagators}

\begin{definition}[Free Dirac functional-integral datum]
\label{def:free-dirac-functional-integral-datum}
For a classical Grassmann spinor field \(\psi_\alpha(x)\), define
\[
  \bar\psi^\alpha(x)
  =
  \bigl(\psi^\dagger(x)\beta\bigr)^\alpha .
\]
Here \(\beta=\ii\gamma^0\), so
\[
  (\gamma^\mu)^\dagger\beta=-\beta\gamma^\mu .
\]
This is the adjoint convention for which \(\bar\psi(\not\partial+m)\psi\)
is a Lorentz scalar density.
The free Dirac Lagrangian density in the conventions of this volume is
\[
  \mathcal L_{\mathrm D}
  =
  -\bar\psi(\not\partial+m)\psi .
\]
Its Euler--Lagrange equation is
\[
  (\not\partial+m)\psi=0.
\]
The momentum conjugate to \(\psi\), using the right derivative appropriate
for odd variables, is
\[
  \Pi_\psi{}^\alpha
  =
  \mathcal L_{\mathrm D}
  \frac{\overleftarrow\partial}{\partial(\partial_0\psi_\alpha)}
  =
  -\bar\psi^\beta(\gamma^0)_\beta{}^\alpha .
\]
The equal-time anticommutator computed from the mode expansion above,
\[
  \{\widehat\psi_\alpha(t,\vec x),
    \widehat{\bar\psi}^{\beta}(t,\vec y)\}
  =
  \ii(\gamma^0)_\alpha{}^\beta\,\delta^{(3)}(\vec x-\vec y),
\]
therefore gives the canonical odd bracket
\[
  \{\widehat\psi_\alpha(t,\vec x),
    \widehat\Pi_\psi{}^\beta(t,\vec y)\}
  =
  \ii\delta_\alpha{}^\beta\,\delta^{(3)}(\vec x-\vec y).
\]
The regulated Lorentzian generating functional, written in continuum
shorthand, is normalized so that
\[
\begin{aligned}
  Z[\bar J,J]
  &=
  \int D\bar\psi\,D\psi\,
  \exp\left[
    \ii\int\dd^4x\,\mathcal L_{\mathrm D}
    +
    \ii\int\dd^4x\,
    \bigl(\bar J\psi+\bar\psi J\bigr)
  \right]  \\
  &=
  Z[0,0]\,
  \exp\left[
    -\ii\int\dd^4x\,\dd^4y\,
    \bar J^\alpha(x)
    S_{F,\alpha}{}^\beta(x-y)
    J_\beta(y)
  \right],
\end{aligned}
\]
where \(J\) and \(\bar J\) are Grassmann sources. The Feynman two-point
function is
\[
  S_{F,\alpha}{}^\beta(x-y)
  =
  \bra{\vac}
  \operatorname T\{
    \widehat\psi_\alpha(x)\widehat{\bar\psi}^{\beta}(y)
  \}
  \ket{\vac} .
\]
Charge conservation gives
\[
  \bra{\vac}\operatorname T\{
    \widehat\psi_\alpha(x)\widehat\psi_\beta(y)\}\ket{\vac}=0,
  \qquad
  \bra{\vac}\operatorname T\{
    \widehat{\bar\psi}^{\alpha}(x)
    \widehat{\bar\psi}^{\beta}(y)\}\ket{\vac}=0
\]
for the free charged Dirac field.
\end{definition}

\begin{proposition}[Free Dirac propagator as Feynman inverse]
\label{prop:free-dirac-propagator-feynman-inverse}
The free charged Dirac two-point function in
Definition~\ref{def:free-dirac-functional-integral-datum} is the Feynman
inverse of \(\not\partial+m\):
\[
  S_F(x;m)
  =
  (-\not\partial_x+m)\Delta_F(x;m^2).
\]
Using the scalar convention
\[
  \Delta_F(x;m^2)
  =
  -\ii
  \int\frac{\dd^4k}{(2\pi)^4}
  \frac{\ee^{\ii k\cdot x}}{k^2+m^2-\ii\epsilon},
\]
one obtains
\[
  S_F(k)
  =
  \frac{-\ii(-\ii\not k+m)}
       {k^2+m^2-\ii\epsilon}.
\]
The numerator is the spinor projector algebra of
Chapter~\ref{ch:massive-particles-spin}, while the denominator is the scalar
mass-shell pole with the Feynman boundary value.
\end{proposition}

\begin{proof}
Since \(\Delta_F\) is the scalar Feynman inverse in the convention
\[
  (-\partial^2+m^2)\Delta_F(x;m^2)=-\ii\delta^{(4)}(x),
\]
and the Clifford relation gives
\[
  (\not\partial+m)(-\not\partial+m)
  =
  -\not\partial^2+m^2
  =
  -\partial^2+m^2,
\]
the distribution \(S_F=(-\not\partial+m)\Delta_F\) obeys the Dirac inverse
equation with the same Feynman boundary value.  Fourier transforming
\(\partial_\mu\mapsto\ii k_\mu\) in the convention used in the scalar
chapter gives the displayed numerator
\(-\ii(-\ii\not k+m)\) over \(k^2+m^2-\ii0\).
\end{proof}

\section{Spinor Spectral Poles and LSZ}

\begin{definition}[Isolated spinor one-particle pole datum]
\label{def:isolated-spinor-one-particle-pole-datum}
Let an interacting theory have a global \(U(1)\) charge \(\widehat Q\), a
vacuum satisfying \(\widehat Q\vac=0\), and an isolated massive
spin-\(\frac12\) one-particle sector of mass \(m\) and charge \(+1\). Let
\(\widehat\psi_\alpha\) be a local spinor field of charge \(+1\) with nonzero
overlap with this sector:
\[
  [\widehat Q,\widehat\psi_\alpha]=\widehat\psi_\alpha,
  \qquad
  [\widehat Q,\widehat{\bar\psi}^{\alpha}]
  =
  -\widehat{\bar\psi}^{\alpha}.
\]
In the delta-normalized one-particle convention, Lorentz covariance permits
the matrix element
\[
  \langle\vec p,\sigma,+|\widehat\psi_\alpha(x)\ket{\vac}
  =
  \frac{\ee^{-\ii p\cdot x}}{(2\pi)^{3/2}}
  \left[
    Z_v^{1/2}\mathcal V_\alpha^\sigma(p)
    +
    Z_u^{1/2}\mathcal U_\alpha^\sigma(p)
  \right],
\]
where \(\mathcal U,\mathcal V\) are the two massive spinor intertwiners
constructed from the one-particle representation. A constant local
redefinition inside the span of \(\psi\) and \(\gamma_5\psi\), together with a
phase convention for one-particle states, chooses an interpolating field whose
isolated pole residue is proportional to one chosen intertwiner. After making
that choice and passing to the covariant one-particle normalization used in
the LSZ formulas, write
\[
  \langle\vec p,\sigma,+|\widehat\psi_\alpha(x)\ket{\vac}
  =
  Z_\psi^{1/2}\,
  \ee^{-\ii p\cdot x}v_\alpha^\sigma(p),
  \qquad
  Z_\psi\ge0 .
\]
\end{definition}

\begin{proposition}[Locality fixes conjugate spinor pole residue]
\label{prop:locality-fixes-conjugate-spinor-pole-residue}
The conjugate field has the corresponding overlap with the charge \(-1\)
sector,
\[
  \langle\vec p,\sigma,-|
  \widehat{\bar\psi}^{\alpha}(x)\ket{\vac}
  =
  Z_\psi^{1/2}\,
  \ee^{-\ii p\cdot x}\bar u^{\sigma,\alpha}(p),
\]
with the same pole residue after the phase convention is fixed.  The
time-ordered two-point function then has the one-particle pole
\[
  Z_\psi
  (-\not\partial_x+m)_\alpha{}^\beta
  \Delta_F(x-y;m^2).
\]
\end{proposition}

\begin{proof}
Let \(Z_{\bar\psi}\) denote the conjugate-sector residue before imposing
locality.  The one-particle contribution to the vacuum anticommutator has the
form
\[
\begin{aligned}
  C_\alpha{}^\beta(x-y)
  &=
  \sum_\sigma\int\dd\Pi_p\,
  \bigl[
    Z_\psi\,
    \ee^{-\ii p\cdot(x-y)}
    v_\alpha^\sigma(p)\bar v^{\sigma,\beta}(p)  \\
  &\hspace{3.2cm}
    +
    Z_{\bar\psi}\,
    \ee^{\ii p\cdot(x-y)}
    u_\alpha^\sigma(p)\bar u^{\sigma,\beta}(p)
  \bigr].
\end{aligned}
\]
The spin sums of Chapter~\ref{ch:massive-particles-spin} convert this to
\[
  C_\alpha{}^\beta(x-y)
  =
  Z_\psi(-\not\partial_x+m)_\alpha{}^\beta
  \Delta(x-y;m^2)
\]
precisely when \(Z_{\bar\psi}=Z_\psi\). Here \(\Delta\) is the scalar
Pauli--Jordan distribution, hence the pole contribution has support in the
closed light cone. The positive- and negative-frequency terms separately do
not have this support property, so the equality of residues is fixed by
locality and the chosen normalization.

The time-ordered two-point function then has the one-particle pole
\[
  \bra{\vac}
  \operatorname T\{
    \widehat\psi_\alpha(x)\widehat{\bar\psi}^{\beta}(y)
  \}
  \ket{\vac}
  =
  Z_\psi
  (-\not\partial_x+m)_\alpha{}^\beta
  \Delta_F(x-y;m^2)
  +
  G_{\alpha}{}^{\beta,\mathrm{cont}}(x-y),
\]
where \(G^{\mathrm{cont}}\) is the contribution from states outside the
isolated one-particle mass shell. In momentum space the pole term is
\[
  Z_\psi
  \frac{-\ii(-\ii\not k+m)_\alpha{}^\beta}
       {k^2+m^2-\ii\epsilon}.
\]
\end{proof}

\begin{proposition}[Spinor mass-shell residue and external factors]
\label{prop:spinor-mass-shell-residue-external-factors}
The normalization content of this pole fixes the external spinor factor.  Put
\(p=(\omega_{\vec p},\vec p)\).  The
delta-normalized spinor intertwiners obey
\begin{equation}
  \sum_{\sigma=1}^2
  u_\alpha^\sigma(p)\bar u^{\sigma,\beta}(p)
  =
  \frac{(m-\ii\not p)_\alpha{}^\beta}{2\omega_{\vec p}},
  \qquad
  \sum_{\sigma=1}^2
  v_\alpha^\sigma(p)\bar v^{\sigma,\beta}(p)
  =
  -\,\frac{(m+\ii\not p)_\alpha{}^\beta}{2\omega_{\vec p}} .
  \label{eq:spinor-sum-lsz-normalization}
\end{equation}
Thus the numerator of the Dirac pole contains the mass-shell Jacobian:
\begin{equation}
  m-\ii\not p
  =
  2\omega_{\vec p}
  \sum_{\sigma=1}^2 u^\sigma(p)\bar u^\sigma(p),
  \qquad
  m+\ii\not p
  =
  -2\omega_{\vec p}
  \sum_{\sigma=1}^2 v^\sigma(p)\bar v^\sigma(p).
  \label{eq:spinor-numerator-jacobian}
\end{equation}
Since
\[
  k^2+m^2-\ii0
  =
  -2\omega_{\vec k}\bigl(k^0-\omega_{\vec k}+\ii0\bigr)
  +
  O\bigl((k^0-\omega_{\vec k})^2\bigr)
\]
near the positive-energy pole, the pole contribution is
\begin{equation}
  Z_\psi
  \frac{-\ii(m-\ii\not k)_\alpha{}^\beta}
       {k^2+m^2-\ii0}
  =
  \frac{\ii Z_\psi}
       {k^0-\omega_{\vec k}+\ii0}
  \sum_{\sigma=1}^2
  u_\alpha^\sigma(k)\bar u^{\sigma,\beta}(k)
  +\text{terms regular at }k^0=\omega_{\vec k}.
  \label{eq:positive-energy-spinor-pole-residue}
\end{equation}
The negative-energy pole has the same form with \(v^\sigma\) and
\(k^0+\omega_{\vec k}-\ii0\).  Therefore the factor \(2\omega_{\vec k}\) from
the mass-shell denominator is already accounted for by the spinor completeness
relation \eqref{eq:spinor-sum-lsz-normalization}.  In the
\(\delta^{(3)}(\vec p-\vec q)\)-normalized external basis, each external
wavefunction carries the Fourier factor \((2\pi)^{-3/2}\), while the spinor
intertwiner carries the energy normalization in
\(\bar u^\sigma u^\tau=(m/\omega_{\vec p})\delta^{\sigma\tau}\) and
\(\bar v^\sigma v^\tau=-(m/\omega_{\vec p})\delta^{\sigma\tau}\).  Hence a
spinor external leg contributes
\begin{equation}
  \frac{Z_\psi^{1/2}}{(2\pi)^{3/2}}\,u^\sigma_\alpha(p),
  \qquad
  \frac{Z_\psi^{1/2}}{(2\pi)^{3/2}}\,\bar u^{\sigma,\alpha}(p),
  \qquad
  \frac{Z_\psi^{1/2}}{(2\pi)^{3/2}}\,v^\sigma_\alpha(p),
  \qquad
  \frac{Z_\psi^{1/2}}{(2\pi)^{3/2}}\,\bar v^{\sigma,\alpha}(p),
  \label{eq:spinor-external-normalization-factors}
\end{equation}
with the choice among them fixed by charge and by the in/out convention.  This
is the spinorial normalization counterpart of the scalar one-particle
normalization in Chapter~\ref{ch:volume-ii-local-data}.
\end{proposition}

\begin{proof}
The spin sums \eqref{eq:spinor-sum-lsz-normalization} express the Dirac
numerator as the mass-shell projector together with the Jacobian
\(2\omega_{\vec p}\), giving
\eqref{eq:spinor-numerator-jacobian}.  Expanding the scalar denominator near
the positive-energy shell gives the linear factor displayed before
\eqref{eq:positive-energy-spinor-pole-residue}; substituting the projector
identity gives the residue formula
\eqref{eq:positive-energy-spinor-pole-residue}.  The negative-energy pole is
the same calculation with the \(v\)-spin sum and
\(k^0+\omega_{\vec k}-\ii0\).  Therefore no additional factor of
\(2\omega_{\vec p}\) is inserted by hand in the external spinor; it has
already appeared in the projector identity.  The remaining Fourier
normalization of the delta-normalized one-particle wavefunction is
\((2\pi)^{-3/2}\), yielding
\eqref{eq:spinor-external-normalization-factors}.
\end{proof}

\begin{definition}[Spinorial LSZ pole assignment]
\label{def:spinorial-lsz-pole-assignment}
Spinorial LSZ reduction applies the same Haag--Ruelle and pole-extraction
logic as the scalar case. The difference is that the residue has spinor
indices. With Fourier transform convention
\[
  F(k)=\int\dd^4x\,\ee^{\ii k\cdot x}F(x),
\]
the pole assignments are:
\[
\renewcommand{\arraystretch}{1.25}
\begin{array}{c|c|c}
  \text{field insertion} & \text{mass-shell limit}
  & \text{external one-particle state and residue} \\ \hline
  \widehat{\bar\psi}^{\alpha}(k)
  & k^0\to+\omega_{\vec k}
  & \text{outgoing charge }+,\quad
    Z_\psi^{1/2}\bar v^{\sigma,\alpha}(k)\\
  \widehat{\bar\psi}^{\alpha}(k)
  & k^0\to-\omega_{\vec k}
  & \text{incoming charge }-,\quad
    Z_\psi^{1/2}\bar u^{\sigma,\alpha}(k)\\
  \widehat\psi_\alpha(k)
  & k^0\to+\omega_{\vec k}
  & \text{outgoing charge }-,\quad
    Z_\psi^{1/2}u_\alpha^\sigma(k)\\
  \widehat\psi_\alpha(k)
  & k^0\to-\omega_{\vec k}
  & \text{incoming charge }+,\quad
    Z_\psi^{1/2}v_\alpha^\sigma(k).
\end{array}
\]
Thus amputation of an external spinor leg consists of:
\[
  \text{multiply by }(k^2+m^2),\qquad
  \text{take }k^0\to\omega_{\vec k},\qquad
  \text{contract with }u^\sigma(p)\text{ or }v^\sigma(p),
\]
with the choice of \(u\) or \(v\) determined by the external charge and by
whether the particle is incoming or outgoing. Thus external spinor factors in
perturbative scattering are the one-particle pole intertwiners of local
spinor fields.
\end{definition}

\section{A First Fermionic Vertex}

At this point the S-matrix has already been defined through asymptotic
one-particle states, and LSZ has related its matrix elements to pole
residues of time-ordered Green functions. Perturbation theory now enters as a
method for computing those Green functions.

\begin{proposition}[First four-fermion direct-minus-exchange vertex]
\label{prop:first-four-fermion-direct-minus-exchange-vertex}
Consider
\[
  \mathcal L
  =
  -\bar\psi(\not\partial+m)\psi
  -
  \frac g2(\bar\psi\psi)^2 .
\]
To first order in \(g\), the connected four-point function has one local
vertex. With all propagators oriented from \(\psi\) to \(\bar\psi\), the
spinor-index kernel of the vertex is
\[
  K_{\alpha\beta}{}^{\gamma\delta}
  =
  \ii g\,
  \left(
    \delta_\alpha{}^\gamma\delta_\beta{}^\delta
    -
    \delta_\alpha{}^\delta\delta_\beta{}^\gamma
  \right).
\]
The two terms are the direct and exchange contractions. Their relative sign
is the sign obtained by moving odd fields through one another before applying
the Gaussian contraction.  Figure~\ref{fig:four-fermion-direct-exchange-vertex}
draws the two index contractions; the arrows are a bookkeeping convention for
the ordered contraction from \(\psi\) to \(\bar\psi\), not an additional
dynamical orientation.

For external spinors \(\zeta_i\) selected by the LSZ table, the corresponding
tree-level four-fermion amplitude has the index contraction
\[
  \mathcal M
  =
  \bar\zeta_3{}^\gamma\bar\zeta_4{}^\delta\,
  K_{\alpha\beta}{}^{\gamma\delta}\,
  \zeta_1^\alpha\zeta_2^\beta .
\]
For two identical incoming and two identical outgoing charge-\(+\) particles
this reads
\[
  \mathcal M
  =
  \ii g
  \left[
    (\bar\zeta_3\zeta_1)(\bar\zeta_4\zeta_2)
    -
    (\bar\zeta_4\zeta_1)(\bar\zeta_3\zeta_2)
  \right],
\]
with \(\zeta_i\) replaced by the charge and in/out spinors appropriate to the
chosen process. This example is the first template for fermionic
Feynman diagrams: every vertex has a local index tensor, every internal line
is the time-ordered two-point function, and every exchange of odd fields
contributes its parity sign.
\end{proposition}

\begin{proof}
At first order in \(g\), the interaction insertion is the local product
\(-\frac g2(\bar\psi\psi)(\bar\psi\psi)\).  Contracting two incoming
\(\psi\)-fields with two outgoing \(\bar\psi\)-fields gives two pairings.
The first pairs \((\alpha,\gamma)\) and \((\beta,\delta)\) and gives
\(\delta_\alpha{}^\gamma\delta_\beta{}^\delta\).  The second pairs
\((\alpha,\delta)\) and \((\beta,\gamma)\).  To put the odd fields in that
pairing order one transposes one pair of fermionic variables, producing the
minus sign.  The factor \(1/2\) in the Lagrangian cancels the two identical
ways of choosing the two \(\bar\psi\psi\) factors, leaving the displayed
vertex normalization.  The external expression is obtained by contracting
this local kernel with the spinorial LSZ residues from
Definition~\ref{def:spinorial-lsz-pole-assignment}.
\end{proof}
