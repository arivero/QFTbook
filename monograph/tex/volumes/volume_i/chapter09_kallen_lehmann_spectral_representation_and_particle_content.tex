\ConventionsLorentzian


\section{Spectral Data of a Scalar Field}

\begin{definition}[Centered scalar two-point spectral datum]
\label{def:kallen-lehmann-centered-scalar-spectral-datum}
Let \(\Hilb\) carry a strongly continuous unitary representation of spacetime
translations with commuting self-adjoint generators \(P^\mu\). Let
\[
  E(\Delta)
\]
be the joint projection-valued spectral measure of \(P^\mu\), where
\(\Delta\subset\mathbb R^D\) is Borel. Assume the spectrum condition
\[
  \operatorname{Spec}(P)\subset\overline V_+
  =
  \{p\in\mathbb R^D:p^0\ge0,\ p^2\le0\},
\]
with \(p^2=\eta_{\mu\nu}p^\mu p^\nu\). Let \(\vac\) be a translation-invariant
vacuum vector in the pure vacuum sector,
\[
  E(\{0\})\Hilb=\mathbb C\vac .
\]

Let \(\widehat\phi\) be a scalar operator-valued distribution. If
\[
  \langle\vac|\widehat\phi(x)|\vac\rangle
\]
is nonzero, replace \(\widehat\phi\) by
\(\widehat\phi-\langle\vac|\widehat\phi|\vac\rangle\). This removes the
vacuum atom from the two-point function and keeps the connected spectral data:
because the zero-momentum spectral projection is \(|\vac\rangle\langle\vac|\),
there is no remaining translation-invariant component after this subtraction.
Point-field notation below is distributional notation; after smearing, the
matrix elements are ordinary Hilbert-space pairings.
\end{definition}

\begin{proposition}[Two-point spectral measure of a scalar field]
\label{prop:kallen-lehmann-two-point-spectral-measure}
In the datum of
Definition~\ref{def:kallen-lehmann-centered-scalar-spectral-datum}, the
centered two-point distribution determines a unique positive tempered Borel
measure \(\nu_\phi\) on \(\mathbb R^D\) by
\begin{equation}
  W_2(f)
  =
  \int_{\mathbb R^D}\widetilde f(-p)\,\dd\nu_\phi(p),
  \qquad
  f\in\mathcal S(\mathbb M^D),
  \label{eq:kallen-lehmann-two-point-measure-characterization}
\end{equation}
where
\[
  \widetilde f(p)=\int\dd^D x\,f(x)\ee^{-\ii p\cdot x}.
\]
Equivalently, for smeared fields,
\begin{equation}
  \langle\widehat\phi(f)\vac,\widehat\phi(g)\vac\rangle
  =
  \int_{\mathbb R^D}
    \overline{\widetilde f(p)}\,\widetilde g(p)\,
    \dd\nu_\phi(p),
  \qquad f,g\in\mathcal S(\mathbb M^D).
  \label{eq:kallen-lehmann-smeared-inner-product}
\end{equation}
The measure is supported in \(\overline V_+\).  If \(\widehat\phi\) is
Lorentz scalar and the vacuum is Lorentz invariant, \(\nu_\phi\) is invariant
under \(SO^+(1,D-1)\).  In formal point-field notation one writes
\[
  \nu_\phi(\Delta)
  =
  \langle\vac|
  \widehat\phi(0)E(\Delta)\widehat\phi(0)
  |\vac\rangle ,
\]
but this line is shorthand for the smeared characterization above, not a
literal assertion that \(\widehat\phi(0)\vac\) is a Hilbert-space vector.  The
Wightman two-point distribution is the Fourier--Stieltjes transform
\[
  W_2(x-y)
  =
  \int_{\overline V_+}\ee^{\ii p\cdot(x-y)}\,\dd\nu_\phi(p).
\]
\end{proposition}

\begin{proof}
The centered Wightman distribution
\[
  W_2(x-y)=
  \langle\vac|\widehat\phi(x)\widehat\phi(y)|\vac\rangle
\]
is a tempered distribution of positive type.  Indeed, for every
\(f\in\mathcal S(\mathbb M^D)\),
\[
  \int\dd^D x\,\dd^D y\,
    \overline{f(x)}\,f(y)\,W_2(x-y)
  =
  \|\widehat\phi(f)\vac\|^2
  \ge0 .
\]
The Bochner--Schwartz theorem for positive-type tempered distributions
therefore gives a unique positive tempered Borel measure \(\nu_\phi\) whose
Fourier--Stieltjes transform is \(W_2\), namely
\eqref{eq:kallen-lehmann-two-point-measure-characterization}.  Positivity is
thus a Hilbert-space positivity statement before it is a statement about a
measure.

The support is fixed by the spectrum condition.  In the present translation
convention,
\[
  U(a)^{-1}\widehat\phi(0)U(a)=\widehat\phi(a),
  \qquad
  U(a)=\ee^{\ii a^\mu P_\mu},
\]
so \(W_2(x)\) has spectral Fourier transform supported in
\(\operatorname{Spec}(P)\).  Since
\(E(\mathbb R^D\setminus\overline V_+)=0\), this support lies in
\(\overline V_+\).  Equivalently, after smearing, the spectral theorem gives
the same measure by the identity
\eqref{eq:kallen-lehmann-smeared-inner-product}; the unsmeared expression
\(\langle\vac|\widehat\phi(0)E(\Delta)\widehat\phi(0)|\vac\rangle\) is the
customary notation for the corresponding spectral-measure matrix element.

Finally suppose the field is a scalar and the vacuum is Lorentz invariant.
Then \(W_2(\Lambda x)=W_2(x)\) for every
\(\Lambda\in SO^+(1,D-1)\).  Applying this equality to test functions and
using uniqueness in the Bochner--Schwartz representation gives
\(\nu_\phi(\Lambda\Delta)=\nu_\phi(\Delta)\) for every Borel set \(\Delta\).
\end{proof}

Equivalently, if one writes the same statement in generalized
energy-momentum eigenvector notation, with
\[
  P^\mu|\alpha\rangle=p_\alpha^\mu|\alpha\rangle,
\]
then
\[
  W_2(x-y)
  =
  \int \dd\alpha\,
  \ee^{\ii p_\alpha\cdot(x-y)}
  \left|
    \langle\alpha|\widehat\phi(0)|\vac\rangle
  \right|^2 .
\]
This formula is only notation for the spectral measure above; the measure
\(d\nu_\phi\) is the invariant object.

\section{Invariant-Mass Decomposition}

For a scalar field and invariant vacuum, \(\nu_\phi\) is invariant under the
proper orthochronous Lorentz group. The Lorentz-invariant parameter on the
forward cone is the invariant mass squared
\[
  \mu^2=-p^2\ge0.
\]
Let
\[
  \pi:\overline V_+\to[0,\infty),
  \qquad
  \pi(p)=-p^2,
\]
which is a Borel map between standard Borel spaces.  The spectral density
measure below is a quotient measure for the Lorentz orbits of this map, not the
literal pushforward \(\pi_*\nu_\phi\): a single massive shell has infinite
Lorentz-invariant volume, so the raw pushforward would assign infinite mass to
an isolated one-particle shell.  The normalization of the invariant measure on
each shell must be fixed first.

\begin{proposition}[Invariant mass-shell disintegration with endpoint split]
\label{prop:kallen-lehmann-mass-shell-disintegration}
Let \(\nu\) be a positive locally finite Borel measure on
\(\overline V_+\) that is invariant under \(SO^+(1,D-1)\).  Assume that
\(\nu\) has no atom at \(p=0\).  If \(D\ge3\), then there is a positive Borel
measure \(\rho\) on \([0,\infty)\) such that for every
\(F\in C_c(\overline V_+)\),
\[
  \int_{\overline V_+}F(p)\,\dd\nu(p)
  =
  \int_{[0,\infty)}\dd\rho(\mu^2)
  \int\frac{\dd^Dp}{(2\pi)^{D-1}}\,
  \theta(p^0)\delta(p^2+\mu^2)F(p).
\]
If \(D=2\), the same formula holds with the spectral integral over
\((0,\infty)\).  Equivalently one may use a positive Borel measure on
\([0,\infty)\) with
\[
  \rho(\{0\})=0 .
\]
The two-dimensional punctured null cone consists of two proper-orthochronous
Lorentz orbits; a positive locally finite scalar measure on the full closed
cone carries zero coefficient on both null-ray orbit measures.  The measure
\(\rho\) is the spectral density measure after the normalization of the
invariant measure on each shell has been fixed.
\end{proposition}

\begin{proof}
Let
\[
  G=SO^+(1,D-1),\qquad X=\overline V_+\setminus\{0\}.
\]
The point \(p=0\) has zero \(\nu\)-measure by hypothesis.  We first disintegrate
\(\nu|_X\), where \(X\) is locally compact, second countable,
\(\sigma\)-compact, and standard Borel.  The group \(G\) is a locally compact
second countable Lie group, and its action on \(X\) is continuous.

We first verify the regular-orbit hypotheses in the Mackey--Effros
decomposition theorem.  The massive \(G\)-orbits in \(X\) are
\[
  \Sigma_s^+
  =
  \{p\in\overline V_+: -p^2=s\},
  \qquad s>0,
\]
with representative \(k_s=(\sqrt s,0,\ldots,0)\) and compact stabilizer
\(SO(D-1)\).  For \(D\ge3\), the null fiber is one orbit,
\[
  \Sigma_0^{+\times}
  =
  \{p\in\overline V_+:p^2=0,\ p\ne0\}.
\]
It has representative \(\ell=(1,1,0,\ldots,0)\) and stabilizer the Euclidean
massless little group
\[
  ISO(D-2)=\mathbb R^{D-2}\rtimes SO(D-2)
\]
embedded in \(G\).  The orbit maps identify the orbits with the homogeneous
spaces
\[
  \Sigma_s^+\simeq G/SO(D-1),
  \qquad
  \Sigma_0^{+\times}\simeq G/ISO(D-2).
\]
For \(D\ge3\) the orbit space is the standard Borel space \([0,\infty)\), with
orbit map \(\pi(p)=-p^2\), after removing the origin from the fiber over
\(0\).  A Borel cross-section is
\[
  c(0)=\ell,\qquad c(s)=k_s\quad(s>0).
\]

For \(D=2\) the massive orbits remain \(\Sigma_s^+\), \(s>0\), but the
punctured null fiber splits into two \(SO^+(1,1)\)-orbits,
\[
  R_{\mathrm R}=\{(k,k):k>0\},
  \qquad
  R_{\mathrm L}=\{(k,-k):k>0\}.
\]
In light-cone coordinates \(p^\pm=p^0\pm p^1\), a boost of rapidity \(\eta\)
acts by
\[
  p^+\mapsto \ee^\eta p^+,\qquad
  p^-\mapsto \ee^{-\eta}p^- .
\]
Thus \(R_{\mathrm R}\), defined by \(p^-=0\), and \(R_{\mathrm L}\), defined by
\(p^+=0\), are rescaled separately.  A Borel cross-section for the
two-dimensional action is obtained by adjoining the two representatives
\((1,1)\) and \((1,-1)\) to the massive representatives \(k_s\), \(s>0\).

In both cases the orbit equivalence relation is regularly embedded, the
quotient is standard Borel, and a Borel section exists.  The stabilizers above
are closed unimodular subgroups, so each homogeneous orbit carries a nonzero
\(G\)-invariant Radon measure, unique up to a positive scalar.  The
Mackey--Effros disintegration theorem for smooth Borel actions of locally
compact second countable groups therefore applies to the \(\sigma\)-finite
invariant Radon measure \(\nu|_X\).

Consequently there are conditional invariant Radon measures
\(\nu_{\mathcal O}\), supported on the corresponding orbits, and a positive
Borel measure \(\lambda\) on the orbit quotient such that
\[
  \int F(p)\,\dd\nu(p)
  =
  \int
  \left(\int_{\mathcal O}F(p)\,\dd\nu_{\mathcal O}(p)\right)
  \dd\lambda(\mathcal O)
\]
for all \(F\in C_c(X)\).  This \(\lambda\) is an orbit-space measure; it is
determined by the chosen normalization of conditional orbit measures.  A
single massive orbit has infinite invariant volume, so this normalization is
part of the data entering the quotient measure.

It remains to identify the orbit measures and fix their normalization.  For
\(s>0\), the standard invariant measure on the positive mass shell is
\[
  \dd\sigma_s(p)
  =
  \frac{\dd^Dp}{(2\pi)^{D-1}}\,
  \theta(p^0)\delta(p^2+s).
\]
Indeed, the distribution \(\theta(p^0)\delta(p^2+s)\dd^Dp\) is Lorentz
invariant, and in coordinates \(p=(\omega_{\vec p},\vec p)\),
\(\omega_{\vec p}=(\vec p^{\,2}+s)^{1/2}\), it is
\[
  \dd\sigma_s(p)
  =
  \frac{\dd^{D-1}\vec p}{(2\pi)^{D-1}2\omega_{\vec p}} .
\]
For \(D\ge3\) and \(s=0\), the same formula gives the invariant measure on the
punctured future light cone,
\[
  \dd\sigma_0(p)
  =
  \frac{\dd^{D-1}\vec p}{(2\pi)^{D-1}2|\vec p|},
  \qquad \vec p\ne0,
\]
which is locally finite on \(X\).  Near the origin its radial part is
\[
  \frac{|\mathbb S^{D-2}|}{2(2\pi)^{D-1}}\,r^{D-3}\,\dd r ,
\]
so its mass in \(0<r<\epsilon\) is finite and tends to zero as
\(\epsilon\downarrow0\) precisely for \(D\ge3\).

For \(D=2\), the formal \(s=0\) shell formula is instead
\[
  \frac{\dd p^1}{2\pi\,2|p^1|}
  =
  \frac{\dd k}{4\pi k}\Big|_{R_{\mathrm R}}
  +
  \frac{\dd k}{4\pi k}\Big|_{R_{\mathrm L}},
  \qquad k>0.
\]
Each summand is a Radon measure on the corresponding punctured ray.  It fails
local finiteness at the endpoint in the full closed cone: if a compact
neighborhood of the origin contains \(0<k<1\), then its mass contains
\[
  \frac1{4\pi}\int_0^1\frac{\dd k}{k}=\infty .
\]
Equivalently, every dyadic annulus
\([2^{-(n+1)},2^{-n}]\) on either ray has mass
\((4\pi)^{-1}\log2\); the union of the first \(N\) annuli sits in one compact
neighborhood of the origin and has mass growing linearly in \(N\).  Since
\(\nu\) is locally finite on \(\overline V_+\), both two-dimensional scalar
null-ray coefficients are zero.

The uniqueness of invariant Radon measure on each homogeneous orbit gives
\[
  \nu_s=c(s)\sigma_s
\]
for \(\lambda\)-almost every massive orbit, with \(c(s)\ge0\), and similarly
for the \(D\ge3\) null orbit.  The family \(s\mapsto\sigma_s\) is measurable:
for \(F\in C_c(X)\),
\[
  s\longmapsto \int_{\pi^{-1}(s)}F(p)\,\dd\sigma_s(p)
\]
is Borel, and it is continuous on \((0,\infty)\) by the displayed
\(\vec p\)-coordinate formula and dominated convergence on compact supports;
the \(s=0\) value is the corresponding null-shell limit whenever the support
stays away from the origin.  The measurable scalar \(c(s)\) is then fixed
\(\lambda\)-almost everywhere by testing \(\nu_s\) against any countable
measure-determining family of compactly supported continuous functions on
\(X\).  Absorbing \(c(s)\) into \(\dd\lambda(s)\) defines the positive Borel
measure \(\dd\rho(s):=c(s)\dd\lambda(s)\).  Substituting
\(\nu_s=c(s)\sigma_s\) gives the displayed mass-shell formula for every
\(F\in C_c(X)\); in \(D=2\) the preceding endpoint argument removes the two
null-ray terms, leaving \(s>0\).

It remains, for \(D\ge3\), to pass from \(C_c(X)\) to
\(C_c(\overline V_+)\).  Choose continuous cutoffs \(\chi_\epsilon\) on
\(\overline V_+\) with \(0\le\chi_\epsilon\le1\), with \(\chi_\epsilon=0\) near
\(p=0\), and with \(\chi_\epsilon(p)\uparrow1\) for every \(p\ne0\).  For
\(F\in C_c(\overline V_+)\), the functions \(F\chi_\epsilon\) belong to
\(C_c(X)\).  The left side converges to \(\int F\,\dd\nu\) because
\(\nu(\{0\})=0\) and \(\nu\) is finite on \(\operatorname{supp}F\).  On the
right side, apply the same statement first to nonnegative \(F\) and then to
real and imaginary parts.  Monotone convergence handles the positive mass
shells.  The \(D\ge3\) radial estimate above handles the massless shell, since
the lost \(0<r<\epsilon\) contribution tends to zero.  This proves the asserted
formula on the full closed cone.
\end{proof}

\begin{definition}[K\"all\'en--Lehmann spectral measure]
\label{def:kallen-lehmann-spectral-measure}
Applying Proposition~\ref{prop:kallen-lehmann-mass-shell-disintegration} to
\(\nu_\phi\), with the vacuum atom removed as above, gives the following
dimension-dependent convention.  For \(D\ge3\), \(d\rho\) is a positive Borel
measure on \([0,\infty)\).  For \(D=2\), the positive locally finite scalar
case is recorded by the same notation together with
\(\rho(\{0\})=0\).  Thus the displayed shell formula means
\[
  \dd\nu_\phi(p)
  =
  \int_{[0,\infty)}
  \dd\rho(\mu^2)\,
  \frac{\dd^Dp}{(2\pi)^{D-1}}\,
  \theta(p^0)\delta(p^2+\mu^2).
\]
Here \(d\rho\) is a positive Borel measure on \([0,\infty)\). When it is
absolutely continuous one writes \(\dd\rho(\mu^2)=\rho(\mu^2)\dd\mu^2\);
when it has atoms, the atomic weights are part of the same measure.
In two dimensions, a massless scalar boundary object requires extra infrared
data, such as a zero-mode test space, derivative or chiral fields, or a
declared finite-volume or small-mass prescription.  Those choices produce
additional spectral data rather than a positive locally finite scalar shell
measure on the full closed cone.
In the special case in which \(d\nu_\phi\) has a density with respect to
\(\dd^Dp\), scalar covariance and the spectrum condition say equivalently that
the density is of the form
\[
  \frac{\theta(p^0)}{(2\pi)^{D-1}}\rho(-p^2),
\]
with support on \(-p^2\ge0\). The measure notation is retained because the
one-particle contribution is an atom, not an ordinary function.
\end{definition}

\begin{definition}[Positive-frequency scalar shell distribution]
\label{def:positive-frequency-massive-scalar-distribution}
For \(\mu^2>0\) in any dimension, and for \(\mu^2=0\) when \(D\ge3\), define
the positive-frequency scalar two-point function of mass \(\mu\) by
\[
  \Delta_+(x;\mu^2)
  =
  \int\frac{\dd^Dp}{(2\pi)^{D-1}}\,
  \theta(p^0)\delta(p^2+\mu^2)\,
  \ee^{\ii p\cdot x}.
\]
Equivalently,
\[
  \Delta_+(x;\mu^2)
  =
  \int
  \frac{\dd^{D-1}\vec p}{(2\pi)^{D-1}}\,
  \frac{
    \ee^{\ii\vec p\cdot\vec x-\ii\sqrt{\vec p^{\,2}+\mu^2}\,x^0}
  }{2\sqrt{\vec p^{\,2}+\mu^2}} .
\]
At \(D=2\) and \(\mu^2=0\), the unrestricted scalar shell integral has the
logarithmic endpoint divergence
\(\int_0^\epsilon \dd k/k\) on each null ray.  It becomes a distribution after
an infrared prescription that removes or fixes the zero mode; the locally
finite scalar K\"all\'en--Lehmann measure above has \(\rho(\{0\})=0\), so the
endpoint value is absent from the spectral integral.
\end{definition}

\begin{proposition}[K\"all\'en--Lehmann representation for \(W_2\)]
\label{prop:kallen-lehmann-wightman-representation}
The Wightman two-point function is then
\[
  W_2(x)
  =
  \int_{[0,\infty)}
  \dd\rho(\mu^2)\,
  \Delta_+(x;\mu^2).
\]
This is the K\"all\'en--Lehmann representation for the Wightman two-point
function.  In \(D=2\) it is read with \(\rho(\{0\})=0\) for the positive
locally finite scalar spectral measure; infrared-subtracted massless scalar
objects require their declared test space or regulator.
\end{proposition}

\begin{proof}
Let \(f\in\mathcal S(\mathbb M^D)\).  The spectral Fourier representation of
Proposition~\ref{prop:kallen-lehmann-two-point-spectral-measure} gives
\[
  W_2(f)
  =
  \int_{\overline V_+}\dd\nu(p)\,\widetilde f(-p),
\]
with \(\nu\) the positive translation-spectral measure carried by the vacuum
two-point vector.  By
Definition~\ref{def:kallen-lehmann-spectral-measure}, \(\nu\) disintegrates
over the invariant mass coordinate \(s=-p^2\ge0\) as
\[
  \dd\nu(p)
  =
  \int_{[0,\infty)}\dd\rho(\mu^2)\,
  \frac{\dd^Dp}{(2\pi)^{D-1}}\,
  \theta(p^0)\delta(p^2+\mu^2).
\]
Pairing with \(\widetilde f(-p)\) is legitimate because \(f\) is Schwartz and
the spectral measure is tempered by the Wightman distribution hypothesis.  The
positive measure permits Tonelli for nonnegative test majorants, and the
Schwartz decay supplies the finite signed/complex integrals by decomposing
\(\widetilde f\) into real and imaginary parts.  Hence
\[
  W_2(f)
  =
  \int_{[0,\infty)}\dd\rho(\mu^2)
  \int\frac{\dd^Dp}{(2\pi)^{D-1}}\,
  \theta(p^0)\delta(p^2+\mu^2)\,\widetilde f(-p).
\]
The inner integral is precisely the distributional pairing
\(\Delta_+(\,\cdot\,;\mu^2)(f)\) in
Definition~\ref{def:positive-frequency-massive-scalar-distribution}.
Therefore the displayed formula holds as an identity on every Schwartz test
function, hence as an identity of tempered distributions.
\end{proof}

\paragraph{Spacelike symmetry of the scalar two-point function.}
For each shell value for which
Definition~\ref{def:positive-frequency-massive-scalar-distribution} defines
\(\Delta_+(x;\mu^2)\), the distribution is even at spacelike separation:
\[
  \Delta_+(x;\mu^2)=\Delta_+(-x;\mu^2),
  \qquad x^2>0,
\]
and hence \(W_2(x)=W_2(-x)\) for spacelike \(x\).  No such equality follows
from the spectral representation for timelike \(x\).
For \(x^2>0\), choose a proper orthochronous Lorentz transformation carrying
\(x\) to \((0,\vec r)\).  The mass-shell measure in
Definition~\ref{def:positive-frequency-massive-scalar-distribution} is
Lorentz invariant, so \(\Delta_+(x;\mu^2)=\Delta_+(0,\vec r;\mu^2)\).  In
that frame the integral is
\[
  \int
  \frac{\dd^{D-1}\vec p}{(2\pi)^{D-1}}
  \frac{\ee^{\ii\vec p\cdot\vec r}}{2\sqrt{\vec p^{\,2}+\mu^2}},
\]
which has an even denominator.  Changing variables \(\vec p\mapsto-\vec p\)
gives
\(\Delta_+(0,\vec r;\mu^2)=\Delta_+(0,-\vec r;\mu^2)\), which is
\(\Delta_+(-x;\mu^2)\).  Integrating this equality against the positive
measure \(d\rho(\mu^2)\) gives the statement for \(W_2\).

\section{Time-Ordered and Euclidean Forms}

\begin{definition}[Time-ordered scalar two-point distribution]
\label{def:kallen-lehmann-time-ordered-two-point}
The time-ordered two-point function is
\[
  G_T(x)
  =
  \langle\vac|
  \operatorname{T}\{
    \widehat\phi(x)\widehat\phi(0)
  \}
  |\vac\rangle.
\]
Using the definition of time ordering,
\[
  G_T(x)
  =
  \theta(x^0)W_2(x)+\theta(-x^0)W_2(-x).
\]
\end{definition}

\begin{definition}[Massive Feynman two-point kernel]
\label{def:kallen-lehmann-massive-feynman-kernel}
For \(\mu^2>0\) in any dimension, and for \(\mu^2=0\) when \(D\ge3\), define
\[
  \Delta_F(x;\mu^2)
  =
  -\ii
  \int\frac{\dd^Dk}{(2\pi)^D}
  \frac{\ee^{\ii k\cdot x}}{k^2+\mu^2-\ii\epsilon}.
\]
The \(D=2\), \(\mu^2=0\) scalar kernel uses the same infrared prescription as
\(\Delta_+\).  In the positive locally finite scalar K\"all\'en--Lehmann
measure, the endpoint has zero atomic weight.
\end{definition}

\begin{proposition}[Time-ordered K\"all\'en--Lehmann representation]
\label{prop:kallen-lehmann-time-ordered-representation}
The time-ordered two-point function satisfies
\[
  G_T(x)
  =
  \int_{[0,\infty)}
  \dd\rho(\mu^2)\,
  \Delta_F(x;\mu^2)
\]
as a distribution.  Equivalently,
\[
  \Delta_F(x;\mu^2)
  =
  \theta(x^0)\Delta_+(x;\mu^2)
  +
  \theta(-x^0)\Delta_+(-x;\mu^2).
\]
For \(D=2\), the spectral integral samples the endpoint with zero atomic
weight in the positive locally finite scalar case.
\end{proposition}

\begin{proof}
The first equality follows from
Definition~\ref{def:kallen-lehmann-time-ordered-two-point},
Proposition~\ref{prop:kallen-lehmann-wightman-representation}, and the second
displayed identity for \(\Delta_F\).  It remains to prove that the
\(\ii\epsilon\) kernel in
Definition~\ref{def:kallen-lehmann-massive-feynman-kernel} has this
time-ordered boundary value.

Set
\[
  \omega_{\vec k,\mu}=\sqrt{\vec k^{\,2}+\mu^2}.
\]
Since \(k^2+\mu^2=-(k^0)^2+\omega_{\vec k,\mu}^2\), the \(k^0\)-integral in
\(\Delta_F\) is
\[
  I(x^0,\vec k;\mu^2)
  =
  \int_{-\infty}^{\infty}
  \frac{\dd k^0}{2\pi\ii}\,
  \frac{\ee^{-\ii k^0x^0}}
  {-(k^0)^2+\omega_{\vec k,\mu}^2-\ii\epsilon}.
\]
The two poles are displaced to
\[
  k^0=+\omega_{\vec k,\mu}-\ii0,
  \qquad
  k^0=-\omega_{\vec k,\mu}+\ii0.
\]
For \(x^0>0\), the exponential damps in the lower half-plane and the closed
contour has clockwise orientation.  For \(x^0<0\), the exponential damps in
the upper half-plane.  The residue theorem gives
\[
  I(x^0,\vec k;\mu^2)
  =
  \theta(x^0)\frac{\ee^{-\ii\omega_{\vec k,\mu}x^0}}
  {2\omega_{\vec k,\mu}}
  +
  \theta(-x^0)\frac{\ee^{+\ii\omega_{\vec k,\mu}x^0}}
  {2\omega_{\vec k,\mu}}.
\]
After the remaining spatial Fourier transform this is exactly
\[
  \Delta_F(x;\mu^2)
  =
  \theta(x^0)\Delta_+(x;\mu^2)
  +
  \theta(-x^0)\Delta_+(-x;\mu^2).
\]
\end{proof}

\begin{proposition}[Euclidean K\"all\'en--Lehmann representation]
\label{prop:kallen-lehmann-euclidean-representation}
Let \(G_E\) be the Euclidean two-point distribution obtained from the ordered
Wightman boundary values by \(x^0=-\ii\tau_x\),
\(y^0=-\ii\tau_y\).  Then
\[
  G_E(x_E)
  =
  \int_{[0,\infty)}
  \dd\rho(\mu^2)\,
  \Delta_E(x_E;\mu^2),
\]
where, for the shell values sampled by \(d\rho\),
\begin{equation}
  \Delta_E(x_E;\mu^2)
  =
  \int\frac{\dd^Dk_E}{(2\pi)^D}
  \frac{\ee^{\ii k_E\cdot x_E}}{k_E^2+\mu^2}.
  \label{eq:kallen-lehmann-free-euclidean-kernel}
\end{equation}
Thus the same positive measure \(d\rho\) controls the Wightman,
time-ordered, and Euclidean scalar two-point functions.  In \(D=2\), a
zero-mass scalar atom is absent for a positive locally finite scalar measure on
the full closed cone; an infrared-prescribed massless scalar belongs to the
separately declared prescription.
\end{proposition}

\begin{proof}
The Euclidean function is obtained by analytic continuation of the Wightman
functions in the time variable.  For
\[
  z=x-y,\qquad z^0\in\mathbb C,\qquad \operatorname{Im}z^0<0,
\]
the factor
\(\exp(-\ii p^0z^0)\) is damped by the spectrum condition \(p^0\ge0\), so
\(W_2(z)\) is analytic in this tube after smearing in the spatial variables.
Set \(x^0=-\ii\tau_x\), \(y^0=-\ii\tau_y\). If
\(\tau_x>\tau_y\), then \(z^0=-\ii(\tau_x-\tau_y)\) lies in the lower half
plane and the Euclidean two-point function is the continuation of
\(W_2(x-y)\). If \(\tau_x<\tau_y\), the time-ordered function instead continues
\(W_2(y-x)\). This gives a single symmetric Euclidean distribution.
For each fixed shell value sampled by \(d\rho\), the same continuation of
\(\Delta_+(x;\mu^2)\) gives the free Euclidean kernel
\eqref{eq:kallen-lehmann-free-euclidean-kernel}; this is the one-particle
Wick rotation already derived in
Chapter~\ref{ch:scalar-path-integrals-green-functions}.  Substituting
the spectral integral of
Proposition~\ref{prop:kallen-lehmann-wightman-representation} and testing
against compactly supported Euclidean test functions permits the continuation
under the positive spectral integral.  The result is the displayed formula for
\(G_E\).  The three Green functions are therefore different boundary values or
analytic continuations of the same positive spectral data.
\end{proof}

\section{Stieltjes Positivity of the Euclidean Propagator}

\begin{definition}[Stieltjes datum of a scalar Euclidean two-point function]
\label{def:kallen-lehmann-stieltjes-datum}
The Euclidean momentum-space two-point function contains the same information
in a form that is often the most robust for constructive and lattice
arguments.  In this section \(z\) denotes the complex continuation of the
Euclidean invariant \(Q^2\).  On the physical Euclidean axis set
\[
  Q^2=k_E^2\ge0 .
\]
If the spectral measure obeys the Stieltjes integrability condition
\begin{equation}
  \int_{[0,\infty)}\frac{\dd\rho(s)}{1+s}<\infty ,
  \label{eq:stieltjes-integrability-condition}
\end{equation}
then, after Fourier transform,
\begin{equation}
  \widetilde G_E(Q^2)
  =
  \int_{[0,\infty)}
  \frac{\dd\rho(s)}{Q^2+s}.
  \label{eq:euclidean-stieltjes-propagator}
\end{equation}
Equation~\eqref{eq:stieltjes-integrability-condition} is automatic if
\(\rho([0,\infty))<\infty\), as in the canonically normalized elementary scalar
case discussed below.  It is not automatic for arbitrary local composite
operators; for them the two-point function may require subtractions, and
polynomial ambiguities in \(Q^2\) are precisely the momentum-space form of
contact terms at coincident Euclidean points.
For \(D=2\) scalar fields governed by
Proposition~\ref{prop:kallen-lehmann-mass-shell-disintegration}, the endpoint
analysis imposes \(\rho(\{0\})=0\).  Continuous spectral support accumulating at
\(s=0\) is still allowed when the Stieltjes integrability condition and the
distributional hypotheses hold.
\end{definition}

\begin{proposition}[Stieltjes positivity and spectral recovery]
\label{prop:kallen-lehmann-stieltjes-positivity}
Under the integrability condition
\eqref{eq:stieltjes-integrability-condition}, the function
\[
  F(z)=\int_{[0,\infty)}\frac{\dd\rho(s)}{z+s}
\]
is holomorphic on \(\mathbb C\setminus(-\infty,0]\), restricts to
\(\widetilde G_E(Q^2)\) on \(Q^2>0\), obeys
\[
  (-1)^r
  \frac{\dd^r}{\dd (Q^2)^r}\widetilde G_E(Q^2)\ge0,
  \qquad Q^2>0,\quad r=0,1,2,\ldots ,
\]
and recovers the absolutely continuous spectral density by the discontinuity
formula
\[
  \rho_{\rm ac}(s)
  =
  -\frac{1}{2\pi\ii}
  \operatorname{Disc}_{z=-s}\widetilde G_E(z).
\]
Atoms of \(d\rho\) are simple poles on the cut with positive residues.
\end{proposition}

\begin{proof}
Under \eqref{eq:stieltjes-integrability-condition}, \(\widetilde G_E\) is a
Stieltjes transform of a positive measure.  More precisely,
\[
  F(z)=\int_{[0,\infty)}\frac{\dd\rho(s)}{z+s}
\]
converges absolutely and uniformly on compact subsets of
\(\mathbb C\setminus(-\infty,0]\): for every compact
\(K\subset\mathbb C\setminus(-\infty,0]\) there is \(c_K>0\) such that
\(|z+s|\ge c_K(1+s)\) for \(z\in K\) and \(s\ge0\).  Dominated convergence
then permits differentiation under the integral sign, so \(F\) is holomorphic
on this cut plane and its restriction to \(z=Q^2>0\) is the Euclidean function in
\eqref{eq:euclidean-stieltjes-propagator}.  The endpoint \(z=0\) is part of
the cut if the spectrum has support arbitrarily near \(s=0\); no continuation
through that endpoint is being assumed.  If the spectral support begins at a
positive mass gap \(s=m_0^2>0\), the cut instead begins at \(z=-m_0^2\), and
the function is analytic in a neighborhood of \(z=0\).
Consequently, for every \(Q^2>0\),
\[
  (-1)^r
  \frac{\dd^r}{\dd (Q^2)^r}\widetilde G_E(Q^2)
  =
  r!\int_{[0,\infty)}
  \frac{\dd\rho(s)}{(Q^2+s)^{r+1}}
  \ge0,
  \qquad r=0,1,2,\ldots .
\]
These inequalities are not perturbative statements.  They are Hilbert-space
positivity written in Euclidean momentum variables.

The measure can be recovered from the discontinuity across the negative real
\(z\)-axis.  The continuation is performed by treating \(z\) as a complex
variable on the cut plane and approaching the point \(z=-s\) from the two
sides of the cut.  If the absolutely continuous part has density
\(\rho_{\rm ac}(s)\), then
\[
  \rho_{\rm ac}(s)
  =
  -\frac{1}{2\pi\ii}
  \operatorname{Disc}_{z=-s}\widetilde G_E(z),
  \qquad s>0,
\]
with atoms appearing as simple poles at \(z=-s_0\) of positive residue.  This
is the Euclidean form of the statement that stable particles are isolated
positive atoms of the spectral measure and multiparticle states are continuous
spectral support.
\end{proof}

\begin{proposition}[Subtracted Stieltjes form and contact terms]
\label{prop:kallen-lehmann-subtracted-stieltjes-contact-terms}
If the spectral measure has tempered growth in the sense that there is
\(N\ge0\) with
\[
  \int_{[0,\infty)}\frac{\dd\rho(s)}{(1+s)^{N+1}}<\infty ,
\]
then the \((N+1)\)-st derivative of the nonlocal Euclidean momentum-space
two-point function is the convergent Stieltjes transform
\[
  F^{(N+1)}(z)
  =
  (-1)^{N+1}(N+1)!
  \int_{[0,\infty)}\frac{\dd\rho(s)}{(z+s)^{N+2}} .
\]
Integrating back determines \(F\) only up to a polynomial of degree \(N\),
which is the momentum-space ambiguity of contact terms at coincident points.
Under inverse Fourier transform this polynomial is precisely a finite linear
combination of derivatives of the delta distribution supported at the
coincident diagonal.
\end{proposition}

\begin{proof}
For a more singular local field, the spectral measure may have only tempered
growth, by which we mean here that there is an integer \(N\ge0\) such that
\[
  \int_{[0,\infty)}\frac{\dd\rho(s)}{(1+s)^{N+1}}<\infty .
\]
The unsubtracted Stieltjes integral need not define an ordinary function.  The
nonlocal part of the Euclidean momentum-space distribution is nevertheless
characterized by its \((N+1)\)-st derivative,
\[
  F^{(N+1)}(z)
  =
  (-1)^{N+1}(N+1)!
  \int_{[0,\infty)}\frac{\dd\rho(s)}{(z+s)^{N+2}} .
\]
The right-hand side is holomorphic on the same cut plane.  If \(F_1\) and
\(F_2\) are two distributional momentum-space representatives with this same
\((N+1)\)-st derivative, their difference \(P=F_1-F_2\) satisfies
\(\partial_z^{N+1}P=0\) on the cut plane.  On every simply connected
subdomain this implies that \(P\) is a polynomial of degree at most \(N\);
the polynomial pieces agree on overlaps, so the same polynomial ambiguity is
global.  In momentum space a polynomial in \(p_E^2\) and, in the nonscalar
case, polynomial tensor structures, Fourier-transform to finite linear
combinations of derivatives of \(\delta(x)\).  These are exactly
coincident-point contact terms.  They have no discontinuity across the cut,
and therefore do not change the spectral measure away from coincident-point
ambiguities.
\end{proof}

\section{Canonical Normalization and Spectral Weight}

\begin{definition}[Canonical equal-time normalization of a scalar field]
\label{def:kallen-lehmann-canonical-equal-time-normalization}
The measure \(d\rho\) depends on the normalization of the local field.  A
canonical equal-time normalization consists of a regularity datum on the
operator-valued distribution together with a smeared commutator identity.
Let \(d=D-1\).  Assume there is a dense common invariant domain
\(\mathcal D\subset\Hilb\), with \(\vac\in\mathcal D\), such that for every
\(f\in\mathcal S(\mathbb R^d)\) the smeared equal-time fields
\[
  \widehat\phi_t(f)=\int\dd^d\vec x\,f(\vec x)\widehat\phi(t,\vec x),
  \qquad
  \widehat\pi_t(f)=\int\dd^d\vec x\,f(\vec x)\widehat\pi(t,\vec x)
\]
preserve \(\mathcal D\), and
\[
  \frac{\dd}{\dd t}\widehat\phi_t(f)\psi
  =
  \widehat\pi_t(f)\psi,
  \qquad \psi\in\mathcal D,
\]
where the derivative is taken in the Hilbert norm.  This is \(C^1\)
distributional time-regularity of \(\widehat\phi\).  The canonical
normalization is the domain identity
\[
  [\widehat\phi_t(f),\widehat\pi_t(g)]\psi
  =
  \ii\int_{\mathbb R^d} f(\vec x)g(\vec x)\,\dd^d\vec x\,\psi,
  \qquad f,g\in\mathcal S(\mathbb R^d),\quad \psi\in\mathcal D .
\]
The formal notation
\[
  [\widehat\phi(t,\vec x),\partial_t\widehat\phi(t,\vec y)]
  =
  \ii\,\delta^{(d)}(\vec x-\vec y)\mathbf 1
\]
means precisely this smeared statement with
\(\widehat\pi_t=\partial_t\widehat\phi_t\) as an operator-valued distribution.
An interacting local field used as an interpolating operator need not satisfy
this canonical identity.  If an analogous equal-time commutator exists with a
central normalization \(C_\phi\ge0\),
\[
  [\widehat\phi_t(f),\widehat\pi_t(g)]\psi
  =
  \ii C_\phi
  \int_{\mathbb R^d} f(\vec x)g(\vec x)\,\dd^d\vec x\,\psi ,
\]
then the same argument below gives \(\int\dd\rho=C_\phi\).  If no such
equal-time normalization is part of the field data, the K\"all\'en--Lehmann
representation by itself gives positivity and support, but not a total-weight
sum rule.
\end{definition}

\begin{proposition}[Canonical spectral-weight sum rule]
\label{prop:kallen-lehmann-canonical-spectral-sum-rule}
Assume the canonical equal-time normalization of
Definition~\ref{def:kallen-lehmann-canonical-equal-time-normalization}, and
assume that the spectral truncations converge after applying
\(\partial_t|_{t=0}\) in \(\mathcal S'(\mathbb R^{D-1})\).  Then
\[
  \int_{[0,\infty)}\dd\rho(s)=1 .
\]
If the equal-time commutator has central normalization \(C_\phi\), then
\[
  \int_{[0,\infty)}\dd\rho(s)=C_\phi .
\]
\end{proposition}

\begin{proof}
The commutator two-point distribution is
\[
  C(x):=
  \langle\vac|[\widehat\phi(x),\widehat\phi(0)]|\vac\rangle
  =
  \int_{[0,\infty)}\dd\rho(s)\,
  \bigl(\Delta_+(x;s)-\Delta_+(-x;s)\bigr).
\]
The sign of the equal-time derivative is fixed by commutator antisymmetry.
The canonical identity gives, in smeared form,
\([\widehat\phi_0(\vec r),\widehat\pi_0(\vec0)]
=\ii\delta^{(d)}(\vec r)\mathbf1\).  Since the derivative
\(\partial_t C(t,\vec r)|_{t=0}\) differentiates the first field in
\([\widehat\phi(t,\vec r),\widehat\phi(0,\vec0)]\), it produces
\[
  [\widehat\pi_0(\vec r),\widehat\phi_0(\vec0)]
  =
  -[\widehat\phi_0(\vec0),\widehat\pi_0(\vec r)]
  =
  -\ii\delta^{(d)}(\vec r)\mathbf1,
\]
where the last equality uses the symmetry of the spatial delta distribution.
The \(C^1\) hypothesis gives the equal-time derivative as a spatial tempered
distribution:
\begin{equation}
  \left.\partial_t C(t,\vec r)\right|_{t=0}
  =
  \langle\vac|[\widehat\pi_0(\vec r),\widehat\phi_0(\vec 0)]|\vac\rangle
  =
  -\ii\,\delta^{(d)}(\vec r).
  \label{eq:kallen-lehmann-equal-time-commutator-derivative}
\end{equation}

It remains to justify the spectral calculation of the same derivative.  Let
\(\rho_R\) be the restriction of \(\rho\) to \([0,R]\), and let
\[
  C_R(t,\vec r)=
  \int_{[0,R]}\dd\rho(s)\,
  \bigl(\Delta_+(t,\vec r;s)-\Delta_+(-t,-\vec r;s)\bigr).
\]
The measure \(\rho_R\) is finite by local finiteness.  For a spatial test
function \(h\in\mathcal S(\mathbb R^d)\), set
\[
  F_s(t;h)=
  \int_{\mathbb R^d}\dd^d\vec r\,h(\vec r)
  \bigl(\Delta_+(t,\vec r;s)-\Delta_+(-t,-\vec r;s)\bigr).
\]
Writing \(\omega_{\vec p,s}=\sqrt{\vec p^{\,2}+s}\), the spatial Fourier
representation gives
\[
\begin{aligned}
  \frac{\dd}{\dd t}F_s(t;h)
  &=
  -\frac{\ii}{2}
  \int\frac{\dd^d\vec p}{(2\pi)^d}
  \left(
    \widetilde h(\vec p)\ee^{-\ii\omega_{\vec p,s}t}
    +
    \widetilde h(-\vec p)\ee^{+\ii\omega_{\vec p,s}t}
  \right),
\end{aligned}
\]
where
\[
  \widetilde h(\vec p)
  =
  \int_{\mathbb R^d}\dd^d\vec r\,
  h(\vec r)\ee^{\ii\vec p\cdot\vec r}.
\]
The bound
\[
  \left|\frac{\dd}{\dd t}F_s(t;h)\right|
  \le
  \int\frac{\dd^d\vec p}{(2\pi)^d}\,|\widetilde h(\vec p)|
\]
is uniform in \(s\ge0\) and \(t\).  Hence, for every finite \(R\), dominated
convergence permits the derivative to pass through the \(s\)-integral.  At
\(t=0\),
\[
  \left.\frac{\dd}{\dd t}F_s(t;h)\right|_{t=0}
  =
  -\ii h(\vec0).
\]
Therefore
\[
  \left\langle
    \left.\partial_t C_R(t,\cdot)\right|_{t=0},
    h
  \right\rangle
  =
  -\ii\,\rho([0,R])\,h(\vec0).
\]

The final limiting step is part of the equal-time regularity assumption: the
spectral truncations \(C_R\) converge to \(C\) after applying the operation
\(\partial_t|_{t=0}\) in \(\mathcal S'(\mathbb R^d)\).  Comparing with
\eqref{eq:kallen-lehmann-equal-time-commutator-derivative} gives
\[
  \rho([0,R])\longrightarrow 1
  \qquad (R\to\infty),
\]
and monotone convergence yields the canonical sum rule
\begin{equation}
  \int_{[0,\infty)}\dd\rho(s)=1.
  \label{eq:kallen-lehmann-canonical-sum-rule}
\end{equation}
It fixes the chosen field normalization; changing the field normalization
rescales the spectral measure.
If the equal-time commutator has central normalization \(C_\phi\), the right
hand side of
\eqref{eq:kallen-lehmann-equal-time-commutator-derivative} is multiplied by
\(C_\phi\), and the same comparison gives
\(\int_{[0,\infty)}\dd\rho(s)=C_\phi\).
\end{proof}

\paragraph{One-particle residue bound for a canonically normalized field.}
If a canonically normalized scalar field has a one-particle atomic
contribution \(Z\,\delta_{m^2}(\dd s)\), then
\[
  0\le Z\le1 .
\]
More generally, if the equal-time commutator has central normalization
\(C_\phi\), then \(0\le Z\le C_\phi\).  Without such a finite equal-time
normalization, the atom weight \(Z\) is positive for the chosen field but has
no universal upper bound.

If the same canonically normalized field has a one-particle atomic contribution
\(Z\,\delta_{m^2}(\dd s)\), where \(\delta_{m^2}\) is the unit point measure at
\(s=m^2\), positivity of \(d\rho\) and
\eqref{eq:kallen-lehmann-canonical-sum-rule} imply
\[
  0\le Z\le1.
\]
The remainder \(1-Z\) is the total spectral weight carried by all continuum
states and by any other atoms in this normalization.  For a noncanonical
interpolating field, a composite operator, or a renormalized field in a
different convention, the same K\"all\'en--Lehmann representation holds, but
\eqref{eq:kallen-lehmann-canonical-sum-rule} is replaced, when an equal-time
commutator normalization exists, by
\[
  \int_{[0,\infty)}\dd\rho(s)=C_\phi .
\]
Then positivity gives \(0\le Z\le C_\phi\).  Without such a finite
normalization, the coefficient \(Z\) is still a positive spectral weight of the
chosen interpolating field, but it has no universal upper bound; rescaling the
field rescales \(Z\).  The canonical free field used later in LSZ is the
asymptotic field on the one-particle Fock space, normalized separately from the
interpolating local field whose correlators contain the pole.

\begin{remark}[Support hypothesis for a single massive scalar channel]
\label{rem:kallen-lehmann-single-massive-support}
The lower endpoint \(0\) in the K\"all\'en--Lehmann integral is conventional:
it is the possible lower endpoint of invariant mass squared under the spectrum
condition, not a claim that the chosen field has spectral weight at zero.  In
the special situation used by the elementary scalar LSZ discussion, assume:

\begin{enumerate}[label=(\roman*)]
  \item the theory has no massless particle seen by \(\widehat\phi\), so
    \(d\rho_\phi(\{0\})=0\);
  \item the field has nonzero overlap with a stable scalar one-particle mass
    \(m>0\), giving the atom \(Z_\phi\delta_{m^2}(\dd s)\);
  \item there are no lighter massive particles or bound-state atoms in this
    channel; and
  \item the non-one-particle part created by \(\widehat\phi\) begins with
    states containing at least two particles of mass \(m\).
\end{enumerate}

Then the non-vacuum spectral measure has the support decomposition
\[
  \operatorname{supp}d\rho_\phi
  \subset
  \{m^2\}\cup[4m^2,\infty),
  \qquad
  \dd\rho_\phi(s)
  =
  Z_\phi\delta_{m^2}(\dd s)+\dd\rho_{\phi,\mathrm c}(s),
\]
with \(\operatorname{supp}d\rho_{\phi,\mathrm c}\subset[4m^2,\infty)\).  This
gap between the atom and the rest of the support is the spectral input used
later by Haag--Ruelle and LSZ.  If massless particles, infraparticle
thresholds, or additional bound-state atoms are present, the displayed support
decomposition must be replaced by the corresponding spectral statement.
\end{remark}

\begin{remark}[Existence of an isolated mass shell is a spectral statement]
\label{rem:kallen-lehmann-mass-shell-existence-status}
The K\"all\'en--Lehmann representation does not imply the existence of a
one-particle state.  For a fixed scalar local field \(\widehat\phi\), a term
\[
  \dd\rho_\phi(s)=Z_\phi\,\delta_{m^2}(\dd s)+\dd\rho_{\phi,\mathrm c}(s),
  \qquad Z_\phi>0,
\]
is exactly the statement that \(\widehat\phi\vac\) has nonzero spectral
projection onto the positive mass shell of invariant mass \(m\).  A positive
measure whose support begins at \(m^2\) but has no atom at \(m^2\) gives a
threshold, not a normalizable Wigner one-particle vector created by that
field.

The field-independent Hilbert-space assertion needed by Haag--Ruelle theory is
stronger and has two parts.  With \(E\) the joint spectral measure of
\(P^\mu\) and
\[
  \Sigma_m^+=\{p:p^0>0,\ p^2=-m^2\},
\]
one must have
\[
  E(\Sigma_m^+)\Hilb\ne\{0\},
  \qquad
  \operatorname{Spec}(P)\cap N=\Sigma_m^+
\]
for some open neighborhood \(N\) of \(\Sigma_m^+\).  The first condition says
that the one-particle subspace is nonzero; the second says that the shell is
isolated from the rest of the energy-momentum spectrum.  If the first condition
fails, the notation \(E(\Sigma_m^+)\Hilb\) denotes the zero subspace and the
corresponding scattering construction is vacuous.  If the second condition
fails, the sharp Wigner particle hypothesis required by ordinary
Haag--Ruelle estimates is absent even when spectral weight accumulates at the
same invariant mass.

An atom may be invisible to a particular interpolating field if that field has
zero matrix element to the corresponding superselection sector or internal
channel.  Conversely, absence of an atom in every physical local
interpolating field is evidence for absence of that stable particle sector, but
the invariant statement is the vanishing of the spectral projection on the
physical Hilbert space.  The nonzero isolated-shell property is a theorem in
free massive fields and in those constructive massive models for which the
Wightman theory, the mass gap, and a separated one-particle spectral component
have actually been proved.  Formal perturbation theory alone supplies
candidate pole data rather than this Hilbert-space spectral theorem.
\end{remark}

\section{Scalar One-Particle States and Normalization}

\begin{definition}[Scalar massive one-particle normalization datum]
\label{def:kallen-lehmann-scalar-one-particle-normalization-datum}
Assume now that the Hilbert space contains an isolated positive-mass
one-particle subspace with mass \(m>0\). In this section ``scalar particle''
means that the corresponding massive Wigner representation has trivial
little-group representation and no internal degeneracy. This is a property of
the one-particle representation; it is not the same statement as saying that a
particular local field is a Lorentz scalar.

Let
\[
  k_R^\mu=(m,\vec 0),
  \qquad
  k^\mu=(\omega_{\vec k},\vec k),
  \qquad
  \omega_{\vec k}=\sqrt{\vec k^{\,2}+m^2}.
\]
Choose, for every \(\vec k\), a standard boost \(L(\vec k)\in SO^+(1,D-1)\)
such that
\[
  L(\vec k)k_R=k.
\]
With generalized momentum eigenvectors normalized by
\[
  {}_\delta\langle\vec k\,{}'|\vec k\rangle_\delta
  =
  \delta^{(D-1)}(\vec k\,{}'-\vec k),
\]
write
\[
  |\vec k\rangle_\delta
  =
  N(\vec k)\,U(L(\vec k))|\vec 0\rangle_\delta,
  \qquad
  N(\vec 0)=1.
\]
\end{definition}

\begin{lemma}[Delta-normalized and invariant scalar momentum states]
\label{lem:kallen-lehmann-one-particle-state-normalization}
For the scalar massive one-particle representation,
\[
  N(\vec k)=\sqrt{\frac{m}{\omega_{\vec k}}},
\]
up to a phase convention.  The invariantly normalized states
\[
  |\vec k\rangle_{\rm inv}
  =
  \sqrt{(2\pi)^{D-1}2\omega_{\vec k}}\,
  |\vec k\rangle_\delta
\]
obey
\[
  {}_{\rm inv}\langle\vec k\,{}'|\vec k\rangle_{\rm inv}
  =
  (2\pi)^{D-1}2\omega_{\vec k}\,
  \delta^{(D-1)}(\vec k\,{}'-\vec k),
  \qquad
  U(\Lambda)|\vec k\rangle_{\rm inv}=|\Lambda\vec k\rangle_{\rm inv}.
\]
\end{lemma}

\begin{proof}
For a Lorentz transformation \(\Lambda\), define the Wigner rotation
\[
  W(\Lambda,k)=L(\Lambda k)^{-1}\Lambda L(k).
\]
It fixes \(k_R\), hence lies in the little group \(SO(D-1)\). For the scalar
one-particle representation, \(U(W(\Lambda,k))\) acts trivially on the rest
state. Therefore
\[
  U(\Lambda)|\vec k\rangle_\delta
  =
  \frac{N(\vec k)}{N(\Lambda\vec k)}
  |\Lambda\vec k\rangle_\delta,
\]
where \(\Lambda\vec k\) denotes the spatial part of \(\Lambda k\).
Unitarity fixes \(N\). Since the on-shell measure
\[
  \frac{\dd^{D-1}\vec k}{2\omega_{\vec k}}
\]
is Lorentz invariant,
\[
  \delta^{(D-1)}(\Lambda\vec k\,{}'-\Lambda\vec k)
  =
  \frac{k^0}{(\Lambda k)^0}
  \delta^{(D-1)}(\vec k\,{}'-\vec k).
\]
Comparing
\[
  {}_\delta\langle\vec k\,{}'|\vec k\rangle_\delta
  =
  {}_\delta\langle\vec k\,{}'|U(\Lambda)^\dagger U(\Lambda)|\vec k\rangle_\delta
\]
gives
\[
  |N(\Lambda\vec k)|^2
  =
  |N(\vec k)|^2\,\frac{k^0}{(\Lambda k)^0}.
\]
Taking \(k=k_R\) and \(\Lambda=L(\vec k)\), together with \(N(\vec0)=1\),
yields
\[
  N(\vec k)=\sqrt{\frac{m}{\omega_{\vec k}}},
\]
after fixing the harmless phase convention.

It is often better to use invariantly normalized states
\[
  |\vec k\rangle_{\rm inv}
  =
  \sqrt{(2\pi)^{D-1}2\omega_{\vec k}}\,
  |\vec k\rangle_\delta,
\]
for which
\[
  {}_{\rm inv}\langle\vec k\,{}'|\vec k\rangle_{\rm inv}
  =
  (2\pi)^{D-1}2\omega_{\vec k}\,
  \delta^{(D-1)}(\vec k\,{}'-\vec k),
  \qquad
  U(\Lambda)|\vec k\rangle_{\rm inv}=|\Lambda\vec k\rangle_{\rm inv}.
\]
\end{proof}

\begin{proposition}[Scalar field matrix element on an isolated scalar shell]
\label{prop:kallen-lehmann-scalar-field-one-particle-matrix-element}
Work in a selected spin-zero isolated one-particle channel with the
delta-normalized and invariantly normalized states of
Lemma~\ref{lem:kallen-lehmann-one-particle-state-normalization}.  If
\(\widehat\phi\) is a scalar local field and the vacuum is invariant, then the
matrix element
\[
  A(\vec k)
  =
  {}_\delta\langle\vec k|\widehat\phi(0)|\vac\rangle
\]
has the same \(\vec k\)-dependence as \(N(\vec k)\).  Equivalently, there is
a nonnegative number \(Z\), depending on the chosen field, such that
\[
  {}_\delta\langle\vec k|\widehat\phi(0)|\vac\rangle
  =
  \left[
    \frac{Z}{(2\pi)^{D-1}2\omega_{\vec k}}
  \right]^{1/2},
\]
or equivalently
\[
  {}_{\rm inv}\langle\vec k|\widehat\phi(0)|\vac\rangle
  =
  Z^{1/2}.
\]
\end{proposition}

\begin{proof}
The point notation is distributional: equivalently one may smear
\(\widehat\phi\) in a small neighborhood of the origin and then read the
coefficient of the isolated shell.  The isolated-shell projection removes
continuum contributions, and the assumption of a selected spin-zero channel
means that the little group acts trivially on the rest vector up to the phase
already fixed in
Lemma~\ref{lem:kallen-lehmann-one-particle-state-normalization}.
The scalar covariance equation
\(U(\Lambda)\widehat\phi(0)U(\Lambda)^{-1}=\widehat\phi(0)\), vacuum
invariance, and
Lemma~\ref{lem:kallen-lehmann-one-particle-state-normalization} imply
the transformation law
\[
  A(\vec k)
  =
  \frac{N(\vec k)}{N(\Lambda\vec k)}
  A(\Lambda\vec k).
\]
Taking
\(\Lambda=L(\vec k)^{-1}\) reduces the matrix element to its rest-frame value
and gives \(A(\vec k)=N(\vec k)A(\vec0)\).  If \(A(\vec0)=0\), then the field
has zero overlap with this channel and \(Z=0\).  Otherwise the remaining phase
freedom of the one-particle states fixes \(A(\vec0)\ge0\).  Writing
\(Z=(2\pi)^{D-1}2m\,|A(\vec0)|^2\) gives the delta-normalized formula, and
the invariant normalization removes the kinematic factor.  If several
spin-zero channels of the same mass are present, the same argument applies to
the vector of rest-frame overlaps in the finite or direct-integral
multiplicity space; choosing a normalized component gives the scalar formula
above, while \(Z\) for the full field is the squared norm of that overlap
vector.
\end{proof}

\section{One-Particle Atoms}

\begin{proposition}[One-particle atom in the scalar spectral measure]
\label{prop:kallen-lehmann-one-particle-atom}
Assume the spectrum contains an isolated scalar one-particle mass \(m>0\), and
assume the local field \(\widehat\phi\) has nonzero projection onto that
one-particle subspace.  With the invariant normalization from
Lemma~\ref{lem:kallen-lehmann-one-particle-state-normalization}, define
\(Z\ge0\) by
\[
  {}_{\rm inv}\langle\vec p|\widehat\phi(0)|\vac\rangle=Z^{1/2}.
\]
Then the one-particle contribution to the spectral measure is
\[
  \dd\rho_1(\mu^2)=Z\,\delta_{m^2}(\dd\mu^2),
\]
so
\[
  \dd\rho(\mu^2)
  =
  Z\,\delta_{m^2}(\dd\mu^2)
  +
  \dd\rho_{\mathrm c}(\mu^2)
\]
with \(d\rho_{\mathrm c}\) positive.
\end{proposition}

\begin{proof}
With the invariant normalization from the one-particle construction,
\[
  \langle\vec p\,{}'|\vec p\rangle
  =
  (2\pi)^{D-1}2\omega_{\vec p}\,
  \delta^{(D-1)}(\vec p\,{}'-\vec p),
  \qquad
  \omega_{\vec p}=\sqrt{\vec p^{\,2}+m^2},
\]
define \(Z\ge0\) by
\[
  {}_{\rm inv}\langle\vec p|\widehat\phi(0)|\vac\rangle=Z^{1/2}
\]
for scalar one-particle states. Lorentz covariance makes the right-hand side
independent of \(\vec p\) in this normalization. Let \(\delta_{m^2}\) denote
the unit point measure at \(\mu^2=m^2\). The one-particle contribution to the
spectral measure is
\[
  \dd\rho_1(\mu^2)=Z\,\delta_{m^2}(\dd\mu^2).
\]
Thus
\[
  \dd\rho(\mu^2)
  =
  Z\,\delta_{m^2}(\dd\mu^2)
  +
  \dd\rho_{\mathrm c}(\mu^2),
\]
where \(d\rho_{\mathrm c}\) is the remaining positive measure.
At the level of the momentum-space measure, using the delta-normalized states
in the previous section, the same statement is the calculation
\[
\begin{aligned}
  \dd\nu_{\phi,1}(p)
  &=
  \int\dd^{D-1}\vec k\,
  \left|
    {}_\delta\langle\vec k|\widehat\phi(0)|\vac\rangle
  \right|^2
  \delta^{(D)}(p-k)\,\dd^Dp
  \\
  &=
  \int\dd^{D-1}\vec k\,
  \frac{Z}{(2\pi)^{D-1}2\omega_{\vec k}}\,
  \delta^{(D)}(p-k)\,\dd^Dp
  =
  \frac{Z\,\dd^Dp}{(2\pi)^{D-1}}\,
  \theta(p^0)\delta(p^2+m^2).
\end{aligned}
\]
Comparison with the invariant-mass decomposition then gives
\(\dd\rho_1(\mu^2)=Z\,\delta_{m^2}(\dd\mu^2)\).
\end{proof}

The spectral decomposition of \(\widehat\phi(f)\vac\) used in the preceding
measure calculation is the orthogonal sum of the vacuum atom
\(E(\{0\})\widehat\phi(f)\vac\), the isolated one-particle atom
\(E(\mu^2=m^2)\widehat\phi(f)\vac\), and the continuum projection
\(E_{\mathrm c}\widehat\phi(f)\vac\).  After subtracting the vacuum
expectation value of \(\widehat\phi\), the vacuum projection is absent.  The
one-particle atom remains precisely when the field has nonzero overlap with
the isolated mass shell.

Under the same assumptions,
\[
  G_T(x)
  =
  Z\,\Delta_F(x;m^2)
  +
  \int_{[0,\infty)}
  \dd\rho_{\mathrm c}(\mu^2)\,
  \Delta_F(x;\mu^2).
\]
If the theory has a single stable particle of mass \(m\), no lighter
particles, and the continuum part is generated by states containing at least
two such particles, then
\[
  \operatorname{supp}d\rho_{\mathrm c}\subset[4m^2,\infty).
\]
When \(d\rho_{\mathrm c}\) has a density \(\sigma(\mu^2)\) in that region,
\[
  G_T(x)
  =
  Z\,\Delta_F(x;m^2)
  +
  \int_{4m^2}^{\infty}
  \dd\mu^2\,\sigma(\mu^2)\Delta_F(x;\mu^2).
\]

Figure~\ref{fig:kallen-lehmann-pole-and-continuum} displays this threshold
structure in the spectral measure: the isolated atom at \(m^2\) and the
continuum support beginning at \(4m^2\) under the stated assumption.

The number \(Z\) is attached to the chosen local field. Rescaling the field
rescales \(Z\). Once the field normalization is fixed, \(Z\) is a definite
matrix element and the one-particle pole residue of its two-point function.
It is therefore not an unphysical datum in the following precise sense: it is
not invariant under changing the operator whose two-point function is being
studied, but for a fixed operator it is an unambiguous spectral weight.

\section{Role in the Construction}

The K\"all\'en--Lehmann representation is nonperturbative. It states that
Hilbert-space positivity and the spectrum condition force the scalar
two-point function to be a positive superposition of free massive two-point
functions. Isolated atoms of the measure identify stable one-particle mass
shells seen by the local field. Continuous spectral support records the
remaining invariant-mass spectrum.

The local field \(\widehat\phi\) in this chapter is an interpolating local
field.  Its time-ordered two-point function contains the full spectral measure
\[
  G_T^\phi(x)
  =
  Z\,\Delta_F(x;m^2)
  +
  \int_{[0,\infty)}\dd\rho_{\mathrm c}(s)\,\Delta_F(x;s)
\]
whenever an isolated one-particle atom of mass \(m\) is present.  The
asymptotic scalar field associated with that atom is a different object: it is
the canonical free scalar field on the asymptotic Fock space generated by the
one-particle Hilbert space \(\Hilb_1\), and its two-point function is
\[
  G_T^{\mathrm{as}}(x)=\Delta_F(x;m^2)
\]
with unit one-particle residue and no continuum part.  The relation between the
two fields is the one-particle projection and normalization
\[
  P_1\widehat\phi(f)\vac
  =
  Z^{1/2}\psi_f,
\]
where, in the Fourier convention
\[
  f_-(p)=\int\dd^D x\,f(x)\ee^{\ii p\cdot x},
\]
\[
  \psi_f=
  \int_{\Sigma_m^+}\dd\mu_m(p)\,f_-(p)|p\rangle
  \in\Hilb_1 .
\]
Haag--Ruelle constructs the asymptotic Fock space from such vectors, and LSZ
extracts the corresponding pole coefficient from correlation functions of the
interpolating local field.

\begin{remark}[Infraparticle spectrum in charged QED]
The word ``atom'' here means a literal point mass in the positive spectral
measure, for example \(Z\,\delta_{m^2}(\dd s)\) with \(Z>0\).  This is exactly
the input needed later by Haag--Ruelle theory and by LSZ pole-coefficient
extraction.  The
charged sector of QED with a massless photon realizes a different spectral
structure.  It contains the charged threshold together with states differing by
photons of arbitrarily small energy; the energy-momentum spectral support
therefore accumulates at the threshold instead of leaving an open neighborhood
containing only \(\Sigma_m^+\).  Equivalently, the charged particle is an
infraparticle: threshold spectral support replaces the simple one-particle
pole, and the charged sector has no normalizable one-particle Wigner subspace
obtained from \(P_1=E(\Sigma_m^+)\).

Gauge-fixed perturbative notation must be interpreted through this physical
spectral statement.  A gauge-dependent electron propagator is not, by itself,
the positive K\"all\'en--Lehmann measure of a compactly localized physical
observable in the Hilbert space.  Physical charged asymptotic data require
either inclusive probabilities or dressed charged states.  The later infrared
chapters develop the corresponding Bloch--Nordsieck cancellation and
Faddeev--Kulish-type dressing constructions in
Chapter~\ref{ch:qed-external-states} and
Chapter~\ref{ch:volume-ii-infrared-inclusive-qed}.
\end{remark}

\begin{remark}[Confinement and physical spectral atoms]
In a confining nonabelian gauge theory the positive Hilbert-space spectral
measure is attached to fields or observables that act on the physical Hilbert
space.  Gauge-invariant color-singlet interpolating fields have
K\"all\'en--Lehmann measures supported on hadronic states; when such a measure
has an isolated atom, the atom corresponds to a stable hadron or glueball seen
by that interpolating field.  Gauge-dependent quark and gluon fields are not
local physical observables with positive spectral measures on the physical
Hilbert space, and confinement excludes isolated quark or gluon mass-shell
atoms from the physical asymptotic spectrum.  Consequently the Haag--Ruelle and
LSZ constructions later in this volume apply in a confining theory only to
stable color-singlet particles, after the corresponding physical interpolating
operators and isolated hadronic atoms have been identified.
\end{remark}

The next step in the construction is the perturbative computation of Green
functions. The construction of scattering states and the S-matrix comes later,
after the spectral one-particle data have been separated from the local
two-point functions.
