\section{The Conformal Killing Equation}

Work first in flat Euclidean space \((\R^D,\delta_{\mu\nu})\) with
\(D\ge3\).  Cartesian coordinates are denoted by \(x^\mu\), and repeated
indices are contracted with \(\delta_{\mu\nu}\).  A smooth vector field
\(\epsilon^\mu(x)\) generates an
infinitesimal coordinate transformation
\[
  x^\mu\longmapsto x^\mu+\epsilon^\mu(x).
\]

\begin{definition}[Euclidean conformal Killing vector]
\label{def:euclidean-conformal-killing-vector}
A smooth vector field \(\epsilon=\epsilon^\mu\partial_\mu\) on an open subset
of \(\R^D\) is a conformal Killing vector when the change of the metric is
pointwise proportional to the metric:
\[
  \mathcal L_\epsilon\delta_{\mu\nu}
  =
  \partial_\mu\epsilon_\nu+\partial_\nu\epsilon_\mu
  =
  2\sigma_\epsilon(x)\delta_{\mu\nu}
\]
for some scalar function \(\sigma_\epsilon\).  Taking the trace gives
\[
  \sigma_\epsilon(x)
  =
  \frac1D\partial_\rho\epsilon^\rho(x).
\]
Thus the conformal Killing equation is
\begin{equation}
  \partial_\mu\epsilon_\nu+\partial_\nu\epsilon_\mu
  -
  \frac2D\delta_{\mu\nu}\partial_\rho\epsilon^\rho
  =
  0.
  \label{eq:conformal-killing-equation}
\end{equation}
\end{definition}

\begin{proposition}[Conformal current from a traceless stress tensor]
\label{prop:conformal-current-from-traceless-stress-tensor}
Let \(T_{\mu\nu}\) be symmetric, conserved, and traceless, and let
\(\epsilon\) obey \eqref{eq:conformal-killing-equation}.  Then
\[
  J^\mu_\epsilon(x)=T^{\mu}{}_{\nu}(x)\epsilon^\nu(x)
\]
is conserved.  The charge
\[
  Q_\epsilon(\Sigma)
  =
  \int_\Sigma \dd\sigma\,n_\mu J^\mu_\epsilon
\]
on a closed codimension-one surface \(\Sigma\) depends only on the homology
class of \(\Sigma\) relative to operator insertions.
\end{proposition}

\begin{proof}
The divergence is
\[
  \partial_\mu J^\mu_\epsilon
  =
  (\partial_\mu T^{\mu}{}_{\nu})\epsilon^\nu
  +
  T^{\mu\nu}\partial_\mu\epsilon_\nu
  =
  \frac12T^{\mu\nu}
  (\partial_\mu\epsilon_\nu+\partial_\nu\epsilon_\mu)
  =
  \sigma_\epsilon T^\mu{}_\mu
  =
  0.
\]
Conservation makes the charge invariant under deformations of \(\Sigma\) that
do not cross operator insertions.
After analytic continuation to Lorentzian signature this formula is read on a
spacelike Cauchy surface \(\Sigma\).  The measure \(\dd\sigma\) is the induced
spatial volume form, \(n^\mu\) is the future-directed unit timelike normal, and
\begin{equation}
  Q_\epsilon(\Sigma)
  =
  \int_\Sigma \dd\sigma\,n_\mu T^\mu{}_\nu\epsilon^\nu .
  \label{eq:lorentzian-conformal-charge}
\end{equation}
If \(\Sigma_1\) and \(\Sigma_2\) bound a spacetime region \(V\) containing no
operator insertions, Stokes' theorem gives
\[
  Q_\epsilon(\Sigma_2)-Q_\epsilon(\Sigma_1)
  =
  \int_V \dd^D x\,\partial_\mu J^\mu_\epsilon
  =
  0.
\]
This is the precise sense in which the charge depends on the homology class of
the surface relative to insertions.
\end{proof}

surface-deformation statement: two spacelike slices homologous relative to the
insertions define the same charge.

\section{Solutions in Dimension at Least Three}

\begin{lemma}[Rigidity of the flat conformal factor]
\label{lem:flat-ckv-conformal-factor-rigidity}
Let \(D\ge3\), and let \(\epsilon\) solve the flat conformal Killing equation
\eqref{eq:conformal-killing-equation} on a connected open set.  Then
\[
  \partial_\mu\partial_\nu\sigma_\epsilon=0 .
\]
Consequently \(\sigma_\epsilon\) is affine-linear in \(x\).
\end{lemma}

\begin{proof}
The conformal Killing equation is
\[
  \partial_\mu\epsilon_\nu+\partial_\nu\epsilon_\mu
  =
  2\sigma_\epsilon\delta_{\mu\nu}.
\]
Differentiate this identity and combine the three equations obtained by
cyclically permuting \((\mu,\nu,\rho)\).  The result is
\begin{equation}
  \partial_\mu\partial_\nu\epsilon_\rho
  =
  \delta_{\rho\nu}\partial_\mu\sigma_\epsilon
  +
  \delta_{\rho\mu}\partial_\nu\sigma_\epsilon
  -
  \delta_{\mu\nu}\partial_\rho\sigma_\epsilon .
  \label{eq:ckv-second-derivative-identity}
\end{equation}
Set
\[
  S_{\mu\nu}:=\partial_\mu\partial_\nu\sigma_\epsilon .
\]
Since third derivatives of \(\epsilon_\rho\) commute, differentiating
\eqref{eq:ckv-second-derivative-identity} with respect to \(x^\kappa\) and
then interchanging \(\kappa\) and \(\mu\) gives
\[
  \delta_{\rho\mu}S_{\kappa\nu}
  -
  \delta_{\mu\nu}S_{\kappa\rho}
  =
  \delta_{\rho\kappa}S_{\mu\nu}
  -
  \delta_{\kappa\nu}S_{\mu\rho}.
\]
Contracting \(\rho\) with \(\mu\) yields
\[
  (D-2)S_{\kappa\nu}
  =
  -\delta_{\kappa\nu}S_\lambda{}^\lambda .
\]
Taking the trace gives
\[
  (2D-2)S_\lambda{}^\lambda=0.
\]
For \(D\ge3\), this implies \(S_\lambda{}^\lambda=0\), and then the preceding
display gives \(S_{\kappa\nu}=0\).  Thus all second derivatives of
\(\sigma_\epsilon\) vanish, so \(\sigma_\epsilon\) is affine-linear on each
connected component.
\end{proof}

\begin{proposition}[Classification of flat conformal Killing vectors]
\label{prop:flat-conformal-killing-vector-classification}
For \(D\ge3\), every flat conformal Killing vector is at most quadratic in the
Cartesian coordinates.  More precisely, the general solution is
\begin{equation}
  \epsilon^\mu(x)
  =
  a^\mu
  +
  \omega^\mu{}_\nu x^\nu
  +
  \lambda x^\mu
  +
  c^\mu x^2-2(c\cdot x)x^\mu,
  \label{eq:ckv-general-solution}
\end{equation}
where \(a^\mu\), \(c^\mu\), and \(\lambda\) are constants and
\(\omega_{\mu\nu}=-\omega_{\nu\mu}\).  These terms generate translations,
rotations, dilatations, and special conformal transformations, respectively.
Equivalently, the Lorentz and dilatation terms may be combined as
\(b^\mu{}_\nu x^\nu\), where \(b_{\mu\nu}\) is either antisymmetric or
proportional to \(\delta_{\mu\nu}\).
\end{proposition}

\begin{proof}
By Lemma~\ref{lem:flat-ckv-conformal-factor-rigidity},
\[
  \sigma_\epsilon(x)=\lambda-2c_\rho x^\rho
\]
for constants \(\lambda\) and \(c_\rho\), where the sign of \(c_\rho\) is
chosen to match \eqref{eq:ckv-general-solution}.  Substituting this affine
form in \eqref{eq:ckv-second-derivative-identity} gives
\[
  \partial_\mu\partial_\nu\epsilon_\rho
  =
  -2\delta_{\rho\nu}c_\mu
  -2\delta_{\rho\mu}c_\nu
  +2\delta_{\mu\nu}c_\rho ,
\]
so the second derivatives of \(\epsilon_\rho\) are constant and all third
derivatives vanish.  Integrating twice gives a polynomial of degree at most
two:
\[
  \epsilon_\rho(x)=a_\rho+B_{\rho\nu}x^\nu
  +c_\rho x^2-2(c\cdot x)x_\rho .
\]
The symmetric part of \(B_{\rho\nu}\) is fixed by the conformal Killing
equation at the origin:
\[
  B_{\rho\nu}+B_{\nu\rho}=2\lambda\delta_{\rho\nu}.
\]
Writing \(B_{\rho\nu}=\omega_{\rho\nu}+\lambda\delta_{\rho\nu}\) with
\(\omega_{\rho\nu}=-\omega_{\nu\rho}\) gives
\eqref{eq:ckv-general-solution}.  Direct substitution verifies that the
displayed vector fields solve \eqref{eq:conformal-killing-equation}.  The
preceding integration shows that no other smooth local solutions exist in
dimension \(D\ge3\).
\end{proof}

\begin{definition}[Differential conformal generators]
\label{def:differential-conformal-generators}
The associated differential operators are
\begin{equation}
  P_\mu=\partial_\mu,\qquad
  M_{\mu\nu}=x_\mu\partial_\nu-x_\nu\partial_\mu,
  \qquad
  D=x^\mu\partial_\mu,
  \label{eq:conformal-differential-generators-1}
\end{equation}
and
\begin{equation}
  K_\mu=x^2\partial_\mu-2x_\mu x^\rho\partial_\rho .
  \label{eq:conformal-differential-generators-2}
\end{equation}
The signs in \eqref{eq:conformal-differential-generators-2} are chosen so
that the finite special conformal transformation below has parameter \(c^\mu\)
with denominator \(1+2c\cdot x+c^2x^2\).
\end{definition}

\section{Charges and Vector Fields}

\begin{convention}[Charge sign convention]
\label{conv:conformal-charge-sign}
Let \(U_\epsilon(s)\) denote the one-parameter unitary family generated by
\(Q_\epsilon\):
\[
  U_\epsilon(s)=\exp(\ii s Q_\epsilon).
\]
The normalization of \(Q_\epsilon\) is fixed by translations and Lorentz
transformations, whose charges are already part of the Poincare
representation of the QFT.  With this normalization, infinitesimal coordinate
transformations compose through the ordinary Lie bracket of vector fields,
\begin{equation}
  [\epsilon_1,\epsilon_2]^\mu
  =
  \epsilon_1^\nu\partial_\nu\epsilon_2^\mu
  -
  \epsilon_2^\nu\partial_\nu\epsilon_1^\mu .
  \label{eq:ckv-lie-bracket}
\end{equation}
\end{convention}

\begin{proposition}[Charge algebra from the vector-field bracket]
\label{prop:conformal-charge-algebra-from-vector-fields}
Assume the charges \(Q_{\epsilon}\) are represented on a common invariant
domain and that no Schwinger contact term contributes to the finite-dimensional
global conformal algebra.  Then
\begin{equation}
  [Q_{\epsilon_1},Q_{\epsilon_2}]
  =
  -\ii Q_{[\epsilon_1,\epsilon_2]} .
  \label{eq:charge-vector-field-commutator}
\end{equation}
\end{proposition}

\begin{proof}
On local operators inserted away from the integration surfaces, the commutator
of the two infinitesimal coordinate transformations is the infinitesimal
transformation generated by the vector-field bracket
\eqref{eq:ckv-lie-bracket}.  The sign in the charge commutator is fixed by
\(U_\epsilon(s)=\exp(\ii sQ_\epsilon)\): the group commutator at order
\(s_1s_2\) is generated by \(-[Q_{\epsilon_1},Q_{\epsilon_2}]\), whereas the
geometric commutator is generated by \(Q_{[\epsilon_1,\epsilon_2]}\).  A
possible discrepancy is a central local contact term; by hypothesis it is
absent in the global algebra.
\end{proof}
Equation \eqref{eq:charge-vector-field-commutator} is an operator statement on
a common invariant domain, or equivalently a Ward identity for correlation
functions with separated insertions.  Possible central extensions or
Schwinger terms are local contact terms; in dimensions \(D\ge3\) the global
finite-dimensional conformal algebra has no central extension.

\section{Finite Transformations}

\begin{definition}[Finite Euclidean conformal map]
\label{def:finite-euclidean-conformal-map}
A finite conformal transformation is a diffeomorphism
\[
  f:\R^D\supset U\longrightarrow f(U)\subset\R^D
\]
whose Jacobian satisfies
\begin{equation}
  \frac{\partial f^\rho}{\partial x^\mu}
  \frac{\partial f^\sigma}{\partial x^\nu}
  \delta_{\rho\sigma}
  =
  \Omega_f(x)^2\delta_{\mu\nu},
  \qquad
  \Omega_f(x)>0.
  \label{eq:finite-conformal-map}
\end{equation}
\end{definition}

\begin{proposition}[Infinitesimal limit and composition law]
\label{prop:finite-conformal-map-infinitesimal-and-composition}
For a transformation close to the identity,
\[
  f^\mu(x)=x^\mu+\epsilon^\mu(x)+O(\epsilon^2),
\]
condition \eqref{eq:finite-conformal-map} reduces at first order to the
conformal Killing equation \eqref{eq:conformal-killing-equation}.  If
\(f\) and \(h\) satisfy \eqref{eq:finite-conformal-map}, then their
composition satisfies it with Weyl factor
\[
  \Omega_{h\circ f}(x)=\Omega_h(f(x))\Omega_f(x).
\]
Thus finite conformal maps form a group wherever the composed maps are
defined.
\end{proposition}

\begin{proof}
Expanding \eqref{eq:finite-conformal-map} to first order gives
\[
  \partial_\mu\epsilon_\nu+\partial_\nu\epsilon_\mu
  =
  2\,\delta\Omega(x)\,\delta_{\mu\nu}.
\]
Taking the trace fixes
\(\delta\Omega=D^{-1}\partial_\rho\epsilon^\rho\), hence
\eqref{eq:conformal-killing-equation}.  For composition, the chain rule gives
the product of the two Jacobians, so the Weyl factor of \(h\circ f\) is
\(\Omega_h(f(x))\Omega_f(x)\).
\end{proof}

\begin{proposition}[Inversion is conformal]
\label{prop:euclidean-inversion-is-conformal}
Translations, rotations, and dilatations are immediate examples.  A further
basic example is inversion,
\begin{equation}
  I(x)^\mu=\frac{x^\mu}{x^2},
  \qquad
  x^2=\delta_{\mu\nu}x^\mu x^\nu .
  \label{eq:inversion-map}
\end{equation}
Its Jacobian is
\[
  \frac{\partial I^\mu}{\partial x^\nu}
  =
  \frac1{x^2}
  \left(
  \delta^\mu{}_\nu-2\frac{x^\mu x_\nu}{x^2}
  \right),
\]
and the matrix in parentheses is orthogonal.  Therefore
\[
  \Omega_I(x)=\frac1{x^2}.
\]
Inversion is conformal but lies in a disconnected component of the full
Euclidean conformal group: it cannot be reached by exponentiating a smooth
path of conformal Killing vectors from the identity.
\end{proposition}

\begin{proof}
The Jacobian is \(1/x^2\) times
\(\delta^\mu{}_\nu-2x^\mu x_\nu/x^2\).  The latter matrix is an orthogonal
reflection, so the pullback metric is \(x^{-4}\delta_{\mu\nu}\), with Weyl
factor \(\Omega_I=1/x^2\).  Inversion reverses the radial direction on the
compactified sphere and is not connected to the identity component generated
by smooth conformal Killing vector flows.
\end{proof}

Figure~\ref{fig:euclidean-inversion-unit-sphere} fixes the inversion geometry:
the unit sphere is mapped to itself, radial direction is reversed, and the Weyl
factor is \(\Omega_I=1/x^2\).

\begin{proposition}[Special conformal transformations from inversion]
\label{prop:finite-special-conformal-from-inversion}
The composition \(I\circ T_c\circ I\), with \(T_c(x)=x+c\), gives the special
conformal transformation
\begin{equation}
  x^\mu
  \longmapsto
  x'^\mu
  =
  \frac{x^\mu+c^\mu x^2}{1+2c\cdot x+c^2x^2}.
  \label{eq:finite-special-conformal}
\end{equation}
Its infinitesimal vector field is the quadratic term in
\eqref{eq:ckv-general-solution}.
\end{proposition}

\begin{proof}
Substituting \(I(x)=x/x^2\), translating by \(c\), and inverting once more
gives
\[
  I(T_c(I(x)))^\mu
  =
  \frac{x^\mu+c^\mu x^2}{1+2c\cdot x+c^2x^2}.
\]
Expanding to first order in \(c\) gives
\(\delta x^\mu=c^\mu x^2-2(c\cdot x)x^\mu\).
\end{proof}

The Poincare transformations, dilatations, and inversion generate all finite
conformal transformations in \(D>2\), after restricting to domains on which
the rational maps are defined.

\section{Euclidean Compactification and the Radial Cylinder}
\label{sec:euclidean-compactification-radial-cylinder}

\begin{definition}[Euclidean conformal compactification]
\label{def:euclidean-conformal-compactification}
The rational formulae above are local expressions for globally defined maps
on the conformal compactification.  The Euclidean compactification is the
one-point compactification
\[
  \widehat{\R^D}=\R^D\cup\{\infty\}\simeq S^D .
\]
In stereographic coordinates \(x^\mu\in\R^D\), the unit sphere
\[
  S^D=\{N\in\R^{D+1}:N\cdot N=1\}
\]
is parametrized away from one pole by
\begin{equation}
  N^\mu=\frac{2x^\mu}{1+x^2},
  \qquad
  N^{D+1}=\frac{1-x^2}{1+x^2},
  \qquad
  x^2=\delta_{\mu\nu}x^\mu x^\nu .
  \label{eq:stereographic-sphere-coordinates}
\end{equation}
The pullback of the round metric is
\begin{equation}
  \dd s^2_{S^D}
  =
  \frac{4\,\delta_{\mu\nu}\dd x^\mu\dd x^\nu}{(1+x^2)^2}.
  \label{eq:stereographic-round-metric}
\end{equation}
Thus the flat metric and the round metric are in the same conformal class on
the stereographic chart.  The point \(x=\infty\) is the omitted pole.
\end{definition}

\begin{quotedtheorem}[Liouville theorem for \(D\ge3\)]
\label{qthm:liouville-conformal-maps-d-ge-three}
In \(D\ge3\), Liouville's theorem implies that any smooth conformal
diffeomorphism between connected open subsets of \(\R^D\) is the restriction
of a Mobius transformation.  Equivalently, the transformations generated by
translations, rotations, dilatations, and inversion act as conformal
diffeomorphisms of \(S^D\).  Singularities in the flat coordinate formulae
are coordinate singularities of the stereographic chart: a point at which the
denominator of a special conformal transformation vanishes is sent to
\(\infty\), and \(\infty\) is sent to an ordinary point in another chart.
For the connected orientation-preserving Euclidean conformal group this
global action is the projective action of \(SO_0(D+1,1)\) on the corresponding
Euclidean projective null cone, or equivalently its action on \(S^D\) by
Mobius transformations.
\end{quotedtheorem}

\begin{proposition}[Radial cylinder as a Weyl representative]
\label{prop:radial-cylinder-weyl-representative}
The radial cylinder used in radial quantization is the same compactification
seen in a different conformal frame.  On \(\R^D\setminus\{0\}\), write
\[
  x^\mu=r n^\mu,\qquad
  r>0,\qquad
  n\in S^{D-1},\qquad
  \tau=\log r .
\]
Then
\begin{equation}
  \delta_{\mu\nu}\dd x^\mu\dd x^\nu
  =
  \ee^{2\tau}
  \left(\dd\tau^2+\dd\Omega_{D-1}^2\right).
  \label{eq:chapter2-radial-cylinder-metric}
\end{equation}
After the Weyl rescaling by \(r^{-2}\), punctured flat space becomes the
Euclidean cylinder \(\R_\tau\times S^{D-1}\).  The origin and infinity are the
two ends \(\tau=-\infty\) and \(\tau=+\infty\).  Inversion acts by
\[
  r\mapsto r^{-1},
  \qquad
  \tau\mapsto-\tau,
  \qquad
  n\mapsto n .
\]
Consequently the cylinder appearing in radial quantization is not an
independent geometric input; it is a Weyl representative of
\(\widehat{\R^D}\setminus\{0,\infty\}\), and radial time translations are
ordinary dilatations in the flat stereographic chart.
\end{proposition}

\begin{proof}
Substituting \(x^\mu=e^\tau n^\mu\), with \(n\cdot n=1\), into the flat metric
gives \eqref{eq:chapter2-radial-cylinder-metric}.  Multiplication by
\(r^{-2}=e^{-2\tau}\) removes the Weyl factor and leaves the product cylinder
metric.  The inversion \(x\mapsto x/x^2\) sends \(r\) to \(r^{-1}\), hence
\(\tau\) to \(-\tau\), and the flat-space dilatation flow is precisely
translation in \(\tau\).
\end{proof}

\section{The Conformal Algebra}

\begin{proposition}[Euclidean conformal charge algebra]
\label{prop:euclidean-conformal-charge-algebra}
Let the quantum charges associated with the vector fields
\eqref{eq:conformal-differential-generators-1} and
\eqref{eq:conformal-differential-generators-2} be denoted by
\[
  \widehat P_\mu,\qquad
  \widehat J_{\mu\nu},\qquad
  \widehat D,\qquad
  \widehat K_\mu .
\]
Their commutators are the charge realization of the Lie bracket of conformal
Killing vector fields.  In the Euclidean convention of this chapter, indices
are contracted with \(\delta_{\mu\nu}\), and the algebra is
\begin{align}
  [\widehat D,\widehat P_\mu]&=\ii\widehat P_\mu,
  &
  [\widehat D,\widehat K_\mu]&=-\ii\widehat K_\mu,
  &
  [\widehat D,\widehat J_{\mu\nu}]&=0,
  \label{eq:conf-alg-dilatation}\\
  [\widehat P_\mu,\widehat K_\nu]
  &=
  2\ii(\delta_{\mu\nu}\widehat D-\widehat J_{\mu\nu}),
  \label{eq:conf-alg-pk}
\end{align}
together with the Poincare commutators
\begin{align}
  [\widehat J_{\mu\nu},\widehat P_\rho]
  &=
  \ii(\delta_{\rho\mu}\widehat P_\nu-\delta_{\rho\nu}\widehat P_\mu),
  \\
  [\widehat J_{\mu\nu},\widehat J_{\rho\sigma}]
  &=
  \ii(
  \delta_{\mu\rho}\widehat J_{\nu\sigma}
  -\delta_{\nu\rho}\widehat J_{\mu\sigma}
  -\delta_{\mu\sigma}\widehat J_{\nu\rho}
  +\delta_{\nu\sigma}\widehat J_{\mu\rho}).
\end{align}
All unlisted commutators among the \(\widehat P_\mu\)'s and among the
\(\widehat K_\mu\)'s vanish.
\end{proposition}

\begin{proof}
Compute the Lie brackets of the vector fields in
Definition~\ref{def:differential-conformal-generators} and then apply
Proposition~\ref{prop:conformal-charge-algebra-from-vector-fields}.  The
factors of \(\ii\) and the overall signs are fixed by
Convention~\ref{conv:conformal-charge-sign}; in particular the sign chosen for
\(K_\mu\) in \eqref{eq:conformal-differential-generators-2} fixes the sign in
\eqref{eq:conf-alg-pk}.
\end{proof}

\begin{proposition}[Identification with \(\mathfrak{so}(D+1,1)\)]
\label{prop:euclidean-conformal-algebra-so-dplusone-one}
The Euclidean conformal algebra is isomorphic to \(\mathfrak{so}(D+1,1)\).
Introduce indices
\[
  A,B=-1,0,1,\ldots,D
\]
and a metric
\[
  \eta_{AB}^{(D+1,1)}=\operatorname{diag}(-1,+1,\ldots,+1),
\]
where the negative entry is the \((-1)\)-direction.  The generators
\(\widehat{\mathcal J}_{AB}=-\widehat{\mathcal J}_{BA}\) are
related to the conformal generators by
\[
  \widehat{\mathcal J}_{\mu\nu}=\widehat J_{\mu\nu},\qquad
  \widehat{\mathcal J}_{0,-1}=\widehat D,\qquad
  \widehat{\mathcal J}_{0,\mu}=\frac12(\widehat P_\mu-\widehat K_\mu),
  \qquad
  \widehat{\mathcal J}_{-1,\mu}=\frac12(\widehat P_\mu+\widehat K_\mu).
\]
Then
\[
  [\widehat{\mathcal J}_{AB},\widehat{\mathcal J}_{CD}]
  =
  \ii(
  \eta_{AC}^{(D+1,1)}\widehat{\mathcal J}_{BD}
  -\eta_{BC}^{(D+1,1)}\widehat{\mathcal J}_{AD}
  -\eta_{AD}^{(D+1,1)}\widehat{\mathcal J}_{BC}
  +\eta_{BD}^{(D+1,1)}\widehat{\mathcal J}_{AC}).
\]
\end{proposition}

\begin{proof}
The displayed change of basis is invertible.  Substituting it into
Proposition~\ref{prop:euclidean-conformal-charge-algebra} gives the
\(\mathfrak{so}(D+1,1)\) commutator: relations with only \(\mu,\nu\) indices
are the rotation algebra, relations involving \(0,-1\) reproduce the
dilatation commutators, and the mixed relations reproduce
\eqref{eq:conf-alg-pk}.  Since the basis change is invertible, this identifies
the Lie algebras.
\end{proof}
The Lorentzian conformal algebra is obtained by analytic continuation of one
physical coordinate and has real form \(\mathfrak{so}(D,2)\); it is not the
real form used in the Euclidean derivation above.

\section{The Lorentzian Conformal Group and Its Cover}
\label{sec:lorentzian-conformal-group-cover}

The preceding algebra statement does not by itself identify the group that
acts on a Lorentzian Hilbert space.  Let
\[
  M=\R^{1,D-1},
  \qquad
  \eta^{\rm L}_{\mu\nu}=\operatorname{diag}(-1,+1,\ldots,+1),
  \qquad
  x^2=\eta^{\rm L}_{\mu\nu}x^\mu x^\nu .
\]
Introduce an ambient vector space
\[
  \R^{D,2}
  =
  \R X^+\oplus\R X^-\oplus \R^{1,D-1}
\]
with bilinear form
\begin{equation}
  X\cdot Y
  =
  \eta^{\rm L}_{\mu\nu}X^\mu Y^\nu
  -\frac12(X^+Y^-+X^-Y^+).
  \label{eq:ambient-d2-bilinear-form}
\end{equation}
Thus \(X\cdot X=\eta^{\rm L}_{\mu\nu}X^\mu X^\nu-X^+X^-\), and the form has
signature \((D,2)\).  The conformal compactification of Minkowski space is
the projective null cone
\begin{equation}
  \overline M
  =
  \{X\in\R^{D,2}\setminus\{0\}:X\cdot X=0\}/\R^\times .
  \label{eq:lorentzian-projective-null-cone}
\end{equation}
\begin{definition}[Lorentzian conformal compactification]
\label{def:lorentzian-conformal-compactification}
The compactified Lorentzian spacetime \(\overline M\) is the projective null
cone \eqref{eq:lorentzian-projective-null-cone} in \(\R^{D,2}\).  The Poincare
patch is the affine chart \(X^+\ne0\).
\end{definition}
The Poincare patch \(X^+\ne0\) is identified with \(M\) by choosing the
representative
\begin{equation}
  X^+=1,\qquad
  X^\mu=x^\mu,\qquad
  X^-=x^2 .
  \label{eq:minkowski-patch-null-cone}
\end{equation}
The null condition follows immediately from
\eqref{eq:ambient-d2-bilinear-form}.  Points with \(X^+=0\) are the conformal
boundary of this patch.

\begin{proposition}[Einstein-cylinder cover]
\label{prop:lorentzian-einstein-cylinder-cover}
The same compactification is the Einstein cylinder.  Write
\[
  x^\mu=(t,r\omega),
  \qquad
  r\ge0,
  \qquad
  \omega\in S^{D-2},
\]
and introduce null coordinates \(u=t-r\), \(v=t+r\).  On the Poincare patch
set
\[
  u=\tan\frac{T-\chi}{2},
  \qquad
  v=\tan\frac{T+\chi}{2},
  \qquad
  0\le\chi\le\pi .
\]
Then
\begin{equation}
  \eta^{\rm L}_{\mu\nu}\dd x^\mu\dd x^\nu
  =
  \Omega_{\rm cyl}(T,\chi)^{-2}
  \left(
    -\dd T^2+\dd\chi^2+\sin^2\chi\,\dd\Omega_{D-2}^2
  \right),
  \label{eq:lorentzian-einstein-cylinder-metric}
\end{equation}
where
\begin{equation}
  \Omega_{\rm cyl}(T,\chi)
  =
  2\cos\frac{T-\chi}{2}\cos\frac{T+\chi}{2}.
  \label{eq:lorentzian-cylinder-weyl-factor}
\end{equation}
The metric in parentheses is the metric on \(\R_T\times S^{D-1}\), the
universal cover of the Lorentzian conformal compactification.  The projective
compactification \eqref{eq:lorentzian-projective-null-cone} is obtained from
\(S^1_T\times S^{D-1}\) by the antipodal projective identification
\((T,n)\sim(T+\pi,-n)\).  Thus local Lorentzian conformal transformations are
restrictions to a Poincare chart of diffeomorphisms of the compactified
space, and their Hilbert-space action is naturally lifted to the cylinder
cover \(\R_T\times S^{D-1}\).
\end{proposition}

\begin{proof}
The coordinate substitution \(u=\tan((T-\chi)/2)\) and
\(v=\tan((T+\chi)/2)\) converts the Minkowski line element into
\eqref{eq:lorentzian-einstein-cylinder-metric}; the trigonometric denominator
is the Weyl factor \eqref{eq:lorentzian-cylinder-weyl-factor}.  Periodic
cylinder time gives the projective compactification with the antipodal
identification displayed above, while the Hilbert-space representation uses
the universal cover in the \(T\)-direction.
\end{proof}

\begin{definition}[Projective Lorentzian conformal group]
\label{def:projective-lorentzian-conformal-group}
The group \(O(D,2)\) acts linearly on \(\R^{D,2}\), preserves the null cone,
and therefore acts on \(\overline M\).  Scalar matrices act trivially after
projectivization, so the spacetime conformal transformations in the identity
component are described by the projective group
\begin{equation}
  G_{\rm conf}^{\rm Lor}
  =
  PO_0(D,2),
  \label{eq:lorentzian-projective-conformal-group}
\end{equation}
the identity component of the projective orthogonal group.  Equivalently, one
may start from \(SO_0(D,2)\) and quotient by the finite central subgroup that
acts trivially on projective null lines.  The Lie algebra of
\(G_{\rm conf}^{\rm Lor}\) is \(\mathfrak{so}(D,2)\).
\end{definition}

\begin{proposition}[Ambient transformations and Poincare-patch formulae]
\label{prop:ambient-transformations-poincare-patch}
The standard finite transformations on the Poincare patch are restrictions of
linear ambient transformations.  Lorentz transformations \(\Lambda\in
SO_0(1,D-1)\) act by
\[
  X^\mu\mapsto \Lambda^\mu{}_\nu X^\nu,\qquad
  X^\pm\mapsto X^\pm .
\]
Translations by \(a^\mu\) act by
\begin{equation}
  X^+\mapsto X^+,\qquad
  X^\mu\mapsto X^\mu+a^\mu X^+,\qquad
  X^-\mapsto X^-+2a\cdot X+a^2X^+,
  \label{eq:ambient-translation}
\end{equation}
where \(a\cdot X=\eta^{\rm L}_{\mu\nu}a^\mu X^\nu\).  Dilatations with
parameter \(\lambda\) act by
\begin{equation}
  X^+\mapsto e^{-\lambda}X^+,\qquad
  X^-\mapsto e^\lambda X^-,\qquad
  X^\mu\mapsto X^\mu ,
  \label{eq:ambient-dilatation}
\end{equation}
which becomes \(x^\mu\mapsto e^\lambda x^\mu\) after rescaling back to
\(X^+=1\).  Special conformal transformations with parameter \(c^\mu\) act by
\begin{equation}
  X^-\mapsto X^-,\qquad
  X^\mu\mapsto X^\mu+c^\mu X^-,\qquad
  X^+\mapsto X^+ +2c\cdot X+c^2X^- .
  \label{eq:ambient-special-conformal}
\end{equation}
On the patch \(X^+=1\), this gives
\begin{equation}
  x^\mu
  \longmapsto
  \frac{x^\mu+c^\mu x^2}{1+2c\cdot x+c^2x^2},
  \label{eq:lorentzian-finite-special-conformal}
\end{equation}
where the transformation is read locally whenever the denominator is nonzero,
and globally as a map of the compactification
\eqref{eq:lorentzian-projective-null-cone}.  These formulae show explicitly
how the infinitesimal generators integrate to the connected projective
conformal group.
\end{proposition}

\begin{proof}
Each displayed ambient transformation preserves the bilinear form
\eqref{eq:ambient-d2-bilinear-form}, hence preserves the null cone and its
projectivization.  On the affine chart \(X^+=1\), divide the transformed
representative by its new \(X^+\)-component.  This gives translations,
dilatations, Lorentz rotations, and
\eqref{eq:lorentzian-finite-special-conformal} for special conformal
transformations.  Differentiating at the identity recovers the Lorentzian real
form of the conformal Killing generators.
\end{proof}

\begin{definition}[Hilbert-space conformal group]
\label{def:hilbert-space-conformal-group}
For Hilbert-space representation theory, the projective spacetime group is
still not the final object.  Let
\begin{equation}
  \widetilde G_{\rm conf}^{\rm Lor}
  \longrightarrow
  G_{\rm conf}^{\rm Lor}
  \label{eq:lorentzian-conformal-universal-cover}
\end{equation}
be the universal covering group.  A unitary Lorentzian CFT carries a
strongly continuous unitary representation
\[
  U:\widetilde G_{\rm conf}^{\rm Lor}\longrightarrow
  \mathcal U(\mathcal H)
\]
whose derived representation gives the self-adjoint conformal charges on a
common invariant finite-energy domain.  The need for the cover is not a
minor global convention.  The maximal compact subalgebra of
\(\mathfrak{so}(D,2)\) is
\(\mathfrak{so}(D)\oplus\mathfrak{so}(2)\).  In the universal cover its
corresponding subgroup is covered by
\begin{equation}
  \operatorname{Spin}(D)\times \R ,
  \label{eq:maximal-compact-cover}
\end{equation}
where the \(\R\) factor is generated by Lorentzian cylinder time.  Under
radial quantization this generator is the nonnegative cylinder Hamiltonian
\(D_{\rm rad}\), and
\begin{equation}
  U_{\rm cyl}(t)=\exp(-\ii t D_{\rm rad}).
  \label{eq:cylinder-time-unitary-cover}
\end{equation}
\end{definition}

\begin{proposition}[Universal-cover necessity]
\label{prop:universal-cover-required-for-conformal-hilbert-space}
If one forced the representation to descend to the finite group with an
\(SO(2)\) cylinder-time factor, the operator
\(\exp(-2\pi\ii D_{\rm rad})\) would have to be the identity.  This would
impose integrality of all scaling dimensions, contrary to generic unitary
CFTs.  Spinorial local operators similarly require the spin cover of spatial
rotations.  Thus later references to the Lorentzian conformal group acting on
the Hilbert space mean the universal cover
\(\widetilde G_{\rm conf}^{\rm Lor}\); its projection to
\(G_{\rm conf}^{\rm Lor}\) is the group acting on compactified spacetime
points.
\end{proposition}

\begin{proof}
The \(\R\) factor in \eqref{eq:maximal-compact-cover} covers the
\(SO(2)\) cylinder-time circle.  A representation descends to the quotient by
the \(2\pi\)-translation kernel precisely when
\(\exp(-2\pi\ii D_{\rm rad})=1\).  If
\(D_{\rm rad}\psi=\Delta\psi\), this condition gives
\(\exp(-2\pi\ii\Delta)=1\).  Thus every scaling dimension in the sector would
have to be integral.  Generic unitary CFTs have nonintegral scaling
dimensions, and spinorial sectors similarly require the spin cover of
rotations.
\end{proof}

\begin{definition}[Local primary state]
\label{def:local-primary-state}
The vacuum vector is invariant under
\(\widetilde G_{\rm conf}^{\rm Lor}\).  A local primary state is a lowest
energy vector for the \(\R\)-generator in
\eqref{eq:maximal-compact-cover}, transforms in a finite-dimensional unitary
representation of \(\operatorname{Spin}(D)\), and generates a positive-energy
module by the action of the noncompact translation generators.  This is the
group-theoretic meaning of the radial primary conditions used in later
chapters.
\end{definition}

\section{Representation-Theoretic Conventions}
\label{sec:conformal-representation-theoretic-conventions}

The same Lie algebra appears in several representation-theoretic roles, and
the distinctions matter.  The Euclidean conformal group
\[
  G_E=SO_0(D+1,1)
\]
acts on the compactified Euclidean sphere \(S^D\), or through a chosen cover
when spin structures are present.

\begin{proposition}[Parabolic stabilizer of a Euclidean point]
\label{prop:euclidean-point-parabolic-stabilizer}
Fix the point \(x=0\) in the flat chart of \(S^D\).  Its stabilizer in the
identity component of the Euclidean conformal group has Lie algebra
\[
  \mathfrak p_E
  =
  \mathfrak{so}(D)\oplus\mathbb R D\oplus
  \operatorname{span}\{K_\mu\}.
\]
Equivalently, at the group level and in the chosen chart, it is generated by
rotations, positive dilatations, and special conformal transformations.  Its
Levi factor is \(SO(D)\times\mathbb R_+\), and the special conformal
transformations form the nilpotent radical.
\end{proposition}

\begin{proof}
Translations do not fix the origin unless their parameter is zero.  A rotation
and a positive dilatation fix \(x=0\) and act on the tangent space by
\(\lambda R\), with \(R\in SO(D)\) and \(\lambda>0\).  The finite special
conformal transformation
\[
  x^\mu\mapsto
  \frac{x^\mu+c^\mu x^2}{1+2c\cdot x+c^2x^2}
\]
also fixes \(x=0\), and its derivative at \(x=0\) is the identity; hence its
generator lies in the nilpotent radical of the stabilizer.  The infinitesimal
classification in Proposition~\ref{prop:flat-conformal-killing-vector-classification}
shows that these exhaust the conformal Killing vectors vanishing at the
origin.  Thus the stabilizer has the displayed Lie algebra and
Levi/nilpotent decomposition.
\end{proof}

\begin{definition}[Euclidean primary type]
\label{def:euclidean-primary-parabolic-type}
A Euclidean primary type at the point \(x=0\) is a finite-dimensional
representation of the point stabilizer \(P_E\) with the following structure.
The nilpotent radical generated by \(K_\mu\) acts trivially; the rotation
factor acts by a finite-dimensional representation \(\rho\) of \(SO(D)\), or
of \(\operatorname{Spin}(D)\) when spin is present; and a positive dilatation
\(x\mapsto\lambda x\) acts by the character \(\lambda^{-\Delta}\).  The real
number \(\Delta\) is the Euclidean scaling dimension in the covariance
convention used below.
\end{definition}

A Euclidean primary operator is encoded by a covariant section of the
homogeneous vector bundle associated to this \(P_E\)-representation.  In a
flat chart, the transformation law has the form
\begin{definition}[Euclidean primary covariance law]
\label{def:euclidean-primary-covariance-law}
\begin{equation}
  U_E(f)\,\mathcal O_i^a(x)\,U_E(f)^{-1}
  =
  \Omega_f(x)^{-\Delta_i}\,
  \rho_i(R_f(x))^a{}_b\,
  \mathcal O_i^b(f(x)),
  \label{eq:euclidean-primary-induced-representation}
\end{equation}
where \(\Delta_i\in\mathbb R\) is the scaling dimension,
\(\rho_i\) is a finite-dimensional representation of \(\operatorname{Spin}(D)\)
or \(SO(D)\), and \(R_f(x)\) is the local rotation obtained from the
Jacobian after extracting the positive Weyl factor \(\Omega_f(x)\).  The
fiber representation of \(P_E\) contains a noncompact dilation character
\(\lambda\mapsto\lambda^{-\Delta_i}\).  For the real scaling dimensions
relevant to reflection-positive CFT, this fiber action is an induced
covariance representation rather than the Hilbert-space unitary
representation.  Euclidean correlators are covariance distributions for the
associated bundles, while reflection positivity is the additional input that
produces the Lorentzian Hilbert-space representation.
\end{definition}

\begin{definition}[Positive-energy representation]
\label{def:positive-energy-lorentzian-conformal-representation}
The Lorentzian Hilbert-space representation is a strongly continuous unitary
representation of the universal cover
\(\widetilde G_{\rm conf}^{\rm Lor}\).  Let
\[
  K_{\rm cov}=\operatorname{Spin}(D)\times\mathbb R
\]
denote the cover of the maximal compact subgroup, with the \(\mathbb R\)
factor generated by the cylinder Hamiltonian \(D_{\rm rad}\).  A positive-energy
representation means a unitary representation \(U\) for which
\[
  U(r(t))=e^{-\ii tD_{\rm rad}},\qquad
  D_{\rm rad}\ge0,
\]
with \(r(t)\in\widetilde G_{\rm conf}^{\rm Lor}\) the lifted cylinder-time
rotation and \(D_{\rm rad}\) self-adjoint.  The finite-energy subspace is
\begin{equation}
  \mathcal H_{\rm fin}
  =
  \bigoplus_{\Delta<\infty}\ker(D_{\rm rad}-\Delta),
  \label{eq:finite-energy-conformal-subspace}
\end{equation}
whenever the spectrum is discrete with finite-dimensional eigenspaces, as
assumed in the radial-reconstruction hypotheses.  This algebraic direct sum is
a common invariant domain for the derived conformal generators.  On this
domain,
\[
  P_\mu^\dagger=K_\mu,\qquad
  D_{\rm rad}^\dagger=D_{\rm rad},\qquad
  J_{\mu\nu}^\dagger=J_{\mu\nu},
\]
with the adjoints understood in the radial Hilbert-space inner product.  The
closure of each one-parameter generator is the self-adjoint generator supplied
by Stone's theorem from the unitary group representation.
\end{definition}

\begin{definition}[Generalized Verma module of a conformal primary]
\label{def:chapter2-conformal-generalized-verma-module}
After complexification, let
\[
  \mathfrak g_\mathbb C=\mathfrak{so}(D+2,\mathbb C).
\]
The radial grading decomposes it as
\[
  \mathfrak g_\mathbb C
  =
  \mathfrak n_-\oplus
  (\mathfrak{so}(D,\mathbb C)\oplus\mathbb C D_{\rm rad})
  \oplus \mathfrak n_+,
  \qquad
  \mathfrak n_+=\operatorname{span}_{\mathbb C}\{P_\mu\},\quad
  \mathfrak n_-=\operatorname{span}_{\mathbb C}\{K_\mu\}.
  \label{eq:conformal-triangular-decomposition}
\]
A primary vector \(v\) of dimension \(\Delta\) and spin representation
\(\rho\) obeys
\[
  D_{\rm rad}v=\Delta v,\qquad
  K_\mu v=0,\qquad
  \operatorname{Spin}(D)\text{ acts on }\operatorname{span}\{v\}\text{ by }\rho .
\]
The corresponding algebraic highest-weight, or generalized Verma, module is
\begin{equation}
  M(\Delta,\rho)
  =
  U(\mathfrak n_+)\otimes V_{\Delta,\rho},
  \label{eq:chapter2-conformal-generalized-verma-module}
\end{equation}
where \(V_{\Delta,\rho}\) is the finite-dimensional
\(\mathfrak{so}(D,\mathbb C)\oplus\mathbb C D_{\rm rad}\)-module on which
\(\mathfrak n_-\) acts trivially.  Its level-\(n\) subspace is generated by
\(n\) translations \(P_\mu\).
\end{definition}

\begin{proposition}[Null quotient and unitarity test]
\label{prop:conformal-null-quotient-unitarity-test}
The radial two-point function defines a
contravariant Hermitian form on \(M(\Delta,\rho)\).  The null radical of this
form is a submodule, and the physical irreducible positive-energy module is
\begin{equation}
  L(\Delta,\rho)=M(\Delta,\rho)/\operatorname{Rad}M(\Delta,\rho)
  \label{eq:chapter2-conformal-irreducible-null-quotient}
\end{equation}
when the quotient form is positive.  Unitarity bounds are precisely the
conditions that the finite Gram matrices at every level be positive
semidefinite, with shortening occurring when a Gram matrix develops a null
submodule.
\end{proposition}

\begin{proof}
Radial reflection sends \(P_\mu\) to \(K_\mu\), so the two-point pairing is
contravariant with respect to the anti-involution \(P_\mu^\dagger=K_\mu\),
\(D_{\rm rad}^\dagger=D_{\rm rad}\), and
\(J_{\mu\nu}^\dagger=J_{\mu\nu}\).  If a vector pairs to zero with every
descendant, contravariance implies that its image under the complexified
conformal algebra also pairs to zero with every descendant; hence the radical
is a submodule.  Quotienting by the radical gives the irreducible module
generated by the primary whenever no further invariant null subspace remains.
Positivity of all finite-level Gram matrices is exactly positivity of the
Hermitian form on the algebraic descendant space; a zero determinant gives a
shortening submodule.
\end{proof}

The passage from the algebraic module \(L(\Delta,\rho)\) to a group
representation is a globalization problem.  In the unitary CFT setting the
globalization is supplied by the strongly continuous representation of
\(\widetilde G_{\rm conf}^{\rm Lor}\) on the radial Hilbert space, and
\(L(\Delta,\rho)\) is the \(K_{\rm cov}\)-finite subspace generated by the
primary.  Descent from the universal cover to a quotient group is an additional
condition: every element of the covering kernel must act trivially.  For
example, descent to an \(SO(2)\) cylinder-time factor would require
\(\exp(-2\pi\ii D_{\rm rad})=1\), hence integral scaling dimensions on the
sector under consideration; spinorial sectors similarly require keeping the
spin cover rather than \(SO(D)\).  Throughout the conformal block chapters,
statements about conformal multiplets refer to these algebraic
\(\mathfrak g_\mathbb C\)-modules together with the Hilbert-space
globalization supplied by radial quantization.

\section{Two Euclidean Dimensions}

\begin{proposition}[Two-dimensional conformal Killing fields]
\label{prop:two-dimensional-conformal-killing-fields}
For \(D=2\), the local conformal Killing equation has infinitely many
solutions.  In complex coordinates
\[
  z=x^1+\ii x^2,\qquad
  \bar z=x^1-\ii x^2,
\]
the local conformal transformations are
\[
  z\mapsto f(z),\qquad
  \bar z\mapsto \bar f(\bar z)
\]
in Euclidean signature, with \(f\) holomorphic in its domain.  The global
conformal transformations of the Riemann sphere are the Möbius maps
\[
  z\mapsto \frac{az+b}{cz+d},
  \qquad
  ad-bc\neq0.
\]
\end{proposition}

\begin{proof}
In two real dimensions the conformal Killing equation is equivalent to the
Cauchy--Riemann equations for the complex components of the infinitesimal
coordinate change.  Integrating locally gives holomorphic maps
\(z\mapsto f(z)\) and antiholomorphic maps in the conjugate coordinate.  A
globally defined biholomorphism of the compact Riemann sphere is a degree-one
rational map, hence the displayed Möbius transformation.
\end{proof}

The infinite-dimensional enhancement has its own representation theory.  The
global Lorentzian conformal algebra factorizes, after passing to a cover, as
\(\mathfrak{sl}(2,\mathbb R)\oplus\mathfrak{sl}(2,\mathbb R)\), while the
local chiral stress-tensor modes generate Virasoro algebras with central
charge.  A global primary is annihilated by the global lowering generators,
whereas a Virasoro primary is annihilated by all positive chiral modes
\(L_n,\bar L_n\) with \(n>0\).  The two notions coincide only after restricting
attention to the global subalgebra.  The present core conformal-block
discussion uses the finite-dimensional global conformal group; a full
two-dimensional CFT volume must keep global blocks, Virasoro blocks, and their
different null-module quotients separate.
