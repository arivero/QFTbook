\chapter{Correlation Functions and Conformal Frames}
\label{ch:volume-iii-correlation-functions-conformal-frames}

\section{Ward Identities for Correlators}

Throughout this chapter, correlators are Euclidean vacuum correlators at
separated insertion points.  The conformal vacuum is assumed to be invariant
under the connected conformal group, so that the global Ward identity of
Chapter~\ref{ch:volume-iii-conformal-charges-ward-identities} applies.  If
\(\epsilon^\mu(x)\partial_\mu\) is a conformal Killing vector and
\(\delta_\epsilon\mathcal O_i\) is the corresponding infinitesimal variation of
the \(i\)th local operator, then
\begin{equation}
  \sum_{i=1}^n
  \left\langle
    \mathcal O_1(x_1)\cdots
    \delta_\epsilon\mathcal O_i(x_i)\cdots
    \mathcal O_n(x_n)
  \right\rangle
  =
  0
  \label{eq:correlator-global-ward-identity}
\end{equation}
as a distribution on the configuration space \(x_i\ne x_j\).  Contact terms on
coincident-point diagonals are part of the local Ward identities and are not
used in the kinematic classification below.

Equivalently, for a finite conformal transformation \(f:x\mapsto \widetilde x\)
implemented by a unitary operator \(U_f\), invariance of the vacuum gives
\[
  \left\langle
    \mathcal O_1(x_1)\cdots\mathcal O_n(x_n)
  \right\rangle
  =
  \left\langle
    (U_f\mathcal O_1U_f^{-1})(x_1)\cdots
    (U_f\mathcal O_nU_f^{-1})(x_n)
  \right\rangle .
\]
The operators on the right are written at the same coordinate labels as before;
their transformation laws supply the Jacobian and spin factors.  For a scalar
primary \(\phi\) of dimension \(\Delta_\phi\),
\begin{equation}
  \widetilde\phi(\widetilde x)
  =
  \left|\det\frac{\partial\widetilde x}{\partial x}\right|^{-\Delta_\phi/D}
  \phi(x).
  \label{eq:scalar-primary-finite-transform-correlators}
\end{equation}

\section{One- and Two-Point Functions}

Let \(\mathcal O_i\) be scalar primary operators of dimensions
\(\Delta_i\).  Translation invariance makes a one-point function constant:
\[
  \left\langle \mathcal O_i(x)\right\rangle=c_i.
\]
Scale covariance gives
\[
  c_i=\lambda^{-\Delta_i}c_i
  \qquad
  \text{for all }\lambda>0.
\]
Hence \(c_i=0\) unless \(\Delta_i=0\).  In a unitary theory satisfying the
vacuum assumptions stated in Chapter~\ref{ch:volume-iii-unitarity-bounds-short-multiplets},
the scalar primary with \(\Delta=0\) is the identity representation.

\begin{theorem}[Scalar two-point selection rule]
\label{thm:scalar-two-point-selection-rule}
Assume the separated-point global conformal Ward identities
\eqref{eq:correlator-global-ward-identity} and scalar primary transformation
law \eqref{eq:scalar-primary-finite-transform-correlators}.  If
\(\mathcal O_i,\mathcal O_j\) are scalar primaries of dimensions
\(\Delta_i,\Delta_j\), then
\begin{equation}
  \left\langle \mathcal O_i(x)\mathcal O_j(y)\right\rangle
  =
  \begin{cases}
    \displaystyle \frac{C_{ij}}{|x-y|^{2\Delta_i}},
    &\Delta_i=\Delta_j,\\[1.0em]
    0,&\Delta_i\ne\Delta_j .
  \end{cases}
  \label{eq:scalar-primary-two-point}
\end{equation}
\end{theorem}

\begin{proof}
Translation and rotation invariance imply
\[
  \left\langle \mathcal O_i(x)\mathcal O_j(y)\right\rangle
  =
  F_{ij}(|x-y|).
\]
Scale covariance gives
\[
  F_{ij}(r)=\frac{C_{ij}}{r^{\Delta_i+\Delta_j}}.
\]
The condition \(\Delta_i=\Delta_j\) for a nonzero coefficient follows from a
conformal transformation that fixes both points but has different local scale
factors at the two points.  Put the two points at \(0\) and \(e_1\), where
\(e_1\) is a fixed unit vector, and define
\[
  f_\lambda(x)^\mu
  =
  \lambda
  \frac{x^\mu+a^\mu x^2}{1+2a\cdot x+a^2x^2},
  \qquad
  a^\mu=(\lambda-1)e_1^\mu,
  \qquad
  \lambda>0 .
\]
Then \(f_\lambda(0)=0\), \(f_\lambda(e_1)=e_1\),
\[
  \left.\frac{\partial f_\lambda^\mu}{\partial x^\nu}\right|_{x=0}
  =
  \lambda\,\delta^\mu{}_\nu,
  \qquad
  \left.\frac{\partial f_\lambda^\mu}{\partial x^\nu}\right|_{x=e_1}
  =
  \lambda^{-1}\delta^\mu{}_\nu .
\]
Applying \eqref{eq:scalar-primary-finite-transform-correlators} to
\(\langle\mathcal O_i(e_1)\mathcal O_j(0)\rangle\) gives
\[
  C_{ij}=\lambda^{\Delta_i-\Delta_j}C_{ij}
  \qquad\text{for all }\lambda>0,
\]
so \(C_{ij}=0\) unless \(\Delta_i=\Delta_j\).  Substituting this condition in
the scale-covariant form gives \eqref{eq:scalar-primary-two-point}.
\end{proof}
Within a finite-dimensional space of primaries with the same dimension and the
same spin, one may choose a basis that diagonalizes the Hermitian matrix
\(C_{ij}\).

\section{Radial Inner Product and BPZ Conjugation}

Let \(x=rn\), \(r>0\), \(n\in S^{D-1}\), and set
\(\tau=\log r\).  The Weyl map from \(\R^D\setminus\{0\}\) to
\(\R_\tau\times S^{D-1}\) relates a scalar primary on flat space to the
cylinder operator by
\begin{equation}
  \mathcal O(x)=r^{-\Delta}\widetilde{\mathcal O}(\tau,n).
  \label{eq:flat-cylinder-scalar-primary}
\end{equation}
The state created by insertion at the origin is
\begin{equation}
  \ket{\mathcal O}
  =
  \lim_{\tau\to-\infty}
  \ee^{-\Delta\tau}
  \widetilde{\mathcal O}(\tau,n)\ket{\vac},
  \label{eq:ket-from-cylinder-limit}
\end{equation}
where \(\ket{\vac}\) is the conformal vacuum.  For a primary the limit is
independent of \(n\).  The conjugate bra is obtained by the opposite
cylinder-time limit,
\begin{equation}
  \langle\!\langle \mathcal O|
  =
  \lim_{\tau\to+\infty}
  \ee^{\Delta\tau}
  \bra{\vac}\widetilde{\mathcal O}(\tau,n).
  \label{eq:bpz-bra-from-cylinder-limit}
\end{equation}
This radial conjugation is often called BPZ conjugation.  In a
reflection-positive theory, after including the ordinary internal adjoint of the
local operator, it is the Hilbert-space adjoint of the state.  Consequently the
two-point coefficient has the inner-product interpretation
\begin{equation}
  \langle\!\langle \mathcal O_i|\mathcal O_j\rangle
  =
  \lim_{\tau_i\to+\infty,\ \tau_j\to-\infty}
  \ee^{\Delta_i\tau_i-\Delta_j\tau_j}
  \left\langle
    \widetilde{\mathcal O}_i(\tau_i,n_i)
    \widetilde{\mathcal O}_j(\tau_j,n_j)
  \right\rangle
  =
  C_{ij},
  \label{eq:bpz-inner-product-two-point-coefficient}
\end{equation}
for primaries with equal dimension.  Positivity of this matrix is the same
positivity used in the unitarity-bound derivation.

\section{Spin Structures}

The inversion tensor
\begin{equation}
  I_{\mu\nu}(x)
  =
  \delta_{\mu\nu}-2\frac{x_\mu x_\nu}{x^2}
  \label{eq:inversion-tensor}
\end{equation}
is the orthogonal part of the inversion Jacobian.  It is the elementary
building block for two-point functions with vector or tensor indices.  Its
action on the line spanned by \(x^\mu\) is multiplication by \(-1\), while its
action on the orthogonal complement of \(x^\mu\) is the identity.  Hence
\(\det I(x)=-1\): inversion lies in \(O(D)\setminus SO(D)\).  Every use of
\(I_{\mu\nu}(x)\) in a spin tensor structure therefore presupposes the extension
of the \(SO(D)\) spin action to the disconnected orthogonal, or pin, action
specified in Chapter~\ref{ch:volume-iii-primary-operators-finite-transformations}.
For
example, a vector primary of dimension \(\Delta_V\) obeys
\begin{equation}
  \left\langle V_\mu(x)V_\nu(0)\right\rangle
  =
  \frac{C_V}{|x|^{2\Delta_V}}I_{\mu\nu}(x),
  \label{eq:vector-primary-two-point}
\end{equation}
after diagonalizing any degeneracy labels.

More generally, let
\(\mathcal O_{\mu_1\cdots\mu_s}\) be a rank-\(s\) symmetric traceless tensor
primary.  Define
\[
  \mathcal I^{(s)}_{\mu_1\cdots\mu_s,\nu_1\cdots\nu_s}(x)
  =
  \Pi_{\operatorname{STT}}
  \left[
    I_{\mu_1\nu_1}(x)\cdots I_{\mu_s\nu_s}(x)
  \right],
\]
where \(\Pi_{\operatorname{STT}}\) symmetrizes separately in the
\(\mu\)-indices and the \(\nu\)-indices and removes all traces with respect to
\(\delta_{\mu\nu}\).  Conformal covariance fixes the two-point function to be
\begin{equation}
  \left\langle
    \mathcal O_{\mu_1\cdots\mu_s}(x)
    \mathcal O_{\nu_1\cdots\nu_s}(0)
  \right\rangle
  =
  \frac{C_{\mathcal O}}{|x|^{2\Delta_{\mathcal O}}}
  \mathcal I^{(s)}_{\mu_1\cdots\mu_s,\nu_1\cdots\nu_s}(x),
  \label{eq:stt-primary-two-point}
\end{equation}
up to the same positive Hermitian coefficient matrix in possible degeneracy
labels.  Two-point functions between inequivalent \(SO(D)\) representations
vanish by rotational covariance.

For the stress tensor, \(s=2\), \(\Delta_T=D\), and conservation gives a
canonical normalization convention through the Ward identity.  Its two-point
function is
\begin{equation}
  \left\langle T_{\mu\nu}(x)T_{\rho\sigma}(0)\right\rangle
  =
  \frac{C_T}{|x|^{2D}}
  I_{\mu\nu,\rho\sigma}(x),
  \label{eq:stress-tensor-two-point}
\end{equation}
where
\[
  I_{\mu\nu,\rho\sigma}(x)
  =
  \frac12\bigl(
    I_{\mu\rho}(x)I_{\nu\sigma}(x)
    +
    I_{\mu\sigma}(x)I_{\nu\rho}(x)
  \bigr)
  -
  \frac1D\delta_{\mu\nu}\delta_{\rho\sigma}.
\]
The coefficient \(C_T\) is part of the intrinsic CFT data because the
normalization of \(T_{\mu\nu}\) is fixed by the translation Ward identity.

\section{Three-Point Functions}

For scalar primaries, conformal covariance fixes the three-point function up
to one coefficient:
\begin{equation}
  \left\langle
    \mathcal O_1(x_1)\mathcal O_2(x_2)\mathcal O_3(x_3)
  \right\rangle
  =
  \frac{C_{123}}
  {|x_{12}|^{\Delta_1+\Delta_2-\Delta_3}
   |x_{23}|^{\Delta_2+\Delta_3-\Delta_1}
   |x_{13}|^{\Delta_1+\Delta_3-\Delta_2}},
  \label{eq:scalar-primary-three-point}
\end{equation}
where
\[
  x_{ij}=x_i-x_j.
\]
The constants \(C_{123}\), after fixing the two-point normalization of
primaries, are the scalar three-point OPE coefficients.  If the primaries
carry spin, let \(a_i\) denote indices in the corresponding finite-dimensional
\(SO(D)\) representations.  Conformal covariance says that
\begin{equation}
  \left\langle
    \mathcal O_{1,a_1}(x_1)
    \mathcal O_{2,a_2}(x_2)
    \mathcal O_{3,a_3}(x_3)
  \right\rangle
  =
  \sum_{r=1}^{n_{123}}
  C_{123}^{(r)}
  \mathcal I_{a_1a_2a_3}^{(r)}(x_{12},x_{13}),
  \label{eq:spinning-three-point-structures}
\end{equation}
where the tensors \(\mathcal I^{(r)}\) form a basis of conformally covariant
three-point tensor structures, including the required homogeneity in the
separations.  The finite integer \(n_{123}\) is representation-theoretic; the
numbers \(C_{123}^{(r)}\) are dynamical data after a convention for two-point
normalization has been fixed.

The stress tensor illustrates why some three-point coefficients are
constrained but not removed from the data.  In a parity-even CFT in
\(D\ge4\), the separated three-point function of a symmetric conserved
traceless stress tensor has the following explicit form.  Set
\[
  X\equiv X_{12,3}
  =
  \frac{x_{13}}{x_{13}^2}
  -
  \frac{x_{23}}{x_{23}^2},
  \qquad
  \widehat X_\mu=\frac{X_\mu}{(X^2)^{1/2}},
\]
and, for a dimensionless tensor
\(t_{\mu\nu,\rho\sigma,\alpha\beta}(\widehat X)\), define
\begin{equation}
  \mathfrak C[t]_{\mu\nu,\rho\sigma,\alpha\beta}
  =
  \frac{
    I_{\mu\nu,\mu'\nu'}(x_{13})
    I_{\rho\sigma,\rho'\sigma'}(x_{23})
    t_{\mu'\nu',\rho'\sigma',\alpha\beta}(\widehat X)}
  {|x_{12}|^D |x_{13}|^D |x_{23}|^D}.
  \label{eq:op-ttt-prefactor}
\end{equation}
Let \(\operatorname{Sym}_{\mu\nu}\) denote the unnormalized symmetrization
obtained by adding the expression with \(\mu\) and \(\nu\) interchanged.  The
elementary symmetric-traceless tensors are
\begin{align}
  h^1_{\mu\nu}
  &=
  \widehat X_\mu\widehat X_\nu-\frac1D\delta_{\mu\nu},
  \nonumber\\
  h^2_{\mu\nu,\rho\sigma}
  &=
  \widehat X_\mu\widehat X_\rho\delta_{\nu\sigma}
  +\widehat X_\nu\widehat X_\rho\delta_{\mu\sigma}
  +\widehat X_\mu\widehat X_\sigma\delta_{\nu\rho}
  +\widehat X_\nu\widehat X_\sigma\delta_{\mu\rho}
  \nonumber\\
  &\quad
  -\frac4D
  \left(
    \widehat X_\mu\widehat X_\nu\delta_{\rho\sigma}
    +
    \widehat X_\rho\widehat X_\sigma\delta_{\mu\nu}
  \right)
  +
  \frac4{D^2}\delta_{\mu\nu}\delta_{\rho\sigma},
  \nonumber\\
  h^3_{\mu\nu,\rho\sigma}
  &=
  \delta_{\mu\rho}\delta_{\nu\sigma}
  +
  \delta_{\mu\sigma}\delta_{\nu\rho}
  -
  \frac2D\delta_{\mu\nu}\delta_{\rho\sigma},
  \nonumber\\
  h^4_{\mu\nu,\rho\sigma,\alpha\beta}
  &=
  \operatorname{Sym}_{\rho\sigma}
  \operatorname{Sym}_{\alpha\beta}
  \left[
    h^3_{\mu\nu,\rho\alpha}
    \widehat X_\sigma\widehat X_\beta
  \right]
  -
  \frac2D\delta_{\rho\sigma}h^2_{\mu\nu,\alpha\beta}
  \nonumber\\
  &\quad
  -
  \frac2D\delta_{\alpha\beta}h^2_{\mu\nu,\rho\sigma}
  -
  \frac8{D^2}\delta_{\rho\sigma}\delta_{\alpha\beta}h^1_{\mu\nu},
  \nonumber\\
  h^5_{\mu\nu,\rho\sigma,\alpha\beta}
  &=
  \operatorname{Sym}_{\mu\nu}
  \operatorname{Sym}_{\rho\sigma}
  \operatorname{Sym}_{\alpha\beta}
  \left[
    \delta_{\mu\rho}\delta_{\nu\alpha}\delta_{\sigma\beta}
  \right]
  \nonumber\\
  &\quad
  -
  \frac4D
  \left(
    \delta_{\mu\nu}h^3_{\rho\sigma,\alpha\beta}
    +
    \delta_{\rho\sigma}h^3_{\mu\nu,\alpha\beta}
    +
    \delta_{\alpha\beta}h^3_{\mu\nu,\rho\sigma}
  \right)
  -
  \frac8{D^2}
  \delta_{\mu\nu}\delta_{\rho\sigma}\delta_{\alpha\beta}.
  \label{eq:op-h-tensors}
\end{align}
Each \(h^r\) is symmetric and traceless in every displayed index pair.  The
most general parity-even separated-point structure satisfying stress-tensor
conservation is the three-parameter tensor
\begin{align}
  t[a,b,c]_{\mu\nu,\rho\sigma,\alpha\beta}
  &=
  a\,h^5_{\mu\nu,\rho\sigma,\alpha\beta}
  +
  b\,h^4_{\alpha\beta,\mu\nu,\rho\sigma}
  -
  (2a+b)
  \left(
    h^4_{\mu\nu,\rho\sigma,\alpha\beta}
    +
    h^4_{\rho\sigma,\mu\nu,\alpha\beta}
  \right)
  \nonumber\\
  &\quad
  +
  c\,h^3_{\mu\nu,\rho\sigma}h^1_{\alpha\beta}
  +
  c
  \left(
    h^3_{\rho\sigma,\alpha\beta}h^1_{\mu\nu}
    +
    h^3_{\mu\nu,\alpha\beta}h^1_{\rho\sigma}
  \right)
  \nonumber\\
  &\quad
  +
  e\,h^2_{\mu\nu,\rho\sigma}h^1_{\alpha\beta}
  +
  e'
  \left(
    h^2_{\rho\sigma,\alpha\beta}h^1_{\mu\nu}
    +
    h^2_{\mu\nu,\alpha\beta}h^1_{\rho\sigma}
  \right)
  \nonumber\\
  &\quad
  +
  f\,h^1_{\mu\nu}h^1_{\rho\sigma}h^1_{\alpha\beta},
  \label{eq:op-ttt-tensor}
\end{align}
where conservation fixes the auxiliary coefficients to be
\begin{align}
  e=(D+2)(Da+b-c),
  \qquad
  e'=-(D+4)(D-2)a-(D-2)b+Dc,
  \nonumber\\
  f=(D+4)(D-2)(4a+2b-c).
  \label{eq:op-ttt-dependent-coefficients}
\end{align}
The three tensor structures in
\eqref{eq:stress-tensor-three-point-structures} are therefore
\[
  \mathcal I^{(1)}=\mathfrak C[t[1,0,0]],
  \qquad
  \mathcal I^{(2)}=\mathfrak C[t[0,1,0]],
  \qquad
  \mathcal I^{(3)}=\mathfrak C[t[0,0,1]].
\]
Equivalently,
\begin{equation}
  \left\langle
    T_{\mu\nu}(x_1)T_{\rho\sigma}(x_2)T_{\alpha\beta}(x_3)
  \right\rangle
  =
  a\,\mathcal I^{(1)}_{\mu\nu,\rho\sigma,\alpha\beta}
  +
  b\,\mathcal I^{(2)}_{\mu\nu,\rho\sigma,\alpha\beta}
  +
  c\,\mathcal I^{(3)}_{\mu\nu,\rho\sigma,\alpha\beta}
  =
  \mathfrak C[t[a,b,c]]_{\mu\nu,\rho\sigma,\alpha\beta},
  \label{eq:stress-tensor-three-point-structures}
\end{equation}
with three independent coefficients \(a,b,c\).  The Ward identity fixes one
linear combination of these coefficients in terms of the two-point coefficient
\(C_T\).  Let
\[
  S_D=\frac{2\pi^{D/2}}{\Gamma(D/2)}
\]
be the area of the unit \((D-1)\)-sphere.  The small-sphere Ward identity uses
the angular averages
\[
  \int_{S^{D-1}}\dd\Omega\,\widehat n_\mu\widehat n_\nu
  =
  \frac{S_D}{D}\delta_{\mu\nu},
  \qquad
  \int_{S^{D-1}}\dd\Omega\,
  \widehat n_\mu\widehat n_\nu\widehat n_\rho\widehat n_\sigma
  =
  \frac{S_D}{D(D+2)}
  \left(
    \delta_{\mu\nu}\delta_{\rho\sigma}
    +
    \delta_{\mu\rho}\delta_{\nu\sigma}
    +
    \delta_{\mu\sigma}\delta_{\nu\rho}
  \right).
\]
Substituting \eqref{eq:op-ttt-tensor} into the flux
\(\int_{\partial B_\epsilon(x_3)}\dd S\,n_\lambda
T^{\lambda\tau}\epsilon_\tau\), taking \(\epsilon\to0\), and comparing with
the conformal variation of the two remaining stress tensors in
\eqref{eq:stress-tensor-two-point} gives
\begin{equation}
  4S_D\,
  \frac{(D-2)(D+3)a-2b-(D+1)c}{D(D+2)}
  =
  C_T .
  \label{eq:ttt-ward-ct-relation}
\end{equation}
The remaining two independent combinations are genuine CFT data.  In lower
dimensions or in parity-violating theories the list of admissible structures
changes; the statement above is tied to the stated assumptions.

As an important mixed example, the Ward identity fixes the coefficient of the
stress-tensor three-point function with two identical scalar primaries.  For
\(\Delta_1=\Delta_2=\Delta\),
\begin{equation}
  \left\langle
    \mathcal O(x_1)\mathcal O(x_2)T_{\mu\nu}(x_3)
  \right\rangle
  =
  \frac{C_{\mathcal O\mathcal O T}}
  {|x_{13}|^D|x_{23}|^D|x_{12}|^{2\Delta-D}}
  \left(
    \frac{X_\mu X_\nu}{X^2}
    -
    \frac1D\delta_{\mu\nu}
  \right),
  \label{eq:scalar-scalar-stress-tensor-three-point}
\end{equation}
with
\[
  X^\mu=\frac{x_{13}^\mu}{x_{13}^2}
       -\frac{x_{23}^\mu}{x_{23}^2}.
\]
More generally, for two scalar primaries \(\mathcal O_1,\mathcal O_2\) of the
same dimension \(\Delta\) with
\[
  \left\langle
    \mathcal O_1(x)\mathcal O_2(0)
  \right\rangle
  =
  \frac{N_{12}}{|x|^{2\Delta}},
\]
the same tensor structure appears with coefficient
\begin{equation}
  C_{12T}
  =
  -\,\frac{D\Delta}{(D-1)A_{D-1}}\,N_{12},
  \qquad
  A_{D-1}=\operatorname{vol}(S^{D-1})
  =
  \frac{2\pi^{D/2}}{\Gamma(D/2)}.
  \label{eq:scalar-scalar-stress-ward-coefficient}
\end{equation}
Here the sign is the sign of the stress-tensor convention in
\eqref{eq:cft-charge-surface} and \eqref{eq:variation-by-small-charge}.  The
derivation is a small-sphere Ward identity for the dilatation vector
\(\epsilon^\nu=x^\nu\).  With \(x_2=0\) and \(r=|x_3|\to0\), the flux
integral contains
\[
  \int_{S^{D-1}}\dd\Omega\,
  n^\mu n^\nu
  \left(
    n_\mu n_\nu-\frac1D\delta_{\mu\nu}
  \right)
  =
  A_{D-1}\left(1-\frac1D\right),
\]
and matching the charge action on \(\mathcal O_2(0)\) gives
\eqref{eq:scalar-scalar-stress-ward-coefficient}.  If
\(\Delta_1\ne\Delta_2\), the two-point coefficient \(N_{12}\) vanishes by
dilatation covariance, and the corresponding stress-tensor three-point
function vanishes.

\section{Four-Point Functions and Cross Ratios}

For four scalar insertions, conformal transformations can remove all position
dependence except two cross ratios:
\begin{equation}
  u=
  \frac{x_{12}^2x_{34}^2}{x_{13}^2x_{24}^2},
  \qquad
  v=
  \frac{x_{14}^2x_{23}^2}{x_{13}^2x_{24}^2}.
  \label{eq:cross-ratios-u-v}
\end{equation}
Equivalently, the four points can be moved into a two-plane and placed at
\[
  z_1=0,\qquad z_2=z,\qquad z_3=1,\qquad z_4=\infty,
\]
with
\[
  z=\frac{z_{12}z_{34}}{z_{13}z_{24}},
  \qquad
  u=z\bar z,\qquad v=(1-z)(1-\bar z).
\]
This frame is obtained by first using the conformal group to put the four
points in a common two-plane and then using the residual fractional-linear
transformation \(z\mapsto(\alpha z+\beta)/(\gamma z+\delta)\) on that plane.

The general scalar four-point function can be written as
\begin{align}
  &\left\langle
    \mathcal O_1(x_1)\mathcal O_2(x_2)
    \mathcal O_3(x_3)\mathcal O_4(x_4)
  \right\rangle
  \notag\\
  &\quad =
  \frac{\mathcal G(u,v)}
  {|x_{12}|^{\Delta_1+\Delta_2}
   |x_{34}|^{\Delta_3+\Delta_4}}
  \left(\frac{|x_{24}|}{|x_{14}|}\right)^{\Delta_1-\Delta_2}
  \left(\frac{|x_{14}|}{|x_{13}|}\right)^{\Delta_3-\Delta_4}.
  \label{eq:scalar-four-point-general}
\end{align}
For identical scalar primaries of dimension \(\Delta\), this simplifies to
\begin{equation}
  \left\langle
    \mathcal O(x_1)\mathcal O(x_2)
    \mathcal O(x_3)\mathcal O(x_4)
  \right\rangle
  =
  \frac{\mathcal G(u,v)}
       {|x_{12}|^{2\Delta}|x_{34}|^{2\Delta}}.
  \label{eq:identical-scalar-four-point}
\end{equation}
The function \(\mathcal G(u,v)\) is dynamical CFT data.  The next stage is to
determine how the operator product expansion encodes local multiplication of
operators and how associativity constrains the allowed spectrum and OPE
coefficients.  Detailed computational machinery for solving crossing equations
is deferred from the core monograph until conformal theory is rebuilt as a
separate systematic development.
