\chapter{Mirror-Channel TBA and Finite-Size Effects}
\label{ch:integrable-mirror-channel-tba-finite-size-wrapping}

\ConventionsMixedCFT

This chapter refines the finite-volume interpretation of the thermodynamic
Bethe ansatz.  The primitive object is a Euclidean two-torus partition
function with two Hamiltonian decompositions.  The mirror-channel
decomposition turns the circumference of the direct theory into the inverse
temperature of a gas in the rotated theory.  Finite-size corrections are then
organized by the number of mirror particles winding around the compact cycle.

\section{Two Hamiltonian Decompositions}

Let the Euclidean spacetime be a flat torus with cycles of lengths \(R\) and
\(L\).  Quantization along the \(L\)-direction gives
\[
  Z(R,L)=\operatorname{Tr}_{\Hilb_R}\exp[-L H_R],
\]
where \(H_R\) is the Hamiltonian on a spatial circle of circumference \(R\).
Assume that \(H_R\) is self-adjoint, bounded below, and has a unique ground
state separated by a gap from the rest of the spectrum.  Then
\[
  -\lim_{L\to\infty}\frac{1}{L}\log Z(R,L)=E_0(R),
\]
where \(E_0(R)\) is the direct-channel vacuum energy.

Quantization after exchanging the two cycles gives
\[
  Z(R,L)=\operatorname{Tr}_{\widetilde{\Hilb}_L}\exp[-R\widetilde H_L].
\]
The Hilbert space \(\widetilde{\Hilb}_L\) is the mirror-channel Hilbert space.
For a relativistic theory obtained by Euclidean rotation, the one-particle
dispersion in the infinite-volume mirror channel is
\[
  \widetilde E_a(\theta)=m_a\cosh\theta,\qquad
  \widetilde p_a(\theta)=m_a\sinh\theta .
\]
The same function \(m_a\cosh\theta\) now appears in the Boltzmann weight
\(\exp[-R\widetilde E_a(\theta)]\).

\begin{remark}[Vacuum energy from mirror pressure]
\label{rem:mirror-pressure-vacuum-energy}
Assume that both torus decompositions exist at a common regulator, that the
large-\(L\) limit of \(L^{-1}\log Z(R,L)\) exists in the direct channel, and
that the mirror channel has a thermodynamic limit at inverse temperature
\(R\).  Let
\[
  f_{\mathrm{mir}}(R)
  =
  -\lim_{L\to\infty}\frac{1}{RL}\log
  \operatorname{Tr}_{\widetilde{\Hilb}_L}\exp[-R\widetilde H_L]
\]
be the mirror free-energy density.  Then
\[
  E_0(R)=R\,f_{\mathrm{mir}}(R).
\]
\end{remark}

The two traces are the same torus functional written in two Hamiltonian
channels.  Dividing the equality of traces by \(L\) after taking the logarithm
gives
\[
  \frac{1}{L}\log Z(R,L)
  =
  \frac{1}{L}\log
  \operatorname{Tr}_{\widetilde{\Hilb}_L}\exp[-R\widetilde H_L].
\]
The direct-channel assumption gives the left-hand limit \(-E_0(R)\).  The
mirror thermodynamic definition gives the right-hand limit
\(-R f_{\mathrm{mir}}(R)\).  Equality of the two limits proves the formula.

\section{Mirror TBA Functional}

Use the kernel convention of
Chapter~\ref{ch:integrable-thermodynamic-bethe-ansatz}:
\[
  \varphi_{ab}(\theta)
  =
  \frac{1}{2\pi\ii}\frac{d}{d\theta}\log S_{ab}(\theta).
\]
Here \(S_{ab}(\theta)\) is the diagonal two-particle scattering phase in the
mirror channel, and the branch of the logarithm is fixed continuously on the
real rapidity line after the particle labels and statistics convention have
been declared.

Let \(\rho_a(\theta)\) and \(\rho_a^h(\theta)\) be particle and hole densities
per unit mirror length, so that
\[
  L\rho_a(\theta)d\theta
\]
is the number of occupied rapidities of species \(a\) in the interval
\((\theta,\theta+d\theta)\).  Differentiating the Bethe--Yang equations gives
\[
  \rho_a(\theta)+\rho_a^h(\theta)
  =
  \frac{m_a}{2\pi}\cosh\theta
  +
  \sum_b\int_{\R}d\theta'\,
  \varphi_{ab}(\theta-\theta')\rho_b(\theta').
\]
The entropy density of a Bethe cell with exclusion occupancy is
\[
  s_a=
  (\rho_a+\rho_a^h)\log(\rho_a+\rho_a^h)
  -\rho_a\log\rho_a-\rho_a^h\log\rho_a^h .
\]
Define the pseudoenergy by
\[
  \epsilon_a(\theta)=\log\frac{\rho_a^h(\theta)}{\rho_a(\theta)} .
\]

\begin{proposition}[Mirror TBA variational equation]
\label{prop:mirror-tba-variational-equation}
For diagonal factorized scattering with the density equation above, the
stationary point of the mirror free-energy functional at inverse temperature
\(R\) satisfies
\[
  \epsilon_a(\theta)
  =
  Rm_a\cosh\theta
  -
  \sum_b\int_{\R}d\theta'\,
  \varphi_{ab}(\theta-\theta')
  \log\!\left(1+e^{-\epsilon_b(\theta')}\right).
\]
At this stationary point,
\[
  E_0(R)=\mathcal E_{\mathrm{bulk}}R
  -
  \sum_a\int_{\R}\frac{d\theta}{2\pi}\,
  m_a\cosh\theta\,
  \log\!\left(1+e^{-\epsilon_a(\theta)}\right).
\]
\end{proposition}

\begin{proof}
The variation is the same variational calculation as in
Proposition~\ref{prop:tba-variational-derivation}, with \(R\) now interpreted
as inverse temperature of the mirror gas.  A variation \(\delta\rho_a\)
induces
\[
  \delta\rho_c^t(\theta)
  =
  \sum_a\int d\theta'\,
  \varphi_{ca}(\theta-\theta')\delta\rho_a(\theta'),
  \qquad
  \rho_c^t=\rho_c+\rho_c^h .
\]
The entropy variation is
\[
  \delta s
  =
  \sum_a\int d\theta\,
  \left[
  \log\frac{\rho_a^h}{\rho_a}\,\delta\rho_a
  +
  \log\frac{\rho_a^t}{\rho_a^h}\,\delta\rho_a^t
  \right].
\]
Substituting the density-constraint variation and using
\(\log(\rho_a^t/\rho_a^h)=\log(1+e^{-\epsilon_a})\) gives the displayed
pseudoenergy equation.  Inserting the stationary relation back into the
mirror free-energy density gives
\[
  f_{\mathrm{mir}}(R)
  =
  -\frac{1}{2\pi R}
  \sum_a m_a\int_{\R}d\theta\,\cosh\theta
  \log(1+e^{-\epsilon_a(\theta)}).
\]
Remark~\ref{rem:mirror-pressure-vacuum-energy} then gives the
subtracted direct-channel vacuum energy.  The additive
\(\mathcal E_{\mathrm{bulk}}R\) is the local extensive coordinate fixed in
Chapter~\ref{ch:integrable-rg-flows-perturbed-cft}.
\end{proof}

\section{Large-Circumference Expansion}

Let
\[
  d_a(\theta)=Rm_a\cosh\theta,\qquad
  q_a(\theta)=e^{-d_a(\theta)},\qquad
  L_a(\theta)=\log(1+e^{-\epsilon_a(\theta)}).
\]
The large-\(R\) expansion is an expansion in mirror occupation factors
\(q_a\), equivalently in windings around the direct spatial circle.  The
first winding is universal; later windings also contain the scattering
kernels.

\paragraph{Two-winding expansion of the vacuum energy.}
Assume that the kernel convolution operator is bounded in a weighted
\(L^\infty\) norm for which all \(q_a\) are small, so that the TBA equation is
solved by iteration.  The TBA equation is
\[
  \epsilon_a=d_a-\sum_b\varphi_{ab}*L_b .
\]
If \(L_b=q_b+O(q^2)\), then
\[
  e^{-\epsilon_a}
  =
  e^{-d_a}\exp\!\left(\sum_b\varphi_{ab}*L_b\right)
  =
  q_a\left(1+\sum_b\varphi_{ab}*q_b\right)+O(q^3).
\]
Using \(\log(1+x)=x-x^2/2+O(x^3)\) gives
\[
  L_a(\theta)
  =
  q_a(\theta)
  +
  q_a(\theta)\sum_b\int_{\R}d\theta'\,
  \varphi_{ab}(\theta-\theta')q_b(\theta')
  -\frac12 q_a(\theta)^2
  +O(q^3).
\]
Consequently
\[
\begin{split}
  E_0(R)-\mathcal E_{\mathrm{bulk}}R
  &=
  -\sum_a\int_{\R}\frac{d\theta}{2\pi}\,
  m_a\cosh\theta\,q_a(\theta)
\\
  &\quad
  -\sum_{a,b}\int_{\R}\frac{d\theta}{2\pi}\int_{\R}d\theta'\,
  m_a\cosh\theta\,
  q_a(\theta)\varphi_{ab}(\theta-\theta')q_b(\theta')
\\
  &\quad
  +\frac12\sum_a\int_{\R}\frac{d\theta}{2\pi}\,
  m_a\cosh\theta\,q_a(\theta)^2
  +O(q^3).
\end{split}
\]

\begin{proposition}[Vacuum Luescher term with a remainder bound]
\label{prop:vacuum-luscher-term-remainder}
Let the set \(\mathsf A\) of mirror particle species be finite, and let
\[
  m_*=\min_{a\in\mathsf A}m_a>0 .
\]
Assume that the diagonal mirror TBA equations of
Proposition~\ref{prop:mirror-tba-variational-equation} have the exclusion
form written above and that their kernels obey
\[
  \Phi
  =
  \sup_{a\in\mathsf A}
  \sum_{b\in\mathsf A}\int_{\R}du\,|\varphi_{ab}(u)|<\infty .
\]
Then there are constants \(R_0\) and \(C\), depending on the masses and on
\(\Phi\), such that for \(R\ge R_0\) the solution in the small-occupation
branch satisfies
\[
  E_0(R)-\mathcal E_{\mathrm{bulk}}R
  =
  -\sum_{a\in\mathsf A}\frac{m_a}{\pi}K_1(m_aR)
  +\mathcal R_2(R),
  \qquad
  |\mathcal R_2(R)|\le C R^{-1/2}e^{-2m_*R}.
\]
Here \(K_1\) is the modified Bessel function normalized by
\[
  K_1(r)=\int_0^\infty d\theta\,e^{-r\cosh\theta}\cosh\theta .
\]
\end{proposition}

\begin{proof}
Write the TBA fixed-point equation directly for
\[
  L_a(\theta)=\log(1+e^{-\epsilon_a(\theta)}).
\]
With
\[
  (K L)_a(\theta)
  =
  \sum_{b\in\mathsf A}\int_{\R}d\theta'\,
  \varphi_{ab}(\theta-\theta')L_b(\theta'),
\]
the equation is
\[
  L_a(\theta)
  =
  \log\!\left(1+q_a(\theta)e^{(K L)_a(\theta)}\right),
  \qquad
  q_a(\theta)=e^{-Rm_a\cosh\theta}.
\]
The integral bound gives
\[
  \|K L\|_\infty\le \Phi\|L\|_\infty .
\]
For \(R\) large enough the map
\[
  T_R(L)_a(\theta)
  =
  \log\!\left(1+q_a(\theta)e^{(K L)_a(\theta)}\right)
\]
maps the closed ball
\(\|L\|_\infty\le 2e^{-Rm_*}\) into itself.  Indeed
\[
  |T_R(L)_a(\theta)|
  \le q_a(\theta)e^{\Phi\|L\|_\infty}
  \le e^{-Rm_*}e^{2\Phi e^{-Rm_*}}
  \le 2e^{-Rm_*}
\]
after increasing \(R_0\).  The mean-value theorem and
\(|\log(1+x)-\log(1+y)|\le |x-y|\) for \(x,y\ge0\) give
\[
  \|T_R(L)-T_R(M)\|_\infty
  \le
  C_\Phi e^{-Rm_*}\|L-M\|_\infty ,
\]
with \(C_\Phi\) independent of \(R\).  Thus \(T_R\) is a contraction for
\(R\ge R_0\), so the small-occupation solution exists uniquely in this ball.

For that solution,
\[
  |(K L)_a(\theta)|\le 2\Phi e^{-Rm_*}.
\]
Using \(e^x=1+O(x)\) uniformly for
\(|x|\le 2\Phi e^{-Rm_*}\) and
\(\log(1+y)=y+O(y^2)\) uniformly for
\(0\le y\le 2e^{-Rm_*}\), one obtains the pointwise estimate
\[
  L_a(\theta)
  =
  q_a(\theta)+\delta L_a(\theta),
  \qquad
  |\delta L_a(\theta)|
  \le
  C q_a(\theta)e^{-Rm_*}+C q_a(\theta)^2 .
\]
Substitution in the vacuum-energy formula gives
\[
  E_0(R)-\mathcal E_{\mathrm{bulk}}R
  =
  -\sum_a\int_{\R}\frac{d\theta}{2\pi}\,
  m_a\cosh\theta\,q_a(\theta)+\mathcal R_2(R),
\]
where
\[
  |\mathcal R_2(R)|
  \le
  C\sum_a
  \int_{\R}\frac{d\theta}{2\pi}\,
  m_a\cosh\theta
  \left(q_a(\theta)e^{-Rm_*}+q_a(\theta)^2\right).
\]
The first integral is
\[
  \int_{\R}d\theta\,\cosh\theta\,e^{-Rm_a\cosh\theta}
  =
  2K_1(m_aR),
\]
so its contribution to the energy is
\(-m_aK_1(m_aR)/\pi\).  It remains only to record the uniform large-\(R\)
bound.  For \(\lambda\ge1\),
\[
  K_1(\lambda)
  =
  \int_0^\infty d\theta\,\cosh\theta\,e^{-\lambda\cosh\theta}
  \le
  C\lambda^{-1/2}e^{-\lambda}.
\]
This elementary estimate follows by splitting the integral at
\(\theta=1\): on \(0\le\theta\le1\), \(\cosh\theta\ge 1+\theta^2/3\), while
on \(\theta\ge1\) the stronger decay of \(e^{-\lambda\cosh\theta}\) gives a
term bounded by \(C e^{-\lambda\cosh 1}\).  Applying this estimate to
\(\lambda=m_aR\) and to \(\lambda=2m_aR\) gives
\[
  |\mathcal R_2(R)|\le C R^{-1/2}e^{-2m_*R}
\]
after another harmless enlargement of \(C\).
\end{proof}

\begin{example}[Free Majorana leading winding]
\label{ex:free-majorana-leading-winding}
For the massive Ising/Majorana theory the derivative kernel vanishes.  The
leading winding contribution is
\[
  E_0(R)-\mathcal E_{\mathrm{bulk}}R
  =
  -\frac{m}{2\pi}\int_{-\infty}^{\infty}d\theta\,
  \cosh\theta\,e^{-mR\cosh\theta}
  +O(e^{-2mR}).
\]
The modified Bessel function has the integral representation
\[
  K_\nu(r)=\int_0^\infty d\theta\,e^{-r\cosh\theta}\cosh(\nu\theta),
  \qquad r>0.
\]
Therefore
\[
  \int_{-\infty}^{\infty}d\theta\,
  \cosh\theta\,e^{-r\cosh\theta}=2K_1(r),
\]
and the first winding gives
\[
  E_0(R)-\mathcal E_{\mathrm{bulk}}R
  =
  -\frac{m}{\pi}K_1(mR)+O(e^{-2mR}).
\]
Using the standard steepest-descent expansion at the saddle \(\theta=0\),
\[
  K_1(r)
  \sim
  \sqrt{\frac{\pi}{2r}}\,e^{-r}
  \left(1+\frac{3}{8r}-\frac{15}{128r^2}+\cdots\right),
\]
so the leading finite-size vacuum correction is exponentially small with
prefactor \(R^{-1/2}e^{-mR}\).
\end{example}

\section{Excited-State F-Term}

A direct-channel finite-volume excited state is described, at leading large
\(R\), by physical rapidities \(\theta_j\) and species \(a_j\) satisfying
Bethe--Yang equations.  A single mirror particle of species \(b\) and rapidity
\(\theta\) winding once around the direct spatial circle scatters through the
physical particles.  Euclidean rotation converts this scattering into the
crossed amplitudes
\[
  S_{b a_j}\!\left(\theta-\theta_j+\frac{\ii\pi}{2}\right).
\]

\begin{definition}[Analytic strip for the F-term]
\label{def:f-term-analytic-strip}
For a state \(\Psi=(a_1,\theta_1;\ldots;a_n,\theta_n)\), an F-term strip is a
connected domain containing the real \(\theta\)-axis on which every function
\[
  S_{b a_j}\!\left(\theta-\theta_j+\frac{\ii\pi}{2}\right)
\]
is meromorphic, the chosen integration contour can be kept away from poles,
and the product of amplitudes has sufficient decay for the contour integral
below.
\end{definition}

\begin{controlledapproximation}[One-winding excited-state F-term]
\label{ca:one-winding-excited-state-f-term}
Assume diagonal scattering, a declared F-term strip with no pole crossed by
the real mirror contour, and a large-\(R\) cluster expansion of the torus trace
in mirror occupation factors.  The one-mirror-particle correction to the
energy of the direct-channel state \(\Psi\), after subtracting the vacuum
mirror loop, is
\[
  \Delta E_{\Psi}^{F}(R)
  =
  -\sum_b\int_{\R}\frac{d\theta}{2\pi}\,
  m_b\cosh\theta\,e^{-Rm_b\cosh\theta}
  \left(
  \prod_{j=1}^{n}
  S_{b a_j}\!\left(\theta-\theta_j+\frac{\ii\pi}{2}\right)
  -1
  \right).
\]
\end{controlledapproximation}

In the mirror channel, the direct excited state is represented by operator
insertions that a mirror particle crosses while winding around the cycle.  At
one winding, the trace contribution of one mirror particle of species \(b\)
and rapidity \(\theta\) has Boltzmann weight
\(\exp[-Rm_b\cosh\theta]\) and density of states \(d\theta/(2\pi)\).  Crossing
the direct-channel physical particles multiplies this contribution by
\[
  \prod_{j=1}^{n}
  S_{b a_j}\!\left(\theta-\theta_j+\frac{\ii\pi}{2}\right).
\]
The logarithm of the trace converts this one-particle contribution into an
energy shift with the overall factor \(-m_b\cosh\theta\); the identical loop
with no physical particles is the vacuum contribution and gives the
subtraction by \(1\).  Summing over mirror species gives the displayed
formula.

If a pole is crossed while moving the contour to the declared F-term strip,
the residue is a separate contribution.  A convenient residue ledger is
\[
  \Delta E_{\Psi}^{\mu}(R)
  =
  -\sum_{\theta_*\in\mathcal P}
  \sigma_*\,
  \ii\,\operatorname*{Res}_{\theta=\theta_*}
  \left[
  \frac{m_b\cosh\theta}{2\pi}
  e^{-Rm_b\cosh\theta}
  \left(
  \prod_j
  S_{b a_j}\!\left(\theta-\theta_j+\frac{\ii\pi}{2}\right)
  -1
  \right)
  \right],
\]
where \(\mathcal P\) is the finite set of crossed poles and
\(\sigma_*=\pm1\) records the orientation of the contour deformation.  In a
bound-state theory those residues are controlled by the same pole data that
enter the factorized scattering bootstrap.  The formula becomes a theorem
only after the analytic strip, contour orientation, and bound-state pole
normalization have been specified.

\section{Wrapping Effects for Local Operators}

A wrapping contribution is a finite-volume term represented in the mirror
channel by virtual propagation around a compact cycle.  Its order is measured
by the number and species of mirror occupation factors \(q_a(\theta)\).  This
ordering is independent of the loop order of any Lagrangian perturbation
series.

Let \(\widehat{\mathcal O}\) be a local operator with a declared infinite-volume
normalization.  A diagonal finite-volume matrix element has an expansion of
the form
\[
  \langle \Psi,R|\widehat{\mathcal O}|\Psi,R\rangle
  =
  \langle \Psi|\widehat{\mathcal O}|\Psi\rangle_{\infty}
  +
  \sum_{k\ge1}\Delta_{\mathcal O}^{(k)}(R),
\]
where \(\Delta_{\mathcal O}^{(k)}(R)\) is assembled from \(k\)-mirror-particle
form factors, crossed scattering factors, and finite-volume density
determinants.  The first term in this series is determined by the connected
one-mirror-particle form factor in the crossed channel, after subtracting
the vacuum wrapping term and the disconnected pieces already accounted for by
the finite-volume normalization of \(|\Psi,R\rangle\).

\begin{definition}[Operator wrapping datum]
\label{def:operator-wrapping-datum}
An operator wrapping datum consists of:
\begin{enumerate}
\item the infinite-volume form factors of \(\widehat{\mathcal O}\) in the
state normalization of Chapter~\ref{ch:integrable-form-factor-bootstrap-local-operators};
\item the analytic strip and crossing prescription for moving mirror
rapidities through the operator insertion and through the physical particles;
\item the finite-volume density determinant for the physical Bethe--Yang
state;
\item a subtraction prescription for vacuum terms and disconnected diagonal
pieces.
\end{enumerate}
\end{definition}

With this datum fixed, each wrapping correction is a contour integral whose
integrand is a declared meromorphic function.  Residues from crossed poles
are then part of the same contour calculation rather than extra rules.

\section{Relation To Planar Wrapping}

In large-\(N\) gauge-theory spectral problems the word wrapping often denotes
corrections whose spin-chain interaction range reaches the length of a
single-trace operator.  The relativistic finite-size mechanism developed here
has a literal spacetime compact cycle and a mirror thermal trace.  Later
planar-integrability chapters build the bridge by identifying the spectral
problem, the worldsheet or spin-chain mirror channel, and the analytic
continuation that maps finite-size corrections between the two settings.

\begin{openproblem}[Excited-state mirror construction]
\label{op:excited-state-mirror-construction}
Give a theorem-level construction of excited-state TBA starting from
factorized scattering, a declared analytic strip, pole data, and finite-volume
Bethe states, including a proof that the contour deformations and residue
terms reproduce the spectral trace on the torus.
\end{openproblem}
