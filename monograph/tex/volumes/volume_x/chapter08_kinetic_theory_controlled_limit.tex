\chapter{Kinetic Theory As A Controlled Limit}
\label{ch:thermal-kinetic-theory-controlled-limit}

\ConventionsMixed

Kinetic theory is the long-wavelength, weakly off-equilibrium description in
which selected real-time QFT correlators are represented by on-shell
phase-space densities.  The construction has three inputs: a quasiparticle
spectral regime, an on-shell collision kernel extracted from microscopic
scattering or finite-temperature amplitudes, and a hierarchy separating the
microscopic correlation scale from the spacetime scale of variation.  This
chapter develops the relativistic Boltzmann framework with the assumptions
kept explicit, derives detailed balance and the entropy inequality, and
records the precise point at which kinetic theory becomes hydrodynamics.

Let spacetime dimension be \(D=d+1\), with spatial momentum
\(\mathbf p\in\mathbb R^d\).  For species \(a\),
\[
  p^\mu=(E_a(\mathbf p),\mathbf p),
  \qquad
  E_a(\mathbf p)=\sqrt{\mathbf p^2+m_a^2}.
\]
The Lorentz-invariant on-shell measure used below is
\begin{equation}
  d\Pi_a(p)
  =
  \frac{d^d\mathbf p}{(2\pi)^d\,2E_a(\mathbf p)}.
  \label{eq:kinetic-onshell-measure}
\end{equation}
The factor \(2p^\mu d\Pi_a\) has the standard equal-time interpretation:
\[
  2E_a\,d\Pi_a=\frac{d^d\mathbf p}{(2\pi)^d}.
\]
Internal degeneracies may be included by replacing the species label \(a\)
with a refined label, or by inserting an explicit degeneracy \(g_a\).

\section{Quasiparticle Distribution From Spectral Data}

Let \(G^<_a(x,y)\) be a Wightman two-point function for a microscopic field
or for a gauge-invariant quasiparticle interpolating operator in a regulated
theory.  Introduce center and relative coordinates
\[
  X=\frac{x+y}{2},\qquad s=x-y,
\]
and define the Wigner transform
\[
  G^<_a(X,p)=\int d^Ds\,e^{\ii p\cdot s}G^<_a(X+s/2,X-s/2).
\]
The quasiparticle distribution \(f_a(X,\mathbf p)\) is the coefficient of
the positive-energy pole contribution when the spectral density has the form
\begin{equation}
  \rho_a(X,p)
  =
  2\pi Z_a(X,\mathbf p)\operatorname{sgn}(p^0)
  \delta\!\left((p^0)^2-E_a(\mathbf p)^2\right)
  +\rho^{\rm off}_a(X,p),
  \label{eq:kinetic-narrow-spectral-form}
\end{equation}
with the off-shell remainder small in the topology used for the gradient
expansion.  Equivalently, near \(p^0=E_a(\mathbf p)\),
\[
  G^<_a(X,p)
  =
  2\pi Z_a(X,\mathbf p)
  f_a(X,\mathbf p)
  \delta(p^0-E_a(\mathbf p))
  +\text{controlled off-shell remainder}.
\]
The pole approximation requires
\begin{equation}
  \Gamma_a(\mathbf p)\ll E_a(\mathbf p),
  \qquad
  \Gamma_a(\mathbf p)\,\ell_{\rm variation}\ll1,
  \label{eq:kinetic-quasiparticle-scale-assumptions}
\end{equation}
where \(\Gamma_a\) is the damping width and
\(\ell_{\rm variation}\) is the spacetime scale over which \(f_a\) changes.
These inequalities are hypotheses on the state and on the coupling regime;
they are not consequences of conservation laws alone.

\section{Quasiparticle Projection of the Two-Point Equation}
\label{sec:kinetic-quasiparticle-projection-two-point-equation}

The Boltzmann drift term is the on-shell projection of the leading
Schwinger--Keldysh two-point equation.  We record the projection explicitly
because it is the point where an operator statement about correlators becomes
a phase-space equation.

\begin{controlledapproximation}[Force-free quasiparticle projection]
\label{ca:kinetic-force-free-quasiparticle-projection}
Consider one scalar species in flat spacetime, with constant mass and a
single narrow positive-energy peak.  Let
\begin{equation}
  F(p)=(p^0)^2-E(\mathbf p)^2,\qquad
  E(\mathbf p)=\sqrt{\mathbf p^2+m^2},
  \label{eq:kinetic-free-mass-shell-function}
\end{equation}
and assume that the leading-gradient Kadanoff--Baym equation has the form
\begin{equation}
  \frac12\{F,G^<\}_{\rm kin}=I^<[G],
  \label{eq:kinetic-leading-kb-force-free}
\end{equation}
where \(I^<\) is the collision functional and
\begin{equation}
  \{A,B\}_{\rm kin}
  =
  \partial_{p^0}A\,\partial_tB
  -\partial_tA\,\partial_{p^0}B
  -\partial_{\mathbf p}A\cdot\nabla_{\mathbf x}B
  +\nabla_{\mathbf x}A\cdot\nabla_{\mathbf p}B .
  \label{eq:kinetic-poisson-bracket-convention}
\end{equation}
Use the positive-energy ansatz
\begin{equation}
  G^<(X,p)=2\pi Z(\mathbf p)f(X,\mathbf p)\delta(p^0-E(\mathbf p))
  \label{eq:kinetic-positive-energy-lesser-ansatz}
\end{equation}
with \(Z\) independent of \(X\) at this order.  Terms from mean-field
forces, Berry curvature, spatially varying masses, and off-shell memory
kernels are omitted in this controlled approximation; each of them modifies
the left-hand side by additional explicitly computable phase-space terms.
\end{controlledapproximation}

\begin{remark}[Projection of the drift term]
\label{rem:kinetic-quasiparticle-drift-projection}
Under
Controlled Approximation~\ref{ca:kinetic-force-free-quasiparticle-projection},
the positive-energy shell projection of
\eqref{eq:kinetic-leading-kb-force-free} is
\begin{equation}
  p^\mu\partial_\mu f(X,\mathbf p)=\mathcal C[f](X,\mathbf p),
  \qquad
  p^\mu=(E(\mathbf p),\mathbf p),
  \label{eq:kinetic-projected-boltzmann-equation}
\end{equation}
with
\begin{equation}
  \mathcal C[f](X,\mathbf p)
  =
  \frac{1}{Z(\mathbf p)}
  \int_0^\infty\frac{dp^0}{2\pi}\,I^<[G](X,p).
  \label{eq:kinetic-projected-collision-functional}
\end{equation}

Since \(F\) is independent of \(X\),
\begin{equation}
  \{F,G^<\}_{\rm kin}
  =
  2p^0\partial_tG^<
  +2\mathbf p\cdot\nabla_{\mathbf x}G^< .
  \label{eq:kinetic-free-bracket}
\end{equation}
Using \eqref{eq:kinetic-positive-energy-lesser-ansatz} and the
\(X\)-independence of \(Z\) and \(E\),
\begin{equation}
  \frac12\{F,G^<\}_{\rm kin}
  =
  2\pi Z(\mathbf p)
  \bigl(p^0\partial_t+\mathbf p\cdot\nabla_{\mathbf x}\bigr)
  f(X,\mathbf p)\,\delta(p^0-E(\mathbf p)).
  \label{eq:kinetic-projected-bracket-before-integral}
\end{equation}
Integrating over \(p^0>0\) and dividing by \(Z(\mathbf p)\) gives
\begin{equation}
  E(\mathbf p)\partial_tf+\mathbf p\cdot\nabla_{\mathbf x}f
  =
  \frac{1}{Z(\mathbf p)}
  \int_0^\infty\frac{dp^0}{2\pi}I^<[G](X,p),
  \label{eq:kinetic-projected-bracket-after-integral}
\end{equation}
which is \eqref{eq:kinetic-projected-boltzmann-equation}.  Dividing by
\(E\) gives the coordinate-time form
\((\partial_t+\mathbf v\cdot\nabla_{\mathbf x})f=C^{(t)}[f]\) with
\(\mathbf v=\mathbf p/E\).
\end{remark}

\section{Boltzmann Equation and Collision Kernel}

The covariant Boltzmann equation is
\begin{equation}
  p^\mu\partial_\mu f_a(X,\mathbf p)=\mathcal C_a[f](X,\mathbf p).
  \label{eq:kinetic-covariant-boltzmann}
\end{equation}
In coordinate time this is
\[
  \left(\partial_t+\mathbf v_a\cdot\nabla_{\mathbf x}\right)f_a
  =
  C^{(t)}_a[f],
  \qquad
  \mathbf v_a=\nabla_{\mathbf p}E_a,
  \qquad
  C^{(t)}_a=\frac{\mathcal C_a}{E_a}.
\]

For \(a(p_1)b(p_2)\to c(p_3)d(p_4)\), define
\[
  \eta_a=
  \begin{cases}
    +1,& a\ \text{bosonic},\\
    -1,& a\ \text{fermionic}.
  \end{cases}
\]
The \(2\to2\) contribution to \(\mathcal C_a\) is
\begin{align}
  \mathcal C_a(p_1)
  &=
  \frac12
  \sum_{bcd}\int d\Pi_b(p_2)d\Pi_c(p_3)d\Pi_d(p_4)\,
  (2\pi)^D\delta^{(D)}(p_1+p_2-p_3-p_4)
  |\mathcal M_{ab\to cd}|^2
  \nonumber\\
  &\quad\times
  \Big[
  f_c(p_3)f_d(p_4)
  (1+\eta_a f_a(p_1))(1+\eta_b f_b(p_2))
  \nonumber\\
  &\qquad\qquad
  -
  f_a(p_1)f_b(p_2)
  (1+\eta_c f_c(p_3))(1+\eta_d f_d(p_4))
  \Big].
  \label{eq:kinetic-2to2-collision}
\end{align}
The factor \(1/2\) converts the coordinate-time normalization to the
covariant equation \eqref{eq:kinetic-covariant-boltzmann}; equivalently,
\(C^{(t)}_a\) carries the familiar prefactor \(1/(2E_1)\).  Symmetry factors
for identical particles are included in the definition of
\(|\mathcal M|^2\) and in the channel sum.

The matrix element in \eqref{eq:kinetic-2to2-collision} is the on-shell
scattering amplitude in the normalization of the scattering chapters, after
the regulator and medium corrections relevant at the kinetic scale have been
specified.  In gauge theories this last clause is essential because soft
exchange and collinear splitting can contribute at the same parametric order
as \(2\to2\) hard scattering.

\section{Detailed Balance}

The local equilibrium distribution is
\begin{equation}
  f_a^{(0)}(X,\mathbf p)
  =
  \frac{1}{\exp[\beta(X)(-u_\mu(X)p^\mu-\mu_A(X)q^A_a)]-\eta_a}.
  \label{eq:kinetic-local-equilibrium-distribution}
\end{equation}
At a fixed point \(X\), the identity
\begin{equation}
  1+\eta_a f_a^{(0)}
  =
  e^{\beta(-u_\mu p^\mu-\mu_Aq^A_a)}f_a^{(0)}
  \label{eq:kinetic-one-plus-eta-f}
\end{equation}
implies detailed balance.  Indeed, for an allowed reaction satisfying
\[
  p_1+p_2=p_3+p_4,
  \qquad
  q_a^A+q_b^A=q_c^A+q_d^A,
\]
the loss product
\[
  X_{12\to34}
  =
  f_a^{(0)}(p_1)f_b^{(0)}(p_2)
  (1+\eta_c f_c^{(0)}(p_3))(1+\eta_d f_d^{(0)}(p_4))
\]
equals the gain product
\[
  Y_{34\to12}
  =
  f_c^{(0)}(p_3)f_d^{(0)}(p_4)
  (1+\eta_a f_a^{(0)}(p_1))(1+\eta_b f_b^{(0)}(p_2)).
\]
This is an algebraic consequence of
\eqref{eq:kinetic-one-plus-eta-f}: the exponential multiplying
\(f_1f_2f_3f_4\) is the same on the two sides by energy-momentum and charge
conservation.  Therefore the collision term vanishes on global equilibrium,
and on local equilibrium it vanishes up to gradients of \(\beta u^\mu\) and
\(\beta\mu_A\).

\section{Kinetic Entropy and the H Theorem}

Define the entropy current
\begin{equation}
  S^\mu_{\rm kin}
  =
  -\sum_a\int 2d\Pi_a\,p^\mu
  \left[
    f_a\log f_a
    -\eta_a(1+\eta_af_a)\log(1+\eta_af_a)
  \right].
  \label{eq:kinetic-entropy-current}
\end{equation}
Using the Boltzmann equation,
\begin{equation}
  \partial_\mu S^\mu_{\rm kin}
  =
  -\sum_a\int 2d\Pi_a\,\mathcal C_a[f]\,
  \log\!\left(\frac{f_a}{1+\eta_af_a}\right).
  \label{eq:kinetic-entropy-divergence-before-sym}
\end{equation}
For the \(2\to2\) kernel, symmetrizing over the four external legs gives
\begin{align}
  \partial_\mu S^\mu_{\rm kin}
  &=
  \frac14
  \sum_{abcd}\int
  d\Pi_1d\Pi_2d\Pi_3d\Pi_4\,
  (2\pi)^D\delta^{(D)}(p_1+p_2-p_3-p_4)
  |\mathcal M_{ab\to cd}|^2
  \nonumber\\
  &\quad\times
  (X_{12\to34}-Y_{34\to12})
  \log\!\left(\frac{X_{12\to34}}{Y_{34\to12}}\right)\ge0.
  \label{eq:kinetic-h-theorem}
\end{align}
The inequality follows from
\[
  (x-y)\log(x/y)\ge0
  \qquad
  (x>0,\ y>0).
\]
Equality holds, within the set of reactions included in the kernel, exactly
when each allowed gain product equals the corresponding loss product.  When
the reactions connect the species space sufficiently, this condition forces
\(\log[f_a/(1+\eta_af_a)]\) to be a linear combination of the conserved
collision invariants \(1\), \(p^\mu\), and \(q_a^A\), giving
\eqref{eq:kinetic-local-equilibrium-distribution}.

\section{Hydrodynamic Moments}

The kinetic stress tensor and conserved currents are
\begin{align}
  T^{\mu\nu}(X)
  &=
  \sum_a\int 2d\Pi_a\,p^\mu p^\nu f_a(X,\mathbf p),
  \label{eq:kinetic-stress-tensor}\\
  J^\mu_A(X)
  &=
  \sum_a q_{A,a}\int 2d\Pi_a\,p^\mu f_a(X,\mathbf p).
  \label{eq:kinetic-current}
\end{align}
Multiplying \eqref{eq:kinetic-covariant-boltzmann} by \(2p^\nu\) and
integrating gives
\[
  \partial_\mu T^{\mu\nu}
  =
  \sum_a\int 2d\Pi_a\,p^\nu\mathcal C_a[f].
\]
The right side vanishes because each collision conserves total
four-momentum.  Similarly,
\[
  \partial_\mu J^\mu_A
  =
  \sum_a q_{A,a}\int 2d\Pi_a\,\mathcal C_a[f]=0
\]
when the collision channels conserve charge \(q_A\).  Thus the hydrodynamic
Ward identities are the collision-invariant moments of the kinetic equation.

At local equilibrium \eqref{eq:kinetic-local-equilibrium-distribution}, the
moments have the ideal-fluid form
\[
  T^{\mu\nu}_{(0)}=\varepsilon u^\mu u^\nu+p\Delta^{\mu\nu},
  \qquad
  J^\mu_{A(0)}=n_Au^\mu,
\]
with \(p\), \(\varepsilon\), and \(n_A\) obtained from the phase-space
integrals.  First-order transport arises from the first gradient correction
to \(f_a\).

\section{Linearized Collision Operator}

Write
\begin{equation}
  f_a=f_a^{(0)}+f_a^{(0)}(1+\eta_af_a^{(0)})\chi_a .
  \label{eq:kinetic-linearized-ansatz}
\end{equation}
To first order,
\[
  \mathcal C_a[f]=-\sum_b\mathcal L_{ab}\chi_b+O(\chi^2).
\]
The natural inner product on perturbations is
\begin{equation}
  \langle \phi,\psi\rangle
  =
  \sum_a\int 2d\Pi_a\,
  f_a^{(0)}(1+\eta_af_a^{(0)})\,\phi_a(p)\psi_a(p).
  \label{eq:kinetic-linearized-inner-product}
\end{equation}
For the \(2\to2\) collision kernel,
\begin{align}
  \langle \chi,\mathcal L\chi\rangle
  &=
  \frac14
  \sum_{abcd}\int
  d\Pi_1d\Pi_2d\Pi_3d\Pi_4\,
  (2\pi)^D\delta^{(D)}(p_1+p_2-p_3-p_4)
  |\mathcal M_{ab\to cd}|^2
  \nonumber\\
  &\quad\times
  f_1^{(0)}f_2^{(0)}
  (1+\eta_c f_3^{(0)})(1+\eta_d f_4^{(0)})
  \bigl(\chi_1+\chi_2-\chi_3-\chi_4\bigr)^2\ge0.
  \label{eq:kinetic-linearized-positive-form}
\end{align}
The null space consists of perturbations for which
\(\chi_1+\chi_2-\chi_3-\chi_4=0\) for every included reaction.  Under the
same connectivity assumption used in the entropy argument, this null space
is spanned by conserved collision invariants.  Transport coefficients are
quadratic forms involving the inverse of \(\mathcal L\) on the orthogonal
complement of this null space.

\section{Finite Collision Algebra Behind the Integral Formula}
\label{sec:finite-collision-algebra-behind-kinetic-integral}

The continuum formulas above are compact notations for a positive measure on
reaction kinematics.  The sign, null-space, and detailed-balance statements
are already present at the finite algebraic level, so we record that level
explicitly.  This also fixes the precise content of the phrase ``collision
invariant'' before functional analysis is added.

\begin{definition}[Finite reversible collision datum]
\label{def:finite-reversible-collision-datum}
A finite reversible collision datum consists of:
\begin{enumerate}
\item a finite set \(I\) of one-particle cells, each with a statistics sign
  \(\eta_i\in\{+1,-1\}\);
\item an open state domain
  \[
    \mathcal D
    =
    \{f\in\mathbb R^I:\ f_i>0,\ 1+\eta_i f_i>0\ \text{for all }i\};
  \]
\item a finite set \(\mathcal R\) of reversible reactions
  \(r=(i,j;k,l)\), interpreted as \(i+j\rightleftarrows k+l\), with weights
  \(w_r>0\);
\item additive labels \(P_i\) taking values in a real vector space
  \(\mathcal P\), such that
  \begin{equation}
    P_i+P_j=P_k+P_l
    \label{eq:finite-collision-additive-conservation}
  \end{equation}
  for every reaction \(r=(i,j;k,l)\).
\end{enumerate}
For \(f\in\mathcal D\), define
\begin{equation}
  L_r(f)=f_if_j(1+\eta_k f_k)(1+\eta_l f_l),
  \qquad
  G_r(f)=f_kf_l(1+\eta_i f_i)(1+\eta_j f_j),
  \label{eq:finite-collision-loss-gain-products}
\end{equation}
and the stoichiometric vector
\begin{equation}
  \nu_r=-e_i-e_j+e_k+e_l\in\mathbb R^I .
  \label{eq:finite-collision-stoichiometric-vector}
\end{equation}
The collision vector field is
\begin{equation}
  \dot f
  =
  \sum_{r\in\mathcal R}w_r\bigl(L_r(f)-G_r(f)\bigr)\nu_r .
  \label{eq:finite-collision-vector-field}
\end{equation}
\end{definition}

\begin{proposition}[Finite collision conservation and entropy production]
\label{prop:finite-collision-conservation-h-theorem}
For every finite reversible collision datum, the vector field
\eqref{eq:finite-collision-vector-field} preserves every additive conserved
quantity
\begin{equation}
  Q_A(f)=\sum_{i\in I}A(P_i)f_i,
  \qquad A\in\mathcal P^* .
  \label{eq:finite-collision-additive-charge}
\end{equation}
Moreover the entropy
\begin{equation}
  S(f)
  =
  -\sum_{i\in I}
  \left[
    f_i\log f_i
    -\eta_i(1+\eta_if_i)\log(1+\eta_if_i)
  \right]
  \label{eq:finite-collision-entropy}
\end{equation}
satisfies
\begin{equation}
  \dot S(f)
  =
  \sum_{r\in\mathcal R}
  w_r\bigl(L_r(f)-G_r(f)\bigr)
  \log\!\left(\frac{L_r(f)}{G_r(f)}\right)
  \ge0 .
  \label{eq:finite-collision-entropy-production}
\end{equation}
Equality holds if and only if \(L_r(f)=G_r(f)\) for every reaction with
\(w_r>0\).
\end{proposition}

\begin{proof}
The finite state space lets us work entirely inside the open domain
\(\mathcal D\), where every factor in \(L_r\) and \(G_r\) is positive.  Thus
the logarithms below are honest finite-dimensional functions, and the
boundary of \(\mathcal D\) is not used in the argument.

For the conserved quantity,
\[
  \frac{d}{dt}Q_A(f)
  =
  \sum_{r}w_r(L_r-G_r)
  \bigl[-A(P_i)-A(P_j)+A(P_k)+A(P_l)\bigr],
\]
and the bracket vanishes by
\eqref{eq:finite-collision-additive-conservation}.

For the entropy, differentiate the one-cell entropy.  The same formula covers
the classical, Bose, and Fermi signs because
\[
  \frac{\dd}{\dd f}
  \left[
    -f\log f+\eta(1+\eta f)\log(1+\eta f)
  \right]
  =
  -\log\!\left(\frac f{1+\eta f}\right),
\]
with the \(\eta=0\) case understood by taking the limit.  Hence
\begin{equation}
  \frac{\partial S}{\partial f_i}
  =
  -\log\!\left(\frac{f_i}{1+\eta_if_i}\right).
  \label{eq:finite-collision-entropy-derivative}
\end{equation}
For \(r=(i,j;k,l)\),
\[
  \nu_r\cdot\nabla S
  =
  \log\!\left(
  \frac{f_if_j(1+\eta_kf_k)(1+\eta_lf_l)}
       {f_kf_l(1+\eta_if_i)(1+\eta_jf_j)}
  \right)
  =
  \log\!\left(\frac{L_r}{G_r}\right).
  \]
Substitution into \(\dot S=\nabla S\cdot\dot f\) gives
\eqref{eq:finite-collision-entropy-production}.  The sign is fixed reaction
by reaction: for \(x,y>0\),
\[
  (x-y)\log(x/y)\ge0,\qquad x>0,\quad y>0 .
  \]
Indeed \(x>y\) implies \(\log x-\log y>0\), while \(x<y\) makes both factors
negative.  Equality in the sum holds exactly when every positive-weight
reaction has \(L_r=G_r\), because all summands are nonnegative.
\end{proof}

\begin{definition}[Finite local equilibrium and collision invariants]
\label{def:finite-local-equilibrium-collision-invariants}
A state \(f^{(0)}\in\mathcal D\) is a finite equilibrium for the datum when
\begin{equation}
  \frac{f_i^{(0)}}{1+\eta_i f_i^{(0)}}
  =
  \exp[-\alpha-A(P_i)]
  \label{eq:finite-collision-equilibrium-form}
\end{equation}
for some \(\alpha\in\mathbb R\) and \(A\in\mathcal P^*\).  A function
\(\chi:I\to\mathbb R\) is a collision invariant when
\begin{equation}
  \chi_i+\chi_j-\chi_k-\chi_l=0
  \label{eq:finite-collision-invariant}
\end{equation}
for every reaction \(i+j\rightleftarrows k+l\).
\end{definition}

\paragraph{Finite linearized collision operator.}
Let \(f^{(0)}\) be a finite equilibrium.  Write
\begin{equation}
  f_i=f_i^{(0)}+f_i^{(0)}(1+\eta_i f_i^{(0)})\chi_i .
  \label{eq:finite-collision-linearization}
\end{equation}
To first order in \(\chi\),
\begin{equation}
  L_r(f)-G_r(f)
  =
  M_r
  \bigl(\chi_i+\chi_j-\chi_k-\chi_l\bigr)+O(\chi^2),
  \label{eq:finite-collision-linearized-rate}
\end{equation}
where
\begin{equation}
  M_r=L_r(f^{(0)})=G_r(f^{(0)})>0 .
  \label{eq:finite-collision-equilibrium-weight}
\end{equation}
Consequently the linearized entropy production is
\begin{equation}
  \dot S^{(2)}
  =
  \sum_{r\in\mathcal R}
  w_rM_r
  \bigl(\chi_i+\chi_j-\chi_k-\chi_l\bigr)^2\ge0.
  \label{eq:finite-collision-linearized-entropy-production}
\end{equation}
Its null space is precisely the space of collision invariants.  If the
reaction graph is connected enough that every collision invariant is of the
form \(\chi_i=\alpha+A(P_i)\), then the only zero modes are the conserved
hydrodynamic variables.

The linearization is finite-dimensional algebra.  At equilibrium, detailed
balance follows from
\eqref{eq:finite-collision-equilibrium-form} and additive conservation:
\[
  \frac{L_r(f^{(0)})}{G_r(f^{(0)})}
  =
  \exp[-A(P_i)-A(P_j)+A(P_k)+A(P_l)]
  =
  1 .
  \]
For a variation \(\delta f_i=f_i^{(0)}(1+\eta_i f_i^{(0)})\chi_i\),
\[
  \delta\log\!\left(\frac{f_i}{1+\eta_i f_i}\right)=\chi_i .
  \]
Therefore
\[
  \delta\log(L_r/G_r)
  =
  \chi_i+\chi_j-\chi_k-\chi_l .
  \]
Since \(L_r=G_r=M_r\) at equilibrium, this is equivalent to
\eqref{eq:finite-collision-linearized-rate}: if \(R_r=L_r/G_r\), then
\[
  \delta(L_r-G_r)
  =
  M_r\,\delta\log R_r .
\]
The second variation of the entropy-production summand is obtained from
\[
  (L_r-G_r)\log(L_r/G_r)
  =
  M_r(\delta\log R_r)^2+O(\chi^3),
\]
because both \(L_r-G_r\) and \(\log(L_r/G_r)\) vanish to first order at
equilibrium with the same linear coefficient \(M_r^{-1}\delta(L_r-G_r)\).
Summing the reactions gives
\eqref{eq:finite-collision-linearized-entropy-production}.  Its null space
is the intersection of the kernels of the linear forms
\(\chi_i+\chi_j-\chi_k-\chi_l\), which is precisely the space of collision
invariants by Definition~\ref{def:finite-local-equilibrium-collision-invariants}.
Under the stated connectivity hypothesis this space is the span of the
constant invariant and the additive conserved quantities \(A(P_i)\).

The continuum \(2\to2\) formulas are obtained from
Definition~\ref{def:finite-reversible-collision-datum} by replacing the
finite reaction sum by the invariant positive measure
\[
  d\Pi_1d\Pi_2d\Pi_3d\Pi_4\,
  (2\pi)^D\delta^{(D)}(p_1+p_2-p_3-p_4)
  |\mathcal M_{ab\to cd}|^2 .
\]
The finite theorem is not a derivation of this measure from QFT; it is the
exact algebraic part of the kinetic construction once the collision measure
has been derived or declared.

\section{Worked Example: Relaxation-Time Shear Viscosity}

\begin{controlledapproximation}[Constant relaxation time]
\label{ca:kinetic-constant-relaxation-time}
Consider a single massless classical species in \(d=3\), with
\(f^{(0)}=e^{-p/T}\), no conserved charge, and a collision operator replaced
on the shear channel by
\[
  C^{(t)}_{\rm RTA}[\delta f]=-\frac{\delta f}{\tau_R},
\]
where \(\tau_R\) is a constant.  This replacement preserves the shear
tensor structure and gives a closed worked example.  It is a model for the
positive inverse collision operator on the shear subspace, not a derivation
of a microscopic collision kernel.
\end{controlledapproximation}

Let
\[
  \delta f
  =
  -\tau_R
  f^{(0)}
  \frac{p^ip^j}{2TE}
  \sigma_{ij},
  \qquad
  \sigma_{ij}
  =
  \partial_i v_j+\partial_j v_i
  -\frac23\delta_{ij}\partial_kv_k .
\]
Substitution in the kinetic stress tensor gives
\[
  \delta T^{ij}
  =
  -\eta\,\sigma^{ij},
\]
where angular averaging in three dimensions uses
\[
  \int d\Omega\,\hat p_i\hat p_j\hat p_k\hat p_l
  =
  \frac{4\pi}{15}
  (\delta_{ij}\delta_{kl}+\delta_{ik}\delta_{jl}
  +\delta_{il}\delta_{jk}).
\]
Since \(\sigma^i{}_i=0\), the coefficient is
\begin{equation}
  \eta_{\rm RTA}
  =
  \frac{\tau_R}{15T}
  \int\frac{d^3\mathbf p}{(2\pi)^3}
  p^2e^{-p/T}
  =
  \frac{4}{5}p_{\rm therm}\tau_R,
  \label{eq:kinetic-rta-shear-viscosity}
\end{equation}
with
\[
  p_{\rm therm}
  =
  \frac13\int\frac{d^3\mathbf p}{(2\pi)^3}p\,e^{-p/T}
  =
  \frac{T^4}{\pi^2}.
\]
This example displays the general structure of kinetic transport: a source
term from gradients, inversion of the collision operator on a nonconserved
subspace, and a positive quadratic form for the coefficient.

\section{Gauge Theories and Soft Scales}

In weakly coupled gauge theories, hard particles with momentum of order \(T\)
interact through soft gauge exchange of order \(gT\).  Electric fields are
screened by the Debye scale of Chapter~\ref{ch:thermal-gauge-theory-screening};
magnetic fields at scale \(g^2T\) remain nonperturbative in the static
sector.  Kinetic theory for gauge plasmas therefore contains several
coupled effective descriptions:
\[
  \text{hard quasiparticles}
  \quad\leftrightarrow\quad
  \text{soft collective gauge fields}
  \quad\leftrightarrow\quad
  \text{ultrasoft hydrodynamic modes}.
\]
Small-angle scattering, Landau damping, and collinear splitting processes
must be matched in the same source and regulator convention before a
transport coefficient is a well-defined weak-coupling prediction.

\section{Controlled-Limit Ledger}

A kinetic derivation from QFT requires:
\[
\begin{array}{ll}
\text{spectral input} &
  \text{narrow quasiparticle peaks with a controlled off-shell remainder},\\
\text{scale separation} &
  \ell_{\rm micro}\ll \ell_{\rm variation}
  \ \text{and}\ \Gamma\ell_{\rm variation}\ll1,\\
\text{collision input} &
  \text{on-shell amplitudes with regulator and medium corrections},\\
\text{conservation laws} &
  \text{collision invariants matching QFT Ward identities},\\
\text{error estimate} &
  \text{bounds on neglected gradients, memory terms, and off-shell terms}.
\end{array}
\]

\begin{openproblem}[QFT derivation of kinetic theory with errors]
\label{op:qft-kinetic-theory-error-derivation}
Derive relativistic kinetic theory from Schwinger--Keldysh QFT with explicit
error bounds, including quasiparticle projection, collision kernels,
screening, collinear processes in gauge theory, and the passage from kinetic
theory to hydrodynamic transport.
\end{openproblem}
