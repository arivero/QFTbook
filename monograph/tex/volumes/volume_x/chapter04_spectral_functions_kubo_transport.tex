\chapter{Spectral Functions, Kubo Formulae, and Transport}
\label{ch:thermal-spectral-kubo-transport}

\ConventionsRealTime

Transport coefficients are low-frequency limits of real-time response
functions.  The definitions use a thermal state, a conserved current or
stress tensor, and a source coupled to the corresponding operator.  The
derivation is linear response from the Schwinger--Keldysh generating
functional of Chapter~\ref{ch:thermal-schwinger-keldysh}.  This chapter
keeps separate three pieces that are often conflated: the commutator
spectral density, possible local contact terms in the source response, and
the order of limits needed to define a transport coefficient.

\section{Retarded Functions and Spectral Densities}

Let \(\rho_\beta\) be a KMS state at inverse temperature \(\beta\).  For
bosonic Hermitian operators \(A(t,\mathbf x)\) and \(B(t,\mathbf x)\), define
the retarded function
\begin{equation}
  G^R_{AB}(t,\mathbf x)
  =
  -\ii\,\Theta(t)\,
  \langle [A(t,\mathbf x),B(0,\mathbf 0)]\rangle_\beta
  \label{eq:kubo-retarded-definition}
\end{equation}
and the commutator spectral density
\begin{equation}
  \rho_{AB}(t,\mathbf x)
  =
  \langle [A(t,\mathbf x),B(0,\mathbf 0)]\rangle_\beta .
  \label{eq:kubo-commutator-spectral-density}
\end{equation}
Fourier transforms are defined distributionally by
\begin{equation}
  G^R_{AB}(\omega,\mathbf k)
  =
  \int dt\,d^{D-1}x\,
  e^{\ii\omega t-\ii\mathbf k\cdot\mathbf x}
  G^R_{AB}(t,\mathbf x).
  \label{eq:kubo-fourier-convention}
\end{equation}
With this convention, a retarded distribution that obeys the required
tempered-growth hypothesis is the boundary value of a function holomorphic
for \(\operatorname{Im}\omega>0\).

\begin{proposition}[Finite-volume Lehmann representation]
\label{prop:kubo-finite-volume-lehmann}
Work in a finite spatial volume.  Let \(H|n\rangle=E_n|n\rangle\), let
\(Z=\sum_n e^{-\beta E_n}<\infty\), and let \(A_{\mathbf k}\) obey
\(A_{\mathbf k}^{\dagger}=A_{-\mathbf k}\).  Then
\begin{equation}
  G^>_{AA}(\omega,\mathbf k)
  =
  \frac{2\pi}{Z}\sum_{m,n}
  e^{-\beta E_m}
  |\langle m|A_{\mathbf k}|n\rangle|^2
  \delta(\omega-E_n+E_m),
  \label{eq:kubo-greater-lehmann}
\end{equation}
and
\begin{equation}
  \rho_{AA}(\omega,\mathbf k)
  =
  \frac{2\pi}{Z}\sum_{m,n}
  e^{-\beta E_m}
  \left(1-e^{-\beta\omega}\right)
  |\langle m|A_{\mathbf k}|n\rangle|^2
  \delta(\omega-E_n+E_m).
  \label{eq:kubo-spectral-lehmann}
\end{equation}
Consequently \(\omega\,\rho_{AA}(\omega,\mathbf k)\ge0\) as a
distributional positivity statement.
\end{proposition}

\begin{proof}
Insert a complete set of energy eigenstates into
\[
  G^>_{AA}(t,\mathbf k)
  =
  Z^{-1}\operatorname{Tr}
  \left(e^{-\beta H}A_{\mathbf k}(t)A_{-\mathbf k}(0)\right).
\]
Since
\[
  A_{\mathbf k}(t)
  =
  e^{\ii Ht}A_{\mathbf k}e^{-\ii Ht},
\]
the matrix element carries the phase
\[
  e^{\ii(E_m-E_n)t}
  \langle m|A_{\mathbf k}|n\rangle
  \langle n|A_{-\mathbf k}|m\rangle .
\]
The adjoint relation makes the product the squared modulus displayed in
\eqref{eq:kubo-greater-lehmann}, and the Fourier integral gives the delta
function \(\delta(\omega-E_n+E_m)\).

The lesser function has the same delta support with the Boltzmann weight
\(e^{-\beta E_n}\) after relabelling \(m\leftrightarrow n\).  On the support
of the delta function,
\[
  E_n-E_m=\omega,
  \qquad
  e^{-\beta E_n}=e^{-\beta\omega}e^{-\beta E_m}.
\]
Subtracting \(G^<\) from \(G^>\) gives
\eqref{eq:kubo-spectral-lehmann}.  For \(\omega>0\) every coefficient in
\eqref{eq:kubo-spectral-lehmann} is nonnegative; for \(\omega<0\) it is
nonpositive.  This is exactly \(\omega\rho_{AA}\ge0\).
\end{proof}

\begin{proposition}[Retarded boundary value and sign convention]
\label{prop:kubo-retarded-spectral-sign}
Assume the finite-volume spectral measure above has a thermodynamic limit for
which the displayed integrals remain meaningful as tempered boundary values.
Then
\begin{equation}
  G^R_{AA}(z,\mathbf k)
  =
  \int_{\mathbb R}\frac{d\omega'}{2\pi}\,
  \frac{\rho_{AA}(\omega',\mathbf k)}{z-\omega'},
  \qquad
  \operatorname{Im}z>0,
  \label{eq:kubo-retarded-spectral-representation}
\end{equation}
up to real polynomial contact terms, and therefore
\begin{equation}
  \rho_{AA}(\omega,\mathbf k)
  =
  -2\,\operatorname{Im}G^R_{AA}(\omega+\ii0,\mathbf k)
  \label{eq:kubo-rho-minus-two-im-gr}
\end{equation}
for the nonlocal part of the response.
\end{proposition}

\begin{proof}
Write the inverse Fourier representation of \(\rho\) in
\eqref{eq:kubo-retarded-definition}.  For \(\operatorname{Im}z>0\),
\[
  -\ii\int_0^\infty dt\,e^{\ii zt}e^{-\ii\omega't}
  =
  \frac{1}{z-\omega'} .
\]
This gives \eqref{eq:kubo-retarded-spectral-representation}.  Taking
\(z=\omega+\ii0\) and using
\[
  \frac{1}{\omega+\ii0-\omega'}
  =
  \operatorname{PV}\frac{1}{\omega-\omega'}
  -
  \ii\pi\delta(\omega-\omega')
\]
gives \eqref{eq:kubo-rho-minus-two-im-gr}.  Local source counterterms add
real polynomials in \(\omega\) and \(\mathbf k\) to the real-time kernel in
the absence of dissipation; they are not part of the commutator spectral
measure.
\end{proof}

\section{KMS and Fluctuation--Dissipation}

Define the Wightman functions
\[
  G^>_{AB}(t)=\langle A(t)B(0)\rangle_\beta,\qquad
  G^<_{AB}(t)=\langle B(0)A(t)\rangle_\beta .
\]
The KMS condition gives
\[
  G^>_{AB}(t-\ii\beta)=G^<_{AB}(t)
\]
for operators in the analytic domain of the thermal state.  In frequency
space this implies detailed balance.  For \(A=B^\dagger\), the finite-volume
formula above gives
\begin{equation}
  G^<_{AA}(\omega,\mathbf k)
  =
  e^{-\beta\omega}G^>_{AA}(\omega,\mathbf k),
  \qquad
  \rho_{AA}=G^>_{AA}-G^<_{AA}.
  \label{eq:kubo-detailed-balance}
\end{equation}
Therefore the symmetrized correlator
\[
  G^{\rm sym}_{AA}(\omega,\mathbf k)
  =
  \frac12\bigl(G^>_{AA}(\omega,\mathbf k)
  +G^<_{AA}(\omega,\mathbf k)\bigr)
\]
satisfies
\begin{equation}
  G^{\rm sym}_{AA}(\omega,\mathbf k)
  =
  \frac12\coth\!\left(\frac{\beta\omega}{2}\right)
  \rho_{AA}(\omega,\mathbf k).
  \label{eq:kubo-fluctuation-dissipation}
\end{equation}
The apparent singularity at \(\omega=0\) is interpreted distributionally:
the product is meaningful when the small-\(\omega\) behavior of
\(\rho_{AA}\) is specified.  In a diffusive or viscous channel,
\(\rho_{AA}(\omega)=O(\omega)\), so the symmetrized noise has a finite
zero-frequency limit.  If a ballistic channel produces a Drude delta
function, this limit is a different observable and the dc transport
coefficient is infinite rather than a finite number.

\section{Linear Response from the Closed Time Path}

Perturb the Hamiltonian by a source \(h_B(t,\mathbf x)\) coupled to \(B\):
\begin{equation}
  H(t)=H_0-\int d^{D-1}x\,h_B(t,\mathbf x)B(t,\mathbf x).
  \label{eq:kubo-source-hamiltonian}
\end{equation}
To first order in \(h_B\), the interaction-picture evolution operator is
\[
  U_h(t,-\infty)
  =
  1+\ii\int_{-\infty}^{t}dt'\int d^{D-1}x'\,
  h_B(t',\mathbf x')B(t',\mathbf x')+O(h_B^2).
\]
Consequently
\[
  U_h(t,-\infty)^{-1}A(t,\mathbf x)U_h(t,-\infty)
  =
  A(t,\mathbf x)
  +
  \ii\int_{-\infty}^{t}dt'\,d^{D-1}x'\,
  h_B(t',\mathbf x')
  [A(t,\mathbf x),B(t',\mathbf x')]
  +O(h_B^2).
\]
Taking the thermal expectation value gives
\begin{equation}
  \delta\langle A(t,\mathbf x)\rangle
  =
  -\int dt' d^{D-1}x'\,
  G^R_{AB}(t-t',\mathbf x-\mathbf x')h_B(t',\mathbf x').
  \label{eq:kubo-linear-response}
\end{equation}
Equivalently, the source-response kernel for the convention
\(H=H_0-h_BB\) is
\[
  K^R_{AB}
  =
  \ii\Theta(t)\langle[A(t),B(0)]\rangle_\beta
  =
  -G^R_{AB}.
\]
This is the branch-difference identity obtained from the
Schwinger--Keldysh functional in
Chapter~\ref{ch:thermal-schwinger-keldysh}: differentiating with respect to
the \(a\)-type source and then setting \(J_a=0\) leaves the causal
source-response commutator.  Transport formulae below are stated in terms of
the commutator spectral density \(\rho_{AB}\), which is independent of this
source-sign convention.

\section{Contact Terms and Response Kernels}

The commutator retarded function is not always the full response kernel to a
background source.  If the operator itself depends on the source, or if the
regulated generating functional contains local source counterterms, then
\[
  \delta\langle A\rangle
  =
  (K^R_{AB}+C_{AB})\,h_B
\]
in Fourier space, where \(C_{AB}(\omega,\mathbf k)\) is local in spacetime
and therefore polynomial, or a distributional limit of polynomials, in
\(\omega,\mathbf k\).  For gauge and metric sources these terms are not
optional: they are fixed by Ward identities, equilibrium thermodynamics, and
the chosen operator scheme.

The dissipative transport coefficient is extracted from the spectral slope
of the nonlocal commutator part.  Real local contact terms can change static
susceptibilities and Euclidean zero-mode correlators, but they do not change
the positive spectral measure \(\rho_{AA}\) at nonzero frequency.  This is
why Kubo formulae below are stated using \(\rho\), with a separate warning
about Drude weights and conserved-density projections.

\section{Transport Limits and Zero-Frequency Singular Terms}

Transport is a statement about an infinite system.  In finite spatial volume
the spectrum of the Hamiltonian is discrete, so real-time correlators are
almost periodic sums of phases rather than decaying functions.  The
thermodynamic limit is therefore part of the definition of a finite
dissipative coefficient.

\begin{definition}[Transport limit datum]
\label{def:kubo-transport-limit-datum}
A transport limit datum for a pair of local operators \(A,B\) consists of
\[
  (\Sigma_L,\rho_{\beta,L},A_L,B_L,\mathcal R,\mathcal O),
\]
where \(\Sigma_L\) is a sequence of spatial regions or lattices of volume
\(V_L\), \(\rho_{\beta,L}\) is a KMS or finite Gibbs state on the regulated
system, \(A_L,B_L\) are regulated extensive or density operators with their
normalization declared, \(\mathcal R\) is the prescription for subtracting
local contact terms and conserved-density projections, and \(\mathcal O\) is
the order of limits.  For ordinary dc transport the order is
\[
  L\to\infty,\qquad \mathbf k\to\mathbf0\ \text{in the channel specified by
  symmetry},\qquad \omega\downarrow0 .
  \label{eq:kubo-transport-order-of-limits}
\]
Changing \(\mathcal O\) can change the answer; for example static
susceptibilities and dc conductivities use different zero-frequency limits.
\end{definition}

The most elementary obstruction to a finite dc coefficient is an exactly
conserved operator that has nonzero overlap with the current.  Let \(I_i\) be
the total current in direction \(i\) in finite volume \(V\), and let
\(\bar I_i=I_i-\langle I_i\rangle_\beta\).  Suppose
\(\{Q_a\}_{a=1}^N\) are Hermitian conserved operators with zero thermal
expectation and positive covariance matrix
\[
  C_{ab}
  =
  \frac1{2V}\langle Q_aQ_b+Q_bQ_a\rangle_\beta .
  \label{eq:kubo-conserved-covariance}
\]
Define the current--charge overlap vector
\[
  v_{ia}
  =
  \frac1{2V}\langle \bar I_iQ_a+Q_a\bar I_i\rangle_\beta .
  \label{eq:kubo-current-charge-overlap}
\]
The projection of \(\bar I_i\) onto the span of conserved charges is
\[
  P\bar I_i
  =
  Q_a(C^{-1})^{ab}v_{ib}.
  \label{eq:kubo-mazur-projection}
\]

\begin{proposition}[Finite-volume Mazur projection]
\label{prop:kubo-mazur-projection}
In the finite-dimensional Gibbs setting above, for every real vector \(u_i\),
the infinite-time averaged connected symmetrized current autocorrelation
satisfies
\[
  \lim_{T\to\infty}\frac1T\int_0^T dt\,
  \frac1{2V}
  \left\langle
    \{u_i\bar I_i(t),u_j\bar I_j(0)\}
  \right\rangle_\beta
  \geq
  u_i v_{ia}(C^{-1})^{ab}v_{jb}u_j .
  \label{eq:kubo-mazur-bound}
\]
If the right hand side has a nonzero thermodynamic limit, the conductivity
has a zero-frequency singular contribution and the ordinary dc limit is not
a finite dissipative number.
\end{proposition}

\begin{proof}
Use the real Hilbert-space inner product on Hermitian operators
\[
  (X,Y)_{\beta,{\rm sym}}
  =
  \frac1{2V}\langle XY+YX\rangle_\beta .
\]
Time evolution is orthogonal for this inner product because \(\rho_\beta\)
commutes with \(H\).  For \(X=u_i\bar I_i\), let
\[
  X_\parallel
  =
  Q_a(C^{-1})^{ab}(Q_b,X)_{\beta,{\rm sym}} .
\]
Then \(X_\parallel\) is conserved and is the orthogonal projection of \(X\)
onto the conserved subspace spanned by the \(Q_a\).  The time average of the
symmetrized correlation is the squared norm of the component of \(X\) in the
zero-frequency subspace of the Liouville operator \(L=[H,\cdot]\).  This
zero-frequency subspace contains \(X_\parallel\), hence the time average is
at least
\[
  (X_\parallel,X_\parallel)_{\beta,{\rm sym}}
  =
  u_i v_{ia}(C^{-1})^{ab}v_{jb}u_j .
\]
If this quantity survives per unit volume as \(V\to\infty\), the symmetrized
current spectrum contains a term
\[
  \frac{2\pi}{\beta}D_{ij}\delta(\omega),
  \qquad
  D_{ij}=\beta\lim_{t\to\infty}\lim_{V\to\infty}
  \frac1{2V}\langle\{\bar I_i(t),\bar I_j(0)\}\rangle_\beta ,
  \label{eq:kubo-drude-weight-definition}
\]
in the convention of \eqref{eq:kubo-fluctuation-dissipation}.  Equivalently
\(\operatorname{Re}\sigma_{ij}\) contains
\(\pi D_{ij}\delta(\omega)\).  This is a singular transport sector, not the
regular slope appearing in the finite dc Kubo formula.
\end{proof}

Figure~\ref{fig:kubo-order-of-limits} records the operational order in which
the singular and regular parts of the low-frequency response are separated.
The dc Kubo limit is not a finite-volume operation: one first takes the
thermodynamic limit, separates possible \(\delta(\omega)\) terms produced by
conserved projections, and only then extracts the regular low-frequency
slope.

\section{Conductivity}

For a conserved current \(J^\mu\) coupled to a background gauge field
\(A_\mu^{\rm ext}\), the electric field in the Fourier convention
\eqref{eq:kubo-fourier-convention} is
\[
  E_i(\omega,\mathbf k)=\ii\omega A_i^{\rm ext}(\omega,\mathbf k)
  -\ii k_i A_0^{\rm ext}(\omega,\mathbf k).
\]
At zero spatial momentum and in a gauge with \(A_0^{\rm ext}=0\), define the
full transverse response kernel \(K_{ij}\) by the conventional
background-gauge-field sign
\[
  \langle J_i(\omega)\rangle
  =
  -K_{ij}(\omega,\mathbf0)A_j^{\rm ext}(\omega)
  =
  \sigma_{ij}(\omega)E_j(\omega).
\]
Thus
\[
  \sigma_{ij}(\omega)
  =
  -\frac{1}{\ii\omega}K_{ij}(\omega,\mathbf0),
  \qquad
  K_{ij}=G^R_{J_iJ_j}+C_{ij}.
\]
For \(\omega\ne0\), the dissipative real part is determined by the spectral
density:
\begin{equation}
  \operatorname{Re}\sigma_{ij}(\omega)
  =
  \frac{1}{2\omega}\,
  \rho_{J_iJ_j}(\omega,\mathbf0)
  \quad
  \text{in a transverse channel after contact terms are separated.}
  \label{eq:kubo-conductivity-spectral-slope}
\end{equation}
The dc conductivity, when finite, is therefore
\[
  \sigma_{ij}^{\rm dc}
  =
  \lim_{\omega\downarrow0}
  \frac{1}{2\omega}\rho_{J_iJ_j}(\omega,\mathbf0),
\]
with the thermodynamic limit taken before the \(\omega\to0\) limit.  If
momentum conservation or another exactly conserved quantity overlaps with
the current, a \(\delta(\omega)\) Drude contribution is present and the dc
conductivity is infinite unless the conserved contribution is relaxed or
projected out.

\section{Shear and Bulk Viscosity}

Couple the theory to a spatial metric perturbation
\[
  g_{ij}(t,\mathbf x)=\delta_{ij}+h_{ij}(t,\mathbf x).
\]
The stress tensor is defined by the first variation of the regulated
generating functional with respect to the metric source.  In an isotropic
thermal state, take a shear perturbation \(h_{xy}(t)\) at zero spatial
momentum.  The dissipative part of the stress response is characterized by
the shear viscosity \(\eta\).  With the retarded sign convention
\eqref{eq:kubo-retarded-definition} and the spectral convention
\eqref{eq:kubo-rho-minus-two-im-gr},
\begin{equation}
  \eta
  =
  \lim_{\omega\downarrow0}
  \frac{1}{2\omega}\,
  \rho_{T_{xy}T_{xy}}(\omega,\mathbf0)
  =
  -\lim_{\omega\downarrow0}
  \frac{1}{\omega}\,
  \operatorname{Im}G^R_{T_{xy}T_{xy}}(\omega,\mathbf0).
  \label{eq:kubo-shear-viscosity-sign-correct}
\end{equation}
The equality between the two expressions is exactly
\eqref{eq:kubo-rho-minus-two-im-gr}.  Positivity of
\(\omega\rho_{T_{xy}T_{xy}}\) gives \(\eta\ge0\) when the limit exists and
no singular zero-frequency contribution is present.

The bulk channel requires a projection.  For a neutral relativistic fluid in
\(d=D-1\) spatial dimensions, define
\[
  \mathcal P
  =
  \frac{1}{d}T^i_{\ i}-c_s^2 T^{00},
  \qquad
  c_s^2=\left(\frac{\partial p}{\partial\varepsilon}\right)_{\rm eq}.
\]
The subtraction removes the overlap of the pressure fluctuation with the
conserved energy density.  In charged fluids one must project out all
conserved densities using the susceptibility matrix.  After this projection,
the bulk viscosity is
\[
  \zeta
  =
  \lim_{\omega\downarrow0}
  \frac{1}{2\omega}\rho_{\mathcal P\mathcal P}(\omega,\mathbf0),
\]
again with the thermodynamic and zero-momentum limits taken before the
zero-frequency limit.

\section{Dispersion Relations and Sum-Rule Boundaries}

Retardedness gives analyticity in the upper half \(\omega\)-plane.  If the
response kernel is sufficiently bounded after subtracting its local
polynomial terms, then Cauchy's theorem gives a subtracted dispersion
relation.  A typical once-subtracted form is
\begin{equation}
  G^R(\omega)-G^R(0)
  =
  \omega
  \int_{\mathbb R}\frac{d\omega'}{2\pi}\,
  \frac{\rho(\omega')}
       {\omega'(\omega-\omega'+\ii0)} .
  \label{eq:kubo-once-subtracted-dispersion}
\end{equation}
This formula is not a definition.  It follows only after the large semicircle
is controlled and after all contact terms have been declared.  Sum rules for
conductivity or viscosity compare the small-\(\omega\) transport slope with
Euclidean or equal-time data only under these additional analytic and
ultraviolet assumptions.

\section{Hydrodynamic Pole Boundary}

Kubo formulae define transport coefficients when the indicated limits exist
and commute with the thermodynamic and zero-momentum limits in the stated
order.  Hydrodynamics begins after these coefficients are inserted into the
long-wavelength Ward identities for conserved densities.  The microscopic
Kubo formula by itself does not prove local equilibration, the derivative
expansion, or the dominance of hydrodynamic poles; those are additional
dynamical statements developed in the following chapters.

\begin{openproblem}[Rigorous Kubo-to-hydrodynamics bridge]
\label{op:kubo-hydrodynamics-bridge}
Give sufficient QFT hypotheses under which retarded stress-tensor and current
correlators have the analyticity, spectral positivity, clustering, and
low-frequency limits required by the Kubo formulae, and prove that the same
coefficients control the long-time hydrodynamic effective theory.
\end{openproblem}
