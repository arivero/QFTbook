\chapter{Chern--Simons Theory}
\label{ch:tqft-chern-simons-theory}

\ChapterConventions{oriented smooth three-manifold \(M\)}
{no metric is part of the classical action}
{\(1\), unless explicitly displayed}
{a vacuum vector appears only after choosing a Hamiltonian splitting or a
boundary state space}
{gauge transformations act on connections by pullback and conjugation}

Let \(G\) be a compact Lie group with Lie algebra \(\mathfrak g\).  Fix an
invariant symmetric bilinear form \(\operatorname{tr}\) on \(\mathfrak g\).
For a connection \(A\) on a principal \(G\)-bundle over an oriented
three-manifold \(M\), the Chern--Simons functional in a local trivialization is
\[
  S_{\rm CS}(A)
  =
  \frac{k}{4\pi}\int_M
  \operatorname{tr}\!\left(A\wedge dA+\frac23 A\wedge A\wedge A\right).
\]
The level \(k\) is not a free real parameter once large gauge transformations
are included.

\section{Variation and Classical Equations}

Let \(F_A=dA+A\wedge A\).  For a compactly supported variation \(\delta A\)
on a closed manifold,
\[
  \delta S_{\rm CS}
  =
  \frac{k}{2\pi}\int_M \operatorname{tr}(\delta A\wedge F_A).
\]
Thus the Euler--Lagrange equation is
\[
  F_A=0 .
\]
On a manifold with boundary, the integration by parts also gives
\[
  \frac{k}{4\pi}\int_{\partial M}\operatorname{tr}(A\wedge\delta A),
\]
so a boundary condition, boundary action, or boundary phase space must be part
of the definition.

\section{Level Quantization}

For simply connected compact simple \(G\), choose the bilinear form so that
\[
  \frac{1}{24\pi^2}\int_{S^3}
  \operatorname{tr}\bigl(g^{-1}dg\bigr)^3\in\mathbb Z
\]
for maps \(g:S^3\to G\) representing the generator of \(\pi_3(G)\).  Under a
gauge transformation \(g:M\to G\), use the right action convention
\[
  A^g=gAg^{-1}-(dg)g^{-1}.
\]
Put \(\theta_g=dg\,g^{-1}\).  The right Maurer--Cartan equation is
\[
  d\theta_g=\theta_g\wedge\theta_g .
\]

\paragraph{Chern--Simons transgression under a finite gauge transformation.}
With
\[
  \operatorname{cs}(A)
  =
  \operatorname{tr}\!\left(A\wedge dA+\frac23A\wedge A\wedge A\right),
\]
one has the identity of three-forms
\begin{equation}
  \operatorname{cs}(A^g)-\operatorname{cs}(A)
  =
  d\,\operatorname{tr}(\theta_g\wedge A)
  +\frac13\operatorname{tr}(\theta_g\wedge\theta_g\wedge\theta_g).
  \label{eq:cs-finite-gauge-transgression}
\end{equation}

Write \(B=gAg^{-1}\).  Then \(A^g=B-\theta_g\),
\[
  dB=g(dA)g^{-1}+\theta_g\wedge B-B\wedge\theta_g,
  \qquad
  d\theta_g=\theta_g\wedge\theta_g .
\]
Substitute these two identities into
\(\operatorname{tr}((B-\theta_g)\wedge d(B-\theta_g)
+\frac23(B-\theta_g)^3)\).  The terms containing only \(B\) give
\(\operatorname{cs}(A)\) by invariance of \(\operatorname{tr}\).  The terms
with two \(B\)'s and one \(\theta_g\) cancel after cyclic permutation inside
the invariant trace.  Conjugating the remaining one-\(\theta_g\) mixed terms
back to the original \(A\)-frame by invariance of \(\operatorname{tr}\), they
combine as
\[
  \operatorname{tr}(d\theta_g\wedge A)
  -\operatorname{tr}(\theta_g\wedge dA)
  =
  d\,\operatorname{tr}(\theta_g\wedge A),
\]
and the pure Maurer--Cartan terms give
\(\frac13\operatorname{tr}(\theta_g^3)\).  This is
\eqref{eq:cs-finite-gauge-transgression}.

Therefore, on closed \(M\),
\[
  S_{\rm CS}(A^g)-S_{\rm CS}(A)
  =
  2\pi k\, n(g)
\]
where
\[
  n(g)
  =
  \frac{1}{24\pi^2}
  \int_M\operatorname{tr}(\theta_g\wedge\theta_g\wedge\theta_g)
  \in\mathbb Z
\]
is the winding number with the chosen normalization.  The sign in this
formula is tied to the displayed right-action convention for \(A^g\).

\begin{remark}[Gauge invariance of the exponentiated action]
\label{rem:cs-level-quantization}
With the integral normalization above, \(\exp(\ii S_{\rm CS})\) is invariant
under all gauge transformations on closed \(M\) if \(k\in\mathbb Z\).

The transformation formula gives
\(\exp(\ii S_{\rm CS}(A^g))=\exp(\ii S_{\rm CS}(A))\exp(2\pi\ii k n(g))\).
This equals \(\exp(\ii S_{\rm CS}(A))\) for all \(n(g)\in\mathbb Z\) exactly
when \(k\) is integral in the normalization used for the bilinear form.  The
closed-manifold hypothesis removes boundary terms from the gauge-variation
formula.  With a boundary, the gauge-variation formula contains the boundary
term obtained by restricting the transgression form to \(\partial M\); its
cancellation requires boundary degrees of freedom, boundary conditions, or a
specified anomalous boundary theory together with the integral level.
\end{remark}

\section{Phase Space on a Surface}

Let \(M=\mathbb R\times\Sigma\).  Write \(A=A_0dt+A_\Sigma\).  The term
linear in time derivatives is
\[
  \frac{k}{4\pi}\int_{\mathbb R}dt
  \int_\Sigma \operatorname{tr}(A_\Sigma\wedge \partial_t A_\Sigma).
\]
The field \(A_0\) is a Lagrange multiplier imposing \(F_{A_\Sigma}=0\).
Therefore the classical phase space is the moduli space
\[
  \mathcal M_{\rm flat}(\Sigma,G)
  =
  \{A_\Sigma:F_{A_\Sigma}=0\}/\mathcal G_\Sigma
\]
with symplectic form
\[
  \omega_{\rm CS}
  =
  \frac{k}{4\pi}\int_\Sigma
  \operatorname{tr}(\delta A_\Sigma\wedge \delta A_\Sigma).
\]
Quantization of Chern--Simons theory assigns a finite-dimensional state space
to \(\Sigma\) in compact semisimple examples after the level and boundary
data are fixed.

\section{Wilson Lines and Framing}

For an oriented knot or link component \(K\subset M\) and a representation
\(R\) of \(G\), the Wilson line is
\[
  W_R(K)=\operatorname{Tr}_R\operatorname{Pexp}\int_K A .
\]
The quantum expectation value of Wilson links requires a framing: a
nonvanishing normal vector field along each component, or equivalently a
choice of point-splitting direction.  The framing dependence is a local
counterterm effect for the self-linking singularity of the line operator.

\section{\texorpdfstring{\(SU(2)_k\)}{SU(2)k} Modular Data and Link Invariants}

The first fully explicit compact simple example is \(G=SU(2)\) at positive
integer level \(k\).  In the Hamiltonian/modular-functor normalization, the
simple Wilson-line labels are
\[
  a=0,1,\ldots,k,
  \qquad
  \text{spin}(a)=\frac a2 .
\]
These are the integrable highest weights of the affine algebra
\(\widehat{\mathfrak{su}}(2)_k\).  Define
\begin{equation}
  K=k+2,
  \qquad
  S_{ab}
  =
  \sqrt{\frac{2}{K}}\,
  \sin\!\left(\frac{(a+1)(b+1)\pi}{K}\right),
  \qquad 0\le a,b\le k .
  \label{eq:su2k-modular-s-matrix}
\end{equation}
The matrix \(S\) is the modular \(S\)-transformation on the torus state
space in the basis determined by core Wilson lines in a solid torus.

\begin{remark}[Orthogonality of the \(SU(2)_k\) modular \(S\)-matrix]
\label{rem:su2k-s-matrix-orthogonality}
The matrix \eqref{eq:su2k-modular-s-matrix} is real, symmetric, and
orthogonal:
\[
  \sum_{b=0}^k S_{ab}S_{cb}=\delta_{ac}.
\]

Put \(r=a+1\), \(s=c+1\), and \(K=k+2\).  The required identity is
\[
  \frac{2}{K}\sum_{m=1}^{K-1}
  \sin\!\left(\frac{rm\pi}{K}\right)
  \sin\!\left(\frac{sm\pi}{K}\right)
  =
  \delta_{rs},
  \qquad 1\le r,s\le K-1 .
\]
Using \(2\sin x\sin y=\cos(x-y)-\cos(x+y)\), the sum reduces to finite
geometric sums of \(K\)-th roots of unity.  For \(r=s\), it gives
\(\sum_{m=1}^{K-1}\sin^2(rm\pi/K)=K/2\).  For \(r\ne s\), the two cosine
sums cancel.  This proves the orthogonality relation.
\end{remark}

The quantum dimension of the label \(a\) is
\begin{equation}
  d_a=\frac{S_{0a}}{S_{00}}
  =
  \frac{\sin((a+1)\pi/K)}{\sin(\pi/K)} .
  \label{eq:su2k-quantum-dimension}
\end{equation}
This number is the normalized expectation value of a \(0\)-framed unknot in
\(S^3\) colored by \(a\), in the convention in which the empty normalized
expectation value on \(S^3\) is \(1\):
\begin{equation}
  \langle W_a(\text{unknot})\rangle_{S^3}
  =
  d_a .
  \label{eq:su2k-unknot-quantum-dimension}
\end{equation}
The corresponding \(0\)-framed Hopf link with components colored by \(a\) and
\(b\) is
\begin{equation}
  \langle W_a(K_1)W_b(K_2)\rangle_{S^3,\operatorname{Hopf}}
  =
  \frac{S_{ab}}{S_{00}} .
  \label{eq:su2k-hopf-link}
\end{equation}
Equation~\eqref{eq:su2k-hopf-link} is the basic bridge between the path
integral and the modular category: the two linked Wilson loops compute the
matrix element of the modular \(S\)-transformation, normalized by the vacuum
amplitude.

\begin{example}[\(SU(2)_1\)]
\label{ex:su2-level-one-cs}
At \(k=1\), there are two labels \(0,1\) and
\[
  S=\frac{1}{\sqrt2}
  \begin{pmatrix}
  1&1\\
  1&-1
  \end{pmatrix}.
\]
The nontrivial line has quantum dimension \(d_1=1\), and the Hopf link with
both components colored by \(1\) has normalized value
\[
  \frac{S_{11}}{S_{00}}=-1 .
\]
Thus \(SU(2)_1\) is an abelian topological order in this line basis: the
nontrivial line is invertible and has nontrivial mutual braiding with itself
as measured by the Hopf link.
\end{example}

\begin{example}[\(SU(2)_2\)]
\label{ex:su2-level-two-cs}
At \(k=2\), the labels \(0,1,2\) have quantum dimensions
\[
  d_0=1,\qquad d_1=\sqrt2,\qquad d_2=1 .
\]
The label \(1\), the spin-\(\frac12\) Wilson line, is non-invertible.  This
is the Chern--Simons modular tensor category underlying the Ising fusion
rules after the conventional identification of labels.
\end{example}

\section{Verlinde Formula}

The same modular data determine the fusion coefficients.  For \(SU(2)_k\),
the Verlinde formula reads
\begin{equation}
  N_{ab}^{\phantom{ab}c}
  =
  \sum_{x=0}^{k}
  \frac{S_{ax}S_{bx}S_{cx}}{S_{0x}},
  \label{eq:su2k-verlinde-formula}
\end{equation}
where all labels are self-dual.  The result is the truncated \(SU(2)\)
Clebsch--Gordan rule:
\[
  N_{ab}^{\phantom{ab}c}=1
\]
exactly when
\[
  |a-b|\le c\le \min(a+b,\,2k-a-b),
  \qquad
  a+b+c\in2\mathbb Z,
\]
and \(N_{ab}^{\phantom{ab}c}=0\) otherwise.  The truncation
\(c\le2k-a-b\) is the finite-level effect: tensor products that would exceed
the affine alcove are reflected back by the quantum Weyl symmetry.

\begin{remark}[Verlinde dimensions from the modular-functor input]
\label{rem:su2k-verlinde-dimensions}
Let \(\Sigma_g\) be a closed oriented genus-\(g\) surface.  The
\(SU(2)_k\) Chern--Simons state-space dimension assigned by the modular
functor is
\begin{equation}
  \dim\mathcal H_{SU(2)_k}(\Sigma_g)
  =
  \sum_{x=0}^k (S_{0x})^{2-2g}.
  \label{eq:su2k-verlinde-dimension}
\end{equation}
In particular,
\[
  \dim\mathcal H(S^2)=1,
  \qquad
  \dim\mathcal H(T^2)=k+1 .
\]

The dimension formula is the zero-puncture specialization of the Verlinde
formula for conformal blocks,
\[
  \dim\mathcal H(\Sigma_g;a_1,\ldots,a_n)
  =
  \sum_{x=0}^{k}
  (S_{0x})^{2-2g-n}\prod_{j=1}^n S_{a_jx}.
\]
For \(g=0,n=0\), orthogonality of the first row of \(S\) gives
\(\sum_x S_{0x}^2=1\).  For \(g=1,n=0\), the exponent vanishes, so the sum
has \(k+1\) terms.  Higher genus follows from pair-of-pants gluing: each
handle contraction inserts a sum over an intermediate label weighted by the
inverse square of \(S_{0x}\), which is exactly the displayed exponent.
This paragraph records the modular-functor consequence used below; it is not
a derivation of the modular functor from the Chern--Simons path integral.
\end{remark}

\section{Boundary Theory}

If \(M\) has boundary, the boundary term in the variation is the first
datum.  Let \(\partial M=\Sigma\), choose a complex coordinate \(z,\bar z\)
on \(\Sigma\), and write
\[
  A_\Sigma=A_z\,\dd z+A_{\bar z}\,\dd\bar z,
  \qquad
  \dd^2z:=\dd z\wedge\dd\bar z .
\]
On solutions of \(F_A=0\), the variation of the bulk Chern--Simons action is
\[
  \delta S_{\rm CS}\big|_{F=0}
  =
  \frac{k}{4\pi}\int_\Sigma
  \operatorname{tr}(A_\Sigma\wedge\delta A_\Sigma)
  =
  \frac{k}{4\pi}\int_\Sigma\dd^2z\,
  \operatorname{tr}
  \left(A_z\delta A_{\bar z}-A_{\bar z}\delta A_z\right).
\]
The holomorphic polarization fixes \(A_{\bar z}\) as the source and adds
\[
  S_{\rm pol}
  =
  \frac{k}{4\pi}\int_\Sigma\dd^2z\,
  \operatorname{tr}(A_zA_{\bar z}).
\]
Then
\begin{equation}
  \delta(S_{\rm CS}+S_{\rm pol})\big|_{F=0}
  =
  \frac{k}{2\pi}\int_\Sigma\dd^2z\,
  \operatorname{tr}(A_z\,\delta A_{\bar z}).
  \label{eq:cs-boundary-polarized-variation}
\end{equation}
Thus \(kA_z/(2\pi)\) is the current conjugate to \(A_{\bar z}\) in this
polarization.

The corresponding boundary field is a map \(g:\Sigma\to G\).  For an
extension of \(g\) to a three-manifold \(B\) with \(\partial B=\Sigma\), set
\begin{equation}
  S_{{\rm WZW},k}(g)
  =
  \frac{k}{4\pi}\int_\Sigma\dd^2z\,
  \operatorname{tr}
  \left(g^{-1}\partial_zg\,g^{-1}\partial_{\bar z}g\right)
  +
  \frac{k}{12\pi}\int_B
  \operatorname{tr}(g^{-1}\dd g)^3 .
  \label{eq:wzw-action-boundary-cs}
\end{equation}
The Wess--Zumino term is defined modulo \(2\pi\mathbb Z\) in
\(\exp(\ii S)\) exactly when the same integral normalization that quantizes
the Chern--Simons level is used.

\begin{lemma}[Polyakov--Wiegmann identity]
\label{lem:polyakov-wiegmann-boundary}
With the convention \eqref{eq:wzw-action-boundary-cs},
\[
  S_{{\rm WZW},k}(gh)
  =
  S_{{\rm WZW},k}(g)+S_{{\rm WZW},k}(h)
  +
  \frac{k}{2\pi}\int_\Sigma\dd^2z\,
  \operatorname{tr}
  \left(g^{-1}\partial_{\bar z}g\,
  \partial_zh\,h^{-1}\right).
\]
\end{lemma}

\begin{proof}
The Maurer--Cartan form of the product is
\[
  (gh)^{-1}\dd(gh)
  =
  h^{-1}(g^{-1}\dd g)h+h^{-1}\dd h .
\]
Cyclicity and invariance of \(\operatorname{tr}\) give the three-form
identity
\[
  \operatorname{tr}\!\left((gh)^{-1}\dd(gh)\right)^3
  =
  \operatorname{tr}(g^{-1}\dd g)^3
  +
  \operatorname{tr}(h^{-1}\dd h)^3
  -
  3\,\dd\operatorname{tr}
  \left(g^{-1}\dd g\wedge \dd h\,h^{-1}\right).
\]
Stokes' theorem converts the final term to a boundary integral on
\(\Sigma\).  To see the coefficient and sign, put
\[
  X_z=g^{-1}\partial_zg,\qquad
  X_{\bar z}=g^{-1}\partial_{\bar z}g,\qquad
  Y_z=\partial_zh\,h^{-1},\qquad
  Y_{\bar z}=\partial_{\bar z}h\,h^{-1}.
\]
The kinetic term of \(gh\) has the cross contribution
\[
  \frac{k}{4\pi}\int_\Sigma\dd^2z\,
  \operatorname{tr}(X_zY_{\bar z}+X_{\bar z}Y_z).
\]
The boundary contribution from the Wess--Zumino three-form identity is
\[
  \frac{k}{4\pi}\int_\Sigma\dd^2z\,
  \operatorname{tr}(-X_zY_{\bar z}+X_{\bar z}Y_z).
\]
Adding the two cross contributions cancels the \(X_zY_{\bar z}\) term and
doubles the \(X_{\bar z}Y_z\) term:
\[
  \frac{k}{2\pi}\int_\Sigma\dd^2z\,
  \operatorname{tr}(g^{-1}\partial_{\bar z}g\,\partial_zh\,h^{-1}).
\]
The non-cross terms are exactly \(S_{{\rm WZW},k}(g)\) and
\(S_{{\rm WZW},k}(h)\).  This proves the identity.
\end{proof}

The gauged chiral boundary action in the same polarization is
\begin{equation}
  S_{\partial,k}(g;A_{\bar z})
  =
  S_{{\rm WZW},k}(g)
  +
  \frac{k}{2\pi}\int_\Sigma\dd^2z\,
  \operatorname{tr}
  \left(\partial_zg\,g^{-1}A_{\bar z}\right).
  \label{eq:chiral-gauged-wzw-boundary-action}
\end{equation}
If a flat bulk connection is written near the boundary as
\[
  A_z=-\partial_zg\,g^{-1},
\]
then
\[
  \delta_{A_{\bar z}}S_{\partial,k}
  =
  -\frac{k}{2\pi}\int_\Sigma\dd^2z\,
  \operatorname{tr}(A_z\,\delta A_{\bar z}),
\]
which cancels the polarized Chern--Simons variation
\eqref{eq:cs-boundary-polarized-variation}.  This is the precise sense in
which the chiral Wess--Zumino--Witten boundary degree of freedom supplies the
boundary term needed by the Chern--Simons variational problem.

The holomorphic current is
\[
  J(z)=-\frac{k}{2\pi}\,\partial_zg\,g^{-1}.
\]
Writing \(J=J^at_a\), with
\(\operatorname{tr}(t_at_b)=\kappa_{ab}\) and
\([t_a,t_b]=\ii f^c{}_{ab}t_c\), quantization gives the affine current
algebra at level \(k\),
\[
  J^a(z)J^b(w)
  \sim
  \frac{k\,\kappa^{ab}}{(z-w)^2}
  +
  \frac{\ii f^{ab}{}_{c}J^c(w)}{z-w},
\]
equivalently
\[
  [J_m^a,J_n^b]
  =
  \ii f^{ab}{}_{c}J_{m+n}^c
  +
  k\,m\,\kappa^{ab}\delta_{m+n,0}.
\]
The precise Hilbert space still depends on the boundary polarization and
global boundary condition, but the local current algebra and its level are
fixed by the same integer \(k\) that quantizes the bulk Chern--Simons action.

\begin{openproblem}[Chern--Simons path integral as a QFT object]
\label{op:chern-simons-qft-object}
Give a regulator-level construction of Chern--Simons theory that simultaneously
controls gauge fixing, framing, Wilson-line renormalization, boundary
conditions, and functorial gluing.  Perturbative configuration-space
integrals, geometric quantization, and modular tensor category constructions
each capture part of this structure; a single local-QFT construction should
state the comparison maps among them.
\end{openproblem}
