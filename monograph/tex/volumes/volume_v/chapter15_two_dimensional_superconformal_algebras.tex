\chapter{Two-Dimensional Superconformal Algebras}
\label{ch:two-dimensional-superconformal-algebras}

\ConventionsEuclideanCFT

\section{Spin and Sector Data}

The superconformal algebras in two dimensions are chiral symmetry algebras.
They are not, by themselves, definitions of a continuum quantum field theory.
The underlying QFT must supply local operator domains, correlation functions,
positivity when the theory is unitary, and sewing data.  This chapter isolates
the chiral algebraic layer because it is shared infrastructure for
two-dimensional CFT.  The dynamical construction of supersymmetric field
theories, including Landau--Ginzburg and gauged linear sigma models, belongs
to the supersymmetric-QFT development in Volume~VII.  The statements below are
therefore algebraic consequences and consistency tests for any such model,
not a replacement for the supersymmetric RG analysis.

\begin{definition}[Spin two-dimensional CFT structure]
\label{def:spin-two-dimensional-cft-structure}
A spin two-dimensional CFT structure consists of the following items.
\begin{enumerate}
\item An oriented two-dimensional Euclidean CFT \(\mathcal C\), with
holomorphic stress tensor \(T(z)\), antiholomorphic stress tensor
\(\bar T(\bar z)\), and central charges \(c,\bar c\).
\item For each spin surface \((\Sigma,\mathfrak s)\), correlation functions
of local fields which may be even or odd under the fermion-parity operator
\((-1)^F\).
\item Chiral state spaces on the circle decomposed into Neveu--Schwarz and
Ramond sectors,
\[
  \Hilb^{\mathrm{ch}}
  =
  \Hilb_{\mathrm{NS}}^{\mathrm{ch}}
  \oplus
  \Hilb_{\mathrm R}^{\mathrm{ch}},
\]
where a chiral fermionic field has antiperiodic boundary condition in the
NS sector and periodic boundary condition in the R sector.
\item A locality rule: even fields have ordinary single-valued OPEs with all
local fields, while odd chiral fields are mutually local only after the spin
structure and fermion-parity signs have been specified.
\end{enumerate}
The same chiral algebra may occur in several full spin CFTs, because the full
theory also requires an antiholomorphic sector, a choice of spin-structure
sum or projection, and modular/sewing data.
\end{definition}

Let \(a(z)\) be a chiral field of holomorphic weight \(h_a\).  In a sector
where \(a\) has monodromy \(e^{-2\pi \ii \nu_a}\) around the spatial circle,
we write
\begin{equation}
  a(z)=\sum_{r\in\mathbb Z+\nu_a} a_r z^{-r-h_a}.
  \label{eq:spin-cft-chiral-mode-expansion}
\end{equation}
For the supercurrent of an \(\mathcal N=1\) or \(\mathcal N=2\)
superconformal algebra, \(h_a=3/2\), \(\nu_a=1/2\) in the NS sector, and
\(\nu_a=0\) in the R sector.  The Ramond sector therefore contains
supercharge zero modes.  This is the operator form of the difference between
antiperiodic and periodic spin structures on the cylinder.

\section{\texorpdfstring{\(\mathcal N=1\)}{N=1} Superconformal Algebra}

\begin{definition}[\(\mathcal N=1\) chiral superconformal algebra]
\label{def:n1-superconformal-algebra}
An \(\mathcal N=1\) chiral superconformal algebra of central charge \(c\)
consists of a chiral stress tensor \(T(z)\), an odd chiral field \(G(z)\) of
weight \(3/2\), and a sector choice
\[
  r\in\mathbb Z+\frac12\quad(\mathrm{NS}),\qquad
  r\in\mathbb Z\quad(\mathrm R)
\]
for the modes
\[
  T(z)=\sum_{n\in\mathbb Z}L_n z^{-n-2},
  \qquad
  G(z)=\sum_{r}G_r z^{-r-3/2}.
\]
The singular OPEs are
\begin{align}
  T(z)T(w)
  &\sim
  \frac{c/2}{(z-w)^4}
  +\frac{2T(w)}{(z-w)^2}
  +\frac{\partial T(w)}{z-w},
  \label{eq:n1-tt-ope}\\
  T(z)G(w)
  &\sim
  \frac{(3/2)G(w)}{(z-w)^2}
  +\frac{\partial G(w)}{z-w},
  \label{eq:n1-tg-ope}\\
  G(z)G(w)
  &\sim
  \frac{2c/3}{(z-w)^3}
  +\frac{2T(w)}{z-w}.
  \label{eq:n1-gg-ope}
\end{align}
\end{definition}

\begin{proposition}[Mode algebra from the \(\mathcal N=1\) OPEs]
\label{prop:n1-mode-algebra}
The modes in Definition~\ref{def:n1-superconformal-algebra} obey
\begin{align}
  [L_m,L_n]
  &=
  (m-n)L_{m+n}
  +\frac{c}{12}(m^3-m)\delta_{m+n,0},
  \label{eq:n1-virasoro-mode-algebra}\\
  [L_m,G_r]
  &=
  \left(\frac m2-r\right)G_{m+r},
  \label{eq:n1-lg-mode-algebra}\\
  \{G_r,G_s\}
  &=
  2L_{r+s}
  +\frac c3\left(r^2-\frac14\right)\delta_{r+s,0}.
  \label{eq:n1-gg-mode-algebra}
\end{align}
\end{proposition}

\begin{proof}
The Virasoro bracket is the stress-tensor calculation of
Proposition~\ref{prop:tt-ope-virasoro-algebra}.  For the second formula,
write
\[
  [L_m,G_r]
  =
  \oint_0\frac{\dd w}{2\pi\ii}w^{r+1/2}
  \oint_w\frac{\dd z}{2\pi\ii}z^{m+1}T(z)G(w).
\]
Using \eqref{eq:n1-tg-ope}, the inner residue is
\[
  (m+1)w^m\frac32G(w)+w^{m+1}\partial G(w).
\]
The outer residue of the first term is
\(\frac32(m+1)G_{m+r}\).  For the second term, integration by parts gives
\[
  \oint_0\frac{\dd w}{2\pi\ii}w^{m+r+3/2}\partial G(w)
  =
  -\left(m+r+\frac32\right)G_{m+r}.
\]
The coefficient is therefore \(m/2-r\).

For the anticommutator, the odd fields require the supercommutator contour.
The singular part \eqref{eq:n1-gg-ope} gives
\[
  \{G_r,G_s\}
  =
  \oint_0\frac{\dd w}{2\pi\ii}w^{s+1/2}
  \operatorname{Res}_{z=w}
  z^{r+1/2}
  \left(
    \frac{2c/3}{(z-w)^3}+\frac{2T(w)}{z-w}
  \right).
\]
The \(T\)-term gives \(2L_{r+s}\).  The cubic pole gives
\[
  \frac{2c}{3}\cdot
  \frac12
  \left(r+\frac12\right)\left(r-\frac12\right)
  \delta_{r+s,0}
  =
  \frac c3\left(r^2-\frac14\right)\delta_{r+s,0}.
\]
\end{proof}

\paragraph{Ramond zero-mode bound.}
In a unitary Ramond representation of the \(\mathcal N=1\) algebra,
\begin{equation}
  L_0-\frac{c}{24}\ge0.
  \label{eq:n1-ramond-bound}
\end{equation}
A Ramond ground state is a state annihilated by \(G_0\), equivalently a state
with \(L_0=c/24\).
Set \(r=s=0\) in \eqref{eq:n1-gg-mode-algebra}.  Then
\[
  \{G_0,G_0\}=2L_0-\frac c{12}
  =
  2\left(L_0-\frac c{24}\right).
\]
If the representation is unitary, \(G_0^\dagger=G_0\), so the expectation
value of \(\{G_0,G_0\}=2G_0^2\) on any vector is nonnegative.  Equality
holds precisely on the kernel of \(G_0\), because
\(\langle v,G_0^2v\rangle=\|G_0v\|^2\).  Hence the bottom of a unitary Ramond
module is bounded by \(c/24\), and states at the bound are exactly the
Ramond states annihilated by the zero mode.

\section{\texorpdfstring{\(\mathcal N=2\)}{N=2} Superconformal Algebra}

The \(\mathcal N=2\) algebra adds a holomorphic \(U(1)\) current.  This extra
current is the reason that spectral flow, chiral-primary bounds, and
topological twists are algebraic operations rather than model-dependent
accidents.

\begin{definition}[\(\mathcal N=2\) chiral superconformal algebra]
\label{def:n2-superconformal-algebra}
An \(\mathcal N=2\) chiral superconformal algebra of central charge \(c\)
consists of a chiral stress tensor \(T(z)\), a weight-one even current
\(J(z)\), and two odd weight-\(3/2\) chiral fields \(G^+(z)\) and \(G^-(z)\).
Their modes are
\[
  J(z)=\sum_{n\in\mathbb Z}J_n z^{-n-1},
  \qquad
  G^\pm(z)=\sum_r G^\pm_r z^{-r-3/2},
\]
with \(r\in\mathbb Z+1/2\) in the NS sector and \(r\in\mathbb Z\) in the R
sector.  The singular OPEs, in addition to the Virasoro OPE for \(T\), are
\begin{align}
  T(z)J(w)
  &\sim
  \frac{J(w)}{(z-w)^2}
  +\frac{\partial J(w)}{z-w},
  \label{eq:n2-tj-ope}\\
  T(z)G^\pm(w)
  &\sim
  \frac{(3/2)G^\pm(w)}{(z-w)^2}
  +\frac{\partial G^\pm(w)}{z-w},
  \label{eq:n2-tg-ope}\\
  J(z)J(w)
  &\sim
  \frac{c/3}{(z-w)^2},
  \label{eq:n2-jj-ope}\\
  J(z)G^\pm(w)
  &\sim
  \pm\frac{G^\pm(w)}{z-w},
  \label{eq:n2-jg-ope}\\
  G^+(z)G^-(w)
  &\sim
  \frac{2c/3}{(z-w)^3}
  +\frac{2J(w)}{(z-w)^2}
  +\frac{2T(w)+\partial J(w)}{z-w},
  \label{eq:n2-gplus-gminus-ope}\\
  G^\pm(z)G^\pm(w)&\sim0.
  \label{eq:n2-same-sign-g-ope}
\end{align}
\end{definition}

\begin{proposition}[Mode algebra for \(\mathcal N=2\)]
\label{prop:n2-mode-algebra}
The modes in Definition~\ref{def:n2-superconformal-algebra} obey
\begin{align}
  [L_m,J_n]&=-nJ_{m+n},
  \label{eq:n2-lj-mode-algebra}\\
  [J_m,J_n]&=\frac c3 m\delta_{m+n,0},
  \label{eq:n2-jj-mode-algebra}\\
  [L_m,G^\pm_r]&=\left(\frac m2-r\right)G^\pm_{m+r},
  \label{eq:n2-lg-mode-algebra}\\
  [J_m,G^\pm_r]&=\pm G^\pm_{m+r},
  \label{eq:n2-jg-mode-algebra}\\
  \{G^+_r,G^-_s\}
  &=
  2L_{r+s}
  +(r-s)J_{r+s}
  +\frac c3\left(r^2-\frac14\right)\delta_{r+s,0},
  \label{eq:n2-gplus-gminus-mode-algebra}\\
  \{G^\pm_r,G^\pm_s\}&=0.
  \label{eq:n2-same-sign-g-mode-algebra}
\end{align}
\end{proposition}

\begin{proof}
The formulas involving \(L_m\) are the primary-field residue calculation
used in Proposition~\ref{prop:n1-mode-algebra}, with weights \(1\) and
\(3/2\).  The current commutator follows from the double-pole residue in
\eqref{eq:n2-jj-ope}:
\[
  [J_m,J_n]
  =
  \oint_0\frac{\dd w}{2\pi\ii}w^n
  \operatorname{Res}_{z=w}
  z^m\frac{c/3}{(z-w)^2}
  =
  \frac c3 m\delta_{m+n,0}.
\]
Equation~\eqref{eq:n2-jg-mode-algebra} is the simple-pole residue of
\eqref{eq:n2-jg-ope}.  For the mixed supercurrent anticommutator, the
triple pole in \eqref{eq:n2-gplus-gminus-ope} gives
\(\frac c3(r^2-1/4)\delta_{r+s,0}\), exactly as in the
\(\mathcal N=1\) calculation.  The simple pole gives \(2L_{r+s}\).  The
double pole and the derivative term combine as
\[
  2\left(r+\frac12\right)J_{r+s}
  -
  (r+s+1)J_{r+s}
  =
  (r-s)J_{r+s}.
\]
\end{proof}

\begin{proposition}[Spectral flow automorphism]
\label{prop:n2-spectral-flow}
For any \(\eta\in\frac12\mathbb Z\), the assignment
\begin{align}
  L_n^{(\eta)}
  &=
  L_n+\eta J_n+\frac c6\eta^2\delta_{n,0},
  \label{eq:n2-spectral-flow-l}\\
  J_n^{(\eta)}
  &=
  J_n+\frac c3\eta\delta_{n,0},
  \label{eq:n2-spectral-flow-j}\\
  (G^\pm_r)^{(\eta)}
  &=
  G^\pm_{r\pm\eta}
  \label{eq:n2-spectral-flow-g}
\end{align}
preserves the \(\mathcal N=2\) mode algebra, with the same central charge
\(c\).  Integer \(\eta\) preserves NS and R sectors separately; half-integer
\(\eta\) exchanges them.
\end{proposition}

\begin{proof}
It is enough to check the brackets in
Proposition~\ref{prop:n2-mode-algebra}.  For example,
\[
  [J_m^{(\eta)},J_n^{(\eta)}]
  =
  [J_m,J_n]
  =
  \frac c3m\delta_{m+n,0},
\]
which is the same formula in the flowed generators because the central shifts
commute.  Also
\[
  [J_m^{(\eta)},(G^\pm_r)^{(\eta)}]
  =
  [J_m,G^\pm_{r\pm\eta}]
  =
  \pm G^\pm_{m+r\pm\eta}
  =
  \pm(G^\pm_{m+r})^{(\eta)}.
\]
For the mixed supercurrent relation,
\[
  \{G^+_{r+\eta},G^-_{s-\eta}\}
  =
  2L_{r+s}
  +(r-s+2\eta)J_{r+s}
  +\frac c3\left((r+\eta)^2-\frac14\right)\delta_{r+s,0}.
\]
Substituting
\[
  L_{r+s}=L^{(\eta)}_{r+s}
  -\eta J_{r+s}
  -\frac c6\eta^2\delta_{r+s,0},
  \qquad
  J_{r+s}=J^{(\eta)}_{r+s}
  -\frac c3\eta\delta_{r+s,0},
\]
the \(J\)-coefficient becomes \(r-s\), and the central coefficient reduces
to \(\frac c3(r^2-1/4)\).  The remaining brackets are checked in the same
residue algebra.
\end{proof}

If a simultaneous \(L_0,J_0\) eigenstate has eigenvalues \(h,q\), then
spectral flow sends them to
\begin{equation}
  h^{(\eta)}
  =
  h+\eta q+\frac c6\eta^2,
  \qquad
  q^{(\eta)}
  =
  q+\frac c3\eta .
  \label{eq:n2-spectral-flow-hq}
\end{equation}
This formula is often the cleanest way to see why NS chiral primaries and
Ramond ground states are the same representation-theoretic object viewed in
different spin sectors.

\begin{proposition}[NS unitarity bound and chiral primaries]
\label{prop:n2-chiral-primary-bound}
Let \(v\) be a simultaneous \(L_0,J_0\) eigenvector in a unitary NS
representation, with
\[
  L_0v=hv,\qquad J_0v=qv.
\]
Then
\begin{equation}
  h\ge\frac{|q|}{2}.
  \label{eq:n2-ns-unitarity-bound}
\end{equation}
If \(q\ge0\), equality \(h=q/2\) holds precisely when
\begin{equation}
  G^+_{-1/2}v=0.
  \label{eq:n2-chiral-primary-condition}
\end{equation}
If \(q\le0\), equality \(h=-q/2\) holds precisely when
\begin{equation}
  G^-_{-1/2}v=0.
  \label{eq:n2-antichiral-primary-condition}
\end{equation}
\end{proposition}

\begin{proof}
Unitarity gives \((G^\pm_{-1/2})^\dagger=G^\mp_{1/2}\).  Taking expectation
values of \eqref{eq:n2-gplus-gminus-mode-algebra} gives
\[
  \|G^+_{-1/2}v\|^2
  =
  \langle v,\{G^-_{1/2},G^+_{-1/2}\}v\rangle
  =
  (2h-q)\|v\|^2,
\]
and
\[
  \|G^-_{-1/2}v\|^2
  =
  \langle v,\{G^+_{1/2},G^-_{-1/2}\}v\rangle
  =
  (2h+q)\|v\|^2.
\]
Both norms are nonnegative, so \(2h\ge q\) and \(2h\ge -q\).  Equality in
the corresponding inequality is equivalent to the vanishing of the
corresponding norm.
\end{proof}

\paragraph{NS--R relation for chiral primaries.}
If \(v\) is an NS chiral primary with \(h=q/2\), then spectral flow by
\(\eta=-1/2\) maps it to a Ramond state with
\begin{equation}
  h^{(-1/2)}=\frac c{24}.
  \label{eq:n2-chiral-to-ramond-ground}
\end{equation}
Thus it saturates the Ramond ground-state energy.
This is obtained by inserting \(h=q/2\) and \(\eta=-1/2\) into
\eqref{eq:n2-spectral-flow-hq}:
\[
  h^{(-1/2)}
  =
  \frac q2-\frac q2+\frac c{24}
  =
  \frac c{24}.
\]

\section{Extended \texorpdfstring{\(\mathcal N=2\)}{N=2} Spectral-Flow Operators}
\label{sec:extended-n2-spectral-flow-operators}

Proposition~\ref{prop:n2-spectral-flow} is a statement about the abstract
mode algebra.  It does not say that spectral flow is implemented by a local
operator in a particular CFT.  That extra assertion needs a charge lattice
and locality hypothesis.  This distinction is small in notation and large in
logic: the extended \(\mathcal N=2\) algebra is an additional chiral-algebra
structure requiring more than the singular OPEs in
Definition~\ref{def:n2-superconformal-algebra}.

Put
\[
  \kappa=\frac c3 .
\]
In this section assume \(\kappa>0\) and assume that the \(U(1)_R\) current
exponentiates to a rank-one Heisenberg/lattice chiral algebra.  Concretely,
choose a chiral boson \(\varphi\) with
\begin{equation}
  \varphi(z)\varphi(0)\sim-\log z,
  \qquad
  J(z)=\ii\sqrt{\kappa}\,\partial\varphi(z).
  \label{eq:n2-r-current-bosonization}
\end{equation}
Then \(J(z)J(0)\sim\kappa z^{-2}\), matching
\eqref{eq:n2-jj-ope}.  A field of \(R\)-charge \(q\) has a Heisenberg
factor
\[
  \exp\left(\ii\frac{q}{\sqrt\kappa}\varphi\right)
\]
up to a neutral factor.  The charge lattice of the actual CFT specifies
which such exponentials exist and which are mutually local.

\begin{definition}[Spectral-flow vertex]
\label{def:n2-spectral-flow-vertex}
Let \(\eta\in\frac12\mathbb Z\).  A spectral-flow vertex of parameter
\(\eta\) is a chiral field
\begin{equation}
  X_\eta(z)
  =
  :\exp\left(\ii\eta\sqrt{\kappa}\,\varphi(z)\right):
  \label{eq:n2-spectral-flow-vertex}
\end{equation}
provided the momentum \(\eta\sqrt{\kappa}\) belongs to the chosen
Heisenberg/lattice extension.  It is a local chiral operator only when its
monodromy with the fields of the chiral algebra is trivial.  If its
monodromy is spin-dependent, it is instead an operator between spin sectors.
\end{definition}

\paragraph{Operator implementation of spectral flow.}
Assume the factorization above, and let
\[
  \mathcal O(z)
  =
  :\exp\left(\ii\frac{q}{\sqrt\kappa}\varphi(z)\right):
  \mathcal O_0(z)
\]
where \(\mathcal O_0\) is neutral under \(J\).  If \(X_\eta\) exists, then
\begin{equation}
  X_\eta(z)\mathcal O(0)
  \sim
  z^{\eta q}\,
  \mathcal O^{(\eta)}(0)+\cdots
  \label{eq:n2-spectral-flow-vertex-ope-with-field}
\end{equation}
where \(\mathcal O^{(\eta)}\) has
\begin{equation}
  q^{(\eta)}=q+\eta\kappa,
  \qquad
  h^{(\eta)}=h+\eta q+\frac{\eta^2\kappa}{2}.
  \label{eq:n2-spectral-flow-vertex-hq}
\end{equation}
Since \(\kappa=c/3\), this is exactly
\eqref{eq:n2-spectral-flow-hq}.  The branch factor \(z^{\eta q}\) is the
locality test: integer powers give ordinary chiral locality, while
half-integer powers signal the NS/R spin-sector change.

The OPE of bosonic exponentials gives
\[
  :e^{\ii a\varphi(z)}:
  :e^{\ii b\varphi(0)}:
  \sim
  z^{ab}:e^{\ii(a+b)\varphi(0)}:+\cdots .
\]
With \(a=\eta\sqrt\kappa\) and \(b=q/\sqrt\kappa\), the branch factor is
\(z^{\eta q}\) and the new charge is
\[
  \sqrt\kappa\left(\frac q{\sqrt\kappa}+\eta\sqrt\kappa\right)
  =
  q+\eta\kappa .
\]
The Heisenberg part of the conformal weight changes by
\[
  \frac{(q+\eta\kappa)^2-q^2}{2\kappa}
  =
  \eta q+\frac{\eta^2\kappa}{2}.
\]
The neutral factor \(\mathcal O_0\) is unchanged, so the same difference is
the total weight shift.

For \(\eta=\pm1\), write
\[
  X^\pm=X_{\pm1}.
\]
These fields, when present as local chiral fields, are the integer
spectral-flow generators of the extended \(\mathcal N=2\) algebra.  They
are obtained from the vacuum by spectral flow, hence
\begin{equation}
  h(X^\pm)=\frac c6,
  \qquad
  q(X^\pm)=\pm\frac c3 .
  \label{eq:n2-integer-spectral-flow-operator-hq}
\end{equation}
In a unitary NS representation \(X^+\) is chiral and \(X^-\) is antichiral:
\begin{equation}
  G^+_{-1/2}X^+=0,
  \qquad
  G^-_{-1/2}X^-=0,
  \label{eq:n2-spectral-flow-operator-shortening}
\end{equation}
because \eqref{eq:n2-integer-spectral-flow-operator-hq} saturates the
bound of Proposition~\ref{prop:n2-chiral-primary-bound}.  Define the
opposite superdescendants
\begin{equation}
  Y^+=G^-_{-1/2}X^+,
  \qquad
  Y^-=G^+_{-1/2}X^- .
  \label{eq:n2-extended-y-operators}
\end{equation}
Then
\begin{equation}
  h(Y^\pm)=\frac c6+\frac12,
  \qquad
  q(Y^\pm)=\pm\left(\frac c3-1\right).
  \label{eq:n2-extended-y-operator-hq}
\end{equation}
These four fields \(X^\pm,Y^\pm\), together with
\(T,J,G^\pm\), are the finite generator set usually meant by the extended
\(\mathcal N=2\) chiral algebra.

The singular part of the spectral-flow-generator OPE is also fixed by the
Heisenberg current.  From \eqref{eq:n2-spectral-flow-vertex},
\begin{equation}
  X_\eta(z)X_{-\eta}(0)
  \sim
  z^{-\eta^2\kappa}
  \left[
    1+\eta zJ(0)
    +z^2\left(
      \frac{\eta}{2}\partial J(0)
      +\frac{\eta^2}{2}:J J:(0)
    \right)
    +O(z^3)
  \right].
  \label{eq:n2-spectral-flow-opposite-ope}
\end{equation}
The pole order, the coefficient of \(J\), and the two second-order
coefficients are therefore convention-sensitive consequences of
\eqref{eq:n2-r-current-bosonization}, not independent structure constants.

\begin{example}[Calabi--Yau-type charge lattice]
\label{ex:calabi-yau-type-spectral-flow-operators}
If \(c=3n\), then \(\kappa=n\).  The integer spectral-flow fields have
\[
  h(X^\pm)=\frac n2,
  \qquad
  q(X^\pm)=\pm n,
\]
and the half-flow fields \(X_{\pm1/2}\), when present on the spin cover,
have
\begin{equation}
  h(X_{\pm1/2})=\frac n8,
  \qquad
  q(X_{\pm1/2})=\pm\frac n2 .
  \label{eq:calabi-yau-type-half-flow-hq}
\end{equation}
In a large-radius Calabi--Yau sigma-model representative these are realized
by the holomorphic top-form fermion product and the Ramond spin fields,
respectively.  The present chapter uses only the algebraic implication:
their existence is a charge-lattice and spin-sector input whose weights and
charges must be exactly those displayed above.
\end{example}

\section{Elliptic Genus and Spectral-Flow Covariance}
\label{sec:elliptic-genus-spectral-flow-covariance}

The elliptic genus is the protected genus-one observable that links the
chiral algebraic data above to the supersymmetric models of Volume~VII.  Its
definition requires a full spin CFT, not just a chiral algebra.  We therefore
state the hypotheses before using the trace.

\begin{definition}[Elliptic genus setup]
\label{def:elliptic-genus-setup}
Let \(\mathcal C\) be a compact unitary \(\mathcal N=(2,2)\) spin CFT with
discrete Ramond--Ramond Hilbert space
\[
  \Hilb_{\mathrm{RR}}
  =
  \bigoplus_{h,\bar h,q} \Hilb_{h,\bar h,q}
\]
of finite multiplicities.  Assume \(c=\bar c\), a left \(U(1)_R\) current
\(J\), a fermion parity \((-1)^F\), and convergence of the Ramond trace for
\(\operatorname{Im}\tau>0\).  The elliptic genus is
\begin{equation}
  \varphi_{\mathcal C}(\tau,z)
  =
  \operatorname{Tr}_{\Hilb_{\mathrm{RR}}}
  (-1)^F y^{J_0}
  q^{L_0-c/24}\bar q^{\bar L_0-c/24},
  \qquad
  y=\ee^{2\pi\ii z},\quad q=\ee^{2\pi\ii\tau},\quad
  \bar q=\ee^{-2\pi\ii\bar\tau}.
  \label{eq:elliptic-genus-definition}
\end{equation}
\end{definition}

\begin{proposition}[Right-moving localization]
\label{prop:elliptic-genus-right-moving-localization}
Under Definition~\ref{def:elliptic-genus-setup}, and assuming the usual
trace-class cyclicity for the Ramond supertrace, the elliptic genus is
independent of \(\bar\tau\) and receives contributions only from right-moving
Ramond ground states:
\begin{equation}
  \bar L_0=\frac c{24}.
  \label{eq:elliptic-genus-right-ground-condition}
\end{equation}
\end{proposition}

\begin{proof}
In the right Ramond sector the zero-mode relation is
\begin{equation}
  \{\bar G^+_0,\bar G^-_0\}
  =
  2\left(\bar L_0-\frac c{24}\right).
  \label{eq:right-ramond-zero-mode-elliptic-genus}
\end{equation}
Differentiating \eqref{eq:elliptic-genus-definition} with respect to
\(\bar\tau\) inserts \(\bar L_0-c/24\).  By
\eqref{eq:right-ramond-zero-mode-elliptic-genus}, the derivative is a
supertrace of a right-moving supercommutator.  Cyclicity of the convergent
supertrace makes this vanish.  Equivalently, states with
\(\bar L_0>c/24\) are paired by the right zero modes with opposite fermion
parity and identical left quantum numbers, so their contributions cancel.
The unpaired states are precisely the right Ramond ground states.
\end{proof}

The elliptic variable \(z\) remembers left \(R\)-charge, so spectral flow
imposes a finite functional equation.  Put
\begin{equation}
  m_{\rm ell}:=\frac c6 .
  \label{eq:elliptic-genus-index}
\end{equation}
Assume that integer left spectral flow is implemented on the Ramond Hilbert
space and that the chosen left \(R\)-charge lattice has a multiplier
\(\zeta(\mu)\in U(1)\) under \(z\mapsto z+\mu\), meaning that
\(\ee^{2\pi\ii\mu J_0}\) acts by this scalar on the Ramond states contributing
to the trace.  Then
\begin{equation}
  \varphi_{\mathcal C}(\tau,z+\lambda\tau+\mu)
  =
  \exp\!\left[
    -2\pi\ii m_{\rm ell}\left(\lambda^2\tau+2\lambda z\right)
  \right]
  \zeta(\mu)
  \varphi_{\mathcal C}(\tau,z),
  \label{eq:elliptic-genus-elliptic-law}
\end{equation}
for \(\lambda,\mu\in\mathbb Z\).  If all Ramond left \(R\)-charges in the
trace are integral, the last multiplier is \(1\).

Here is the trace-level derivation of the \(\lambda\)-shift.  Replacing \(z\)
by
\(z+\lambda\tau\) multiplies each state by \(q^{\lambda q_L}\).  Under left
spectral flow by \(\lambda\),
\begin{equation}
  h' = h+\lambda q_L+\frac c6\lambda^2,
  \qquad
  q'_L=q_L+\frac c3\lambda .
  \label{eq:elliptic-genus-spectral-flow-reindex}
\end{equation}
Thus
\begin{equation}
  q^{h-c/24+\lambda q_L}y^{q_L}
  =
  q^{-c\lambda^2/6}y^{-c\lambda/3}
  q^{h'-c/24}y^{q'_L}.
  \label{eq:elliptic-genus-elliptic-law-weight}
\end{equation}
Reindexing the trace by the spectral-flow bijection gives
\eqref{eq:elliptic-genus-elliptic-law}, since
\(m_{\rm ell}=c/6\).  The \(\mu\)-shift is exactly the assumed
charge-lattice scalar \(\zeta(\mu)\).

\begin{quotedtheorem}[Elliptic genus as a Jacobi form]
\label{qthm:elliptic-genus-jacobi-form}
Assume, in addition to Definition~\ref{def:elliptic-genus-setup}, the spin
modular-functor covariance of the Ramond trace, anomaly cancellation for the
chosen \(R\)-symmetry background, and the charge-lattice multiplier specified
above.  Then \(\varphi_{\mathcal C}(\tau,z)\) is a holomorphic weak Jacobi
form of weight \(0\) and index \(m_{\rm ell}=c/6\), with the corresponding
charge-lattice multiplier when the index is half-integral.  In particular,
for
\begin{equation}
  \gamma=\begin{pmatrix}a&b\\ c_\gamma&d\end{pmatrix}\in SL_2(\mathbb Z),
  \label{eq:elliptic-genus-modular-matrix}
\end{equation}
one has
\begin{equation}
  \varphi_{\mathcal C}\!\left(
    \frac{a\tau+b}{c_\gamma\tau+d},
    \frac{z}{c_\gamma\tau+d}
  \right)
  =
  \exp\!\left(
    2\pi\ii m_{\rm ell}\frac{c_\gamma z^2}{c_\gamma\tau+d}
  \right)
  \varphi_{\mathcal C}(\tau,z)
  \label{eq:elliptic-genus-modular-law}
\end{equation}
up to that multiplier.  The modular law is a sewing theorem for the full spin
CFT with an \(R\)-symmetry background; the left-moving mode algebra supplies
only the spectral-flow part of this statement.
\end{quotedtheorem}

There is a finite Landau--Ginzburg shadow of the elliptic genus which can be
checked before any infrared-flow theorem is invoked.  Suppose a
quasihomogeneous isolated polynomial \(W\) of
Definition~\ref{def:quasihomogeneous-lg-chiral-polynomial} is realized by a
compact unitary \(\mathcal N=2\) fixed point, and suppose the Jacobi ring
classes are represented by NS chiral primaries with charges
\[
  Q_R(X^a)=\sum_i a_i q_i .
\]
Spectral flow by \(\eta=-1/2\) gives the corresponding left Ramond charges
\begin{equation}
  Q_R^{\mathrm R}(X^a)
  =
  Q_R(X^a)-\frac{c_{\rm LG}}6 .
  \label{eq:lg-ramond-charge-from-ns-chiral-primary}
\end{equation}
Therefore the \(q^0\) protected charge polynomial has the form
\begin{equation}
  \chi_y^{\rm LG}
  =
  \sum_{[X^a]\in\operatorname{Jac}(W)}
  y^{Q_R(X^a)-c_{\rm LG}/6}.
  \label{eq:lg-chi-y-polynomial}
\end{equation}
For \(W=X^{k+2}\), this becomes
\begin{equation}
  \chi_y^{A_k}
  =
  \sum_{\ell=0}^k
  y^{\frac{\ell}{k+2}-\frac{k}{2(k+2)}}.
  \label{eq:a-series-chi-y-polynomial}
\end{equation}
Setting \(y=1\) gives the Witten index \(k+1=\dim\operatorname{Jac}(W)\).
The equality between this protected finite polynomial and the infrared
elliptic genus is precisely the kind of LG/GLSM construction statement
assigned to Volume~VII.

\section{Protected Algebraic Tests for Supersymmetric Dynamics}

The full construction of two-dimensional \(\mathcal N=(2,2)\)
Landau--Ginzburg and gauged linear sigma-model flows belongs to the
supersymmetric-QFT chapters.  The part needed in the CFT volume is narrower:
the chiral-ring and central-charge data that any proposed
\(\mathcal N=2\) superconformal fixed point must reproduce.  We state this
as an algebraic interface between the CFT and supersymmetric-QFT volumes, not
as a proof of an infrared fixed point.

\begin{definition}[Quasihomogeneous Landau--Ginzburg chiral polynomial]
\label{def:quasihomogeneous-lg-chiral-polynomial}
Let \(X_1,\ldots,X_N\) be complex variables with rational weights
\(q_i\in(0,1/2]\).  A quasihomogeneous Landau--Ginzburg chiral polynomial consists
of a polynomial
\[
  W\in\mathbb C[X_1,\ldots,X_N]
\]
such that
\begin{equation}
  W(\lambda^{q_1}X_1,\ldots,\lambda^{q_N}X_N)=\lambda W(X_1,\ldots,X_N)
  \qquad
  \lambda\in\mathbb C^\times,
  \label{eq:lg-quasihomogeneous-condition}
\end{equation}
and such that the critical locus
\[
  \partial_1W=\cdots=\partial_NW=0
\]
is isolated at the origin.  Its Jacobi ring is
\begin{equation}
  \operatorname{Jac}(W)
  =
  \mathbb C[X_1,\ldots,X_N]/
  (\partial_1W,\ldots,\partial_NW).
  \label{eq:lg-jacobi-ring}
\end{equation}
\end{definition}

The \(U(1)\) charge of a monomial
\[
  X^a=X_1^{a_1}\cdots X_N^{a_N}
\]
is
\begin{equation}
  Q_R(X^a)=\sum_{i=1}^N a_iq_i.
  \label{eq:lg-monomial-r-charge}
\end{equation}
If the polynomial is realized by a unitary compact \(\mathcal N=2\)
superconformal fixed point with these fields as chiral primaries, the
chiral-primary relation gives
\begin{equation}
  h(X^a)=\frac12 Q_R(X^a).
  \label{eq:lg-chiral-weight-from-charge}
\end{equation}
The central charge predicted by the \(U(1)_R\) anomaly is
\begin{equation}
  c_{\mathrm{LG}}
  =
  3\sum_{i=1}^N (1-2q_i).
  \label{eq:lg-central-charge}
\end{equation}
In this chapter \eqref{eq:lg-central-charge} is used only as a protected
chiral-algebra consistency check.  The assertion that a given ultraviolet
Landau--Ginzburg functional flows to a particular continuum SCFT requires
the regulator and RG construction requested in the Volume~VII
supersymmetric-QFT program.

\begin{example}[\(A\)-series algebra]
\label{ex:a-series-lg-algebra}
For
\[
  W=X^{k+2},
  \qquad k\in\mathbb Z_{\ge0},
\]
the quasihomogeneous weight is \(q=1/(k+2)\).  The Jacobi ring is
\begin{equation}
  \operatorname{Jac}(W)
  =
  \mathbb C[X]/(X^{k+1}),
  \label{eq:a-series-jacobi-ring}
\end{equation}
with basis \(1,X,\ldots,X^k\).  The central charge predicted by
\eqref{eq:lg-central-charge} is
\begin{equation}
  c=\frac{3k}{k+2},
  \label{eq:a-series-lg-central-charge}
\end{equation}
and the basis element \(X^\ell\), \(0\le\ell\le k\), has
\begin{equation}
  q_\ell=\frac{\ell}{k+2},
  \qquad
  h_\ell=\frac{\ell}{2(k+2)}
  \label{eq:a-series-lg-chiral-weights}
\end{equation}
when represented by a chiral primary.
\end{example}

\begin{example}[Quintic algebraic check]
\label{ex:quintic-lg-algebraic-check}
For
\[
  W=X_1^5+\cdots+X_5^5
\]
with all weights \(q_i=1/5\), the anomaly formula gives
\begin{equation}
  c=3\cdot 5\left(1-\frac25\right)=9.
  \label{eq:quintic-lg-central-charge}
\end{equation}
This is an algebraic \(U(1)_R\)-anomaly calculation.  Interpreting it as a

\end{example}

\section{Supersymmetric Rank-One Coset Interfaces}
\label{sec:supersymmetric-rank-one-coset-interfaces}

The rank-one cosets of
Section~\ref{subsec:rank-one-solvable-cosets} have supersymmetric
refinements which are too important to leave as informal shorthand.  This section
records the exact chiral-algebraic interface needed by the CFT volume.  It
does not assert a construction of a UV Landau--Ginzburg or GLSM flow.  The
latter requires the regulator, supersymmetric Ward identities, and
infrared-comparison topology of Volume~VII.  The point here is narrower and
sharper: once the supersymmetric WZW coset exists as an \(\mathcal N=2\)
chiral algebra, its central charge, primary weights, \(R\)-charges, spectral
flow, and basic field-identification arithmetic are fixed.

We use the following conventions.  A supersymmetric \(SU(2)_k\) WZW model,
with \(k\in\mathbb Z_{\ge2}\), is represented by a bosonic
\(\widehat{\mathfrak{su}}(2)_{k-2}\) current algebra with currents
\(k^a(z)\), \(a=1,2,3\), together with three free Majorana fermions
\(\chi^a(z)\).  The total Cartan current is denoted \(K^3\).  The
supersymmetric \(SL(2,\mathbb R)_k\) model, with \(k>0\), is represented by
a bosonic \(\widehat{\mathfrak{sl}}(2,\mathbb R)_{k+2}\) current algebra with
currents \(j^a(z)\), three free fermions \(\psi^a(z)\) with the
noncompact-signature bilinear form, and total Cartan current \(J^3\).  In
both cases the coset removes the \(\mathcal N=1\) abelian sector generated
by the Cartan fermion and total Cartan current.  This is an operator
statement: the displayed coset fields are required to have regular OPE with
the removed \(\mathcal N=1\) \(U(1)\) sector.

\begin{definition}[Compact supersymmetric rank-one coset]
\label{def:compact-supersymmetric-rank-one-coset}
For \(k\in\mathbb Z_{\ge2}\), the compact supersymmetric rank-one coset is
the chiral algebra obtained from the supersymmetric \(SU(2)_k\) model by
decoupling the \(\mathcal N=1\) \(U(1)\) sector generated by
\(\chi^3\) and \(K^3\).  Put
\[
  k^\pm=k^1\pm\ii k^2,
  \qquad
  \chi^\pm=\frac{\chi^1\pm\ii\chi^2}{\sqrt2}.
\]
In the standard free-fermion realization the \(\mathcal N=2\) generators of
the coset are
\begin{equation}
  G^\pm
  =
  \sqrt{\frac2k}\,k^\mp\chi^\pm,
  \qquad
  J_R
  =
  -\frac2k k^3+\frac{k-2}{k}\chi^+\chi^-
  =
  -\frac2k K^3+\chi^+\chi^- .
  \label{eq:compact-susy-coset-n2-generators}
\end{equation}
\end{definition}

\begin{proposition}[Compact coset central charge and primary data]
\label{prop:compact-susy-coset-primary-data}
The compact coset of
Definition~\ref{def:compact-supersymmetric-rank-one-coset} has
\begin{equation}
  c_{\mathrm{comp}}(k)=\frac{3(k-2)}{k}.
  \label{eq:compact-susy-coset-central-charge}
\end{equation}
Let \(V^{su}_{j,m,\bar m}\) be a diagonal bosonic
\(\widehat{\mathfrak{su}}(2)_{k-2}\) primary, where
\[
  j\in\left\{0,\frac12,\ldots,\frac{k-2}{2}\right\},
  \qquad
  m,\bar m\in\{-j,-j+1,\ldots,j\}.
\]
Bosonize the total Cartan current by
\[
  K^3\simeq \ii\sqrt{\frac k2}\,\partial Y
\]
and factor the Cartan vertex by
\begin{equation}
  V^{su}_{j,m,\bar m}
  \simeq
  \exp\left[
    \ii\sqrt{\frac2k}\,(mY+\bar m\,\bar Y)
  \right]
  \Psi^{su}_{j,m,\bar m}.
  \label{eq:compact-susy-coset-factorization}
\end{equation}
Then the coset primary \(\Psi^{su}_{j,m,\bar m}\) has
\begin{equation}
  h=\frac{j(j+1)-m^2}{k},
  \qquad
  q=-\frac{2m}{k},
  \qquad
  \bar h=\frac{j(j+1)-\bar m^2}{k},
  \qquad
  \bar q=-\frac{2\bar m}{k}.
  \label{eq:compact-susy-coset-unflowed-hq}
\end{equation}
\end{proposition}

\begin{proof}
The central-charge calculation is a useful place to keep the level shifts
visible.  The bosonic \(SU(2)_{k-2}\) algebra has central charge
\[
  \frac{3(k-2)}{(k-2)+2}=\frac{3(k-2)}{k}.
\]
The three free fermions contribute \(3/2\).  The removed abelian
\(\mathcal N=1\) sector contributes \(1+1/2=3/2\).  Hence the coset central
charge is \(\frac{3(k-2)}{k}+3/2-3/2\), which is
\eqref{eq:compact-susy-coset-central-charge}.

For the weights, the bosonic \(SU(2)_{k-2}\) primary of spin \(j\) has
Sugawara weight \(j(j+1)/k\), while the Cartan factor in
\eqref{eq:compact-susy-coset-factorization} has holomorphic weight
\(m^2/k\).  Subtracting the Cartan part gives the displayed \(h\), and the
antiholomorphic sector gives \(\bar h=(j(j+1)-\bar m^2)/k\) with \(\bar m\)
in place of \(m\).  The \(R\)-charge follows by taking the
simple pole of \(J_R\) with the coset factor after the \(K^3\)-vertex has
been removed.  With the normalization above, this pole is
\(-2m/k\) times the field; the antiholomorphic formula is identical.
\end{proof}

The compact coset is the \(\mathcal N=2\) minimal model whose usual
minimal-model level is
\[
  K=k-2 .
\]
Thus \(c_{\mathrm{comp}}=3K/(K+2)\).  The chiral-primary representatives in
the convention \eqref{eq:compact-susy-coset-unflowed-hq} are obtained by
setting \(m=-j\).  Then
\[
  h=\frac{j}{k},
  \qquad
  q=\frac{2j}{k},
  \qquad
  h=\frac q2 .
\]
Writing \(\ell=2j\), with \(0\le\ell\le K\), gives
\[
  q_\ell=\frac{\ell}{K+2},
  \qquad
  h_\ell=\frac{\ell}{2(K+2)}.
\]
These are exactly the \(A\)-series algebraic values in
Example~\ref{ex:a-series-lg-algebra}, after identifying
\(W=X^{K+2}\).  The equality of protected chiral data is a consistency
interface.  A proof that a chosen UV Landau--Ginzburg regulator flows to this
coset requires control of the corresponding RG trajectory.

\begin{proposition}[Compact coset chiral ring]
\label{prop:compact-susy-coset-chiral-ring}
Let \(K=k-2\).  For \(0\le \ell\le K\), put
\[
  \Phi_\ell=\Psi^{su}_{\ell/2,-\ell/2}
\]
for the holomorphic compact-coset chiral primary in the NS sector.  In the
protected chiral-primary fusion ring,
\begin{equation}
  \Phi_a\Phi_b=
  \begin{cases}
    \Phi_{a+b}, & a+b\le K,\\
    0, & a+b>K,
  \end{cases}
  \qquad
  0\le a,b\le K,
  \label{eq:compact-susy-coset-chiral-ring-product}
\end{equation}
after normalizing \(\Phi_0=1\) and the generator \(\Phi_1\).  Hence
\begin{equation}
  \mathcal R_{\rm ch}\bigl(SU(2)_k/U(1)\bigr)
  \cong
  \mathbb C[x]/(x^{K+1}),
  \qquad
  x^\ell\longleftrightarrow\Phi_\ell .
  \label{eq:compact-susy-coset-chiral-ring}
\end{equation}
Here the \(K=0\) endpoint means \(\mathbb C[x]/(x)\cong \mathbb C\), and the
normalization of \(\Phi_1\) applies only when \(K\ge1\).
\end{proposition}

\begin{proof}
The fields \(\Phi_\ell\) are chiral because
\[
  h_\ell=\frac{\ell}{2k},
  \qquad
  q_\ell=\frac{\ell}{k},
  \qquad
  h_\ell=q_\ell/2 .
\]
The OPE of two NS chiral primaries has no negative powers in the chiral
channel.  Indeed any term of \(R\)-charge \(q_a+q_b\) has, by the unitarity
bound, weight at least \((q_a+q_b)/2=h_a+h_b\).  Thus the chiral-primary
projection of the product has additive \(R\)-charge.

When \(a+b\le K\), the parent \(\widehat{\mathfrak{su}}(2)_{k-2}\) fusion of
spins \(a/2\) and \(b/2\) contains the top spin \((a+b)/2\) with
multiplicity one.  The Cartan charges add from \(-a/2\) and \(-b/2\) to
\(-(a+b)/2\), so the coset branching contains precisely the chiral primary
\(\Phi_{a+b}\) in the BPS channel.  The one-dimensionality of this channel
allows the displayed normalization.

If \(a+b>K\), then the additive charge would be \((a+b)/k>K/k=c/3\).  The
compact coset has no chiral primary with that charge: the complete chiral
list is \(0\le \ell\le K\).  Equivalently, the affine fusion truncation
removes the would-be top spin before it can appear as a chiral coset branch.
The BPS projection of the product is therefore zero.  The resulting
associative algebra is the truncated polynomial algebra
\(\mathbb C[x]/(x^{K+1})\).  This proves the equality of the compact-coset
protected chiral ring with the \(A\)-series Jacobi algebra as a chiral-data
statement, while leaving the dynamical LG-to-coset flow theorem to
Volume~VII.
\end{proof}

The compact coset has independent holomorphic and antiholomorphic spectral
flow operations.  Applying \eqref{eq:n2-spectral-flow-hq} with
\eqref{eq:compact-susy-coset-central-charge} to
\eqref{eq:compact-susy-coset-unflowed-hq} gives the flowed labels
\begin{align}
  h^{(\eta)}
  &=
  \frac{j(j+1)-(m+\eta)^2}{k}+\frac{\eta^2}{2},
  &
  q^{(\eta)}
  &=
  -\frac{2(m+\eta)}{k}+\eta,
  \label{eq:compact-susy-coset-flowed-hq}\\
  \bar h^{(\bar\eta)}
  &=
  \frac{j(j+1)-(\bar m+\bar\eta)^2}{k}
  +\frac{\bar\eta^2}{2},
  &
  \bar q^{(\bar\eta)}
  &=
  -\frac{2(\bar m+\bar\eta)}{k}+\bar\eta,
  \nonumber
\end{align}
where \(\eta,\bar\eta\in\frac12\mathbb Z\).  The compact simple-current
identification becomes, for the corresponding spectral-flowed fields,
\begin{equation}
  \Psi^{su,(\eta,\bar\eta)}_{j,m,\bar m}
  \simeq
  \Psi^{su,(\eta-1,\bar\eta-1)}
       _{\frac{k-2}{2}-j,\,
         m-\frac{k-2}{2},\,\bar m-\frac{k-2}{2}}
  \simeq
  \Psi^{su,(\eta-1,\bar\eta+1)}
       _{\frac{k-2}{2}-j,\,
         m-\frac{k-2}{2},\,\bar m+\frac{k-2}{2}} .
  \label{eq:compact-susy-coset-field-identification}
\end{equation}
The full diagonal compact theory is obtained only after the left and right
representations are paired.  In the conventional diagonal pairing one
imposes
\begin{equation}
  m+\eta-\bar m-\bar\eta\in k\mathbb Z,
  \qquad
  \eta-\bar\eta\in\mathbb Z .
  \label{eq:compact-susy-coset-diagonal-condition}
\end{equation}
The remaining angular symmetry of the disc sigma-model representative is a
\(\mathbb Z_k\) symmetry acting by
\begin{equation}
  \Psi^{su,(\eta,\bar\eta)}_{j,m,\bar m}
  \longmapsto
  \exp\left[
    \frac{2\pi\ii}{k}(m+\eta+\bar m+\bar\eta)
  \right]
  \Psi^{su,(\eta,\bar\eta)}_{j,m,\bar m}.
  \label{eq:compact-susy-coset-zk-action}
\end{equation}
This last statement is a warning against a common shortcut: the disc metric
is not the exact CFT data.  The finite-level CFT remembers the simple-current
identification and the \(\mathbb Z_k\) selection rule.

\begin{definition}[Noncompact supersymmetric rank-one coset]
\label{def:noncompact-supersymmetric-rank-one-coset}
For \(k>0\), the noncompact supersymmetric rank-one coset is obtained from
the supersymmetric \(SL(2,\mathbb R)_k\) model by decoupling the
\(\mathcal N=1\) abelian sector generated by \(\psi^3\) and \(J^3\).  Put
\[
  j^\pm=j^1\pm\ii j^2,
  \qquad
  \psi^\pm=\frac{\psi^1\pm\ii\psi^2}{\sqrt2}.
\]
In the standard realization its \(\mathcal N=2\) generators are
\begin{equation}
  G^\pm
  =
  \sqrt{\frac2k}\,j^\mp\psi^\pm,
  \qquad
  J_R
  =
  \frac2k j^3+\frac{k+2}{k}\psi^+\psi^-
  =
  \frac2k J^3+\psi^+\psi^- .
  \label{eq:noncompact-susy-coset-n2-generators}
\end{equation}
\end{definition}

\begin{proposition}[Cigar central charge and primary data]
\label{prop:noncompact-susy-coset-primary-data}
The noncompact coset of
Definition~\ref{def:noncompact-supersymmetric-rank-one-coset} has
\begin{equation}
  c_{\mathrm{cig}}(k)=\frac{3(k+2)}{k}.
  \label{eq:noncompact-susy-coset-central-charge}
\end{equation}
Let \(V^{sl}_{j,m,\bar m}\) be a bosonic
\(\widehat{\mathfrak{sl}}(2,\mathbb R)_{k+2}\) primary, with \(j\) the
standard \(SL(2,\mathbb R)\) spin parameter and \(m,\bar m\) the total
Cartan eigenvalues.  Bosonize the total Cartan current by
\[
  J^3\simeq-\sqrt{\frac k2}\,\partial X
\]
and factor
\begin{equation}
  V^{sl}_{j,m,\bar m}
  \simeq
  \exp\left[
    \sqrt{\frac2k}\,(mX+\bar m\,\bar X)
  \right]
  \Psi^{sl}_{j,m,\bar m}.
  \label{eq:noncompact-susy-coset-factorization}
\end{equation}
Then
\begin{equation}
  h=\frac{-j(j-1)+m^2}{k},
  \qquad
  q=\frac{2m}{k},
  \qquad
  \bar h=\frac{-j(j-1)+\bar m^2}{k},
  \qquad
  \bar q=\frac{2\bar m}{k}.
  \label{eq:noncompact-susy-coset-unflowed-hq}
\end{equation}
\end{proposition}

\begin{proof}
The bosonic \(SL(2,\mathbb R)_{k+2}\) current algebra has central charge
\(3(k+2)/k\) in the convention of
Section~\ref{subsec:rank-one-solvable-cosets}.  Adding three fermions and
removing the abelian \(\mathcal N=1\) Cartan sector changes the central
charge by \(+3/2-3/2\), so the coset central charge is
\eqref{eq:noncompact-susy-coset-central-charge}.  The bosonic primary has
Sugawara weight \(-j(j-1)/k\).  The Cartan boson in
\eqref{eq:noncompact-susy-coset-factorization} is timelike with respect to
the noncompact bilinear form, so removing the Cartan factor adds \(m^2/k\)
to the coset weight.  The \(R\)-charge is the simple-pole coefficient of
the \(J_R\) OPE after the \(J^3\)-vertex has been factored out, giving
\(2m/k\).  The antiholomorphic sector uses \(\bar m\) in the same
factorization and gives
\(\bar h=(-j(j-1)+\bar m^2)/k\) and \(\bar q=2\bar m/k\).
\end{proof}

The noncompact spectrum is not determined by
\eqref{eq:noncompact-susy-coset-unflowed-hq} alone.  In the conventional
cigar theory the continuous series has
\[
  j=\frac12+\ii s,
  \qquad
  s\in\mathbb R,
  \qquad
  m-\bar m\in\mathbb Z,
\]
while discrete normalizable states are residues in an allowed strip such as
\[
  \frac12<j<\frac{k+1}{2},
  \qquad
  m,\bar m\in j+\mathbb Z_{\ge0}
  \quad\text{or}\quad
  m,\bar m\in -j-\mathbb Z_{\ge0},
\]
together with the winding condition
\begin{equation}
  m+\bar m\in k\mathbb Z .
  \label{eq:noncompact-susy-coset-unflowed-winding}
\end{equation}
The exact reflection coefficient, its poles, and the spectral density are
analytic CFT data.  They should not be replaced by the cigar metric.

Spectral flow gives
\begin{align}
  h^{(\eta)}
  &=
  \frac{-j(j-1)+(m+\eta)^2}{k}+\frac{\eta^2}{2},
  &
  q^{(\eta)}
  &=
  \frac{2(m+\eta)}{k}+\eta,
  \label{eq:noncompact-susy-coset-flowed-hq}\\
  \bar h^{(\bar\eta)}
  &=
  \frac{-j(j-1)+(\bar m+\bar\eta)^2}{k}
  +\frac{\bar\eta^2}{2},
  &
  \bar q^{(\bar\eta)}
  &=
  \frac{2(\bar m+\bar\eta)}{k}+\bar\eta .
  \nonumber
\end{align}
The momentum and winding labels in the asymptotic cigar circle are shifted
to
\begin{equation}
  n=m+\eta-\bar m-\bar\eta,
  \qquad
  w=\frac{m+\eta+\bar m+\bar\eta}{k},
  \label{eq:noncompact-susy-coset-momentum-winding}
\end{equation}
so the winding condition becomes
\begin{equation}
  m+\eta+\bar m+\bar\eta\in k\mathbb Z .
  \label{eq:noncompact-susy-coset-flowed-winding}
\end{equation}
For discrete states one extends the range by the identification
\begin{equation}
  \Psi^{sl,(\eta,\bar\eta)}_{j,m,\bar m}
  \simeq
  \Psi^{sl,(\eta+1,\bar\eta+1)}
       _{\frac{k+2}{2}-j,\,
         m-\frac{k+2}{2},\,\bar m-\frac{k+2}{2}} .
  \label{eq:noncompact-susy-coset-field-identification}
\end{equation}
There is no compact-coset analogue shifting \(\eta\) and \(\bar\eta\) with
opposite signs.  The reason is structural: the cigar has an exact circle
translation quantum number \(n\), whereas the compact disc representative has
only the finite \(\mathbb Z_k\) remnant in
\eqref{eq:compact-susy-coset-zk-action}.

\begin{remark}[Spectral-flow derivation of the rank-one formulas]
\label{rem:supersymmetric-rank-one-spectral-flow-check}
Equations~\eqref{eq:compact-susy-coset-flowed-hq} and
\eqref{eq:noncompact-susy-coset-flowed-hq} follow from the general
\(\mathcal N=2\) spectral-flow law
\eqref{eq:n2-spectral-flow-hq} and the unflowed data
\eqref{eq:compact-susy-coset-unflowed-hq},
\eqref{eq:noncompact-susy-coset-unflowed-hq}.
For the compact coset,
\[
  h+\eta q+\frac{c_{\mathrm{comp}}\eta^2}{6}
  =
  \frac{j(j+1)-m^2}{k}
  -\frac{2\eta m}{k}
  +\frac{k-2}{2k}\eta^2 .
\]
Since
\[
  -\frac{(m+\eta)^2}{k}+\frac{\eta^2}{2}
  =
  -\frac{m^2+2m\eta}{k}
  +\frac{k-2}{2k}\eta^2,
\]
this is the first formula in
\eqref{eq:compact-susy-coset-flowed-hq}.  The charge identity is
\[
  q+\frac{c_{\mathrm{comp}}\eta}{3}
  =
  -\frac{2m}{k}+\frac{k-2}{k}\eta
  =
  -\frac{2(m+\eta)}{k}+\eta .
\]
The barred compact formulas are identical.  For the noncompact coset,
\[
  h+\eta q+\frac{c_{\mathrm{cig}}\eta^2}{6}
  =
  \frac{-j(j-1)+m^2}{k}
  +\frac{2\eta m}{k}
  +\frac{k+2}{2k}\eta^2 ,
\]
and
\[
  \frac{(m+\eta)^2}{k}+\frac{\eta^2}{2}
  =
  \frac{m^2+2m\eta}{k}
  +\frac{k+2}{2k}\eta^2.
\]
The charge formula follows from
\[
  q+\frac{c_{\mathrm{cig}}\eta}{3}
  =
  \frac{2m}{k}+\frac{k+2}{k}\eta
  =
  \frac{2(m+\eta)}{k}+\eta .
\]
\end{remark}

\section{Small \texorpdfstring{\(\mathcal N=4\)}{N=4} Boundary}

For completeness, we record the boundary between what is used in this
monograph and the full small \(\mathcal N=4\) representation theory.  The
small \(\mathcal N=4\) algebra contains an affine \(\mathfrak{su}(2)\) current
algebra and four supercurrents.  It includes \(\mathcal N=2\) subalgebras
after choosing a Cartan \(U(1)\subset SU(2)_R\), but the representation
theory involves the full \(SU(2)_R\) spin.

\begin{definition}[Small \(\mathcal N=4\) chiral algebra]
\label{def:small-n4-chiral-algebra}
A small \(\mathcal N=4\) chiral algebra at level \(k\in\mathbb Z_{\ge0}\)
consists of:
\begin{enumerate}
\item a stress tensor \(T(z)\) of central charge \(c=6k\);
\item affine \(\mathfrak{su}(2)\) currents \(J^a(z)\), \(a=1,2,3\), with
level \(k\);
\item four weight-\(3/2\) supercurrents transforming as two doublets under
\(\mathfrak{su}(2)\);
\item OPEs such that the anticommutator of opposite supercurrents closes on
\(T\), \(J^a\), and the central term, while the current OPEs act on the
supercurrents by the doublet representation.
\end{enumerate}
\end{definition}

Definition~\ref{def:small-n4-chiral-algebra} records only the part of the
small-\(\mathcal N=4\) algebra used in this volume.  The present monograph
uses two consequences: the central charge is tied to the affine level by
\(c=6k\), and every Cartan choice supplies an \(\mathcal N=2\) subalgebra
whose spectral-flow operation is inherited from the affine \(SU(2)_R\)
structure.  A later chapter that needs small-\(\mathcal N=4\) multiplet
classification must state the full OPE tensor and derive the corresponding
unitarity bounds.

\end{openproblem}
