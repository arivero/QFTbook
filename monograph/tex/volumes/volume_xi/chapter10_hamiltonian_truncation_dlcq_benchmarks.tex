\chapter{Hamiltonian Truncation, DLCQ, And Benchmark Protocols}
\label{ch:constructive-hamiltonian-truncation-dlcq-benchmarks}

\ConventionsLorentzian

Hamiltonian numerical methods are regulator frameworks for QFT.  This
chapter formulates Hamiltonian truncation, light-front discretization, and
benchmark scripts as controlled approximations with explicitly declared
Hilbert spaces, cutoffs, counterterms, and convergence tests.

\section{Projected Hamiltonian Datum}

Let \(H\) be a self-adjoint Hamiltonian on a Hilbert space \(\Hilb\), bounded
below.  Choose finite-dimensional subspaces
\[
  \Hilb_\Lambda\subset\Hilb,\qquad P_\Lambda:\Hilb\to\Hilb_\Lambda,
\]
with \(P_\Lambda\to I\) strongly.  The projected Hamiltonian is
\[
  H_\Lambda=P_\Lambda H P_\Lambda
  \]
on \(\Hilb_\Lambda\).  If \(H\) is defined by a QFT regulator, \(H_\Lambda\)
must be supplemented by local counterterms:
\[
  H_\Lambda^{\mathrm{ren}}
  =
  P_\Lambda H P_\Lambda
  +
  \sum_a c_a(\Lambda)P_\Lambda\mathcal O_aP_\Lambda.
\]
The pair \((\Hilb_\Lambda,H_\Lambda^{\mathrm{ren}})\) is the numerical
regulator datum.

\section{Rayleigh--Ritz Bound}

\begin{proposition}[Rayleigh--Ritz bound at fixed Hamiltonian]
\label{prop:rayleigh-ritz-fixed-hamiltonian}
Let \(H\) be a self-adjoint operator on \(\Hilb\), bounded below, and suppose
that
\[
  E_0\le E_1\le\cdots\le E_n
\]
are isolated eigenvalues below the bottom of the essential spectrum, counted
with multiplicity.  Let \(\mathcal Q_H\) be the closed quadratic form of
\(H\).  For a finite-dimensional subspace
\(\Hilb_\Lambda\subset\operatorname{Dom}\mathcal Q_H\), define
\[
  E_n(\Lambda)
  =
  \min_{\substack{W\subset\Hilb_\Lambda\\ \dim W=n+1}}
  \max_{\substack{\psi\in W\\ \|\psi\|=1}}\mathcal Q_H(\psi,\psi).
\]
Then \(E_n(\Lambda)\ge E_n\).  If \(\{\Hilb_\Lambda\}\) is an increasing net
and, for some orthonormal eigenbasis
\(\varphi_0,\ldots,\varphi_n\) of the first \(n+1\) eigenspaces, there are
\(\eta_j^\Lambda\in\Hilb_\Lambda\) with
\[
  \eta_j^\Lambda\to\varphi_j
  \quad\text{in the form norm}\quad
  \|\psi\|_{\mathcal Q}^2
  =
  \|\psi\|^2+\mathcal Q_H(\psi,\psi)-c\|\psi\|^2
\]
for a constant \(c<\inf\operatorname{spec}H\), then
\(E_n(\Lambda)\downarrow E_n\).
\end{proposition}

\begin{proof}
The min--max characterization of the \(n\)-th discrete eigenvalue gives
\[
  E_n
  =
  \inf_{\substack{W\subset\operatorname{Dom}\mathcal Q_H\\ \dim W=n+1}}
  \sup_{\substack{\psi\in W\\ \|\psi\|=1}}\mathcal Q_H(\psi,\psi).
\]
The variational problem defining \(E_n(\Lambda)\) restricts the same infimum
to the smaller family of subspaces contained in \(\Hilb_\Lambda\), hence
\(E_n(\Lambda)\ge E_n\).  By the approximation hypothesis, Gram--Schmidt in
the ordinary Hilbert norm produces an \((n+1)\)-dimensional subspace
\(W_\Lambda\subset\Hilb_\Lambda\) whose unit sphere converges in form norm to
the unit sphere of
\(\operatorname{span}\{\varphi_0,\ldots,\varphi_n\}\).  The Gram--Schmidt
coefficients are rational functions of inner products whose limiting
denominators are nonzero, so the form-norm convergence is preserved.  The
supremum of \(\mathcal Q_H\) on that unit sphere therefore converges to
\(E_n\).
The upper bound \(E_n(\Lambda)\le \sup_{\psi\in W_\Lambda,\|\psi\|=1}
\mathcal Q_H(\psi,\psi)\) gives \(\limsup_\Lambda E_n(\Lambda)\le E_n\).
Together with the lower bound this proves convergence.  Monotonic decrease
follows because increasing \(\Hilb_\Lambda\) enlarges the family of trial
subspaces.
\end{proof}

This proposition concerns a fixed self-adjoint operator \(H\).  It does not
assert monotonicity for a family \(H_\Lambda^{\mathrm{ren}}\) whose
counterterms change with \(\Lambda\), nor for a sequence of finite matrices
obtained from different regulators.  Those comparisons require their own
cutoff-error estimates.

\begin{proposition}[Gap-stable Ritz counting for a fixed Hamiltonian]
\label{prop:gap-stable-ritz-counting-fixed-hamiltonian}
Let \(H\) and \(\mathcal Q_H\) be as in
Proposition~\ref{prop:rayleigh-ritz-fixed-hamiltonian}.  Let
\(\{\Hilb_\Lambda\}\) be an increasing net of finite-dimensional subspaces of
\(\operatorname{Dom}\mathcal Q_H\) which is dense in the form norm.  Let
\[
  E_0\leq E_1\leq\cdots
\]
be the discrete eigenvalues of \(H\) below the bottom of the essential
spectrum, counted with multiplicity, and let
\[
  E_0(\Lambda)\leq E_1(\Lambda)\leq\cdots
       \leq E_{\dim\Hilb_\Lambda-1}(\Lambda)
\]
be the Ritz eigenvalues obtained by restricting \(\mathcal Q_H\) to
\(\Hilb_\Lambda\).  Suppose \(a<b<\inf\operatorname{spec}_{\rm ess}H\) and
that \(a,b\notin\operatorname{spec}H\).  Define
\[
  N_H(c)=\#\{j:E_j<c\},
  \qquad c\in\{a,b\},
\]
and \(m=N_H(b)-N_H(a)\).  Then, for all sufficiently large \(\Lambda\),
exactly \(m\) Ritz eigenvalues lie in the interval \((a,b)\).
\end{proposition}

\begin{proof}
Since \(b\) lies below the essential spectrum and is not an eigenvalue, there
are only finitely many eigenvalues below \(b\).  Put
\[
  M=N_H(b),\qquad r=N_H(a).
\]
For each \(j<M\), the form-norm density of \(\bigcup_\Lambda\Hilb_\Lambda\)
implies the approximation hypothesis in
Proposition~\ref{prop:rayleigh-ritz-fixed-hamiltonian} for the first
\(j+1\) eigenspaces; hence
\[
  E_j(\Lambda)\downarrow E_j .
  \label{eq:fixed-H-ritz-eigenvalue-convergence}
\]
If \(j<r\), then \(E_j<a\), and
\eqref{eq:fixed-H-ritz-eigenvalue-convergence} gives
\[
  E_j(\Lambda)<a
\]
for all sufficiently large \(\Lambda\).  If \(r\leq j<M\), then
\[
  a<E_j<b .
\]
The lower inequality \(E_j(\Lambda)\geq E_j>a\) follows from
Proposition~\ref{prop:rayleigh-ritz-fixed-hamiltonian}; convergence gives
\[
  E_j(\Lambda)<b
\]
for all sufficiently large \(\Lambda\).  Thus, eventually, at least
\(M-r=m\) Ritz values lie in \((a,b)\), namely those with labels
\[
  r,\ldots,M-1 .
\]
It remains to exclude extra Ritz values below \(b\).  If
\(E_M(\Lambda)<b\) for some \(\Lambda\), then the min--max definition of
\(E_M(\Lambda)\) supplies an \((M+1)\)-dimensional subspace of
\(\operatorname{Dom}\mathcal Q_H\) on whose unit sphere the quadratic form is
strictly less than \(b\).  Applying the min--max characterization in the full
form domain would give \(E_M<b\), contradicting \(M=N_H(b)\).  Therefore no
Ritz value with label \(j\geq M\) can lie below \(b\).  Combining the two
statements gives exactly \(m\) Ritz values in \((a,b)\) for all sufficiently
large \(\Lambda\).
\end{proof}

The proposition is a theorem about Galerkin approximation to one fixed
self-adjoint Hamiltonian and one fixed quadratic form.  Its hypotheses are
visible: a lower-bounded self-adjoint operator, an isolated finite spectral
window, and form-norm density of the trial spaces.  It does not justify a
counterterm-dependent truncation unless one has first proved that the
renormalized finite matrices converge to a fixed limiting quadratic form in a
sense strong enough to replace the form-norm density assumption.

\section{Truncated Conformal Space}

For a two-dimensional QFT built as a relevant deformation of a CFT, choose the
CFT Hilbert space on a circle and keep states satisfying
\[
  L_0+\bar L_0-\frac{c}{12}\le E_{\mathrm{cut}}\,\frac{L}{2\pi}.
\]
The Hamiltonian matrix elements are
\[
  \langle i|H|j\rangle
  =
  \frac{2\pi}{L}
  \left(\Delta_i-\frac{c}{12}\right)\delta_{ij}
  +
  \sum_a\lambda_a
  \left(\frac{L}{2\pi}\right)^{1-\Delta_a}
  \langle i|\mathcal O_a(1)|j\rangle.
\]
High-energy states generate cutoff dependence.  The local Hamiltonian
counterterms are determined by matching low-energy observables as
\(E_{\mathrm{cut}}\) is varied.

\section{Exactly Solvable TCSA Benchmark: Ising Energy Deformation}

A useful benchmark must distinguish the finite Hamiltonian matrix from the
continuum theory it approximates.  The thermal deformation of the two
dimensional Ising CFT is special because the deformed theory is a free massive
Majorana fermion.  On a circle of circumference \(L\), in the
Neveu--Schwarz sector, positive modes are labelled by
\[
  r\in \mathbb Z_{\geq0}+\frac12,
  \qquad
  p_r=\frac{2\pi r}{L}.
\]
For each \(r\), the finite-mode Hamiltonian reduces to the real symmetric
Bogoliubov block
\[
  H_r=
  \begin{pmatrix}
    -p_r&m\\
    m&p_r
  \end{pmatrix},
\]
where \(m\) is the Majorana mass induced by the energy-operator coupling in a
fixed normalization.  The corresponding finite truncation with \(N\) modes is
the direct sum
\[
  H_N=\bigoplus_{r=1/2}^{N-1/2} H_r .
\]

\begin{remark}[Finite-mode Majorana benchmark]
\label{rem:ising-energy-tcsa-benchmark}
The eigenvalues of \(H_N\) are
\[
  \left\{
    \pm\sqrt{p_r^2+m^2}\ :\
    r=\frac12,\frac32,\ldots,N-\frac12
  \right\}.
\]
For a fixed mode \(r\),
\[
  \det(H_r-\lambda I)
  =
  (-p_r-\lambda)(p_r-\lambda)-m^2
  =
  \lambda^2-(p_r^2+m^2).
\]
Thus the two eigenvalues of \(H_r\) are
\(\lambda=\pm\sqrt{p_r^2+m^2}\).  The matrix \(H_N\) is a direct sum of the
blocks \(H_r\), so its characteristic polynomial is the product of the block
characteristic polynomials.  The benchmark is therefore exact at every finite
cutoff \(N\); any numerical deviation from these values measures an
implementation or diagonalization error rather than a truncation effect.
\end{remark}

The script
\[
  \texttt{qft\_scripts/tcsa\_ising\_energy\_benchmark.py}
\]
diagonalizes \(H_N\) and checks Remark
\ref{rem:ising-energy-tcsa-benchmark}.  Its role is narrow and rigorous: it
is a public finite-matrix benchmark for Hamiltonian-truncation conventions
before nonintegrable perturbations, counterterm fitting, and cutoff
extrapolation are introduced.  The magnetic-field Ising TCSA problem and the
\(E_8\) mass-spectrum computation require additional sectors and perturbing
operators.

\section{Ising TCSA And TFFSA With Spin Deformation}

The two-dimensional Ising field theory with both thermal and magnetic
deformations is the basic test case in which Hamiltonian truncation is useful
and conceptually delicate.  Start from the critical Ising CFT on a spatial
circle of circumference \(L\).  Let \(H_{\rm CFT}(L)\) be the self-adjoint
Hamiltonian generated by cylinder time translations,
\[
  H_{\rm CFT}(L)
  =
  \frac{2\pi}{L}
  \left(L_0+\bar L_0-\frac{c}{12}\right),
  \qquad c=\frac12 .
\]
Let \(\varepsilon\) be the energy operator of scaling dimension
\(\Delta_\varepsilon=1\), and let \(\sigma\) be the spin operator of scaling
dimension \(\Delta_\sigma=1/8\).  A TCSA regulator for the two-coupling Ising
family is the finite matrix
\begin{equation}
  H_{\rm TCSA}(L;E_{\rm cut})
  =
  P_{E_{\rm cut}}
  \left[
    H_{\rm CFT}(L)
    +
    \tau\int_0^L\varepsilon_{\rm cyl}(0,x)\,\dd x
    +
    h\int_0^L\sigma_{\rm cyl}(0,x)\,\dd x
  \right]
  P_{E_{\rm cut}}
  +
  H_{\rm ct}(E_{\rm cut}).
  \label{eq:ising-two-coupling-tcsa-hamiltonian}
\end{equation}
Here \(P_{E_{\rm cut}}\) projects onto CFT states satisfying
\((2\pi/L)(L_0+\bar L_0-c/12)\le E_{\rm cut}\).  The symbol
\(\mathcal O_{\rm cyl}\) means the cylinder field obtained from the plane
field by the conformal map.  After fixing a plane normalization, the
finite-volume matrix elements contain the factors \((2\pi/L)^{\Delta_a}\)
dictated by the state-operator correspondence.  The couplings have
engineering dimensions
\[
  [\tau]=1,\qquad [h]=\frac{15}{8}.
\]
The counterterm matrix \(H_{\rm ct}(E_{\rm cut})\) is part of the regulator.
It is determined by a declared matching prescription; omitting it is a choice
of truncation scheme, not a theorem that no high-energy effects exist.

The same physical family admits a second truncation.  Treat the thermal
deformation exactly.  For \(\tau\ne0\), the unperturbed theory is the free
massive Majorana theory with mass \(m\) proportional to \(\tau\) in the chosen
normalization.  The magnetic perturbation is then
\[
  H_{\rm IFT}(L)
  =
  H_{\rm Maj}(m,L)
  +
  h\int_0^L\sigma(0,x)\,\dd x .
  \label{eq:ising-tffsa-hamiltonian}
\]
Truncating \eqref{eq:ising-tffsa-hamiltonian} in the finite-volume fermion
Fock space is the truncated free-fermion-space approach.  Its advantage is
not a new definition of the continuum theory; rather, it moves the relevant
thermal deformation from the perturbation into the basis.  This can reduce
cutoff sensitivity when the low-energy states are better approximated by
massive fermion states than by a small number of conformal states.

We now spell out the finite matrix.  Let \(\mathfrak s\) denote a fermion
spin-structure sector on the circle, with allowed momenta
\[
  p_n=\frac{2\pi n}{L},\qquad n\in Q_{\mathfrak s},
\]
where \(Q_{\mathfrak s}\) is either \(\mathbb Z+\frac12\) or
\(\mathbb Z\), depending on the sector.  A finite-volume rapidity
\(\theta_n\) is defined by
\[
  p_n=m\sinh\theta_n,\qquad
  \omega_n=m\cosh\theta_n .
\]
For a finite set \(I=\{n_1,\ldots,n_k\}\subset Q_{\mathfrak s}\), write
\[
  \ket{I}_{\mathfrak s}
  =
  a_{n_1}^\dagger\cdots a_{n_k}^\dagger\ket{0}_{\mathfrak s},
  \qquad
  E_I^{(0)}(L)=E_{\mathfrak s}^{\rm vac}(L)+\sum_{n\in I}\omega_n,
  \qquad
  P_I=\sum_{n\in I}p_n .
\]
The TFFSA cutoff is an energy cutoff in this massive basis:
\[
  \Hilb_{\rm FF}(L,E_{\rm cut})
  =
  \operatorname{span}\{\ket{I}_{\mathfrak s}:E_I^{(0)}(L)\le E_{\rm cut}\}.
\]
The diagonal part of \(H_{\rm Maj}(m,L)\) is exact in this basis.  The
nontrivial matrix is the spin perturbation.

\begin{definition}[TFFSA spin matrix element]
\label{def:tffsa-spin-matrix-element}
For two finite-volume fermion states \(I,J\), the connected infinite-volume
asymptotic spin matrix element is
\[
  V_{IJ}^{\rm conn}
  =
  hL\,\delta_{P_I,P_J}\,
  \frac{
    F_{\rm conn}^{\sigma}(\{\theta_i:i\in I\}\mid \{\theta_j:j\in J\})
  }{
    \sqrt{\rho_I\rho_J}
  },
  \qquad
  \rho_I=\prod_{i\in I}mL\cosh\theta_i .
  \label{eq:tffsa-connected-spin-matrix-element}
\]
Here \(F_{\rm conn}^{\sigma}\) is the crossed spin-field form factor derived
in Chapter~\ref{ch:integrable-form-factor-bootstrap-local-operators}, and
\(\rho_I\) is the free Bethe-state density of states.  The full finite-volume
matrix element also contains disconnected contraction terms when \(I\) and
\(J\) share rapidities, and exponentially small finite-size corrections of
order \(e^{-mL}\) in a massive theory.  A numerical TFFSA scheme must specify
whether it uses exact finite-volume Ising form factors, the asymptotic formula
\eqref{eq:tffsa-connected-spin-matrix-element}, or a hybrid prescription.
\end{definition}

This definition makes the status of the computation explicit.  At fixed
\((L,E_{\rm cut})\), after all matrix elements have been declared, TFFSA is
ordinary finite-dimensional spectral theory.  The physical assertions concern
the removal or extrapolation of the cutoff and, if
\eqref{eq:tffsa-connected-spin-matrix-element} is used, the large-\(mL\)
control of the finite-volume form-factor approximation.

\begin{definition}[Zero-momentum connected TFFSA block]
\label{def:zero-momentum-connected-tffsa-block}
Fix \(m>0\), \(L>0\), an integer \(N_{\rm pair}\geq0\), a spin-field
normalization \(\bar\sigma\), and a magnetic coupling \(h\).  For
\[
  r_n=n+\frac12,\qquad
  p_n=\frac{2\pi r_n}{L},\qquad
  p_n=m\sinh\theta_n,\qquad
  \omega_n=m\cosh\theta_n ,
  \quad n=0,\ldots,N_{\rm pair}-1,
\]
let
\[
  \ket{0},\qquad
  \ket{n}:=\ket{\theta_n,-\theta_n},\quad
  n=0,\ldots,N_{\rm pair}-1,
\]
be the finite zero-total-momentum basis consisting of the vacuum and
two-particle pairs.  Set
\[
  \rho_n=(mL\cosh\theta_n)^2 .
\]
The connected-block TFFSA matrix is the Hermitian
\((N_{\rm pair}+1)\times(N_{\rm pair}+1)\) matrix
\[
  H_{ij}=E_i^{(0)}\delta_{ij}+hL\,V_{ij},
\]
where \(E_0^{(0)}=0\), \(E_n^{(0)}=2\omega_n\) for \(n\geq1\), and
\[
  V_{ii}=\bar\sigma .
\]
For \(i<j\), \(V_{ij}\) is defined by the crossed connected form factor
\[
  V_{ij}
  =
  \frac{
    F_{\rm conn}^{\sigma}(
      \{\theta_a+\ii\pi:a\in I_i\}\mid \{\theta_b:b\in I_j\})
  }{
    \sqrt{\rho_i\rho_j}
  },
  \qquad
  V_{ji}=\overline{V_{ij}},
  \label{eq:zero-momentum-tffsa-block}
\]
with \(\rho_0=1\) and \(I_0=\varnothing\).  The diagonal convention
\(V_{ii}=\bar\sigma\) keeps only the disconnected vacuum-expectation
contribution.  The omitted diagonal connected finite parts are singular
crossed limits and require a separate diagonal finite-volume prescription.
\end{definition}

\begin{remark}[Connected TFFSA block]
\label{rem:zero-momentum-connected-tffsa-hermitian}
The matrix in Definition~\ref{def:zero-momentum-connected-tffsa-block} is
Hermitian.  If \(h=0\), its eigenvalues are
\[
  0,\quad 2\omega_0,\quad 2\omega_1,\quad\ldots,\quad
  2\omega_{N_{\rm pair}-1}.
\]
The diagonal entries \(E_i^{(0)}+hL\bar\sigma\) are real.  For \(i<j\) the
lower-triangular entry is defined to be \(\overline{V_{ij}}\), hence
\[
  H_{ji}=hL\,\overline{V_{ij}}=\overline{hL\,V_{ij}}=\overline{H_{ij}}
\]
for real \(h\) and \(L\).  Thus \(H^\dagger=H\).  At \(h=0\), all
off-diagonal entries vanish and the matrix is diagonal with the displayed
free finite-volume energies.
\end{remark}

The script
\[
  \texttt{qft\_scripts/tffsa\_ising\_spin\_connected.py}
\]
implements Definition~\ref{def:zero-momentum-connected-tffsa-block}.  It
checks Hermiticity in smoke mode and prints the finite eigenvalues together
with the corresponding free energies.  The purpose is to make the
form-factor normalization, density factors, and diagonal convention
executable.  It is not a claim that the small connected block is the full
magnetic-Ising spectrum.

\begin{proposition}[Finite TFFSA spectral-flow derivative]
\label{prop:finite-tffsa-hellmann-feynman}
Let
\[
  H(h)=H_0+hW,\qquad H_0=H_0^\dagger,\quad W=W^\dagger ,
  \label{eq:finite-affine-hermitian-flow}
\]
be a finite Hermitian matrix family, with \(h\in\mathbb R\).  Suppose
\(\lambda_*(h_*)\) is a simple eigenvalue of \(H(h_*)\), and let \(u_*\) be a
normalized eigenvector.  If
\[
  g=\operatorname{dist}\bigl(\lambda_*(h_*),
      \operatorname{spec}H(h_*)\setminus\{\lambda_*(h_*)\}\bigr)>0 ,
  \label{eq:finite-spectral-flow-gap}
\]
then for \(|h-h_*|\|W\|<g/2\) there is a unique eigenvalue branch
\(\lambda_*(h)\) near \(\lambda_*(h_*)\), counted with its normalized
eigenvector \(u(h)\) after fixing a differentiable phase.  Its derivative at
\(h=h_*\) is
\[
  \frac{\dd\lambda_*}{\dd h}(h_*)
  =
  u_*^\dagger W u_* .
  \label{eq:finite-hellmann-feynman-derivative}
\]
If all eigenvalues of \(H(h_*)\) are simple and
\(\lambda_a'(h_*)=u_a^\dagger W u_a\), then
\[
  \sum_a\lambda_a'(h_*)=\operatorname{tr} W .
  \label{eq:finite-spectral-flow-trace-identity}
\]
\end{proposition}

\begin{proof}
Choose a positively oriented circle \(C\) centered at \(\lambda_*(h_*)\) with
radius smaller than \(g/2\).  For \(z\in C\),
\[
  \|(h-h_*)W(H(h_*)-z)^{-1}\|
  \leq
  |h-h_*|\|W\|\,\|(H(h_*)-z)^{-1}\|
  <
  1 ,
\]
when \(|h-h_*|\|W\|<g/2\).  Thus the resolvent identity and the Neumann series
give a well-defined resolvent \((H(h)-z)^{-1}\) on \(C\), analytic in \(h\).
The Riesz projection
\[
  P(h)=\frac{1}{2\pi\ii}\oint_C (z-H(h))^{-1}\,\dd z
\]
therefore has rank one for \(h\) in this neighborhood and depends
differentiably on \(h\).  Taking \(u(h)=P(h)u_*/\|P(h)u_*\|\), with the phase
chosen so that \(u_*^\dagger u(h)\) is positive real near \(h_*\), gives a
differentiable normalized eigenvector for the unique eigenvalue branch inside
\(C\).

Differentiate
\[
  H(h)u(h)=\lambda_*(h)u(h)
\]
at \(h=h_*\).  Since \(H'(h)=W\),
\[
  Wu_*+H(h_*)u'(h_*)
  =
  \lambda_*'(h_*)u_*+\lambda_*(h_*)u'(h_*) .
\]
Taking the inner product with \(u_*\), and using
\(u_*^\dagger H(h_*)=\lambda_*(h_*)u_*^\dagger\), cancels the terms
containing \(u'(h_*)\) and gives
\eqref{eq:finite-hellmann-feynman-derivative}.  If all eigenvalues are simple,
their normalized eigenvectors form an orthonormal basis, so
\[
  \sum_a\lambda_a'(h_*)
  =
  \sum_a u_a^\dagger W u_a
  =
  \operatorname{tr}W .
\]
\end{proof}

For the connected Ising TFFSA block,
\eqref{eq:finite-affine-hermitian-flow} is exactly the finite matrix
\[
  H(h)=H_0+hL\,V .
\]
The companion script
\[
  \texttt{qft\_scripts/tffsa\_ising\_spectral\_flow.py}
\]
diagonalizes this finite block on a small grid of \(h\)-values and checks
\eqref{eq:finite-hellmann-feynman-derivative} against centered finite
differences at a point where the finite spectrum is separated.  This is a
useful nonintegrable benchmark because it checks the actual finite spectral
flow generated by spin-field form factors.  It remains a finite-matrix
certificate: comparison with the continuum magnetic-Ising spectrum requires
the cutoff, finite-volume, diagonal-form-factor, and counterterm estimates
described around \eqref{eq:tffsa-connected-spin-matrix-element}.

\begin{proposition}[Schur complement form of truncation error]
\label{prop:hamiltonian-truncation-schur-complement}
In a finite regulated Hilbert space, let \(H=H^\dagger\) be a Hermitian
matrix and let \(\Hilb=P\Hilb\oplus Q\Hilb\), where \(P\) is an orthogonal
projection and \(Q=1-P\).  Suppose \(E\) is not an eigenvalue of the
finite matrix \(QHQ:Q\Hilb\to Q\Hilb\).  Then \(E\) is an eigenvalue of
\(H\) with \(P\psi\ne0\) only if
\[
  \left[
    PHP
    -
    PHQ(QHQ-E)^{-1}QHP
  \right]P\psi
  =
  E\,P\psi .
  \label{eq:feshbach-schur-truncation-effective-hamiltonian}
\]
Conversely, if \(u\in P\Hilb\) satisfies
\eqref{eq:feshbach-schur-truncation-effective-hamiltonian}, then
\[
  \psi
  =
  u
  -
  (QHQ-E)^{-1}QHPu
\]
is an eigenvector of \(H\) with eigenvalue \(E\).
\end{proposition}

\begin{proof}
Write the eigenvalue equation \(H\psi=E\psi\) as two block equations:
\[
  (PHP-E)P\psi+PHQ\,Q\psi=0,
  \qquad
  QHP\,P\psi+(QHQ-E)Q\psi=0 .
\]
Since \(E\) is in the resolvent set of \(QHQ\), the second equation gives
\[
  Q\psi=-(QHQ-E)^{-1}QHP\,P\psi .
\]
Substituting this into the first equation gives
\eqref{eq:feshbach-schur-truncation-effective-hamiltonian}.  Conversely,
suppose \(u\in P\Hilb\) obeys the Schur-complement equation and define
\[
  \psi=u-(QHQ-E)^{-1}QHPu .
\]
Then \(P\psi=u\) and the second block equation is satisfied by construction,
because \((QHQ-E)(QHQ-E)^{-1}=Q\) on \(Q\Hilb\).  Substituting the
Schur-complement equation into the first block equation gives zero there as
well.  Hence \(H\psi=E\psi\).
\end{proof}

Equation~\eqref{eq:feshbach-schur-truncation-effective-hamiltonian} is the
precise operator identity behind cutoff counterterms in TCSA and TFFSA.  In
practice one expands the \(Q\)-space resolvent at high energy and replaces its
low-energy action by local or quasilocal operators allowed by the symmetries.
For the Ising spin perturbation this expansion is governed by the short-
distance operator product of \(\sigma\) with itself; for the free-fermion
basis it is equivalently encoded in the high-rapidity behavior of the spin
form factors.  A claimed numerical spectrum is therefore meaningful only
after the truncation scheme states the cutoff, the finite-volume form-factor
prescription, the symmetry sector, the counterterm coordinates, and the
extrapolation model.

\section{Finite-Matrix Spectral Certificates}
\label{sec:finite-matrix-spectral-certificates}

Every Hamiltonian-truncation or DLCQ computation ultimately diagonalizes a
finite Hermitian matrix.  This elementary fact is useful only if the finite
linear-algebra output is certified separately from the continuum claim.  We
therefore record the minimal certificate used by the companion scripts.

\begin{definition}[Residual certificate for a finite truncation]
\label{def:finite-truncation-residual-certificate}
Let \(A=A^\dagger\) be a finite Hermitian matrix on a finite-dimensional
Hilbert space \(V\).  A residual certificate for a proposed eigenpair is a
triple
\[
  (\lambda,u,\rho),\qquad \|u\|=1,\qquad
  \rho=\|(A-\lambda)u\|.
  \label{eq:finite-matrix-residual-certificate}
\]
It is a certificate for the finite matrix \(A\) only.  The number
\(\lambda\) becomes a QFT energy or invariant mass only after the regulator
label, counterterm prescription, symmetry sector, finite-volume convention,
and continuum-extrapolation hypothesis have also been supplied.
\end{definition}

\begin{proposition}[Spectral meaning of a finite residual]
\label{prop:finite-residual-spectral-certificate}
Let \(A=A^\dagger\) be a finite Hermitian matrix, let
\(\|u\|=1\), and set \(\rho=\|(A-\lambda)u\|\).
Then
\[
  \operatorname{dist}(\lambda,\operatorname{spec}A)\leq \rho .
  \label{eq:residual-distance-to-spectrum}
\]
More generally, let \(S\subset\operatorname{spec}A\) and let \(P_S\) be the
orthogonal spectral projection onto the span of eigenvectors whose
eigenvalues lie in \(S\).  If
\[
  g=\operatorname{dist}\bigl(\lambda,\operatorname{spec}A\setminus S\bigr)>0,
  \label{eq:finite-spectral-gap-from-complement}
\]
then
\[
  \|(1-P_S)u\|\leq \frac{\rho}{g}.
  \label{eq:finite-projector-leakage-bound}
\]
\end{proposition}

\begin{proof}
Choose an orthonormal eigenbasis \(e_a\) of \(A\),
\[
  Ae_a=E_a e_a,\qquad u=\sum_a c_a e_a,\qquad \sum_a |c_a|^2=1.
\]
Then
\[
  \rho^2
  =
  \|(A-\lambda)u\|^2
  =
  \sum_a |c_a|^2 |E_a-\lambda|^2 .
  \label{eq:residual-spectral-expansion}
\]
If every eigenvalue obeyed \(|E_a-\lambda|>\rho\), the right hand side of
\eqref{eq:residual-spectral-expansion} would be strictly larger than
\(\rho^2\), a contradiction.  This proves
\eqref{eq:residual-distance-to-spectrum}.  For the projection bound, split
the sum into \(E_a\in S\) and \(E_a\notin S\).  On the complement,
\(|E_a-\lambda|\geq g\), hence
\[
  \rho^2
  \geq
  g^2\sum_{E_a\notin S}|c_a|^2
  =
  g^2\|(1-P_S)u\|^2 .
\]
Taking square roots gives
\eqref{eq:finite-projector-leakage-bound}.
\end{proof}

The first inequality says that a small residual forces the existence of some
finite-matrix eigenvalue near \(\lambda\).  The second says that, when a
target spectral cluster is separated from the rest of the finite spectrum,
the approximate eigenvector has quantitatively small leakage out of that
cluster.  These statements do not estimate the distance between different
cutoffs.  Cutoff comparison belongs to the Feshbach identity above, to
counterterm matching, and to the extrapolation bounds below.

For a Hermitian block decomposition \(V=PV\oplus QV\), the same finite
linear algebra gives a practical diagnostic for a truncation counterterm.
At a chosen low-energy value \(E\), the omitted-space correction is exactly
\[
  \Sigma_P(E)=PHQ(QHQ-E)^{-1}QHP .
  \label{eq:finite-schur-self-energy}
\]
Approximating \(\Sigma_P(E)\) by a low-dimensional local counterterm is a
separate approximation; the identity itself is exact only before this
replacement is made.  A controlled truncation must therefore state a norm,
an energy window, and a bound on
\[
  \Sigma_P(E)-\Sigma_P^{\rm ct}(E)
  \label{eq:schur-counterterm-remainder}
\]
in that window.

The companion check
\[
  \texttt{calculation-checks/hamiltonian\_truncation\_dlcq\_checks.py}
\]
verifies the finite residual bound
\eqref{eq:residual-distance-to-spectrum}, the projector leakage estimate
\eqref{eq:finite-projector-leakage-bound}, the Feshbach determinant identity
behind \eqref{eq:feshbach-schur-truncation-effective-hamiltonian}, the
Ising-energy Bogoliubov benchmark, the connected Ising TFFSA block
normalization, the finite Ising TFFSA spectral-flow derivative identity, the
finite large-\(N\) QCD DLCQ quadratic-form identity, and the exact-intercept
cases of the finite large-\(K\) extrapolation diagnostic.

\section{DLCQ Datum}

Light-front coordinates are
\[
  x^\pm=\frac1{\sqrt2}(x^0\pm x^1).
\]
DLCQ compactifies
\[
  x^-\sim x^-+2\pi R
\]
so that
\[
  p^+=\frac{n}{R},\qquad n\in\mathbb Z_{\ge0}.
\]
The total longitudinal momentum is
\[
  P^+=\frac{K}{R},
\]
where \(K\) is the harmonic resolution.  The finite DLCQ Hilbert space at
fixed \(K\) consists of partitions of \(K\) among partons, with transverse and
internal cutoffs added as needed.  Zero modes with \(n=0\) are constrained
variables or additional degrees of freedom depending on the theory; their
treatment is part of the regulator.

\section{\texorpdfstring{Large-\(N\) Two-Dimensional QCD: A Finite DLCQ Matrix}{Large-N Two-Dimensional QCD: A Finite DLCQ Matrix}}

The standard large-\(N\) two-dimensional QCD meson equation involves the
principal-value operator
\[
  (\mathcal K\phi)(x)
  =
  \operatorname{PV}\int_0^1
  \frac{\phi(x)-\phi(y)}{(x-y)^2}\,\dd y ,
  \qquad 0<x<1,
\]
together with endpoint mass terms
\[
  \frac{m_1^2}{x}\phi(x)+\frac{m_2^2}{1-x}\phi(x).
\]
This section records a finite regulator, not a continuum theorem.  Fix a
harmonic resolution \(K\geq3\) and grid points \(x_n=n/K\),
\(n=1,\ldots,K-1\).  For \(\gamma>0\), define the real symmetric matrix
\[
  (M_K^2)_{nm}
  =
  \delta_{nm}
  \left[
    \frac{m_1^2}{x_n}
    +
    \frac{m_2^2}{1-x_n}
    +
    \gamma K\sum_{\substack{j=1\\ j\neq n}}^{K-1}
      \frac1{(n-j)^2}
  \right]
  -
  (1-\delta_{nm})\frac{\gamma K}{(n-m)^2}.
\]
The normalization of \(\gamma\) depends on the convention for
\(g^2N/\pi\).  The finite object above is sufficient to illustrate harmonic
resolution, positivity, and extrapolation obligations.

\begin{proposition}[Positivity of the finite principal-value matrix]
\label{prop:thooft-dlcq-finite-positivity}
For \(m_1^2,m_2^2,\gamma>0\) and any
\(v=(v_1,\ldots,v_{K-1})\in\mathbb R^{K-1}\),
\[
  v^T M_K^2 v
  =
  \sum_{n=1}^{K-1}
  \left(
    \frac{m_1^2}{x_n}
    +
    \frac{m_2^2}{1-x_n}
  \right)v_n^2
  +
  \gamma K
  \sum_{1\leq n<m\leq K-1}
  \frac{(v_n-v_m)^2}{(n-m)^2}.
\]
In particular \(M_K^2\) is positive definite.
\end{proposition}

\begin{proof}
Substitute the definition of \(M_K^2\) into \(v^TM_K^2v\).  The mass terms
give the first displayed sum.  The off-diagonal kernel gives
\[
  \gamma K
  \sum_{n=1}^{K-1}
  \sum_{\substack{m=1\\m\neq n}}^{K-1}
  \frac{v_n^2-v_nv_m}{(n-m)^2}.
\]
Pair the term \((n,m)\) with \((m,n)\).  For each unordered pair
\(\{n,m\}\),
\[
  \frac{v_n^2-v_nv_m}{(n-m)^2}
  +
  \frac{v_m^2-v_mv_n}{(m-n)^2}
  =
  \frac{(v_n-v_m)^2}{(n-m)^2}.
\]
This gives the stated quadratic form.  Since \(0<x_n<1\), the mass
coefficients are positive; hence \(v^TM_K^2v>0\) for \(v\neq0\).
\end{proof}

The script
\[
  \texttt{qft\_scripts/thooft\_dlcq.py}
\]
constructs this matrix and diagonalizes it at finite \(K\).  A continuum
claim would require an analysis of the \(K\to\infty\) limit, endpoint
behavior, zero modes, normalization of \(\gamma\), and comparison with a
Lorentz-covariant reconstruction.  The script therefore prints finite
eigenvalues and a quadratic-form positivity check rather than presenting the
numbers as physical meson masses.

\section{Finite-Regulator Extrapolation}
\label{sec:finite-regulator-extrapolation}

The output of a lattice, Hamiltonian-truncation, or DLCQ script is a number
attached to a regulator.  A continuum claim is a separate mathematical
statement about a limiting procedure.  This section records the minimal
finite-dimensional facts used in the companion scripts and explains what an
extrapolation hypothesis must contain.

\begin{definition}[Finite-regulator observable datum]
\label{def:finite-regulator-observable-datum}
A finite-regulator observable datum consists of:
\[
  \bigl(I,\preceq;\ X_i,\ A_i,\ \sigma_i,\ \mathcal E_i\bigr) .
\]
Here \((I,\preceq)\) is a directed set of regulator labels, \(X_i\) is the
finite object being computed at label \(i\), \(A_i\in\mathbb R\) is the
finite observable extracted from \(X_i\), \(\sigma_i\ge0\) is the statistical
standard error if \(A_i\) is estimated by a random algorithm, and
\(\mathcal E_i\) is the declared systematic error model.  For deterministic
TCSA or DLCQ diagonalization one has \(\sigma_i=0\), but \(\mathcal E_i\) is
not zero: it contains cutoff, finite-volume, counterterm, endpoint, or
zero-mode errors.  For a Markov-chain computation, \(A_i\) itself is an
estimator and \(\sigma_i\) must include autocorrelation.
\end{definition}

\begin{remark}[Interpolation ambiguity of finite cutoff data]
\label{rem:finite-data-do-not-determine-continuum-limit}
Let \(K_1,\ldots,K_N\) be distinct positive regulator values and let
\(A_1,\ldots,A_N\in\mathbb R\) be finite outputs.  For every proposed
continuum value \(A_\infty\in\mathbb R\), there is a polynomial \(p(t)\) of
degree at most \(N\) such that
\[
  p(0)=A_\infty,
  \qquad
  p(K_j^{-1})=A_j,\quad j=1,\ldots,N .
\]
Consequently finitely many cutoff values, by themselves, cannot prove a
continuum limit or its value.
The \(N+1\) points
\[
  0,\quad K_1^{-1},\ldots,K_N^{-1}
\]
are distinct.  Lagrange interpolation gives a unique polynomial of degree at
most \(N\) taking the prescribed values \(A_\infty,A_1,\ldots,A_N\) at those
points.  Since \(A_\infty\) was arbitrary, the same finite data are
compatible with arbitrarily many candidate limits unless additional
information is supplied.  The obstruction is the missing analytic ansatz,
monotonicity estimate, remainder bound, or convergence theorem that would
restrict the allowed interpolating functions near \(K^{-1}=0\).
\end{remark}

\begin{definition}[Power-law cutoff expansion with a remainder bound]
\label{def:power-law-cutoff-expansion}
Let \(K\) be a positive regulator parameter tending to infinity.  A
deterministic sequence \(A_K\) has an \(m\)-term power-law cutoff expansion
with base exponent \(\omega>0\) and controlled remainder if there are real
numbers \(A_\infty,c_1,\ldots,c_m\), a constant \(C\), and \(K_0\) such that
\[
  \left|
    A_K
    -
    A_\infty
    -
    \sum_{a=1}^m c_a K^{-a\omega}
  \right|
  \le
  C K^{-(m+1)\omega},
  \qquad K\ge K_0 .
  \label{eq:power-law-cutoff-remainder-bound}
\]
The same definition applies to a subsequence, for example
\(K,Ks_1,\ldots,Ks_m\), provided these are admissible regulator labels and
the displayed bound holds on that subsequence.  In a TCSA computation the
exponent \(\omega\) and the counterterm coordinates should be derived from
the high-energy tail or OPE; in a DLCQ computation they should be derived
from endpoint and zero-mode analysis.  A fit without
\eqref{eq:power-law-cutoff-remainder-bound} is a diagnostic, not a theorem.
\end{definition}

\begin{proposition}[Two-cutoff Richardson cancellation]
\label{prop:two-cutoff-richardson-cancellation}
Assume
\[
  A_K=A_\infty+cK^{-\omega}+r_K,
  \qquad
  |r_K|\le C K^{-\eta},
  \qquad
  \eta>\omega>0 ,
\]
for \(K\ge K_0\).  Define
\[
  R_K
  =
  \frac{2^\omega A_{2K}-A_K}{2^\omega-1}.
  \label{eq:two-cutoff-richardson-estimator}
\]
Then
\[
  |R_K-A_\infty|
  \le
  \frac{2^\omega 2^{-\eta}+1}{2^\omega-1}\,
  C K^{-\eta}.
  \label{eq:two-cutoff-richardson-bound}
\]
\end{proposition}

\begin{proof}
Substitute the assumed expansion into
\eqref{eq:two-cutoff-richardson-estimator}:
\[
  2^\omega A_{2K}-A_K
  =
  (2^\omega-1)A_\infty
  +
  2^\omega c(2K)^{-\omega}
  -
  cK^{-\omega}
  +
  2^\omega r_{2K}-r_K .
\]
The two terms proportional to \(c\) cancel because
\(2^\omega(2K)^{-\omega}=K^{-\omega}\).  Dividing by \(2^\omega-1\) gives
\[
  R_K-A_\infty
  =
  \frac{2^\omega r_{2K}-r_K}{2^\omega-1}.
\]
Using \(|r_{2K}|\le C(2K)^{-\eta}\) gives
\eqref{eq:two-cutoff-richardson-bound}.
\end{proof}

\begin{proposition}[Integer-power extrapolation with explicit weights]
\label{prop:integer-power-extrapolation-weights}
Let \(s_0,\ldots,s_m\) be distinct positive real numbers and suppose
\(A_K\) satisfies \eqref{eq:power-law-cutoff-remainder-bound} on the
subsequence \(s_jK\), \(j=0,\ldots,m\).  Let \(w_0,\ldots,w_m\) be the unique
solution of
\[
  \sum_{j=0}^m w_j=1,
  \qquad
  \sum_{j=0}^m w_j s_j^{-a\omega}=0,
  \quad a=1,\ldots,m .
  \label{eq:extrapolation-weight-moments}
\]
Then the extrapolated value
\[
  R_K^{(m)}
  =
  \sum_{j=0}^m w_j A_{s_jK}
\]
satisfies
\[
  |R_K^{(m)}-A_\infty|
  \le
  C K^{-(m+1)\omega}
  \sum_{j=0}^m |w_j|\,s_j^{-(m+1)\omega}.
  \label{eq:integer-power-extrapolation-bound}
\]
\end{proposition}

\begin{proof}
Set \(q_j=s_j^{-\omega}\).  The numbers \(q_0,\ldots,q_m\) are distinct,
so the Vandermonde matrix \((q_j^a)_{0\le a,j\le m}\) is invertible.  Hence
\eqref{eq:extrapolation-weight-moments} has a unique solution.  Insert
\eqref{eq:power-law-cutoff-remainder-bound} into \(R_K^{(m)}\).  The
coefficient of \(A_\infty\) is \(\sum_jw_j=1\).  For
\(a=1,\ldots,m\), the coefficient of \(c_aK^{-a\omega}\) is
\(\sum_jw_js_j^{-a\omega}=0\).  The remaining error is bounded by the
triangle inequality:
\[
  \left|\sum_{j=0}^m w_j r_{s_jK}\right|
  \le
  \sum_{j=0}^m |w_j|\,C(s_jK)^{-(m+1)\omega}.
\]
This is \eqref{eq:integer-power-extrapolation-bound}.
\end{proof}

\paragraph{\texorpdfstring{Large-\(K\) diagnostic for the finite 't Hooft matrices}{Large-K diagnostic for the finite 't Hooft matrices}.}
For the finite matrices \(M_K^2\) in
Proposition~\ref{prop:thooft-dlcq-finite-positivity}, write their eigenvalues
in nondecreasing order as
\[
  \lambda_0(K)\leq\lambda_1(K)\leq\cdots\leq\lambda_{K-2}(K).
\]
For a fixed label \(\ell\), the numerical sequence
\[
  A_K=\lambda_\ell(K)
\]
is a finite-regulator sequence in the sense of
Definition~\ref{def:finite-regulator-observable-datum}.  Keeping the label
\(\ell\) fixed is only a diagnostic convention until one proves that the
corresponding continuum state is isolated and that nearby finite-\(K\) levels
do not interchange in a way that changes the limiting object.  Choose distinct
resolutions \(K_1<\cdots<K_N\), a base exponent \(\omega>0\), and an integer
\(p<N\).  Set
\[
  q_r=K_r^{-\omega},\qquad
  X_{ra}=q_r^a,\qquad
  r=1,\ldots,N,\quad a=0,\ldots,p .
\]
The least-squares polynomial fit is
\[
  \widehat c=(X^T X)^{-1}X^T A,\qquad
  A=(A_{K_1},\ldots,A_{K_N})^T ,
  \label{eq:dlcq-large-K-least-squares-fit}
\]
provided \(X\) has full column rank.  The reported continuum diagnostic is
\(\widehat c_0\), the fitted intercept at \(q=0\).

\begin{proposition}[Finite large-\(K\) fit with an explicit remainder amplifier]
\label{prop:finite-large-K-fit-remainder-amplifier}
Assume that the finite data used in
\eqref{eq:dlcq-large-K-least-squares-fit} have the form
\[
  A_{K_r}
  =
  A_\infty+\sum_{a=1}^p c_a K_r^{-a\omega}+\rho_r ,
  \qquad r=1,\ldots,N ,
  \label{eq:large-K-fit-model-with-remainder}
\]
with \(X\) of full column rank.  Let
\[
  B=(X^T X)^{-1}X^T .
\]
Then the fitted intercept satisfies
\[
  |\widehat c_0-A_\infty|
  \leq
  \left(\sum_{r=1}^N |B_{0r}|\right)\max_{1\leq r\leq N}|\rho_r| .
  \label{eq:large-K-intercept-remainder-amplification}
\]
In particular, if all \(\rho_r\) vanish, the fit recovers \(A_\infty\)
exactly, independently of the overdetermination \(N>p+1\).
\end{proposition}

\begin{proof}
Let
\[
  c=(A_\infty,c_1,\ldots,c_p)^T,
  \qquad
  \rho=(\rho_1,\ldots,\rho_N)^T .
\]
Equation~\eqref{eq:large-K-fit-model-with-remainder} is the vector identity
\[
  A=Xc+\rho .
\]
Since \(X\) has full column rank, \(BX=I_{p+1}\).  Hence
\[
  \widehat c
  =
  BA
  =
  B(Xc+\rho)
  =
  c+B\rho .
\]
Taking the zeroth component gives
\[
  \widehat c_0-A_\infty
  =
  \sum_{r=1}^N B_{0r}\rho_r .
\]
The triangle inequality gives
\eqref{eq:large-K-intercept-remainder-amplification}.  If \(\rho=0\), then
\(\widehat c=c\) and the intercept is exact.
\end{proof}

of \(X\), and the factor
\(\sum_r |B_{0r}|\) appearing in
\eqref{eq:large-K-intercept-remainder-amplification}.  These quantities
measure how a chosen finite data set responds to a chosen ansatz.  They do not
prove that the large-\(K\) limit of the large-\(N\) QCD meson equation exists
or that the selected exponent \(\omega\) is correct.  Those conclusions require
separate endpoint and zero-mode estimates producing a bound of the form
\eqref{eq:large-K-fit-model-with-remainder}.

\section{Light-Front Hamiltonian}

The invariant mass operator is
\[
  M^2=2P^+P^- -P_\perp^2.
\]
At fixed \(P^+\) and \(P_\perp\), diagonalizing \(P^-\) is equivalent to
diagonalizing \(M^2\).  Gauge theories require solving constraint equations
for nondynamical fields and preserving residual gauge constraints.  A
light-front truncation without the constraint algebra and zero-mode sector is
not a definition of the gauge theory.
