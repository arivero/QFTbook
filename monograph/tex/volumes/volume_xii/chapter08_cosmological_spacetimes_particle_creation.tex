\chapter{Cosmological Spacetimes and Particle Creation}
\label{ch:curved-cosmological-spacetimes-particle-creation}

\ConventionsLorentzian

This chapter studies free scalar QFT on time-dependent globally hyperbolic
backgrounds.  The central point is not that ``particles are ambiguous'' as a
slogan, but the precise mathematical statement: a particle interpretation is
a choice of positive-frequency subspace, equivalently a compatible complex
structure on the real symplectic space of solutions.  When the geometry has
static asymptotic regions, the in and out complex structures are selected by
the asymptotic time translations and their mismatch is measured by
Bogoliubov coefficients.  When no such asymptotic regions exist, one uses
state properties such as the Hadamard condition, adiabatic ultraviolet
behavior, and detector response.

\section{Spatially Flat Robertson--Walker Metrics}

Let \(M=I\times\mathbb R^{d-1}\), with \(d\geq2\), and let
\begin{equation}
  g=-dt^2+a(t)^2d\mathbf x^2
  =
  a(\eta)^2(-d\eta^2+d\mathbf x^2),
  \qquad
  d\eta=\frac{dt}{a(t)}.
  \label{eq:flrw-conformal-metric}
\end{equation}
Assume \(a(\eta)>0\) is smooth on the conformal-time interval under
consideration.  We write
\begin{equation}
  \mathcal H(\eta)=\frac{a'(\eta)}{a(\eta)}
  \label{eq:conformal-hubble-definition}
\end{equation}
for the conformal Hubble function.  The scalar curvature of
\eqref{eq:flrw-conformal-metric} is
\begin{equation}
  R
  =
  (d-1)a^{-2}\bigl(2\mathcal H'+(d-2)\mathcal H^2\bigr).
  \label{eq:flrw-scalar-curvature}
\end{equation}

Consider a real scalar field satisfying
\begin{equation}
  (\Box_g-m^2-\xi R)\phi=0.
  \label{eq:curved-scalar-klein-gordon}
\end{equation}
The Fourier convention is
\begin{equation}
  \phi(\eta,\mathbf x)
  =
  \int\frac{d^{d-1}\mathbf k}{(2\pi)^{d-1}}\,
  \phi_{\mathbf k}(\eta)e^{\ii\mathbf k\cdot\mathbf x},
  \qquad
  k=|\mathbf k|.
  \label{eq:flrw-fourier-convention}
\end{equation}

\begin{remark}[Robertson--Walker mode equation]
\label{rem:flrw-mode-equation}
Let
\begin{equation}
  p=\frac{d-2}{2},
  \qquad
  \chi_{\mathbf k}(\eta)=a(\eta)^p\phi_{\mathbf k}(\eta).
  \label{eq:flrw-rescaled-mode}
\end{equation}
Then \(\chi_{\mathbf k}\) obeys
\begin{equation}
  \chi_{\mathbf k}''+\Omega_k(\eta)^2\chi_{\mathbf k}=0,
  \label{eq:flrw-chi-mode-equation}
\end{equation}
where
\begin{equation}
  \Omega_k^2
  =
  k^2+a^2(m^2+\xi R)-p\mathcal H'-p^2\mathcal H^2 .
  \label{eq:flrw-effective-frequency}
\end{equation}
For a massless conformally coupled scalar, with
\begin{equation}
  m=0,
  \qquad
  \xi=\xi_c=\frac{d-2}{4(d-1)},
  \label{eq:conformal-coupling-d-dimensions}
\end{equation}
one has \(\Omega_k^2=k^2\).
\end{remark}

For the metric \eqref{eq:flrw-conformal-metric},
\[
  \Box_g\phi
  =
  a^{-2}\left[
  -\partial_\eta^2\phi
  -(d-2)\mathcal H\,\partial_\eta\phi
  +\Delta_{\mathbf x}\phi
  \right].
\]
Substituting \eqref{eq:flrw-fourier-convention} into
\eqref{eq:curved-scalar-klein-gordon} gives
\begin{equation}
  \phi_{\mathbf k}''
  +(d-2)\mathcal H\phi_{\mathbf k}'
  +
  \bigl(k^2+a^2(m^2+\xi R)\bigr)\phi_{\mathbf k}=0.
  \label{eq:flrw-unrescaled-mode-equation}
\end{equation}
Writing \(\phi_{\mathbf k}=a^{-p}\chi_{\mathbf k}\), with
\(2p=d-2\), gives
\[
  \phi_{\mathbf k}'
  =
  a^{-p}(\chi_{\mathbf k}'-p\mathcal H\chi_{\mathbf k}),
\]
and
\[
  \phi_{\mathbf k}''
  =
  a^{-p}\left[
  \chi_{\mathbf k}''
  -2p\mathcal H\chi_{\mathbf k}'
  +(p^2\mathcal H^2-p\mathcal H')\chi_{\mathbf k}
  \right].
\]
The first-derivative terms cancel in
\eqref{eq:flrw-unrescaled-mode-equation}, leaving
\eqref{eq:flrw-chi-mode-equation} and
\eqref{eq:flrw-effective-frequency}.  Finally,
\[
  a^2\xi_c R
  =
  \frac{d-2}{4}
  \bigl(2\mathcal H'+(d-2)\mathcal H^2\bigr)
  =
  p\mathcal H'+p^2\mathcal H^2,
\]
so the curvature and rescaling terms cancel when \(m=0\) and
\(\xi=\xi_c\).

The cancellation in the conformally coupled massless case is an exact
calculation, not an adiabatic statement.  It is the local manifestation of
conformal covariance of the classical wave equation.

\section{Symplectic Space and Complex Structures}

Let \(\mathcal S\) be the real vector space of smooth solutions of
\eqref{eq:curved-scalar-klein-gordon} with compact Cauchy data.  On any
Cauchy surface \(\Sigma\), define
\begin{equation}
  \sigma(\phi_1,\phi_2)
  =
  \int_\Sigma d\Sigma^\mu\,
  \bigl(
  \phi_1\nabla_\mu\phi_2
  -
  \phi_2\nabla_\mu\phi_1
  \bigr).
  \label{eq:covariant-symplectic-form}
\end{equation}
The current inside the integral is conserved whenever both fields satisfy
the equation of motion, so \(\sigma\) is independent of \(\Sigma\).

\begin{definition}[Compatible complex structure]
\label{def:compatible-complex-structure-cosmology}
A compatible complex structure on \((\mathcal S,\sigma)\) is a real linear
map \(J:\mathcal S\to\mathcal S\) such that
\begin{equation}
  J^2=-1,
  \qquad
  \sigma(Ju,Jv)=\sigma(u,v),
  \qquad
  \sigma(u,Ju)>0
  \quad (u\neq0).
  \label{eq:compatible-complex-structure}
\end{equation}
It defines a one-particle inner product
\begin{equation}
  \langle u,v\rangle_J
  =
  \sigma(u,Jv)+\ii\sigma(u,v).
  \label{eq:one-particle-inner-product-J}
\end{equation}
Completing \((\mathcal S,\langle\cdot,\cdot\rangle_J)\) gives the
one-particle Hilbert space \(\mathcal H_J\), and the symmetric Fock space
\(\mathcal F_J=\operatorname{Sym}\mathcal H_J\) gives the associated
quasifree representation.
\end{definition}

Equivalently, after complexification, \(J\) selects the \(+\ii\)-eigenspace as
the positive-frequency subspace.  This is the precise datum behind a
particle interpretation.  Different choices of \(J\) can be unitarily
inequivalent in infinite volume.

\begin{theorem}[Shale--Stinespring criterion, diagonal discrete form]
\label{qthm:shale-stinespring-diagonal}
Let the one-particle space be a separable bosonic Hilbert space with
orthonormal mode basis indexed by a countable set \(I\), paired by an
involution \(n\mapsto\bar n\).  Suppose two quasifree complex structures are
related mode-by-mode by Bogoliubov coefficients
\begin{equation}
  u_n^{(1)}
  =
  \alpha_n u_n^{(2)}
  +
  \beta_n\overline{u_{\bar n}^{(2)}},
  \label{eq:diagonal-bogoliubov-relation}
\end{equation}
with
\[
  |\alpha_n|^2-|\beta_n|^2=1 .
\]
Then the two Fock representations are unitarily equivalent precisely when
\begin{equation}
  \sum_{n\in I_+}|\beta_n|^2<\infty ,
  \label{eq:shale-stinespring-diagonal-condition}
\end{equation}
where \(I_+\) contains one representative from each non-fixed pair
\(\{n,\bar n\}\), with the evident one-oscillator modification for fixed
points.  In that case the Bogoliubov transformation is implemented by a
unitary operator on Fock space.
\end{theorem}

\begin{proof}
For a finite set of modes the statement is the standard squeeze
construction.  On one two-mode block write
\[
  b_n=\alpha_n a_n+\beta_n a_{\bar n}^{\dagger},
  \qquad
  b_{\bar n}=\alpha_n a_{\bar n}+\beta_n a_n^\dagger ,
\]
after absorbing harmless phases into the mode vectors.  The relation
\(|\alpha_n|^2-|\beta_n|^2=1\) is exactly the condition that
\([b_n,b_n^\dagger]=[b_{\bar n},b_{\bar n}^\dagger]=1\).  Put
\[
  \gamma_n=-\frac{\beta_n}{\alpha_n}.
\]
The vector in the old Fock space annihilated by \(b_n\) and \(b_{\bar n}\)
is
\begin{equation}
  \Omega_{\gamma_n}
  =
  (1-|\gamma_n|^2)^{1/2}
  \sum_{m=0}^\infty \gamma_n^m
  |m\rangle_n\otimes |m\rangle_{\bar n},
  \label{eq:two-mode-squeezed-vacuum}
\end{equation}
because applying \(a_n\) to the \(m\)-th term and applying
\(\gamma_n a_{\bar n}^\dagger\) to the \((m-1)\)-st term cancel.  The
normalization follows from the geometric series and the identity
\[
  1-|\gamma_n|^2
  =
  \frac{|\alpha_n|^2-|\beta_n|^2}{|\alpha_n|^2}
  =
  |\alpha_n|^{-2}.
  \]
The fixed-point one-oscillator block is the usual one-mode squeezed vector
and gives the same square-summability condition, with a different harmless
normalization convention.

For an infinite countable set, the transformed vacuum is the infinite tensor
product of the block vacua \(\Omega_{\gamma_n}\).  Von Neumann's criterion
for incomplete infinite tensor products says that this tensor product defines
a nonzero vector in the reference Fock space exactly when
\[
  \sum_{n\in I_+}
  \bigl(1-|\langle 0_n0_{\bar n},\Omega_{\gamma_n}\rangle|\bigr)<\infty .
\]
Here
\[
  |\langle 0_n0_{\bar n},\Omega_{\gamma_n}\rangle|
  =
  (1-|\gamma_n|^2)^{1/2}.
\]
Since \(1-\sqrt{1-x}\) is comparable to \(x\) for \(0\le x\le1\), the
criterion is
\[
  \sum_{n\in I_+}|\gamma_n|^2<\infty .
\]
Because \(|\gamma_n|^2=|\beta_n|^2/(1+|\beta_n|^2)\), this is equivalent to
\(\sum_{n\in I_+}|\beta_n|^2<\infty\): if the \(\gamma_n\)-sum converges,
only finitely many \(|\beta_n|\) can exceed \(1\), and on the remaining tail
the two sequences are comparable.

When the sum is finite, the finite-mode squeeze implementers converge
strongly on the finite-particle domain to an isometry whose range contains
the finite-particle domain of the transformed representation; its extension
is the implementing unitary.  When the sum diverges, the candidate
transformed vacuum has zero norm in the reference incomplete tensor product,
so no unitary can implement the transformation: a unitary implementer would
have to send the transformed vacuum to a nonzero vector annihilated by all
\(b_n\), and the block equations above determine that vector uniquely up to
phase.
\end{proof}

For a field on a finite spatial box, \eqref{eq:shale-stinespring-diagonal-condition}
is the literal finite-volume Shale--Stinespring condition.  On a noncompact
homogeneous spatial slice the momentum label is continuous, and the diagonal
map
\[
  (\beta \bar f)(\mathbf k)=\beta_{\mathbf k}\,\overline{f(-\mathbf k)}
\]
is a multiplication operator, composed with parity and conjugation, on
\(L^2(d^{d-1}\mathbf k)\).  As an operator on a non-atomic \(L^2\)-space,
such a diagonal multiplication operator is Hilbert--Schmidt only when it
vanishes almost everywhere.  Thus
\[
  \int d^{d-1}\mathbf k\,|\beta_{\mathbf k}|^2<\infty
\]
is a finite particle-density or local/box-normalized diagnostic; it is not,
by itself, a statement of global unitary equivalence of the two infinite-volume
Fock representations.  This distinction is one of the main reasons the
complex-structure formulation is cleaner than informal particle language.

\section{In and Out Modes}

Assume now that the scale factor and curvature approach static limits as
\(\eta\to\pm\infty\), so that
\begin{equation}
  \Omega_k(\eta)\longrightarrow
  \omega_k^{\mathrm{in/out}}>0
  \qquad
  (\eta\to\mp\infty).
  \label{eq:in-out-frequency-limits}
\end{equation}
The in and out positive-frequency modes are normalized by the Wronskian
condition
\begin{equation}
  \chi_k\overline{\chi_k'}-\chi_k'\overline{\chi_k}=\ii
  \label{eq:mode-wronskian-normalization}
\end{equation}
and by
\begin{equation}
  \chi_k^{\mathrm{in/out}}(\eta)
  \sim
  \frac{e^{-\ii\omega_k^{\mathrm{in/out}}\eta}}
       {\sqrt{2\omega_k^{\mathrm{in/out}}}}
  \qquad
  \hbox{in the corresponding asymptotic region.}
  \label{eq:in-out-positive-frequency-asymptotics}
\end{equation}

\paragraph{Bogoliubov normalization and particle number.}
Let
\begin{equation}
  \chi_k^{\mathrm{in}}
  =
  \alpha_k\chi_k^{\mathrm{out}}
  +
  \beta_k\overline{\chi_k^{\mathrm{out}}}.
  \label{eq:cosmological-bogoliubov-expansion}
\end{equation}
Then
\begin{equation}
  |\alpha_k|^2-|\beta_k|^2=1.
  \label{eq:cosmological-bogoliubov-normalization}
\end{equation}
If \(\ket{0,\mathrm{in}}\) is the in-vacuum, the expected out-particle number
in the \(\mathbf k\)-mode is
\begin{equation}
  \langle0,\mathrm{in}|
  a_{\mathbf k}^{\mathrm{out}\,*}a_{\mathbf k}^{\mathrm{out}}
  |0,\mathrm{in}\rangle
  =
  |\beta_k|^2
  \label{eq:cosmological-particle-number}
\end{equation}
in finite volume, with the usual delta-function density normalization in
infinite volume.
The Wronskian
\[
  W(f,g)=f g'-f'g
\]
is conserved for any two solutions of
\(\chi''+\Omega_k^2\chi=0\).  The normalization
\eqref{eq:mode-wronskian-normalization} gives
\[
  W(\chi_k^{\mathrm{out}},\overline{\chi_k^{\mathrm{out}}})=\ii,
  \qquad
  W(\overline{\chi_k^{\mathrm{out}}},\chi_k^{\mathrm{out}})=-\ii.
  \]
Substitution of \eqref{eq:cosmological-bogoliubov-expansion} therefore gives
\[
  \ii
  =
  W(\chi_k^{\mathrm{in}},\overline{\chi_k^{\mathrm{in}}})
  =
  \ii(|\alpha_k|^2-|\beta_k|^2),
  \]
which proves \eqref{eq:cosmological-bogoliubov-normalization}.

The field mode admits both decompositions
\[
  \widehat\chi_{\mathbf k}
  =
  a_{\mathbf k}^{\mathrm{out}}\chi_k^{\mathrm{out}}
  +
  a_{-\mathbf k}^{\mathrm{out}\,*}\overline{\chi_k^{\mathrm{out}}}
  =
  a_{\mathbf k}^{\mathrm{in}}\chi_k^{\mathrm{in}}
  +
  a_{-\mathbf k}^{\mathrm{in}\,*}\overline{\chi_k^{\mathrm{in}}}.
  \]
Comparing coefficients gives
\begin{equation}
  a_{\mathbf k}^{\mathrm{out}}
  =
  \overline{\alpha_k}\,a_{\mathbf k}^{\mathrm{in}}
  +
  \overline{\beta_k}\,a_{-\mathbf k}^{\mathrm{in}\,*},
  \label{eq:out-annihilator-in-terms-of-in}
\end{equation}
with the phase convention induced by
\eqref{eq:cosmological-bogoliubov-expansion}.  Since
\(a_{\mathbf k}^{\mathrm{in}}\ket{0,\mathrm{in}}=0\) and
\([a_{-\mathbf k}^{\mathrm{in}},a_{-\mathbf k}^{\mathrm{in}\,*}]=1\) in
finite volume, \eqref{eq:cosmological-particle-number} follows.

\begin{example}[Instantaneous frequency jump]
\label{ex:instantaneous-frequency-jump}
For a single oscillator mode whose frequency changes suddenly from
\(\omega_i\) to \(\omega_f\), continuity of \(\chi\) and \(\chi'\) at the
jump gives
\begin{equation}
  \alpha
  =
  \frac12\left(
  \sqrt{\frac{\omega_f}{\omega_i}}
  +
  \sqrt{\frac{\omega_i}{\omega_f}}
  \right),
  \qquad
  \beta
  =
  \frac12\left(
  \sqrt{\frac{\omega_f}{\omega_i}}
  -
  \sqrt{\frac{\omega_i}{\omega_f}}
  \right),
  \label{eq:sudden-quench-bogoliubov}
\end{equation}
and therefore
\begin{equation}
  |\beta|^2
  =
  \frac{(\omega_f-\omega_i)^2}{4\omega_i\omega_f}.
  \label{eq:sudden-quench-particle-number}
\end{equation}
The example is not a smooth cosmological spacetime; it is a normalization
check and a limiting model for a transition fast compared with the mode
period.
\end{example}

\section{Adiabatic States and Ultraviolet Control}

When no static asymptotic regions exist, one can still impose a
high-momentum condition on the positive-frequency modes.  The adiabatic
ansatz is
\begin{equation}
  \chi_k(\eta)
  =
  \frac{1}{\sqrt{2W_k(\eta)}}
  \exp\left(-\ii\int^\eta W_k(\eta')\,d\eta'\right),
  \qquad
  W_k(\eta)>0.
  \label{eq:adiabatic-mode-ansatz}
\end{equation}

\begin{remark}[Riccati equation for the adiabatic frequency]
\label{rem:adiabatic-frequency-riccati}
The ansatz \eqref{eq:adiabatic-mode-ansatz} solves
\eqref{eq:flrw-chi-mode-equation} exactly if and only if
\begin{equation}
  W_k^2
  =
  \Omega_k^2
  -\frac12\frac{W_k''}{W_k}
  +\frac34\left(\frac{W_k'}{W_k}\right)^2 .
  \label{eq:adiabatic-riccati-equation}
\end{equation}
\end{remark}

Write
\[
  \chi=(2W)^{-1/2}e^{-\ii\Theta},
  \qquad
  \Theta'=W.
  \]
Then
\[
  \frac{\chi'}{\chi}
  =
  -\frac12\frac{W'}{W}-\ii W
  \]
and
\[
  \frac{\chi''}{\chi}
  =
  -\ii W'
  -\frac12\left(\frac{W'}{W}\right)'
  +
  \left(
  -\frac12\frac{W'}{W}-\ii W
  \right)^2 .
  \]
The imaginary terms cancel, leaving
\[
  \frac{\chi''}{\chi}
  =
  -W^2
  -\frac12\frac{W''}{W}
  +\frac34\left(\frac{W'}{W}\right)^2 .
  \]
Thus \(\chi''+\Omega_k^2\chi=0\) is equivalent to
\eqref{eq:adiabatic-riccati-equation}.

The adiabatic expansion is obtained by assigning one derivative of
\(a(\eta)\) one adiabatic order and solving
\eqref{eq:adiabatic-riccati-equation} recursively.  To second adiabatic
order, if \(\omega_k=\Omega_k\) denotes the zeroth-order positive frequency,
\begin{equation}
  W_k
  =
  \omega_k
  -
  \frac{\omega_k''}{4\omega_k^2}
  +
  \frac{3(\omega_k')^2}{8\omega_k^3}
  +O(\partial_\eta^4).
  \label{eq:adiabatic-second-order-frequency}
\end{equation}
An adiabatic state of sufficiently high order is Hadamard for the scalar
field on Robertson--Walker spacetime.  Conversely, the point-splitting
subtraction for \(\langle T_{\mu\nu}\rangle\) has the same ultraviolet
content as the fourth adiabatic order subtraction in four spacetime
dimensions.  Equality here means equality of the local large-\(k\)
singular terms; finite local curvature counterterms remain the
renormalization freedom described in
Chapter~\ref{ch:point-splitting-stress-tensor-renormalization}.

\section{Detector Response}

Let \(x(\tau)\) be a timelike worldline parametrized by proper time, and let
\(\lambda\chi(\tau)\mu(\tau)\widehat\phi(x(\tau))\) be an Unruh--DeWitt
coupling with compactly supported switching function \(\chi\).  For a
detector energy gap \(E\), first-order perturbation theory gives
\begin{equation}
  P(E)
  =
  \lambda^2
  \int d\tau\,d\tau'\,
  e^{-\ii E(\tau-\tau')}
  \chi(\tau)\chi(\tau')
  W(x(\tau),x(\tau')),
  \label{eq:cosmological-detector-response}
\end{equation}
where \(W\) is the Wightman two-point function of the state.

\paragraph{Positivity of the switched detector response.}
If \(W\) is a positive-type two-point function, then
\[
  P(E)\geq0
  \]
for every real \(E\) and every smooth compactly supported switching function
\(\chi\).
The content is the positive-type property of the state applied to the detector
smearing.  Define the test function on
spacetime
\[
  f_E(x)
  =
  \int d\tau\,
  e^{-\ii E\tau}\chi(\tau)\delta(x,x(\tau))
  \]
in the distributional sense.  Smearing the two-point function gives
\[
  P(E)/\lambda^2
  =
  W(\overline{f_E},f_E).
  \]
Positive type of the quasifree state is precisely the condition
\(W(\overline f,f)\geq0\) for every test function \(f\).  The worldline
smearing can be obtained as the limit of ordinary compactly supported
spacetime test functions in a tubular neighborhood of the curve, so the same
positivity applies.  The approximation by ordinary test functions is part of
the mathematical meaning of the idealized pointlike detector; a finite-size
detector replaces the delta distribution by a smooth spatial profile and
satisfies positivity directly.

Detector response is therefore a local operational probe of a chosen state.
Its definition uses the state two-point distribution, the detector trajectory,
and the switching/profile data.

\section{The Bunch--Davies State in de Sitter Space}

In the spatially flat patch of \(d\)-dimensional de Sitter space,
\begin{equation}
  a(\eta)=\frac{-1}{H\eta},
  \qquad
  \eta<0,
  \qquad
  H>0,
  \label{eq:de-sitter-flat-patch-scale-factor}
\end{equation}
and \(R=d(d-1)H^2\).  Equation
\eqref{eq:flrw-effective-frequency} becomes
\begin{equation}
  \Omega_k^2
  =
  k^2
  -
  \frac{\nu^2-\frac14}{\eta^2},
  \qquad
  \nu^2
  =
  \frac{(d-1)^2}{4}
  -
  \frac{m^2}{H^2}
  -
  \xi d(d-1).
  \label{eq:de-sitter-nu-definition}
\end{equation}

\begin{remark}[Bunch--Davies mode functions]
\label{rem:bunch-davies-hankel-modes}
The mode
\begin{equation}
  \chi_k^{\mathrm{BD}}(\eta)
  =
  \sqrt{\frac{-\pi\eta}{4}}\,
  \exp\!\left[\ii\pi\left(\frac{\nu}{2}+\frac14\right)\right]
  H_\nu^{(1)}(-k\eta)
  \label{eq:bunch-davies-hankel-mode}
\end{equation}
solves the de Sitter mode equation and obeys
\begin{equation}
  \chi_k^{\mathrm{BD}}(\eta)
  \sim
  \frac{e^{-\ii k\eta}}{\sqrt{2k}}
  \qquad
  \eta\to-\infty .
  \label{eq:bunch-davies-past-asymptotic}
\end{equation}
It is normalized by
\begin{equation}
  \chi_k^{\mathrm{BD}}\overline{(\chi_k^{\mathrm{BD}})'}
  -
  (\chi_k^{\mathrm{BD}})'\overline{\chi_k^{\mathrm{BD}}}
  =
  \ii.
  \label{eq:bunch-davies-wronskian}
\end{equation}
\end{remark}

With \(z=-k\eta>0\), equation
\[
  \chi_k''+
  \left(k^2-\frac{\nu^2-\frac14}{\eta^2}\right)\chi_k=0
  \]
is transformed by \(\chi_k(\eta)=(-\eta)^{1/2}f(z)\) into Bessel's equation
for \(f\) of order \(\nu\).  Thus \(H_\nu^{(1)}(z)\) gives a solution.
The large-\(z\) asymptotic formula
\[
  H_\nu^{(1)}(z)
  \sim
  \sqrt{\frac{2}{\pi z}}\,
  \exp\left[\ii\left(z-\frac{\pi\nu}{2}-\frac{\pi}{4}\right)\right]
  \]
turns \eqref{eq:bunch-davies-hankel-mode} into
\eqref{eq:bunch-davies-past-asymptotic}.  The Wronskian identity
\[
  H_\nu^{(1)}(z)\partial_zH_\nu^{(2)}(z)
  -
  \partial_zH_\nu^{(1)}(z)H_\nu^{(2)}(z)
  =
  -\frac{4\ii}{\pi z}
\]
then gives \eqref{eq:bunch-davies-wronskian}, with the additional minus sign
from \(z=-k\eta\).

For \(m=0\) and \(\xi=\xi_c\), \(\nu=1/2\) and the Bunch--Davies mode reduces
to the conformal Minkowski mode.  For a massless minimally coupled scalar in
\(d=4\), \(\nu=3/2\); the zero-mode and infrared behavior require separate
care and cannot be suppressed by the local high-frequency definition alone.

The Bunch--Davies quasifree state is Hadamard and invariant under the
connected de Sitter group for the standard massive range.  Other
de Sitter-invariant two-point functions must be tested against the Hadamard
short-distance condition; de Sitter invariance alone is not a state-selection
principle.

\section{Produced Stress Tensor In An Out Region}

The number \(|\beta_k|^2\) is not yet a source for gravity.  It becomes a
backreaction datum only after it is converted into the renormalized stress
tensor in a specified state and scheme.  In an asymptotically static future
region this conversion has a simple finite part which is useful because it
separates the produced-particle contribution from the vacuum-polarization and
curvature-counterterm pieces of \(\langle T_{\mu\nu}\rangle_{\rm ren}\).

\begin{controlledapproximation}[Out-region produced stress tensor]
\label{ca:cosmology-out-produced-stress-tensor}
Assume \(a(\eta)\to a_+\) in the future, with vanishing curvature derivatives
in the retained out window, and let
\begin{equation}
  \Omega_k^+=\sqrt{k^2+a_+^2m^2}
  \label{eq:cosmology-out-frequency}
\end{equation}
be the conformal-time out frequency.  Normal ordering with respect to the
out complex structure removes the out-vacuum term.  If the produced-particle
energy moment is finite, then the homogeneous produced contribution to the
future stress tensor in \(d\) spacetime dimensions is
\begin{align}
  \rho_\beta
  &=
  a_+^{-d}
  \int
  \frac{d^{d-1}\mathbf k}{(2\pi)^{d-1}}\,
  \Omega_k^+\,|\beta_k|^2,
  \label{eq:cosmology-produced-energy-density}
  \\
  P_\beta
  &=
  a_+^{-d}
  \int
  \frac{d^{d-1}\mathbf k}{(2\pi)^{d-1}}\,
  \frac{k^2}{(d-1)\Omega_k^+}\,|\beta_k|^2 .
  \label{eq:cosmology-produced-pressure}
\end{align}
The factor \(a_+^{-d}\) is not optional: \(a_+^{-(d-1)}\) converts comoving
momentum measure to physical momentum measure, and the additional \(a_+^{-1}\)
converts conformal energy \(\Omega_k^+\) to physical energy.  For a massless
field in the out region, \(P_\beta=\rho_\beta/(d-1)\).  For modes with
\(k^2\ll a_+^2m^2\), the pressure is suppressed relative to the energy
density.

The full stress tensor in the out region is not
\eqref{eq:cosmology-produced-energy-density} alone.  In a declared
renormalization scheme it has the structure
\begin{equation}
  \langle T_{\mu\nu}\rangle_{\rm ren}
  =
  T_{\mu\nu}^{\rm prod}[\beta]
  +
  T_{\mu\nu}^{\rm vac,sub}
  +
  T_{\mu\nu}^{\rm geom,fin}
  +
  T_{\mu\nu}^{\rm tail},
  \label{eq:cosmology-stress-produced-decomposition}
\end{equation}
where the last term records omitted nonstatic-window, endpoint, or adiabatic
matching remainders.  Thus particle production supplies a finite physical
source only after the fourth-order adiabatic or point-splitting subtraction,
finite local curvature coordinates, and tail bounds have also been fixed.

For a spatially flat homogeneous semiclassical Friedmann equation in the same
out window, write \(\kappa_d=8\pi G_d\).  Since
\[
  G_{tt}=\frac{(d-1)(d-2)}2 H_t^2 ,
\]
a retained small produced source changes the Hubble-square coordinate by
\begin{equation}
  \delta H_t^2
  =
  \frac{2\kappa_d}{(d-1)(d-2)}\,\delta\rho_\beta
  \label{eq:cosmology-produced-friedmann-response}
\end{equation}
before stability and state-transport residuals are included.  This is the
minimal bridge from Bogoliubov coefficients to semiclassical backreaction:
\(|\beta_k|^2\) gives a particle diagnostic, while
\(\rho_\beta\), the subtraction convention, and the Friedmann response window
give the gravitational source datum.
\end{controlledapproximation}

\begin{controlledapproximation}[Continuity certificate for produced
stress]
\label{ca:cosmology-produced-stress-continuity}
In a finite time-dependent window one often tracks a time-dependent
occupation number \(n_k(t)\) in a chosen adiabatic or asymptotic particle
basis.  This is a useful diagnostic only if the corresponding stress tensor
is also tracked.  Let \(t\) be physical Robertson--Walker time,
\[
  H=\frac{\dot a}{a},\qquad
  \Omega_k(t)=\sqrt{k^2+a(t)^2m^2}.
\]
In the diagonal particle approximation, after the same vacuum subtraction and
finite curvature counterterms have been fixed, define
\begin{align}
  \rho_n(t)
  &=
  a(t)^{-d}
  \int
  \frac{d^{d-1}\mathbf k}{(2\pi)^{d-1}}\,
  \Omega_k(t)n_k(t),
  \label{eq:cosmology-produced-time-dependent-rho}
  \\
  P_n(t)
  &=
  a(t)^{-d}
  \int
  \frac{d^{d-1}\mathbf k}{(2\pi)^{d-1}}\,
  \frac{k^2}{(d-1)\Omega_k(t)}n_k(t).
  \label{eq:cosmology-produced-time-dependent-pressure}
\end{align}
Then the diagonal produced stress obeys the exact finite-window identity
\begin{equation}
  \dot\rho_n+(d-1)H(\rho_n+P_n)
  =
  a^{-d}
  \int
  \frac{d^{d-1}\mathbf k}{(2\pi)^{d-1}}\,
  \Omega_k\,\dot n_k .
  \label{eq:cosmology-produced-continuity-source}
\end{equation}
Indeed \(\dot\Omega_k=H(\Omega_k-k^2/\Omega_k)\), so the redshift of the
physical volume, the change of the physical energy, and the pressure work
cancel exactly when \(n_k\) is fixed.  The right-hand side of
\eqref{eq:cosmology-produced-continuity-source} is therefore not an optional
correction: it is the stress-energy source carried by ongoing particle
production in the selected basis.

Consequently a time-dependent \(|\beta_k(t)|^2\) curve cannot be used as a
conserved matter component unless the residual
\begin{equation}
  E_{\rm cont}
  =
  \left|
  \dot\rho_n+(d-1)H(\rho_n+P_n)
  -
  a^{-d}
  \int
  \frac{d^{d-1}\mathbf k}{(2\pi)^{d-1}}\,
  \Omega_k\,\dot n_k
  \right|
  \label{eq:cosmology-produced-continuity-residual}
\end{equation}
is small after the omitted coherence terms, nonadiabatic basis-change terms,
vacuum-polarization pieces, and geometric counterterm contributions have been
assigned to the remainder in
\eqref{eq:cosmology-stress-produced-decomposition}.  In an out region
\(\dot n_k=0\), and
\eqref{eq:cosmology-produced-continuity-source} reduces to ordinary
redshifting of a free produced gas.  During production, the source term is the
quantity which must enter any retained Friedmann or semiclassical response
window.  Dropping \(P_n\), using only the number density, or replacing
\eqref{eq:cosmology-produced-continuity-source} by \(\dot\rho_n=0\) is a
different approximation and requires its own error bound.
\end{controlledapproximation}

\section{Backreaction Boundary}

The expectation value \(\langle T_{\mu\nu}\rangle\) in a time-dependent state
is defined by the renormalized stress-tensor prescription of
Chapter~\ref{ch:point-splitting-stress-tensor-renormalization}.  The
semiclassical Einstein equation
\begin{equation}
  G_{\mu\nu}+\Lambda g_{\mu\nu}
  =
  8\pi G\,\langle T_{\mu\nu}\rangle_{\mathrm{ren}}
  \label{eq:semiclassical-einstein-cosmology}
\end{equation}
is an equation for a classical metric and a quantum state.  The right-hand
side depends on the state, on the finite curvature counterterms, and on the
renormalization convention for the cosmological constant and Newton
constant.  Particle production by itself is not a closed backreaction
equation; it is one diagnostic extracted from the state when a particle
decomposition has been selected.

\begin{openproblem}[Interacting cosmological QFT]
\label{op:interacting-cosmological-qft}
Construct interacting QFT on cosmological spacetimes with Hadamard states,
renormalized stress tensor, detector response, particle-production
diagnostics where asymptotic regions exist, and controlled semiclassical
backreaction.  The construction should distinguish the local definition of
fields and states from the additional asymptotic structure needed to define
particles.
\end{openproblem}
